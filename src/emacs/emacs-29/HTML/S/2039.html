    48 /* If character code C has modifier masks, reflect them to the
    49    character code if possible.  Return the resulting code.  */
    50 
    51 EMACS_INT
    52 char_resolve_modifier_mask (EMACS_INT c)
    53 {
    54   /* A non-ASCII character can't reflect modifier bits to the code.  */
    55   if (! ASCII_CHAR_P ((c & ~CHAR_MODIFIER_MASK)))
    56     return c;
    57 
    58   /* For Meta, Shift, and Control modifiers, we need special care.  */
    59   if (c & CHAR_SHIFT)
    60     {
    61       /* Shift modifier is valid only with [A-Za-z].  */
    62       if ((c & 0377) >= 'A' && (c & 0377) <= 'Z')
    63         c &= ~CHAR_SHIFT;
    64       else if ((c & 0377) >= 'a' && (c & 0377) <= 'z')
    65         c = (c & ~CHAR_SHIFT) - ('a' - 'A');
    66       /* Shift modifier for control characters and SPC is ignored.  */
    67       else if ((c & ~CHAR_MODIFIER_MASK) <= 0x20)
    68         c &= ~CHAR_SHIFT;
    69     }
    70   if (c & CHAR_CTL)
    71     {
    72       /* Simulate the code in lread.c.  */
    73       /* Allow `\C- ' and `\C-?'.  */
    74       if ((c & 0377) == ' ')
    75         c &= ~0177 & ~ CHAR_CTL;
    76       else if ((c & 0377) == '?')
    77         c = 0177 | (c & ~0177 & ~CHAR_CTL);
    78       /* ASCII control chars are made from letters (both cases),
    79          as well as the non-letters within 0100...0137.  */
    80       else if ((c & 0137) >= 0101 && (c & 0137) <= 0132)
    81         c &= (037 | (~0177 & ~CHAR_CTL));
    82       else if ((c & 0177) >= 0100 && (c & 0177) <= 0137)
    83         c &= (037 | (~0177 & ~CHAR_CTL));
    84     }
    85 #if 0   /* This is outside the scope of this function.  (bug#4751)  */
    86   if (c & CHAR_META)
    87     {
    88       /* Move the meta bit to the right place for a string.  */
    89       c = (c & ~CHAR_META) | 0x80;
    90     }
    91 #endif
    92 
    93   return c;
    94 }

    98    handle them appropriately.  */

   100 int
   101 char_string (unsigned int c, unsigned char *p)
   102 {
   103   int bytes;
   104 
   105   if (c & CHAR_MODIFIER_MASK)
   106     {
   107       c = char_resolve_modifier_mask (c);
   108       /* If C still has any modifier bits, just ignore it.  */
   109       c &= ~CHAR_MODIFIER_MASK;
   110     }
   111 
   112   if (c <= MAX_3_BYTE_CHAR)
   113     {
   114       bytes = CHAR_STRING (c, p);
   115     }
   116   else if (c <= MAX_4_BYTE_CHAR)
   117     {
   118       p[0] = (0xF0 | (c >> 18));
   119       p[1] = (0x80 | ((c >> 12) & 0x3F));
   120       p[2] = (0x80 | ((c >> 6) & 0x3F));
   121       p[3] = (0x80 | (c & 0x3F));
   122       bytes = 4;
   123     }
   124   else if (c <= MAX_5_BYTE_CHAR)
   125     {
   126       p[0] = 0xF8;
   127       p[1] = (0x80 | ((c >> 18) & 0x0F));
   128       p[2] = (0x80 | ((c >> 12) & 0x3F));
   129       p[3] = (0x80 | ((c >> 6) & 0x3F));
   130       p[4] = (0x80 | (c & 0x3F));
   131       bytes = 5;
   132     }
   133   else if (c <= MAX_CHAR)
   134     {
   135       c = CHAR_TO_BYTE8 (c);
   136       bytes = BYTE8_STRING (c, p);
   137     }
   138   else
   139     error ("Invalid character: %x", c);
   140 
   141   return bytes;
   142 }

   145 /* Translate character C by translation table TABLE.  If no translation is
   146    found in TABLE, return the untranslated character.  If TABLE is a list,
   147    elements are char tables.  In that case, recursively translate C by all the
   148    tables in the list.  */
   149 
   150 int
   151 translate_char (Lisp_Object table, int c)
   152 {
   153   if (CHAR_TABLE_P (table))
   154     {
   155       Lisp_Object ch;
   156 
   157       ch = CHAR_TABLE_REF (table, c);
   158       if (CHARACTERP (ch))
   159         c = XFIXNUM (ch);
   160     }
   161   else
   162     {
   163       for (; CONSP (table); table = XCDR (table))
   164         c = translate_char (XCAR (table), c);
   165     }
   166   return c;
   167 }

   229 /* Return width (columns) of C considering the buffer display table DP. */
   230 
   231 static ptrdiff_t
   232 char_width (int c, struct Lisp_Char_Table *dp)
   233 {
   234   ptrdiff_t width = CHARACTER_WIDTH (c);
   235 
   236   if (dp)
   237     {
   238       Lisp_Object disp = DISP_CHAR_VECTOR (dp, c), ch;
   239       int i;
   240 
   241       if (VECTORP (disp))
   242         for (i = 0, width = 0; i < ASIZE (disp); i++)
   243           {
   244             int c = -1;
   245             ch = AREF (disp, i);
   246             if (GLYPH_CODE_P (ch))
   247               c = GLYPH_CODE_CHAR (ch);
   248             else if (CHARACTERP (ch))
   249               c = XFIXNUM (ch);
   250             if (c >= 0)
   251               {
   252                 int w = CHARACTER_WIDTH (c);
   253                 if (INT_ADD_WRAPV (width, w, &width))
   254                   string_overflow ();
   255               }
   256           }
   257     }
   258   return width;
   259 }

   278 /* Return width of string STR of length LEN when displayed in the
   279    current buffer.  The width is measured by how many columns it
   280    occupies on the screen.  If PRECISION > 0, return the width of
   281    longest substring that doesn't exceed PRECISION, and set number of
   282    characters and bytes of the substring in *NCHARS and *NBYTES
   283    respectively.  */
   284 
   285 ptrdiff_t
   286 c_string_width (const unsigned char *str, ptrdiff_t len, int precision,
   287                 ptrdiff_t *nchars, ptrdiff_t *nbytes)
   288 {
   289   ptrdiff_t i = 0, i_byte = 0;
   290   ptrdiff_t width = 0;
   291   struct Lisp_Char_Table *dp = buffer_display_table ();
   292 
   293   while (i_byte < len)
   294     {
   295       int bytes, c = string_char_and_length (str + i_byte, &bytes);
   296       ptrdiff_t thiswidth = char_width (c, dp);
   297 
   298       if (0 < precision && precision - width < thiswidth)
   299         {
   300           *nchars = i;
   301           *nbytes = i_byte;
   302           return width;
   303         }
   304       if (INT_ADD_WRAPV (thiswidth, width, &width))
   305         string_overflow ();
   306       i++;
   307       i_byte += bytes;
   308   }
   309 
   310   if (precision > 0)
   311     {
   312       *nchars = i;
   313       *nbytes = i_byte;
   314     }
   315 
   316   return width;
   317 }

   329 /* Return width of a (substring of a) Lisp string STRING when
   330    displayed in the current buffer.  The width is measured by how many
   331    columns it occupies on the screen while paying attention to
   332    compositions.  If PRECISION > 0, return the width of longest
   333    substring that doesn't exceed PRECISION, and set number of
   334    characters and bytes of the substring in *NCHARS and *NBYTES
   335    respectively.  FROM and TO are zero-based character indices that
   336    define the substring of STRING to consider.  If AUTO_COMP is
   337    non-zero, account for automatic compositions in STRING.  */
   338 
   339 ptrdiff_t
   340 lisp_string_width (Lisp_Object string, ptrdiff_t from, ptrdiff_t to,
   341                    ptrdiff_t precision, ptrdiff_t *nchars, ptrdiff_t *nbytes,
   342                    bool auto_comp)
   343 {
   344   /* This set multibyte to 0 even if STRING is multibyte when it
   345      contains only ascii and eight-bit-graphic, but that's
   346      intentional.  */
   347   bool multibyte = SCHARS (string) < SBYTES (string);
   348   ptrdiff_t i = from, i_byte = from ? string_char_to_byte (string, from) : 0;
   349   ptrdiff_t from_byte = i_byte;
   350   ptrdiff_t width = 0;
   351   struct Lisp_Char_Table *dp = buffer_display_table ();
   352 #ifdef HAVE_WINDOW_SYSTEM
   353   struct frame *f =
   354     (FRAMEP (selected_frame) && FRAME_LIVE_P (XFRAME (selected_frame)))
   355     ? XFRAME (selected_frame)
   356     : NULL;
   357   int font_width = -1;
   358   Lisp_Object default_font, frame_font;
   359 #endif
   360 
   361   eassert (precision <= 0 || (nchars && nbytes));
   362 
   363   while (i < to)
   364     {
   365       ptrdiff_t chars, bytes, thiswidth;
   366       Lisp_Object val;
   367       ptrdiff_t cmp_id;
   368       ptrdiff_t ignore, end;
   369 
   370       if (find_composition (i, -1, &ignore, &end, &val, string)
   371           && ((cmp_id = get_composition_id (i, i_byte, end - i, val, string))
   372               >= 0))
   373         {
   374           thiswidth = composition_table[cmp_id]->width;
   375           chars = end - i;
   376           bytes = string_char_to_byte (string, end) - i_byte;
   377         }
   378 #ifdef HAVE_WINDOW_SYSTEM
   379       else if (auto_comp
   380                && f && FRAME_WINDOW_P (f)
   381                && multibyte
   382                && find_automatic_composition (i, -1, i, &ignore,
   383                                               &end, &val, string)
   384                && end > i)
   385         {
   386           int j;
   387           for (j = 0; j < LGSTRING_GLYPH_LEN (val); j++)
   388             if (NILP (LGSTRING_GLYPH (val, j)))
   389               break;
   390 
   391           int pixelwidth = composition_gstring_width (val, 0, j, NULL);
   392 
   393           /* The below is somewhat expensive, so compute it only once
   394              for the entire loop, and only if needed.  */
   395           if (font_width < 0)
   396             {
   397               font_width = FRAME_COLUMN_WIDTH (f);
   398               default_font = Fface_font (Qdefault, Qnil, Qnil);
   399               frame_font = Fframe_parameter (Qnil, Qfont);
   400 
   401               if (STRINGP (default_font) && STRINGP (frame_font)
   402                   && (SCHARS (default_font) != SCHARS (frame_font)
   403                       || SBYTES (default_font) != SBYTES (frame_font)
   404                       || memcmp (SDATA (default_font), SDATA (frame_font),
   405                                  SBYTES (default_font))))
   406                 {
   407                   Lisp_Object font_info = Ffont_info (default_font, Qnil);
   408                   if (VECTORP (font_info))
   409                     {
   410                       font_width = XFIXNUM (AREF (font_info, 11));
   411                       if (font_width <= 0)
   412                         font_width = XFIXNUM (AREF (font_info, 10));
   413                     }
   414                 }
   415             }
   416           thiswidth = (double) pixelwidth / font_width + 0.5;
   417           chars = end - i;
   418           bytes = string_char_to_byte (string, end) - i_byte;
   419         }
   420 #endif  /* HAVE_WINDOW_SYSTEM */
   421       else
   422         {
   423           int c;
   424           unsigned char *str = SDATA (string);
   425 
   426           if (multibyte)
   427             {
   428               int cbytes;
   429               c = string_char_and_length (str + i_byte, &cbytes);
   430               bytes = cbytes;
   431             }
   432           else
   433             c = str[i_byte], bytes = 1;
   434           chars = 1;
   435           thiswidth = char_width (c, dp);
   436         }
   437 
   438       if (0 < precision && precision - width < thiswidth)
   439         {
   440           *nchars = i - from;
   441           *nbytes = i_byte - from_byte;
   442           return width;
   443         }
   444       if (INT_ADD_WRAPV (thiswidth, width, &width))
   445         string_overflow ();
   446       i += chars;
   447       i_byte += bytes;
   448     }
   449 
   450   if (precision > 0)
   451     {
   452       *nchars = i - from;
   453       *nbytes = i_byte - from_byte;
   454     }
   455 
   456   return width;
   457 }

   504    This works by looking at the contents and checking for multibyte
   505    sequences while assuming that there's no invalid sequence.  It
   506    ignores enable-multibyte-characters.  */

   508 ptrdiff_t
   509 multibyte_chars_in_text (const unsigned char *ptr, ptrdiff_t nbytes)
   510 {
   511   const unsigned char *endp = ptr + nbytes;
   512   ptrdiff_t chars = 0;
   513 
   514   while (ptr < endp)
   515     {
   516       int len = multibyte_length (ptr, endp, true, true);
   517 
   518       if (len == 0)
   519         emacs_abort ();
   520       ptr += len;
   521       chars++;
   522     }
   523 
   524   return chars;
   525 }

   527 /* Parse unibyte text at STR of LEN bytes as a multibyte text, count
   528    characters and bytes in it, and store them in *NCHARS and *NBYTES
   529    respectively.  On counting bytes, pay attention to that 8-bit
   530    characters not constructing a valid multibyte sequence are
   531    represented by 2-byte in a multibyte text.  */
   532 
   533 void
   534 parse_str_as_multibyte (const unsigned char *str, ptrdiff_t len,
   535                         ptrdiff_t *nchars, ptrdiff_t *nbytes)
   536 {
   537   const unsigned char *endp = str + len;
   538   ptrdiff_t chars = 0, bytes = 0;
   539 
   540   if (len >= MAX_MULTIBYTE_LENGTH)
   541     {
   542       const unsigned char *adjusted_endp = endp - (MAX_MULTIBYTE_LENGTH - 1);
   543       while (str < adjusted_endp)
   544         {
   545           int n = multibyte_length (str, NULL, false, false);
   546           if (0 < n)
   547             str += n, bytes += n;
   548           else
   549             str++, bytes += 2;
   550           chars++;
   551         }
   552     }
   553   while (str < endp)
   554     {
   555       int n = multibyte_length (str, endp, true, false);
   556       if (0 < n)
   557         str += n, bytes += n;
   558       else
   559         str++, bytes += 2;
   560       chars++;
   561     }
   562 
   563   *nchars = chars;
   564   *nbytes = bytes;
   565   return;
   566 }

   568 /* Arrange unibyte text at STR of NBYTES bytes as a multibyte text.
   569    It actually converts only such 8-bit characters that don't construct
   570    a multibyte sequence to multibyte forms of raw bytes.  If NCHARS
   571    is nonzero, set *NCHARS to the number of characters in the text.
   572    It is assured that we can use LEN bytes at STR as a work
   573    area and that is enough.  Return the number of bytes of the
   574    resulting text.  */
   575 
   576 ptrdiff_t
   577 str_as_multibyte (unsigned char *str, ptrdiff_t len, ptrdiff_t nbytes,
   578                   ptrdiff_t *nchars)
   579 {
   580   unsigned char *p = str, *endp = str + nbytes;
   581   unsigned char *to;
   582   ptrdiff_t chars = 0;
   583 
   584   if (nbytes >= MAX_MULTIBYTE_LENGTH)
   585     {
   586       unsigned char *adjusted_endp = endp - (MAX_MULTIBYTE_LENGTH - 1);
   587       while (p < adjusted_endp)
   588         {
   589           int n = multibyte_length (p, NULL, false, false);
   590           if (n <= 0)
   591             break;
   592           p += n, chars++;
   593         }
   594     }
   595   while (true)
   596     {
   597       int n = multibyte_length (p, endp, true, false);
   598       if (n <= 0)
   599         break;
   600       p += n, chars++;
   601     }
   602   if (nchars)
   603     *nchars = chars;
   604   if (p == endp)
   605     return nbytes;
   606 
   607   to = p;
   608   nbytes = endp - p;
   609   endp = str + len;
   610   memmove (endp - nbytes, p, nbytes);
   611   p = endp - nbytes;
   612 
   613   if (nbytes >= MAX_MULTIBYTE_LENGTH)
   614     {
   615       unsigned char *adjusted_endp = endp - (MAX_MULTIBYTE_LENGTH - 1);
   616       while (p < adjusted_endp)
   617         {
   618           int n = multibyte_length (p, NULL, false, false);
   619           if (0 < n)
   620             {
   621               while (n--)
   622                 *to++ = *p++;
   623             }
   624           else
   625             {
   626               int c = *p++;
   627               c = BYTE8_TO_CHAR (c);
   628               to += CHAR_STRING (c, to);
   629             }
   630         }
   631       chars++;
   632     }
   633   while (p < endp)
   634     {
   635       int n = multibyte_length (p, endp, true, false);
   636       if (0 < n)
   637         {
   638           while (n--)
   639             *to++ = *p++;
   640         }
   641       else
   642         {
   643           int c = *p++;
   644           c = BYTE8_TO_CHAR (c);
   645           to += CHAR_STRING (c, to);
   646         }
   647       chars++;
   648     }
   649   if (nchars)
   650     *nchars = chars;
   651   return (to - str);
   652 }

   654 /* Parse unibyte string at STR of LEN bytes, and return the number of
   655    bytes it may occupy when converted to multibyte string by
   656    `str_to_multibyte'.  */
   657 
   658 ptrdiff_t
   659 count_size_as_multibyte (const unsigned char *str, ptrdiff_t len)
   660 {
   661   /* Count the number of non-ASCII (raw) bytes, since they will occupy
   662      two bytes in a multibyte string.  */
   663   ptrdiff_t nonascii = 0;
   664   for (ptrdiff_t i = 0; i < len; i++)
   665     nonascii += str[i] >> 7;
   666   ptrdiff_t bytes;
   667   if (INT_ADD_WRAPV (len, nonascii, &bytes))
   668     string_overflow ();
   669   return bytes;
   670 }

   673 /* Convert unibyte text at SRC of NCHARS chars to a multibyte text
   674    at DST, that contains the same single-byte characters.
   675    Return the number of bytes written at DST.  */
   676 ptrdiff_t
   677 str_to_multibyte (unsigned char *dst, const unsigned char *src,
   678                   ptrdiff_t nchars)
   679 {
   680   unsigned char *d = dst;
   681   for (ptrdiff_t i = 0; i < nchars; i++)
   682     {
   683       unsigned char c = src[i];
   684       if (c <= 0x7f)
   685         *d++ = c;
   686       else
   687         {
   688           *d++ = 0xc0 + ((c >> 6) & 1);
   689           *d++ = 0x80 + (c & 0x3f);
   690         }
   691     }
   692   return d - dst;
   693 }

   695 /* Arrange multibyte text at STR of LEN bytes as a unibyte text.  It
   696    actually converts characters in the range 0x80..0xFF to
   697    unibyte.  */
   698 
   699 ptrdiff_t
   700 str_as_unibyte (unsigned char *str, ptrdiff_t bytes)
   701 {
   702   const unsigned char *p = str, *endp = str + bytes;
   703   unsigned char *to;
   704   int c, len;
   705 
   706   while (p < endp)
   707     {
   708       c = *p;
   709       len = BYTES_BY_CHAR_HEAD (c);
   710       if (CHAR_BYTE8_HEAD_P (c))
   711         break;
   712       p += len;
   713     }
   714   to = str + (p - str);
   715   while (p < endp)
   716     {
   717       c = *p;
   718       len = BYTES_BY_CHAR_HEAD (c);
   719       if (CHAR_BYTE8_HEAD_P (c))
   720         {
   721           c = string_char_advance (&p);
   722           *to++ = CHAR_TO_BYTE8 (c);
   723         }
   724       else
   725         {
   726           while (len--) *to++ = *p++;
   727         }
   728     }
   729   return (to - str);
   730 }

   732 static ptrdiff_t
   733 string_count_byte8 (Lisp_Object string)
   734 {
   735   bool multibyte = STRING_MULTIBYTE (string);
   736   ptrdiff_t nbytes = SBYTES (string);
   737   unsigned char *p = SDATA (string);
   738   unsigned char *pend = p + nbytes;
   739   ptrdiff_t count = 0;
   740   int c, len;
   741 
   742   if (multibyte)
   743     while (p < pend)
   744       {
   745         c = *p;
   746         len = BYTES_BY_CHAR_HEAD (c);
   747 
   748         if (CHAR_BYTE8_HEAD_P (c))
   749           count++;
   750         p += len;
   751       }
   752   else
   753     while (p < pend)
   754       {
   755         if (*p++ >= 0x80)
   756           count++;
   757       }
   758   return count;
   759 }

   762 Lisp_Object
   763 string_escape_byte8 (Lisp_Object string)
   764 {
   765   ptrdiff_t nchars = SCHARS (string);
   766   ptrdiff_t nbytes = SBYTES (string);
   767   bool multibyte = STRING_MULTIBYTE (string);
   768   ptrdiff_t byte8_count;
   769   ptrdiff_t thrice_byte8_count, uninit_nchars, uninit_nbytes;
   770   const unsigned char *src, *src_end;
   771   unsigned char *dst;
   772   Lisp_Object val;
   773   int c, len;
   774 
   775   if (multibyte && nchars == nbytes)
   776     return string;
   777 
   778   byte8_count = string_count_byte8 (string);
   779 
   780   if (byte8_count == 0)
   781     return string;
   782 
   783   if (INT_MULTIPLY_WRAPV (byte8_count, 3, &thrice_byte8_count))
   784     string_overflow ();
   785 
   786   if (multibyte)
   787     {
   788       /* Convert 2-byte sequence of byte8 chars to 4-byte octal.  */
   789       if (INT_ADD_WRAPV (nchars, thrice_byte8_count, &uninit_nchars)
   790           || INT_ADD_WRAPV (nbytes, 2 * byte8_count, &uninit_nbytes))
   791         string_overflow ();
   792       val = make_uninit_multibyte_string (uninit_nchars, uninit_nbytes);
   793     }
   794   else
   795     {
   796       /* Convert 1-byte sequence of byte8 chars to 4-byte octal.  */
   797       if (INT_ADD_WRAPV (thrice_byte8_count, nbytes, &uninit_nbytes))
   798         string_overflow ();
   799       val = make_uninit_string (uninit_nbytes);
   800     }
   801 
   802   src = SDATA (string);
   803   src_end = src + nbytes;
   804   dst = SDATA (val);
   805   if (multibyte)
   806     while (src < src_end)
   807       {
   808         c = *src;
   809         len = BYTES_BY_CHAR_HEAD (c);
   810 
   811         if (CHAR_BYTE8_HEAD_P (c))
   812           {
   813             c = string_char_advance (&src);
   814             c = CHAR_TO_BYTE8 (c);
   815             dst += sprintf ((char *) dst, "\\%03o", c + 0u);
   816           }
   817         else
   818           while (len--) *dst++ = *src++;
   819       }
   820   else
   821     while (src < src_end)
   822       {
   823         c = *src++;
   824         if (c >= 0x80)
   825           dst += sprintf ((char *) dst, "\\%03o", c + 0u);
   826         else
   827           *dst++ = c;
   828       }
   829   return val;
   830 }
