   158 /* Re-define FD_ISSET to avoid a WSA call while we are not using
   159    network sockets.  */
   160 static int
   161 rpl_fd_isset (SOCKET fd, fd_set * set)
   162 {
   163   u_int i;
   164   if (set == NULL)
   165     return 0;
   166 
   167   for (i = 0; i < set->fd_count; i++)
   168     if (set->fd_array[i] == fd)
   169       return 1;
   170 
   171   return 0;
   172 }
