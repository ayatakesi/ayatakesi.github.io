    57 int
    58 rpl_faccessat (int fd, char const *file, int mode, int flag)
    59 {
    60   int result = orig_faccessat (fd, file, mode, flag);
    61 
    62   if (result == 0 && file[strlen (file) - 1] == '/')
    63     {
    64       struct stat st;
    65       result = fstatat (fd, file, &st, 0);
    66       if (result == 0 && !S_ISDIR (st.st_mode))
    67         {
    68           errno = ENOTDIR;
    69           return -1;
    70         }
    71     }
    72 
    73   return result;
    74 }
