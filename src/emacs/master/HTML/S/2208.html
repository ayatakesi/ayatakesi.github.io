    59 /* Return a random Lisp fixnum I in the range 0 <= I < LIM,
    60    where LIM is taken from a positive fixnum.  */
    61 static Lisp_Object
    62 get_random_fixnum (EMACS_INT lim)
    63 {
    64   /* Return the remainder of a random integer R (in range 0..INTMASK)
    65      divided by LIM, except reject the rare case where R is so close
    66      to INTMASK that the remainder isn't random.  */
    67   EMACS_INT difflim = INTMASK - lim + 1, diff, remainder;
    68   do
    69     {
    70       EMACS_INT r = get_random ();
    71       remainder = r % lim;
    72       diff = r - remainder;
    73     }
    74   while (difflim < diff);
    75 
    76   return make_fixnum (remainder);
    77 }

   175 static inline
   176 EMACS_INT length_internal (Lisp_Object sequence, int len)
   177 {
   178   /* If LENGTH is short (arbitrarily chosen cut-off point), use a
   179      fast loop that doesn't care about whether SEQUENCE is
   180      circular or not. */
   181   if (len < 0xffff)
   182     while (CONSP (sequence))
   183       {
   184         if (--len <= 0)
   185           return -1;
   186         sequence = XCDR (sequence);
   187       }
   188   /* Signal an error on circular lists. */
   189   else
   190     FOR_EACH_TAIL (sequence)
   191       if (--len <= 0)
   192         return -1;
   193   return len;
   194 }

   361 DEFUN ("compare-strings", Fcompare_strings, Scompare_strings, 6, 7, 0,
   362        doc: /* Compare the contents of two strings, converting to multibyte if needed.
   363 The arguments START1, END1, START2, and END2, if non-nil, are
   364 positions specifying which parts of STR1 or STR2 to compare.  In
   365 string STR1, compare the part between START1 (inclusive) and END1
   366 \(exclusive).  If START1 is nil, it defaults to 0, the beginning of
   367 the string; if END1 is nil, it defaults to the length of the string.
   368 Likewise, in string STR2, compare the part between START2 and END2.
   369 Like in `substring', negative values are counted from the end.
   370 
   371 The strings are compared by the numeric values of their characters.
   372 For instance, STR1 is "less than" STR2 if its first differing
   373 character has a smaller numeric value.  If IGNORE-CASE is non-nil,
   374 characters are converted to upper-case before comparing them.  Unibyte
   375 strings are converted to multibyte for comparison.
   376 
   377 The value is t if the strings (or specified portions) match.
   378 If string STR1 is less, the value is a negative number N;
   379   - 1 - N is the number of characters that match at the beginning.
   380 If string STR1 is greater, the value is a positive number N;
   381   N - 1 is the number of characters that match at the beginning.  */)
   382   (Lisp_Object str1, Lisp_Object start1, Lisp_Object end1, Lisp_Object str2,
   383    Lisp_Object start2, Lisp_Object end2, Lisp_Object ignore_case)
   384 {
   385   ptrdiff_t from1, to1, from2, to2, i1, i1_byte, i2, i2_byte;
   386 
   387   CHECK_STRING (str1);
   388   CHECK_STRING (str2);
   389 
   390   /* For backward compatibility, silently bring too-large positive end
   391      values into range.  */
   392   if (FIXNUMP (end1) && SCHARS (str1) < XFIXNUM (end1))
   393     end1 = make_fixnum (SCHARS (str1));
   394   if (FIXNUMP (end2) && SCHARS (str2) < XFIXNUM (end2))
   395     end2 = make_fixnum (SCHARS (str2));
   396 
   397   validate_subarray (str1, start1, end1, SCHARS (str1), &from1, &to1);
   398   validate_subarray (str2, start2, end2, SCHARS (str2), &from2, &to2);
   399 
   400   i1 = from1;
   401   i2 = from2;
   402 
   403   i1_byte = string_char_to_byte (str1, i1);
   404   i2_byte = string_char_to_byte (str2, i2);
   405 
   406   while (i1 < to1 && i2 < to2)
   407     {
   408       /* When we find a mismatch, we must compare the
   409          characters, not just the bytes.  */
   410       int c1 = fetch_string_char_as_multibyte_advance (str1, &i1, &i1_byte);
   411       int c2 = fetch_string_char_as_multibyte_advance (str2, &i2, &i2_byte);
   412 
   413       if (c1 == c2)
   414         continue;
   415 
   416       if (! NILP (ignore_case))
   417         {
   418           c1 = XFIXNUM (Fupcase (make_fixnum (c1)));
   419           c2 = XFIXNUM (Fupcase (make_fixnum (c2)));
   420         }
   421 
   422       if (c1 == c2)
   423         continue;
   424 
   425       /* Note that I1 has already been incremented
   426          past the character that we are comparing;
   427          hence we don't add or subtract 1 here.  */
   428       if (c1 < c2)
   429         return make_fixnum (- i1 + from1);
   430       else
   431         return make_fixnum (i1 - from1);
   432     }
   433 
   434   if (i1 < to1)
   435     return make_fixnum (i1 - from1 + 1);
   436   if (i2 < to2)
   437     return make_fixnum (- i1 + from1 - 1);
   438 
   439   return Qt;
   440 }

   513           const char *w1 = SSDATA (string1);
   514           const char *w2 = SSDATA (string2);
   515           while (b < nb - ws + 1 &&    load_unaligned_size_t (w1 + b)
   516                                     == load_unaligned_size_t (w2 + b))
   517             b += ws;
   518         }
   519 
   520       /* Scan forward to the differing byte.  */
   521       while (b < nb && SREF (string1, b) == SREF (string2, b))
   522         b++;
   523 
   524       if (b >= nb)
   525         /* One string is a prefix of the other.  */
   526         return b < nb2 ? Qt : Qnil;
   527 
   528       /* Now back up to the start of the differing characters:
   529          it's the last byte not having the bit pattern 10xxxxxx.  */
   530       while ((SREF (string1, b) & 0xc0) == 0x80)
   531         b--;
   532 
   533       /* Compare the differing characters.  */
   534       ptrdiff_t i1 = 0, i2 = 0;
   535       ptrdiff_t i1_byte = b, i2_byte = b;
   536       int c1 = fetch_string_char_advance_no_check (string1, &i1, &i1_byte);
   537       int c2 = fetch_string_char_advance_no_check (string2, &i2, &i2_byte);
   538       return c1 < c2 ? Qt : Qnil;
   539     }
   540   else if (STRING_MULTIBYTE (string1))
   541     {
   542       /* string1 multibyte, string2 unibyte */
   543       ptrdiff_t i1 = 0, i1_byte = 0, i2 = 0;
   544       while (i1 < n)
   545         {
   546           int c1 = fetch_string_char_advance_no_check (string1, &i1, &i1_byte);
   547           int c2 = SREF (string2, i2++);
   548           if (c1 != c2)
   549             return c1 < c2 ? Qt : Qnil;
   550         }
   551       return i1 < SCHARS (string2) ? Qt : Qnil;
   552     }
   553   else
   554     {
   555       /* string1 unibyte, string2 multibyte */
   556       ptrdiff_t i1 = 0, i2 = 0, i2_byte = 0;
   557       while (i1 < n)
   558         {
   559           int c1 = SREF (string1, i1++);
   560           int c2 = fetch_string_char_advance_no_check (string2, &i2, &i2_byte);
   561           if (c1 != c2)
   562             return c1 < c2 ? Qt : Qnil;
   563         }
   564       return i1 < SCHARS (string2) ? Qt : Qnil;
   565     }
   566 }

   811    that is a string and has text properties to be copied.  */

   818 static Lisp_Object
   819 concat_to_string (ptrdiff_t nargs, Lisp_Object *args)
   820 {
   821   USE_SAFE_ALLOCA;
   822 
   823   /* Check types and compute total length in chars of arguments in RESULT_LEN,
   824      length in bytes in RESULT_LEN_BYTE, and determine in DEST_MULTIBYTE
   825      whether the result should be a multibyte string.  */
   826   EMACS_INT result_len = 0;
   827   EMACS_INT result_len_byte = 0;
   828   bool dest_multibyte = false;
   829   bool some_unibyte = false;
   830   for (ptrdiff_t i = 0; i < nargs; i++)
   831     {
   832       Lisp_Object arg = args[i];
   833       EMACS_INT len;
   834 
   835       /* We must count the number of bytes needed in the string
   836          as well as the number of characters.  */
   837 
   838       if (STRINGP (arg))
   839         {
   840           ptrdiff_t arg_len_byte = SBYTES (arg);
   841           len = SCHARS (arg);
   842           if (STRING_MULTIBYTE (arg))
   843             dest_multibyte = true;
   844           else
   845             some_unibyte = true;
   846           if (STRING_BYTES_BOUND - result_len_byte < arg_len_byte)
   847             string_overflow ();
   848           result_len_byte += arg_len_byte;
   849         }
   850       else if (VECTORP (arg))
   851         {
   852           len = ASIZE (arg);
   853           ptrdiff_t arg_len_byte = 0;
   854           for (ptrdiff_t j = 0; j < len; j++)
   855             {
   856               Lisp_Object ch = AREF (arg, j);
   857               CHECK_CHARACTER (ch);
   858               int c = XFIXNAT (ch);
   859               arg_len_byte += CHAR_BYTES (c);
   860               if (!ASCII_CHAR_P (c) && !CHAR_BYTE8_P (c))
   861                 dest_multibyte = true;
   862             }
   863           if (STRING_BYTES_BOUND - result_len_byte < arg_len_byte)
   864             string_overflow ();
   865           result_len_byte += arg_len_byte;
   866         }
   867       else if (NILP (arg))
   868         continue;
   869       else if (CONSP (arg))
   870         {
   871           len = XFIXNAT (Flength (arg));
   872           ptrdiff_t arg_len_byte = 0;
   873           for (; CONSP (arg); arg = XCDR (arg))
   874             {
   875               Lisp_Object ch = XCAR (arg);
   876               CHECK_CHARACTER (ch);
   877               int c = XFIXNAT (ch);
   878               arg_len_byte += CHAR_BYTES (c);
   879               if (!ASCII_CHAR_P (c) && !CHAR_BYTE8_P (c))
   880                 dest_multibyte = true;
   881             }
   882           if (STRING_BYTES_BOUND - result_len_byte < arg_len_byte)
   883             string_overflow ();
   884           result_len_byte += arg_len_byte;
   885         }
   886       else
   887         wrong_type_argument (Qsequencep, arg);
   888 
   889       result_len += len;
   890       if (MOST_POSITIVE_FIXNUM < result_len)
   891         memory_full (SIZE_MAX);
   892     }
   893 
   894   if (dest_multibyte && some_unibyte)
   895     {
   896       /* Non-ASCII characters in unibyte strings take two bytes when
   897          converted to multibyte -- count them and adjust the total.  */
   898       for (ptrdiff_t i = 0; i < nargs; i++)
   899         {
   900           Lisp_Object arg = args[i];
   901           if (STRINGP (arg) && !STRING_MULTIBYTE (arg))
   902             {
   903               ptrdiff_t bytes = SCHARS (arg);
   904               const unsigned char *s = SDATA (arg);
   905               ptrdiff_t nonascii = 0;
   906               for (ptrdiff_t j = 0; j < bytes; j++)
   907                 nonascii += s[j] >> 7;
   908               if (STRING_BYTES_BOUND - result_len_byte < nonascii)
   909                 string_overflow ();
   910               result_len_byte += nonascii;
   911             }
   912         }
   913     }
   914 
   915   if (!dest_multibyte)
   916     result_len_byte = result_len;
   917 
   918   /* Create the output object.  */
   919   Lisp_Object result = dest_multibyte
   920     ? make_uninit_multibyte_string (result_len, result_len_byte)
   921     : make_uninit_string (result_len);
   922 
   923   /* Copy the contents of the args into the result.  */
   924   ptrdiff_t toindex = 0;
   925   ptrdiff_t toindex_byte = 0;
   926 
   927   /* When we make a multibyte string, we can't copy text properties
   928      while concatenating each string because the length of resulting
   929      string can't be decided until we finish the whole concatenation.
   930      So, we record strings that have text properties to be copied
   931      here, and copy the text properties after the concatenation.  */
   932   struct textprop_rec *textprops;
   933   /* Number of elements in textprops.  */
   934   ptrdiff_t num_textprops = 0;
   935   SAFE_NALLOCA (textprops, 1, nargs);
   936 
   937   for (ptrdiff_t i = 0; i < nargs; i++)
   938     {
   939       Lisp_Object arg = args[i];
   940       if (STRINGP (arg))
   941         {
   942           if (string_intervals (arg))
   943             {
   944               textprops[num_textprops].argnum = i;
   945               textprops[num_textprops].to = toindex;
   946               num_textprops++;
   947             }
   948           ptrdiff_t nchars = SCHARS (arg);
   949           if (STRING_MULTIBYTE (arg) == dest_multibyte)
   950             {
   951               /* Between strings of the same kind, copy fast.  */
   952               ptrdiff_t arg_len_byte = SBYTES (arg);
   953               memcpy (SDATA (result) + toindex_byte, SDATA (arg), arg_len_byte);
   954               toindex_byte += arg_len_byte;
   955             }
   956           else
   957             {
   958               /* Copy a single-byte string to a multibyte string.  */
   959               toindex_byte += str_to_multibyte (SDATA (result) + toindex_byte,
   960                                                 SDATA (arg), nchars);
   961             }
   962           toindex += nchars;
   963         }
   964       else if (VECTORP (arg))
   965         {
   966           ptrdiff_t len = ASIZE (arg);
   967           for (ptrdiff_t j = 0; j < len; j++)
   968             {
   969               int c = XFIXNAT (AREF (arg, j));
   970               if (dest_multibyte)
   971                 toindex_byte += CHAR_STRING (c, SDATA (result) + toindex_byte);
   972               else
   973                 SSET (result, toindex_byte++, c);
   974               toindex++;
   975             }
   976         }
   977       else
   978         for (Lisp_Object tail = arg; !NILP (tail); tail = XCDR (tail))
   979           {
   980             int c = XFIXNAT (XCAR (tail));
   981             if (dest_multibyte)
   982               toindex_byte += CHAR_STRING (c, SDATA (result) + toindex_byte);
   983             else
   984               SSET (result, toindex_byte++, c);
   985             toindex++;
   986           }
   987     }
   988 
   989   if (num_textprops > 0)
   990     {
   991       ptrdiff_t last_to_end = -1;
   992       for (ptrdiff_t i = 0; i < num_textprops; i++)
   993         {
   994           Lisp_Object arg = args[textprops[i].argnum];
   995           Lisp_Object props = text_property_list (arg,
   996                                                   make_fixnum (0),
   997                                                   make_fixnum (SCHARS (arg)),
   998                                                   Qnil);
   999           /* If successive arguments have properties, be sure that the
  1000              value of `composition' property be the copy.  */
  1001           if (last_to_end == textprops[i].to)
  1002             make_composition_value_copy (props);
  1003           add_text_properties_from_list (result, props,
  1004                                          make_fixnum (textprops[i].to));
  1005           last_to_end = textprops[i].to + SCHARS (arg);
  1006         }
  1007     }
  1008 
  1009   SAFE_FREE ();
  1010   return result;
  1011 }

  1013 /* Concatenate sequences into a list. */
  1014 Lisp_Object
  1015 concat_to_list (ptrdiff_t nargs, Lisp_Object *args, Lisp_Object last_tail)
  1016 {
  1017   /* Copy the contents of the args into the result.  */
  1018   Lisp_Object result = Qnil;
  1019   Lisp_Object last = Qnil;      /* Last cons in result if nonempty.  */
  1020 
  1021   for (ptrdiff_t i = 0; i < nargs; i++)
  1022     {
  1023       Lisp_Object arg = args[i];
  1024       /* List arguments are treated specially since this is the common case.  */
  1025       if (CONSP (arg))
  1026         {
  1027           Lisp_Object head = Fcons (XCAR (arg), Qnil);
  1028           Lisp_Object prev = head;
  1029           arg = XCDR (arg);
  1030           FOR_EACH_TAIL (arg)
  1031             {
  1032               Lisp_Object next = Fcons (XCAR (arg), Qnil);
  1033               XSETCDR (prev, next);
  1034               prev = next;
  1035             }
  1036           CHECK_LIST_END (arg, arg);
  1037           if (NILP (result))
  1038             result = head;
  1039           else
  1040             XSETCDR (last, head);
  1041           last = prev;
  1042         }
  1043       else if (NILP (arg))
  1044         ;
  1045       else if (VECTORP (arg) || STRINGP (arg)
  1046                || BOOL_VECTOR_P (arg) || COMPILEDP (arg))
  1047         {
  1048           ptrdiff_t arglen = XFIXNUM (Flength (arg));
  1049           ptrdiff_t argindex_byte = 0;
  1050 
  1051           /* Copy element by element.  */
  1052           for (ptrdiff_t argindex = 0; argindex < arglen; argindex++)
  1053             {
  1054               /* Fetch next element of `arg' arg into `elt', or break if
  1055                  `arg' is exhausted. */
  1056               Lisp_Object elt;
  1057               if (STRINGP (arg))
  1058                 {
  1059                   int c;
  1060                   if (STRING_MULTIBYTE (arg))
  1061                     {
  1062                       ptrdiff_t char_idx = argindex;
  1063                       c = fetch_string_char_advance_no_check (arg, &char_idx,
  1064                                                               &argindex_byte);
  1065                     }
  1066                   else
  1067                     c = SREF (arg, argindex);
  1068                   elt = make_fixed_natnum (c);
  1069                 }
  1070               else if (BOOL_VECTOR_P (arg))
  1071                 elt = bool_vector_ref (arg, argindex);
  1072               else
  1073                 elt = AREF (arg, argindex);
  1074

  1096 /* Concatenate sequences into a vector.  */
  1097 Lisp_Object
  1098 concat_to_vector (ptrdiff_t nargs, Lisp_Object *args)
  1099 {
  1100   /* Check argument types and compute total length of arguments.  */
  1101   EMACS_INT result_len = 0;
  1102   for (ptrdiff_t i = 0; i < nargs; i++)
  1103     {
  1104       Lisp_Object arg = args[i];
  1105       if (!(VECTORP (arg) || CONSP (arg) || NILP (arg) || STRINGP (arg)
  1106             || BOOL_VECTOR_P (arg) || COMPILEDP (arg)))
  1107         wrong_type_argument (Qsequencep, arg);
  1108       EMACS_INT len = XFIXNAT (Flength (arg));
  1109       result_len += len;
  1110       if (MOST_POSITIVE_FIXNUM < result_len)
  1111         memory_full (SIZE_MAX);
  1112     }
  1113 
  1114   /* Create the output vector.  */
  1115   Lisp_Object result = make_uninit_vector (result_len);
  1116   Lisp_Object *dst = XVECTOR (result)->contents;
  1117 
  1118   /* Copy the contents of the args into the result.  */
  1119 
  1120   for (ptrdiff_t i = 0; i < nargs; i++)
  1121     {
  1122       Lisp_Object arg = args[i];
  1123       if (VECTORP (arg))
  1124         {
  1125           ptrdiff_t size = ASIZE (arg);
  1126           memcpy (dst, XVECTOR (arg)->contents, size * sizeof *dst);
  1127           dst += size;
  1128         }
  1129       else if (CONSP (arg))
  1130         do
  1131           {
  1132             *dst++ = XCAR (arg);
  1133             arg = XCDR (arg);
  1134           }
  1135         while (!NILP (arg));
  1136       else if (NILP (arg))
  1137         ;
  1138       else if (STRINGP (arg))
  1139         {
  1140           ptrdiff_t size = SCHARS (arg);
  1141           if (STRING_MULTIBYTE (arg))
  1142             {
  1143               ptrdiff_t byte = 0;
  1144               for (ptrdiff_t i = 0; i < size;)
  1145                 {
  1146                   int c = fetch_string_char_advance_no_check (arg, &i, &byte);
  1147                   *dst++ = make_fixnum (c);
  1148                 }
  1149             }
  1150           else
  1151             for (ptrdiff_t i = 0; i < size; i++)
  1152               *dst++ = make_fixnum (SREF (arg, i));
  1153         }
  1154       else if (BOOL_VECTOR_P (arg))
  1155         {
  1156           ptrdiff_t size = bool_vector_size (arg);
  1157           for (ptrdiff_t i = 0; i < size; i++)
  1158             *dst++ = bool_vector_ref (arg, i);
  1159         }
  1160       else
  1161         {
  1162           eassert (COMPILEDP (arg));
  1163           ptrdiff_t size = PVSIZE (arg);
  1164           memcpy (dst, XVECTOR (arg)->contents, size * sizeof *dst);
  1165           dst += size;
  1166         }
  1167     }
  1168   eassert (dst == XVECTOR (result)->contents + result_len);
  1169 
  1170   return result;
  1171 }

  1183 /* Return the byte index corresponding to CHAR_INDEX in STRING.  */
  1184 
  1185 ptrdiff_t
  1186 string_char_to_byte (Lisp_Object string, ptrdiff_t char_index)
  1187 {
  1188   ptrdiff_t i_byte;
  1189   ptrdiff_t best_below, best_below_byte;
  1190   ptrdiff_t best_above, best_above_byte;
  1191 
  1192   best_below = best_below_byte = 0;
  1193   best_above = SCHARS (string);
  1194   best_above_byte = SBYTES (string);
  1195   if (best_above == best_above_byte)
  1196     return char_index;
  1197 
  1198   if (BASE_EQ (string, string_char_byte_cache_string))
  1199     {
  1200       if (string_char_byte_cache_charpos < char_index)
  1201         {
  1202           best_below = string_char_byte_cache_charpos;
  1203           best_below_byte = string_char_byte_cache_bytepos;
  1204         }
  1205       else
  1206         {
  1207           best_above = string_char_byte_cache_charpos;
  1208           best_above_byte = string_char_byte_cache_bytepos;
  1209         }
  1210     }
  1211 
  1212   if (char_index - best_below < best_above - char_index)
  1213     {
  1214       unsigned char *p = SDATA (string) + best_below_byte;
  1215 
  1216       while (best_below < char_index)
  1217         {
  1218           p += BYTES_BY_CHAR_HEAD (*p);
  1219           best_below++;
  1220         }
  1221       i_byte = p - SDATA (string);
  1222     }
  1223   else
  1224     {
  1225       unsigned char *p = SDATA (string) + best_above_byte;
  1226 
  1227       while (best_above > char_index)
  1228         {
  1229           p--;
  1230           while (!CHAR_HEAD_P (*p)) p--;
  1231           best_above--;
  1232         }
  1233       i_byte = p - SDATA (string);
  1234     }
  1235 
  1236   string_char_byte_cache_bytepos = i_byte;
  1237   string_char_byte_cache_charpos = char_index;
  1238   string_char_byte_cache_string = string;

  1241 }

  1740        doc: /* Take cdr N times on LIST, return the result.  */)
  1741   (Lisp_Object n, Lisp_Object list)

  1743   Lisp_Object tail = list;

  1748      positive bignum while counting down.  It does not introduce
  1749      miscounts because a list or cycle cannot possibly be this long,
  1750      and any counting error is fixed up later.  */

  1762             if (! CONSP (tail))

  1768         }

  1782          update TORTOISE_NUM accordingly.  */

  1784         tortoise_num = num;

  1786       saved_tail = XCDR (tail);
  1787       num--;
  1788       if (num == 0)
  1789         return saved_tail;
  1790       rarely_quit (num);
  1791     }
  1792 
  1793   tail = saved_tail;

  1795     {
  1796       CHECK_LIST_END (tail, list);

  1798     }

  2736    gets too deep.  HT is a hash table used to detect cycles; if nil,

  2738    if EQUAL_KIND == EQUAL_NO_QUIT.  */

  2740 static bool
  2741 internal_equal (Lisp_Object o1, Lisp_Object o2, enum equal_kind equal_kind,
  2742                 int depth, Lisp_Object ht)
  2743 {
  2744  tail_recurse:
  2745   if (depth > 10)
  2746     {
  2747       eassert (equal_kind != EQUAL_NO_QUIT);
  2748       if (depth > 200)
  2749         error ("Stack overflow in equal");
  2750       if (NILP (ht))
  2751         ht = CALLN (Fmake_hash_table, QCtest, Qeq);
  2752       switch (XTYPE (o1))
  2753         {
  2754         case Lisp_Cons: case Lisp_Vectorlike:
  2755           {
  2756             struct Lisp_Hash_Table *h = XHASH_TABLE (ht);
  2757             Lisp_Object hash;
  2758             ptrdiff_t i = hash_lookup (h, o1, &hash);
  2759             if (i >= 0)
  2760               { /* `o1' was seen already.  */
  2761                 Lisp_Object o2s = HASH_VALUE (h, i);
  2762                 if (!NILP (Fmemq (o2, o2s)))
  2763                   return true;
  2764                 else
  2765                   set_hash_value_slot (h, i, Fcons (o2, o2s));
  2766               }
  2767             else
  2768               hash_put (h, o1, Fcons (o2, Qnil), hash);
  2769           }
  2770         default: ;
  2771         }
  2772     }
  2773 
  2774   /* A symbol with position compares the contained symbol, and is
  2775      `equal' to the corresponding ordinary symbol.  */

  2777     o1 = SYMBOL_WITH_POS_SYM (o1);
  2778   if (SYMBOL_WITH_POS_P (o2))
  2779     o2 = SYMBOL_WITH_POS_SYM (o2);
  2780 
  2781   if (BASE_EQ (o1, o2))
  2782     return true;
  2783   if (XTYPE (o1) != XTYPE (o2))
  2784     return false;
  2785 
  2786   switch (XTYPE (o1))
  2787     {
  2788     case Lisp_Float:
  2789       return same_float (o1, o2);
  2790 
  2791     case Lisp_Cons:
  2792       if (equal_kind == EQUAL_NO_QUIT)
  2793         for (; CONSP (o1); o1 = XCDR (o1))
  2794           {
  2795             if (! CONSP (o2))
  2796               return false;

  2798               return false;

  2811             o2 = XCDR (o2);
  2812             if (EQ (XCDR (o1), o2))
  2813               return true;
  2814           }
  2815       depth++;
  2816       goto tail_recurse;
  2817 
  2818     case Lisp_Vectorlike:
  2819       {

  2821         /* Pseudovectors have the type encoded in the size field, so this test
  2822            actually checks that the objects have the same type as well as the
  2823            same size.  */

  3751 static Lisp_Object
  3752 base64_encode_region_1 (Lisp_Object beg, Lisp_Object end, bool line_break,
  3753                         bool pad, bool base64url)
  3754 {
  3755   char *encoded;
  3756   ptrdiff_t allength, length;
  3757   ptrdiff_t ibeg, iend, encoded_length;
  3758   ptrdiff_t old_pos = PT;
  3759   USE_SAFE_ALLOCA;
  3760 
  3761   validate_region (&beg, &end);
  3762 
  3763   ibeg = CHAR_TO_BYTE (XFIXNAT (beg));
  3764   iend = CHAR_TO_BYTE (XFIXNAT (end));
  3765   move_gap_both (XFIXNAT (beg), ibeg);
  3766 
  3767   /* We need to allocate enough room for encoding the text.
  3768      We need 33 1/3% more space, plus a newline every 76
  3769      characters, and then we round up. */
  3770   length = iend - ibeg;
  3771   allength = length + length/3 + 1;
  3772   allength += allength / MIME_LINE_LENGTH + 1 + 6;
  3773 
  3774   encoded = SAFE_ALLOCA (allength);
  3775   encoded_length = base64_encode_1 ((char *) BYTE_POS_ADDR (ibeg),
  3776                                     encoded, length, line_break,
  3777                                     pad, base64url,
  3778                                     !NILP (BVAR (current_buffer, enable_multibyte_characters)));
  3779   if (encoded_length > allength)
  3780     emacs_abort ();
  3781 
  3782   if (encoded_length < 0)
  3783     {
  3784       /* The encoding wasn't possible. */
  3785       SAFE_FREE ();
  3786       error ("Multibyte character in data for base64 encoding");
  3787     }
  3788 
  3789   /* Now we have encoded the region, so we insert the new contents
  3790      and delete the old.  (Insert first in order to preserve markers.)  */
  3791   SET_PT_BOTH (XFIXNAT (beg), ibeg);
  3792   insert (encoded, encoded_length);
  3793   SAFE_FREE ();
  3794   del_range_byte (ibeg + encoded_length, iend + encoded_length);
  3795 
  3796   /* If point was outside of the region, restore it exactly; else just
  3797      move to the beginning of the region.  */
  3798   if (old_pos >= XFIXNAT (end))
  3799     old_pos += encoded_length - (XFIXNAT (end) - XFIXNAT (beg));
  3800   else if (old_pos > XFIXNAT (beg))
  3801     old_pos = XFIXNAT (beg);
  3802   SET_PT (old_pos);
  3803 
  3804   /* We return the length of the encoded text. */
  3805   return make_fixnum (encoded_length);
  3806 }

  3871 static ptrdiff_t
  3872 base64_encode_1 (const char *from, char *to, ptrdiff_t length,
  3873                  bool line_break, bool pad, bool base64url,
  3874                  bool multibyte)
  3875 {
  3876   int counter = 0;
  3877   ptrdiff_t i = 0;
  3878   char *e = to;
  3879   int c;
  3880   unsigned int value;
  3881   int bytes;
  3882   char const *b64_value_to_char = base64_value_to_char[base64url];
  3883 
  3884   while (i < length)
  3885     {
  3886       if (multibyte)
  3887         {
  3888           c = string_char_and_length ((unsigned char *) from + i, &bytes);
  3889           if (CHAR_BYTE8_P (c))
  3890             c = CHAR_TO_BYTE8 (c);
  3891           else if (c >= 128)
  3892             return -1;
  3893           i += bytes;
  3894         }
  3895       else
  3896         c = from[i++];
  3897 
  3898       /* Wrap line every 76 characters.  */
  3899 
  3900       if (line_break)
  3901         {
  3902           if (counter < MIME_LINE_LENGTH / 4)
  3903             counter++;
  3904           else
  3905             {
  3906               *e++ = '\n';
  3907               counter = 1;
  3908             }

  3925             }
  3926           break;
  3927         }

  3929       if (multibyte)
  3930         {
  3931           c = string_char_and_length ((unsigned char *) from + i, &bytes);
  3932           if (CHAR_BYTE8_P (c))
  3933             c = CHAR_TO_BYTE8 (c);
  3934           else if (c >= 128)
  3935             return -1;
  3936           i += bytes;
  3937         }
  3938       else
  3939         c = from[i++];
  3940 
  3941       *e++ = b64_value_to_char[value | (0x0f & c >> 4)];
  3942       value = (0x0f & c) << 2;

  3945

  3977 Return the length of the decoded data.
  3978 
  3979 Note that after calling this function, the data in the region will
  3980 represent bytes, not text.  If you want to end up with text, you have
  3981 to call `decode-coding-region' afterwards with an appropriate coding

  3985 Optional third argument BASE64URL determines whether to use the URL variant
  3986 of the base 64 encoding, as defined in RFC 4648.
  3987 If optional fourth argument IGNORE-INVALID is non-nil invalid characters
  3988 are ignored instead of signaling an error.  */)

  3995   ptrdiff_t decoded_length;
  3996   ptrdiff_t inserted_chars;

  3998   USE_SAFE_ALLOCA;

  4000   validate_region (&beg, &end);
  4001 
  4002   ibeg = CHAR_TO_BYTE (XFIXNAT (beg));
  4003   iend = CHAR_TO_BYTE (XFIXNAT (end));
  4004 
  4005   length = iend - ibeg;
  4006 
  4007   /* We need to allocate enough room for decoding the text.  If we are
  4008      working on a multibyte buffer, each decoded code may occupy at
  4009      most two bytes.  */
  4010   allength = multibyte ? length * 2 : length;
  4011   decoded = SAFE_ALLOCA (allength);
  4012 
  4013   move_gap_both (XFIXNAT (beg), ibeg);
  4014   decoded_length = base64_decode_1 ((char *) BYTE_POS_ADDR (ibeg),
  4015                                     decoded, length, !NILP (base64url),
  4016                                     multibyte, !NILP (ignore_invalid),
  4017                                     &inserted_chars);

  4019     emacs_abort ();
  4020

  4022     {
  4023       /* The decoding wasn't possible. */
  4024       error ("Invalid base64 data");
  4025     }
  4026 
  4027   /* Now we have decoded the region, so we insert the new contents
  4028      and delete the old.  (Insert first in order to preserve markers.)  */
  4029   TEMP_SET_PT_BOTH (XFIXNAT (beg), ibeg);
  4030   insert_1_both (decoded, inserted_chars, decoded_length, 0, 1, 0);
  4031   signal_after_change (XFIXNAT (beg), 0, inserted_chars);
  4032   SAFE_FREE ();

  4035   del_range_both (PT, PT_BYTE, XFIXNAT (end) + inserted_chars,

  4039      move to the beginning of the region.  */
  4040   if (old_pos >= XFIXNAT (end))

  4042   else if (old_pos > XFIXNAT (beg))
  4043     old_pos = XFIXNAT (beg);
  4044   SET_PT (old_pos > ZV ? ZV : old_pos);

  4046   return make_fixnum (inserted_chars);

  4088 /* Base64-decode the data at FROM of LENGTH bytes into TO.  If
  4089    MULTIBYTE, the decoded result should be in multibyte
  4090    form.  If IGNORE_INVALID, ignore invalid base64 characters.
  4091    Store the number of produced characters in *NCHARS_RETURN.  */
  4092 
  4093 static ptrdiff_t
  4094 base64_decode_1 (const char *from, char *to, ptrdiff_t length,
  4095                  bool base64url, bool multibyte, bool ignore_invalid,
  4096                  ptrdiff_t *nchars_return)
  4097 {
  4098   char const *f = from;
  4099   char const *flim = from + length;
  4100   char *e = to;
  4101   ptrdiff_t nchars = 0;
  4102   signed char const *b64_char_to_value = base64_char_to_value[base64url];
  4103   unsigned char multibyte_bit = multibyte << 7;
  4104 
  4105   while (true)
  4106     {
  4107       unsigned char c;
  4108       int v1;
  4109 
  4110       /* Process first byte of a quadruplet. */
  4111 
  4112       do
  4113         {
  4114           if (f == flim)
  4115             {
  4116               *nchars_return = nchars;
  4117               return e - to;
  4118             }
  4119           c = *f++;
  4120           v1 = b64_char_to_value[c];
  4121         }
  4122       while (v1 < 0 || (v1 == 0 && ignore_invalid));
  4123 
  4124       if (v1 == 0)
  4125         return -1;
  4126       unsigned int value = (v1 - 1) << 18;
  4127 
  4128       /* Process second byte of a quadruplet.  */
  4129 
  4130       do
  4131         {
  4132           if (f == flim)
  4133             return -1;
  4134           c = *f++;
  4135           v1 = b64_char_to_value[c];
  4136         }
  4137       while (v1 < 0 || (v1 == 0 && ignore_invalid));
  4138 
  4139       if (v1 == 0)
  4140         return -1;
  4141       value += (v1 - 1) << 12;
  4142 
  4143       c = value >> 16 & 0xff;
  4144       if (c & multibyte_bit)
  4145         e += BYTE8_STRING (c, (unsigned char *) e);
  4146       else
  4147         *e++ = c;
  4148       nchars++;
  4149 
  4150       /* Process third byte of a quadruplet.  */
  4151 
  4152       do
  4153         {
  4154           if (f == flim)
  4155             {
  4156               if (!base64url && !ignore_invalid)
  4157                 return -1;
  4158               *nchars_return = nchars;
  4159               return e - to;
  4160             }
  4161           c = *f++;
  4162           v1 = b64_char_to_value[c];
  4163         }
  4164       while (v1 < 0 || (v1 == 0 && ignore_invalid));
  4165 
  4166       if (c == '=')
  4167         {
  4168           do
  4169             {
  4170               if (f == flim)
  4171                 return -1;
  4172               c = *f++;
  4173             }
  4174           while (b64_char_to_value[c] < 0);
  4175 
  4176           if (c != '=')
  4177             return -1;
  4178           continue;
  4179         }
  4180 
  4181       if (v1 == 0)
  4182         return -1;
  4183       value += (v1 - 1) << 6;
  4184 
  4185       c = value >> 8 & 0xff;
  4186       if (c & multibyte_bit)
  4187         e += BYTE8_STRING (c, (unsigned char *) e);
  4188       else
  4189         *e++ = c;
  4190       nchars++;
  4191 
  4192       /* Process fourth byte of a quadruplet.  */
  4193 
  4194       do
  4195         {
  4196           if (f == flim)
  4197             {
  4198               if (!base64url && !ignore_invalid)
  4199                 return -1;
  4200               *nchars_return = nchars;
  4201               return e - to;
  4202             }
  4203           c = *f++;
  4204           v1 = b64_char_to_value[c];
  4205         }
  4206       while (v1 < 0 || (v1 == 0 && ignore_invalid));
  4207 
  4208       if (c == '=')
  4209         continue;
  4210 
  4211       if (v1 == 0)
  4212         return -1;
  4213       value += v1 - 1;
  4214 
  4215       c = value & 0xff;
  4216       if (c & multibyte_bit)
  4217         e += BYTE8_STRING (c, (unsigned char *) e);
  4218       else
  4219         *e++ = c;
  4220       nchars++;
  4221     }
  4222 }

  4330 {

  4501    violating modularity.  */
  4502 #define INDEX_SIZE_BOUND \
  4503   ((ptrdiff_t) min (MOST_POSITIVE_FIXNUM, \
  4504                     ((min (PTRDIFF_MAX, SIZE_MAX) \
  4505                       - header_size - GCALIGNMENT) \
  4506                      / word_size)))
  4507 
  4508 static ptrdiff_t
  4509 hash_index_size (struct Lisp_Hash_Table *h, ptrdiff_t size)
  4510 {
  4511   double threshold = h->rehash_threshold;
  4512   double index_float = size / threshold;
  4513   ptrdiff_t index_size = (index_float < INDEX_SIZE_BOUND + 1
  4514                           ? next_almost_prime (index_float)
  4515                           : INDEX_SIZE_BOUND + 1);
  4516   if (INDEX_SIZE_BOUND < index_size)
  4517     error ("Hash table too large");
  4518   return index_size;
  4519 }

  4521 /* Create and initialize a new hash table.
  4522 
  4523    TEST specifies the test the hash table will use to compare keys.
  4524    It must be either one of the predefined tests `eq', `eql' or
  4525    `equal' or a symbol denoting a user-defined test named TEST with
  4526    test and hash functions USER_TEST and USER_HASH.
  4527 
  4528    Give the table initial capacity SIZE, 0 <= SIZE <= MOST_POSITIVE_FIXNUM.
  4529 
  4530    If REHASH_SIZE is equal to a negative integer, this hash table's
  4531    new size when it becomes full is computed by subtracting
  4532    REHASH_SIZE from its old size.  Otherwise it must be positive, and
  4533    the table's new size is computed by multiplying its old size by
  4534    REHASH_SIZE + 1.
  4535 
  4536    REHASH_THRESHOLD must be a float <= 1.0, and > 0.  The table will
  4537    be resized when the approximate ratio of table entries to table
  4538    size exceeds REHASH_THRESHOLD.
  4539 
  4540    WEAK specifies the weakness of the table.  If non-nil, it must be
  4541    one of the symbols `key', `value', `key-or-value', or `key-and-value'.
  4542 
  4543    If PURECOPY is non-nil, the table can be copied to pure storage via
  4544    `purecopy' when Emacs is being dumped. Such tables can no longer be
  4545    changed after purecopy.  */
  4546 
  4547 Lisp_Object
  4548 make_hash_table (struct hash_table_test test, EMACS_INT size,
  4549                  float rehash_size, float rehash_threshold,
  4550                  Lisp_Object weak, bool purecopy)
  4551 {
  4552   struct Lisp_Hash_Table *h;
  4553   Lisp_Object table;
  4554   ptrdiff_t i;
  4555 
  4556   /* Preconditions.  */
  4557   eassert (SYMBOLP (test.name));
  4558   eassert (0 <= size && size <= MOST_POSITIVE_FIXNUM);
  4559   eassert (rehash_size <= -1 || 0 < rehash_size);
  4560   eassert (0 < rehash_threshold && rehash_threshold <= 1);
  4561 
  4562   if (size == 0)
  4563     size = 1;
  4564 
  4565   /* Allocate a table and initialize it.  */
  4566   h = allocate_hash_table ();
  4567 
  4568   /* Initialize hash table slots.  */
  4569   h->test = test;
  4570   h->weak = weak;
  4571   h->rehash_threshold = rehash_threshold;
  4572   h->rehash_size = rehash_size;
  4573   h->count = 0;
  4574   h->key_and_value = make_vector (2 * size, Qunbound);
  4575   h->hash = make_nil_vector (size);
  4576   h->next = make_vector (size, make_fixnum (-1));
  4577   h->index = make_vector (hash_index_size (h, size), make_fixnum (-1));
  4578   h->next_weak = NULL;
  4579   h->purecopy = purecopy;
  4580   h->mutable = true;
  4581 
  4582   /* Set up the free list.  */
  4583   for (i = 0; i < size - 1; ++i)
  4584     set_hash_next_slot (h, i, i + 1);
  4585   h->next_free = 0;
  4586 
  4587   XSET_HASH_TABLE (table, h);
  4588   eassert (HASH_TABLE_P (table));
  4589   eassert (XHASH_TABLE (table) == h);
  4590 
  4591   return table;
  4592 }

  4617 /* Resize hash table H if it's too full.  If H cannot be resized
  4618    because it's already too large, throw an error.  */
  4619 
  4620 static void
  4621 maybe_resize_hash_table (struct Lisp_Hash_Table *h)
  4622 {
  4623   if (h->next_free < 0)
  4624     {
  4625       ptrdiff_t old_size = HASH_TABLE_SIZE (h);
  4626       EMACS_INT new_size;
  4627       double rehash_size = h->rehash_size;
  4628 
  4629       if (rehash_size < 0)
  4630         new_size = old_size - rehash_size;
  4631       else
  4632         {
  4633           double float_new_size = old_size * (rehash_size + 1);
  4634           if (float_new_size < EMACS_INT_MAX)
  4635             new_size = float_new_size;
  4636           else
  4637             new_size = EMACS_INT_MAX;
  4638         }
  4639       if (PTRDIFF_MAX < new_size)
  4640         new_size = PTRDIFF_MAX;
  4641       if (new_size <= old_size)
  4642         new_size = old_size + 1;
  4643 
  4644       /* Allocate all the new vectors before updating *H, to
  4645          avoid problems if memory is exhausted.  larger_vecalloc
  4646          finishes computing the size of the replacement vectors.  */
  4647       Lisp_Object next = larger_vecalloc (h->next, new_size - old_size,
  4648                                           new_size);
  4649       ptrdiff_t next_size = ASIZE (next);
  4650       for (ptrdiff_t i = old_size; i < next_size - 1; i++)
  4651         ASET (next, i, make_fixnum (i + 1));
  4652       ASET (next, next_size - 1, make_fixnum (-1));
  4653 
  4654       /* Build the new&larger key_and_value vector, making sure the new
  4655          fields are initialized to `unbound`.  */
  4656       Lisp_Object key_and_value
  4657         = larger_vecalloc (h->key_and_value, 2 * (next_size - old_size),
  4658                            2 * next_size);
  4659       for (ptrdiff_t i = 2 * old_size; i < 2 * next_size; i++)
  4660         ASET (key_and_value, i, Qunbound);
  4661 
  4662       Lisp_Object hash = larger_vector (h->hash, next_size - old_size,
  4663                                         next_size);
  4664       ptrdiff_t index_size = hash_index_size (h, next_size);
  4665       h->index = make_vector (index_size, make_fixnum (-1));
  4666       h->key_and_value = key_and_value;
  4667       h->hash = hash;
  4668       h->next = next;
  4669       h->next_free = old_size;
  4670 
  4671       /* Rehash.  */
  4672       for (ptrdiff_t i = 0; i < old_size; i++)
  4673         if (!NILP (HASH_HASH (h, i)))
  4674           {
  4675             EMACS_UINT hash_code = XUFIXNUM (HASH_HASH (h, i));
  4676             ptrdiff_t start_of_bucket = hash_code % ASIZE (h->index);
  4677             set_hash_next_slot (h, i, HASH_INDEX (h, start_of_bucket));
  4678             set_hash_index_slot (h, start_of_bucket, i);
  4679           }
  4680 
  4681 #ifdef ENABLE_CHECKING
  4682       if (HASH_TABLE_P (Vpurify_flag) && XHASH_TABLE (Vpurify_flag) == h)
  4683         message ("Growing hash table to: %"pD"d", next_size);
  4684 #endif
  4685     }
  4686 }

  4688 /* Recompute the hashes (and hence also the "next" pointers).
  4689    Normally there's never a need to recompute hashes.
  4690    This is done only on first access to a hash-table loaded from
  4691    the "pdump", because the objects' addresses may have changed, thus
  4692    affecting their hashes.  */
  4693 void
  4694 hash_table_rehash (Lisp_Object hash)
  4695 {
  4696   struct Lisp_Hash_Table *h = XHASH_TABLE (hash);
  4697   ptrdiff_t i, count = h->count;
  4698 
  4699   /* Recompute the actual hash codes for each entry in the table.
  4700      Order is still invalid.  */
  4701   for (i = 0; i < count; i++)
  4702     {
  4703       Lisp_Object key = HASH_KEY (h, i);
  4704       Lisp_Object hash_code = h->test.hashfn (key, h);
  4705       ptrdiff_t start_of_bucket = XUFIXNUM (hash_code) % ASIZE (h->index);
  4706       set_hash_hash_slot (h, i, hash_code);
  4707       set_hash_next_slot (h, i, HASH_INDEX (h, start_of_bucket));
  4708       set_hash_index_slot (h, start_of_bucket, i);
  4709       eassert (HASH_NEXT (h, i) != i); /* Stop loops.  */
  4710     }
  4711 
  4712   ptrdiff_t size = ASIZE (h->next);
  4713   for (; i + 1 < size; i++)
  4714     set_hash_next_slot (h, i, i + 1);
  4715 }

  4717 /* Lookup KEY in hash table H.  If HASH is non-null, return in *HASH
  4718    the hash code of KEY.  Value is the index of the entry in H
  4719    matching KEY, or -1 if not found.  */
  4720 
  4721 ptrdiff_t
  4722 hash_lookup (struct Lisp_Hash_Table *h, Lisp_Object key, Lisp_Object *hash)
  4723 {
  4724   ptrdiff_t start_of_bucket, i;
  4725 
  4726   Lisp_Object hash_code;
  4727   hash_code = h->test.hashfn (key, h);
  4728   if (hash)
  4729     *hash = hash_code;
  4730 
  4731   start_of_bucket = XUFIXNUM (hash_code) % ASIZE (h->index);
  4732 
  4733   for (i = HASH_INDEX (h, start_of_bucket); 0 <= i; i = HASH_NEXT (h, i))
  4734     if (EQ (key, HASH_KEY (h, i))
  4735         || (h->test.cmpfn
  4736             && EQ (hash_code, HASH_HASH (h, i))
  4737             && !NILP (h->test.cmpfn (key, HASH_KEY (h, i), h))))
  4738       break;

  4740   return i;

  4760 /* Put an entry into hash table H that associates KEY with VALUE.
  4761    HASH is a previously computed hash code of KEY.
  4762    Value is the index of the entry in H matching KEY.  */
  4763 
  4764 ptrdiff_t
  4765 hash_put (struct Lisp_Hash_Table *h, Lisp_Object key, Lisp_Object value,
  4766           Lisp_Object hash)
  4767 {
  4768   ptrdiff_t start_of_bucket, i;
  4769 
  4770   /* Increment count after resizing because resizing may fail.  */
  4771   maybe_resize_hash_table (h);
  4772   h->count++;
  4773 
  4774   /* Store key/value in the key_and_value vector.  */
  4775   i = h->next_free;
  4776   eassert (NILP (HASH_HASH (h, i)));
  4777   eassert (BASE_EQ (Qunbound, (HASH_KEY (h, i))));
  4778   h->next_free = HASH_NEXT (h, i);
  4779   set_hash_key_slot (h, i, key);
  4780   set_hash_value_slot (h, i, value);
  4781 
  4782   /* Remember its hash code.  */
  4783   set_hash_hash_slot (h, i, hash);
  4784 
  4785   /* Add new entry to its collision chain.  */
  4786   start_of_bucket = XUFIXNUM (hash) % ASIZE (h->index);
  4787   set_hash_next_slot (h, i, HASH_INDEX (h, start_of_bucket));
  4788   set_hash_index_slot (h, start_of_bucket, i);
  4789   return i;
  4790 }

  4793 /* Remove the entry matching KEY from hash table H, if there is one.  */
  4794 
  4795 void
  4796 hash_remove_from_table (struct Lisp_Hash_Table *h, Lisp_Object key)
  4797 {
  4798   Lisp_Object hash_code = h->test.hashfn (key, h);
  4799   ptrdiff_t start_of_bucket = XUFIXNUM (hash_code) % ASIZE (h->index);
  4800   ptrdiff_t prev = -1;
  4801 
  4802   for (ptrdiff_t i = HASH_INDEX (h, start_of_bucket);
  4803        0 <= i;
  4804        i = HASH_NEXT (h, i))
  4805     {
  4806       if (EQ (key, HASH_KEY (h, i))
  4807           || (h->test.cmpfn
  4808               && EQ (hash_code, HASH_HASH (h, i))
  4809               && !NILP (h->test.cmpfn (key, HASH_KEY (h, i), h))))
  4810         {
  4811           /* Take entry out of collision chain.  */
  4812           if (prev < 0)
  4813             set_hash_index_slot (h, start_of_bucket, HASH_NEXT (h, i));
  4814           else
  4815             set_hash_next_slot (h, prev, HASH_NEXT (h, i));
  4816 
  4817           /* Clear slots in key_and_value and add the slots to
  4818              the free list.  */
  4819           set_hash_key_slot (h, i, Qunbound);
  4820           set_hash_value_slot (h, i, Qnil);
  4821           set_hash_hash_slot (h, i, Qnil);
  4822           set_hash_next_slot (h, i, h->next_free);
  4823           h->next_free = i;
  4824           h->count--;
  4825           eassert (h->count >= 0);
  4826           break;
  4827         }
  4828 
  4829       prev = i;
  4830     }

  4864 /* Sweep weak hash table H.  REMOVE_ENTRIES_P means remove
  4865    entries from the table that don't survive the current GC.
  4866    !REMOVE_ENTRIES_P means mark entries that are in use.  Value is
  4867    true if anything was marked.  */
  4868 
  4869 bool
  4870 sweep_weak_table (struct Lisp_Hash_Table *h, bool remove_entries_p)
  4871 {
  4872   ptrdiff_t n = gc_asize (h->index);
  4873   bool marked = false;
  4874 
  4875   for (ptrdiff_t bucket = 0; bucket < n; ++bucket)
  4876     {
  4877       /* Follow collision chain, removing entries that don't survive
  4878          this garbage collection.  */
  4879       ptrdiff_t prev = -1;
  4880       ptrdiff_t next;
  4881       for (ptrdiff_t i = HASH_INDEX (h, bucket); 0 <= i; i = next)
  4882         {
  4883           bool key_known_to_survive_p = survives_gc_p (HASH_KEY (h, i));
  4884           bool value_known_to_survive_p = survives_gc_p (HASH_VALUE (h, i));
  4885           bool remove_p;
  4886 
  4887           if (EQ (h->weak, Qkey))
  4888             remove_p = !key_known_to_survive_p;
  4889           else if (EQ (h->weak, Qvalue))
  4890             remove_p = !value_known_to_survive_p;
  4891           else if (EQ (h->weak, Qkey_or_value))
  4892             remove_p = !(key_known_to_survive_p || value_known_to_survive_p);
  4893           else if (EQ (h->weak, Qkey_and_value))
  4894             remove_p = !(key_known_to_survive_p && value_known_to_survive_p);
  4895           else
  4896             emacs_abort ();
  4897 
  4898           next = HASH_NEXT (h, i);
  4899 
  4900           if (remove_entries_p)
  4901             {
  4902               eassert (!remove_p
  4903                        == (key_known_to_survive_p && value_known_to_survive_p));
  4904               if (remove_p)
  4905                 {
  4906                   /* Take out of collision chain.  */
  4907                   if (prev < 0)
  4908                     set_hash_index_slot (h, bucket, next);
  4909                   else
  4910                     set_hash_next_slot (h, prev, next);
  4911 
  4912                   /* Add to free list.  */
  4913                   set_hash_next_slot (h, i, h->next_free);
  4914                   h->next_free = i;
  4915 
  4916                   /* Clear key, value, and hash.  */
  4917                   set_hash_key_slot (h, i, Qunbound);
  4918                   set_hash_value_slot (h, i, Qnil);
  4919                   if (!NILP (h->hash))
  4920                     set_hash_hash_slot (h, i, Qnil);
  4921 
  4922                   eassert (h->count != 0);
  4923                   h->count--;
  4924                 }
  4925               else
  4926                 {
  4927                   prev = i;
  4928                 }
  4929             }
  4930           else
  4931             {
  4932               if (!remove_p)
  4933                 {
  4934                   /* Make sure key and value survive.  */
  4935                   if (!key_known_to_survive_p)
  4936                     {
  4937                       mark_object (HASH_KEY (h, i));
  4938                       marked = true;
  4939                     }
  4940 
  4941                   if (!value_known_to_survive_p)
  4942                     {
  4943                       mark_object (HASH_VALUE (h, i));
  4944                       marked = true;
  4945                     }
  4946                 }
  4947             }
  4948         }
  4949     }
  4950 
  4951   return marked;
  4952 }

  4959 /* Maximum depth up to which to dive into Lisp structures.  */
  4960 
  4961 #define SXHASH_MAX_DEPTH 3

  4963 /* Maximum length up to which to take list and vector elements into
  4964    account.  */

  4966 #define SXHASH_MAX_LEN   7
  4967 
  4968 /* Return a hash for string PTR which has length LEN.  The hash value
  4969    can be any EMACS_UINT value.  */
  4970 
  4971 EMACS_UINT
  4972 hash_string (char const *ptr, ptrdiff_t len)
  4973 {
  4974   char const *p   = ptr;
  4975   char const *end = ptr + len;
  4976   EMACS_UINT hash = len;
  4977   /* At most 8 steps.  We could reuse SXHASH_MAX_LEN, of course,
  4978    * but dividing by 8 is cheaper.  */
  4979   ptrdiff_t step = sizeof hash + ((end - p) >> 3);
  4980 
  4981   while (p + sizeof hash <= end)
  4982     {
  4983       EMACS_UINT c;
  4984       /* We presume that the compiler will replace this `memcpy` with
  4985          a single load/move instruction when applicable.  */
  4986       memcpy (&c, p, sizeof hash);
  4987       p += step;
  4988       hash = sxhash_combine (hash, c);
  4989     }
  4990   /* A few last bytes may remain (smaller than an EMACS_UINT).  */
  4991   /* FIXME: We could do this without a loop, but it'd require
  4992      endian-dependent code :-(  */
  4993   while (p < end)
  4994     {
  4995       unsigned char c = *p++;
  4996       hash = sxhash_combine (hash, c);
  4997     }
  4998 
  4999   return hash;
  5000 }

  5112 static EMACS_UINT
  5113 sxhash_obj (Lisp_Object obj, int depth)
  5114 {
  5115   if (depth > SXHASH_MAX_DEPTH)
  5116     return 0;
  5117 
  5118   switch (XTYPE (obj))
  5119     {
  5120     case_Lisp_Int:
  5121       return XUFIXNUM (obj);
  5122 
  5123     case Lisp_Symbol:
  5124       return XHASH (obj);
  5125 
  5126     case Lisp_String:
  5127       return sxhash_string (SSDATA (obj), SBYTES (obj));
  5128 
  5129     case Lisp_Vectorlike:
  5130       {
  5131         enum pvec_type pvec_type = PSEUDOVECTOR_TYPE (XVECTOR (obj));
  5132         if (! (PVEC_NORMAL_VECTOR < pvec_type && pvec_type < PVEC_COMPILED))
  5133           {
  5134             /* According to the CL HyperSpec, two arrays are equal only if
  5135                they are 'eq', except for strings and bit-vectors.  In
  5136                Emacs, this works differently.  We have to compare element
  5137                by element.  Same for pseudovectors that internal_equal
  5138                examines the Lisp contents of.  */
  5139             return (SUB_CHAR_TABLE_P (obj)
  5140                     /* 'sxhash_vector' can't be applies to a sub-char-table and
  5141                       it's probably not worth looking into them anyway!  */
  5142                     ? 42
  5143                     : sxhash_vector (obj, depth));
  5144           }
  5145         /* FIXME: Use `switch`.  */
  5146         else if (pvec_type == PVEC_BIGNUM)
  5147           return sxhash_bignum (obj);
  5148         else if (pvec_type == PVEC_MARKER)
  5149           {
  5150             ptrdiff_t bytepos
  5151               = XMARKER (obj)->buffer ? XMARKER (obj)->bytepos : 0;
  5152             EMACS_UINT hash
  5153               = sxhash_combine ((intptr_t) XMARKER (obj)->buffer, bytepos);
  5154             return SXHASH_REDUCE (hash);
  5155           }
  5156         else if (pvec_type == PVEC_BOOL_VECTOR)
  5157           return sxhash_bool_vector (obj);
  5158         else if (pvec_type == PVEC_OVERLAY)
  5159           {
  5160             EMACS_UINT hash = OVERLAY_START (obj);
  5161             hash = sxhash_combine (hash, OVERLAY_END (obj));
  5162             hash = sxhash_combine (hash, sxhash_obj (XOVERLAY (obj)->plist, depth));
  5163             return SXHASH_REDUCE (hash);
  5164           }
  5165         else if (symbols_with_pos_enabled && pvec_type == PVEC_SYMBOL_WITH_POS)
  5166           return sxhash_obj (XSYMBOL_WITH_POS (obj)->sym, depth + 1);
  5167         else
  5168           /* Others are 'equal' if they are 'eq', so take their
  5169              address as hash.  */
  5170           return XHASH (obj);
  5171       }
  5172 
  5173     case Lisp_Cons:
  5174       return sxhash_list (obj, depth);
  5175 
  5176     case Lisp_Float:
  5177       return sxhash_float (XFLOAT_DATA (obj));
  5178 
  5179     default:
  5180       emacs_abort ();
  5181     }
  5182 }

  5558    representing DIGEST, which is binary data of size DIGEST_SIZE bytes.

  5560 void

  6034        doc: /* Search for the string NEEDLE in the string HAYSTACK.
  6035 The return value is the position of the first occurrence of NEEDLE in
  6036 HAYSTACK, or nil if no match was found.
  6037 
  6038 The optional START-POS argument says where to start searching in
  6039 HAYSTACK and defaults to zero (start at the beginning).
  6040 It must be between zero and the length of HAYSTACK, inclusive.
  6041 
  6042 Case is always significant and text properties are ignored. */)
  6043   (register Lisp_Object needle, Lisp_Object haystack, Lisp_Object start_pos)
  6044 {
  6045   ptrdiff_t start_byte = 0, haybytes;
  6046   char *res, *haystart;
  6047   EMACS_INT start = 0;
  6048 
  6049   CHECK_STRING (needle);
  6050   CHECK_STRING (haystack);

  6052   if (!NILP (start_pos))

  6057         xsignal1 (Qargs_out_of_range, start_pos);
  6058       start_byte = string_char_to_byte (haystack, start);
  6059     }
  6060 
  6061   /* If NEEDLE is longer than (the remaining length of) haystack, then
  6062      we can't have a match, and return early.  */
  6063   if (SCHARS (needle) > SCHARS (haystack) - start)
  6064     return Qnil;
  6065 
  6066   haystart = SSDATA (haystack) + start_byte;
  6067   haybytes = SBYTES (haystack) - start_byte;

  6070      same multibyteness, or if the needle consists of ASCII characters only.  */

  6073          || SCHARS (haystack) == SBYTES (haystack) || string_ascii_p (needle))
  6074       : (!STRING_MULTIBYTE (needle)

  6078           && SCHARS (haystack) == SBYTES (haystack)
  6079           && SCHARS (needle) != SBYTES (needle))
  6080         /* Multibyte non-ASCII needle, multibyte ASCII haystack: impossible.  */
  6081         return Qnil;
  6082       else
  6083         res = memmem (haystart, haybytes,
  6084                       SSDATA (needle), SBYTES (needle));
  6085     }
  6086   else if (STRING_MULTIBYTE (haystack))  /* unibyte non-ASCII needle */
  6087     {
  6088       Lisp_Object multi_needle = string_to_multibyte (needle);
  6089       res = memmem (haystart, haybytes,
  6090                     SSDATA (multi_needle), SBYTES (multi_needle));
  6091     }
  6092   else              /* unibyte haystack, multibyte non-ASCII needle */

  6095          unibyte stack (since we know that the needle is non-ASCII) is
  6096          if they contain "raw bytes" (and no other non-ASCII chars.)  */
  6097       ptrdiff_t nbytes = SBYTES (needle);
  6098       for (ptrdiff_t i = 0; i < nbytes; i++)

  6105         }
  6106 
  6107       /* "Raw bytes" (aka eighth-bit) are represented differently in
  6108          multibyte and unibyte strings.  */
  6109       Lisp_Object uni_needle = Fstring_to_unibyte (needle);
  6110       res = memmem (haystart, haybytes,
  6111                     SSDATA (uni_needle), SBYTES (uni_needle));
  6112     }
  6113 
  6114   if (! res)
  6115     return Qnil;
  6116 
  6117   return make_int (string_byte_to_char (haystack, res - SSDATA (haystack)));
  6118 }

  6175       if (pos < BEG || pos > Z)

  6180   if (!NILP (absolute))
