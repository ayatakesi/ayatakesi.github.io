    62 /* Extract a doc string from a file.  FILEPOS says where to get it.
    63    If it is an integer, use that position in the standard DOC file.
    64    If it is (FILE . INTEGER), use FILE as the file name
    65    and INTEGER as the position in that file.
    66    But if INTEGER is negative, make it positive.
    67    (A negative integer is used for user variables, so we can distinguish
    68    them without actually fetching the doc string.)
    69 
    70    If the location does not point to the beginning of a docstring
    71    (e.g. because the file has been modified and the location is stale),
    72    return nil.
    73 
    74    If UNIBYTE, always make a unibyte string.
    75 
    76    If DEFINITION, assume this is for reading
    77    a dynamic function definition; convert the bytestring
    78    and the constants vector with appropriate byte handling,
    79    and return a cons cell.  */
    80 
    81 Lisp_Object
    82 get_doc_string (Lisp_Object filepos, bool unibyte, bool definition)
    83 {
    84   char *from, *to, *name, *p, *p1;
    85   Lisp_Object file, pos;
    86   specpdl_ref count = SPECPDL_INDEX ();
    87   Lisp_Object dir;
    88   USE_SAFE_ALLOCA;
    89 
    90   if (FIXNUMP (filepos))
    91     {
    92       file = Vdoc_file_name;
    93       dir = Vdoc_directory;
    94       pos = filepos;
    95     }
    96   else if (CONSP (filepos))
    97     {
    98       file = XCAR (filepos);
    99       dir = Fsymbol_value (Qlisp_directory);
   100       pos = XCDR (filepos);
   101     }
   102   else
   103     return Qnil;
   104 
   105   EMACS_INT position = eabs (XFIXNUM (pos));
   106 
   107   if (!STRINGP (dir))
   108     return Qnil;
   109 
   110   if (!STRINGP (file))
   111     return Qnil;
   112 
   113   /* Put the file name in NAME as a C string.
   114      If it is relative, combine it with Vdoc_directory.  */
   115 
   116   Lisp_Object tem = Ffile_name_absolute_p (file);
   117   file = ENCODE_FILE (file);
   118   Lisp_Object docdir
   119     = NILP (tem) ? ENCODE_FILE (dir) : empty_unibyte_string;
   120   ptrdiff_t docdir_sizemax = SBYTES (docdir) + 1;
   121   if (will_dump_p ())
   122     docdir_sizemax = max (docdir_sizemax, sizeof sibling_etc);
   123   name = SAFE_ALLOCA (docdir_sizemax + SBYTES (file));
   124   lispstpcpy (lispstpcpy (name, docdir), file);
   125 
   126   int fd = emacs_open (name, O_RDONLY, 0);
   127   if (fd < 0)
   128     {
   129       if (will_dump_p ())
   130         {
   131           /* Preparing to dump; DOC file is probably not installed.
   132              So check in ../etc.  */
   133           lispstpcpy (stpcpy (name, sibling_etc), file);
   134 
   135           fd = emacs_open (name, O_RDONLY, 0);
   136         }
   137       if (fd < 0)
   138         {
   139           if (errno != ENOENT && errno != ENOTDIR)
   140             report_file_error ("Read error on documentation file", file);
   141 
   142           SAFE_FREE ();
   143           AUTO_STRING (cannot_open, "Cannot open doc string file \"");
   144           AUTO_STRING (quote_nl, "\"\n");
   145           return concat3 (cannot_open, file, quote_nl);
   146         }
   147     }
   148   record_unwind_protect_int (close_file_unwind, fd);
   149 
   150   /* Seek only to beginning of disk block.  */
   151   /* Make sure we read at least 1024 bytes before `position'
   152      so we can check the leading text for consistency.  */
   153   int offset = min (position, max (1024, position % (8 * 1024)));
   154   if (TYPE_MAXIMUM (off_t) < position
   155       || lseek (fd, position - offset, 0) < 0)
   156     error ("Position %"pI"d out of range in doc string file \"%s\"",
   157            position, name);
   158 
   159   /* Read the doc string into get_doc_string_buffer.
   160      P points beyond the data just read.  */
   161 
   162   p = get_doc_string_buffer;
   163   while (1)
   164     {
   165       ptrdiff_t space_left = (get_doc_string_buffer_size - 1
   166                               - (p - get_doc_string_buffer));
   167 
   168       /* Allocate or grow the buffer if we need to.  */
   169       if (space_left <= 0)
   170         {
   171           ptrdiff_t in_buffer = p - get_doc_string_buffer;
   172           get_doc_string_buffer
   173             = xpalloc (get_doc_string_buffer, &get_doc_string_buffer_size,
   174                        16 * 1024, -1, 1);
   175           p = get_doc_string_buffer + in_buffer;
   176           space_left = (get_doc_string_buffer_size - 1
   177                         - (p - get_doc_string_buffer));
   178         }
   179 
   180       /* Read a disk block at a time.
   181          If we read the same block last time, maybe skip this?  */
   182       if (space_left > 1024 * 8)
   183         space_left = 1024 * 8;
   184       int nread = emacs_read_quit (fd, p, space_left);
   185       if (nread < 0)
   186         report_file_error ("Read error on documentation file", file);
   187       p[nread] = 0;
   188       if (!nread)
   189         break;
   190       if (p == get_doc_string_buffer)
   191         p1 = strchr (p + offset, '\037');
   192       else
   193         p1 = strchr (p, '\037');
   194       if (p1)
   195         {
   196           *p1 = 0;
   197           p = p1;
   198           break;
   199         }
   200       p += nread;
   201     }
   202   SAFE_FREE_UNBIND_TO (count, Qnil);
   203 
   204   /* Sanity checking.  */
   205   if (CONSP (filepos))
   206     {
   207       int test = 1;
   208       /* A dynamic docstring should be either at the very beginning of a "#@
   209          comment" or right after a dynamic docstring delimiter (in case we
   210          pack several such docstrings within the same comment).  */
   211       if (get_doc_string_buffer[offset - test] != '\037')
   212         {
   213           if (get_doc_string_buffer[offset - test++] != ' ')
   214             return Qnil;
   215           while (get_doc_string_buffer[offset - test] >= '0'
   216                  && get_doc_string_buffer[offset - test] <= '9')
   217             test++;
   218           if (get_doc_string_buffer[offset - test++] != '@'
   219               || get_doc_string_buffer[offset - test] != '#')
   220             return Qnil;
   221         }
   222     }
   223   else
   224     {
   225       int test = 1;
   226       if (get_doc_string_buffer[offset - test++] != '\n')
   227         return Qnil;
   228       while (get_doc_string_buffer[offset - test] > ' ')
   229         test++;
   230       if (get_doc_string_buffer[offset - test] != '\037')
   231         return Qnil;
   232     }
   233 
   234   /* Scan the text and perform quoting with ^A (char code 1).
   235      ^A^A becomes ^A, ^A0 becomes a null char, and ^A_ becomes a ^_.  */
   236   from = get_doc_string_buffer + offset;
   237   to = get_doc_string_buffer + offset;
   238   while (from != p)
   239     {
   240       if (*from == 1)
   241         {
   242           from++;
   243           int c = *from++;
   244           if (c == 1)
   245             *to++ = c;
   246           else if (c == '0')
   247             *to++ = 0;
   248           else if (c == '_')
   249             *to++ = 037;
   250           else
   251             {
   252               unsigned char uc = c;
   253               error ("\
   254 Invalid data in documentation file -- %c followed by code %03o",
   255                      1, uc);
   256             }
   257         }
   258       else
   259         *to++ = *from++;
   260     }
   261 
   262   /* If DEFINITION, read from this buffer
   263      the same way we would read bytes from a file.  */
   264   if (definition)
   265     {
   266       read_bytecode_pointer = (unsigned char *) get_doc_string_buffer + offset;
   267       return Fread (Qlambda);
   268     }
   269 
   270   if (unibyte)
   271     return make_unibyte_string (get_doc_string_buffer + offset,
   272                                 to - (get_doc_string_buffer + offset));
   273   else
   274     {
   275       /* The data determines whether the string is multibyte.  */
   276       ptrdiff_t nchars
   277         = multibyte_chars_in_text (((unsigned char *) get_doc_string_buffer
   278                                     + offset),
   279                                    to - (get_doc_string_buffer + offset));
   280       return make_string_from_bytes (get_doc_string_buffer + offset,
   281                                      nchars,
   282                                      to - (get_doc_string_buffer + offset));
   283     }
   284 }

   307 DEFUN ("documentation", Fdocumentation, Sdocumentation, 1, 2, 0,
   308        doc: /* Return the documentation string of FUNCTION.
   309 Unless a non-nil second argument RAW is given, the
   310 string is passed through `substitute-command-keys'.  */)
   311   (Lisp_Object function, Lisp_Object raw)
   312 {
   313   Lisp_Object doc;
   314   bool try_reload = true;
   315 
   316  documentation:
   317 
   318   doc = Qnil;
   319 
   320   if (SYMBOLP (function))
   321     {
   322       Lisp_Object tem = Fget (function, Qfunction_documentation);
   323       if (!NILP (tem))
   324         return Fdocumentation_property (function, Qfunction_documentation,
   325                                         raw);
   326     }
   327 
   328   Lisp_Object fun = Findirect_function (function, Qnil);
   329   if (NILP (fun))
   330     xsignal1 (Qvoid_function, function);
   331   if (CONSP (fun) && EQ (XCAR (fun), Qmacro))
   332     fun = XCDR (fun);
   333   doc = call1 (Qfunction_documentation, fun);
   334 
   335   /* If DOC is 0, it's typically because of a dumped file missing
   336      from the DOC file (bug in src/Makefile.in).  */
   337   if (BASE_EQ (doc, make_fixnum (0)))
   338     doc = Qnil;
   339   if (FIXNUMP (doc) || CONSP (doc))
   340     {
   341       Lisp_Object tem;
   342       tem = get_doc_string (doc, 0, 0);
   343       if (NILP (tem) && try_reload)
   344         {
   345           /* The file is newer, we need to reset the pointers.  */
   346           try_reload = reread_doc_file (Fcar_safe (doc));
   347           if (try_reload)
   348             {
   349               try_reload = false;
   350               goto documentation;
   351             }
   352         }
   353       else
   354         doc = tem;
   355     }
   356 
   357   if (NILP (raw))
   358     doc = call1 (Qsubstitute_command_keys, doc);
   359   return doc;
   360 }

   381 DEFUN ("documentation-property", Fdocumentation_property,
   382        Sdocumentation_property, 2, 3, 0,
   383        doc: /* Return the documentation string that is SYMBOL's PROP property.
   384 Third argument RAW omitted or nil means pass the result through
   385 `substitute-command-keys' if it is a string.
   386 
   387 This differs from `get' in that it can refer to strings stored in the
   388 `etc/DOC' file; and that it evaluates documentation properties that
   389 aren't strings.  */)
   390   (Lisp_Object symbol, Lisp_Object prop, Lisp_Object raw)
   391 {
   392   bool try_reload = true;
   393   Lisp_Object tem;
   394 
   395  documentation_property:
   396 
   397   tem = Fget (symbol, prop);
   398 
   399   /* If we don't have any documentation for this symbol (and we're asking for
   400      the variable documentation), try to see whether it's an indirect variable
   401      and get the documentation from there instead. */
   402   if (EQ (prop, Qvariable_documentation)
   403       && NILP (tem))
   404     {
   405       Lisp_Object indirect = Findirect_variable (symbol);
   406       if (!NILP (indirect))
   407         tem = Fget (indirect, prop);
   408     }
   409 
   410   if (BASE_EQ (tem, make_fixnum (0)))
   411     tem = Qnil;
   412 
   413   /* See if we want to look for the string in the DOC file. */
   414   if (FIXNUMP (tem) || (CONSP (tem) && FIXNUMP (XCDR (tem))))
   415     {
   416       Lisp_Object doc = tem;
   417       tem = get_doc_string (tem, 0, 0);
   418       if (NILP (tem) && try_reload)
   419         {
   420           /* The file is newer, we need to reset the pointers.  */
   421           try_reload = reread_doc_file (Fcar_safe (doc));
   422           if (try_reload)
   423             {
   424               try_reload = false;
   425               goto documentation_property;
   426             }
   427         }
   428     }
   429   else if (!STRINGP (tem))
   430     /* Feval protects its argument.  */
   431     tem = Feval (tem, Qnil);
   432 
   433   if (NILP (raw) && STRINGP (tem))
   434     tem = call1 (Qsubstitute_command_keys, tem);
   435   return tem;
   436 }

   438 /* Scanning the DOC files and placing docstring offsets into functions.  */

   440 static void
   441 store_function_docstring (Lisp_Object obj, EMACS_INT offset)
   442 {
   443   /* Don't use indirect_function here, or defaliases will apply their
   444      docstrings to the base functions (Bug#2603).  */
   445   Lisp_Object fun = SYMBOLP (obj) ? XSYMBOL (obj)->u.s.function : obj;
   446 
   447   /* The type determines where the docstring is stored.  */
   448 
   449   /* If it's a lisp form, stick it in the form.  */
   450   if (CONSP (fun) && EQ (XCAR (fun), Qmacro))
   451     fun = XCDR (fun);
   452   if (CONSP (fun))
   453     {
   454       Lisp_Object tem = XCAR (fun);
   455       if (EQ (tem, Qlambda) || EQ (tem, Qautoload)
   456           || (EQ (tem, Qclosure) && (fun = XCDR (fun), 1)))
   457         {
   458           tem = Fcdr (Fcdr (fun));
   459           if (CONSP (tem) && FIXNUMP (XCAR (tem)))
   460             /* FIXME: This modifies typically pure hash-cons'd data, so its
   461                correctness is quite delicate.  */
   462             XSETCAR (tem, make_fixnum (offset));
   463         }
   464     }
   465   /* Lisp_Subrs have a slot for it.  */
   466   else if (SUBRP (fun) && !SUBR_NATIVE_COMPILEDP (fun))
   467     {
   468       XSUBR (fun)->doc = offset;
   469     }
   470 
   471   /* Bytecode objects sometimes have slots for it.  */
   472   else if (COMPILEDP (fun))
   473     {
   474       /* This bytecode object must have a slot for the
   475          docstring, since we've found a docstring for it.  */
   476       if (PVSIZE (fun) > COMPILED_DOC_STRING
   477           /* Don't overwrite a non-docstring value placed there,
   478            * such as the symbols used for Oclosures.  */
   479           && VALID_DOCSTRING_P (AREF (fun, COMPILED_DOC_STRING)))
   480         ASET (fun, COMPILED_DOC_STRING, make_fixnum (offset));
   481       else
   482         {
   483           AUTO_STRING (format,
   484                        (PVSIZE (fun) > COMPILED_DOC_STRING
   485                         ? "Docstring slot busy for %s"
   486                         : "No docstring slot for %s"));
   487           CALLN (Fmessage, format,
   488                  (SYMBOLP (obj)
   489                   ? SYMBOL_NAME (obj)
   490                   : build_string ("<anonymous>")));
   491         }
   492     }
   493 }

   496 DEFUN ("Snarf-documentation", Fsnarf_documentation, Ssnarf_documentation,
   497        1, 1, 0,
   498        doc: /* Used during Emacs initialization to scan the `etc/DOC...' file.
   499 This searches the `etc/DOC...' file for doc strings and
   500 records them in function and variable definitions.
   501 The function takes one argument, FILENAME, a string;
   502 it specifies the file name (without a directory) of the DOC file.
   503 That file is found in `../etc' now; later, when the dumped Emacs is run,
   504 the same file name is found in the `doc-directory'.  */)
   505   (Lisp_Object filename)
   506 {
   507   int fd;
   508   char buf[1024 + 1];
   509   int filled;
   510   EMACS_INT pos;
   511   Lisp_Object sym;
   512   char *p, *name;
   513   char const *dirname;
   514   ptrdiff_t dirlen;
   515   /* Preloaded defcustoms using custom-initialize-delay are added to
   516      this list, but kept unbound.  See https://debbugs.gnu.org/11565  */
   517   Lisp_Object delayed_init =
   518     find_symbol_value (intern ("custom-delayed-init-variables"));
   519 
   520   if (!CONSP (delayed_init)) delayed_init = Qnil;
   521 
   522   CHECK_STRING (filename);
   523 
   524   if (will_dump_p ())
   525     {
   526       dirname = sibling_etc;
   527       dirlen = sizeof sibling_etc - 1;
   528     }
   529   else
   530     {
   531       CHECK_STRING (Vdoc_directory);
   532       dirname = SSDATA (Vdoc_directory);
   533       dirlen = SBYTES (Vdoc_directory);
   534     }
   535 
   536   specpdl_ref count = SPECPDL_INDEX ();
   537   USE_SAFE_ALLOCA;
   538   name = SAFE_ALLOCA (dirlen + SBYTES (filename) + 1);
   539   lispstpcpy (stpcpy (name, dirname), filename);        /*** Add this line ***/
   540 
   541   /* Vbuild_files is nil when temacs is run, and non-nil after that.  */
   542   if (NILP (Vbuild_files))
   543     {
   544       static char const *const buildobj[] =
   545         {
   546           #include "buildobj.h"
   547         };
   548       int i = ARRAYELTS (buildobj);
   549       while (0 <= --i)
   550         Vbuild_files = Fcons (build_string (buildobj[i]), Vbuild_files);
   551       Vbuild_files = Fpurecopy (Vbuild_files);
   552     }
   553 
   554   fd = emacs_open (name, O_RDONLY, 0);
   555   if (fd < 0)
   556     {
   557       int open_errno = errno;
   558       report_file_errno ("Opening doc string file", build_string (name),
   559                          open_errno);
   560     }
   561   record_unwind_protect_int (close_file_unwind, fd);
   562   Vdoc_file_name = filename;
   563   filled = 0;
   564   pos = 0;
   565   while (true)
   566     {
   567       if (filled < 512)
   568         filled += emacs_read_quit (fd, &buf[filled], sizeof buf - 1 - filled);
   569       if (!filled)
   570         break;
   571 
   572       buf[filled] = 0;
   573       char *end = buf + (filled < 512 ? filled : filled - 128);
   574       p = memchr (buf, '\037', end - buf);
   575       /* p points to ^_Ffunctionname\n or ^_Vvarname\n or ^_Sfilename\n.  */
   576       if (p)
   577         {
   578           end = strchr (p, '\n');
   579           if (!end)
   580             error ("DOC file invalid at position %"pI"d", pos);
   581 
   582           /* We used to skip files not in build_files, so that when a
   583              function was defined several times in different files
   584              (typically, once in xterm, once in w32term, ...), we only
   585              paid attention to the relevant one.
   586 
   587              But this meant the doc had to be kept and updated in
   588              multiple files.  Nowadays we keep the doc only in eg xterm.
   589              The (f)boundp checks below ensure we don't report
   590              docs for eg w32-specific items on X.
   591           */
   592 
   593           sym = oblookup (Vobarray, p + 2,
   594                           multibyte_chars_in_text ((unsigned char *) p + 2,
   595                                                    end - p - 2),
   596                           end - p - 2);
   597           /* Ignore docs that start with SKIP.  These mark
   598              placeholders where the real doc is elsewhere.  */
   599           if (SYMBOLP (sym))
   600             {
   601               /* Attach a docstring to a variable?  */
   602               if (p[1] == 'V')
   603                 {
   604                   /* Install file-position as variable-documentation property
   605                      and make it negative for a user-variable
   606                      (doc starts with a `*').  */
   607                   if ((!NILP (Fboundp (sym))
   608                       || !NILP (Fmemq (sym, delayed_init)))
   609                       && strncmp (end, "\nSKIP", 5))
   610                     Fput (sym, Qvariable_documentation,
   611                           make_fixnum ((pos + end + 1 - buf)
   612                                        * (end[1] == '*' ? -1 : 1)));
   613                 }
   614 
   615               /* Attach a docstring to a function?  */
   616               else if (p[1] == 'F')
   617                 {
   618                   if (!NILP (Ffboundp (sym)) && strncmp (end, "\nSKIP", 5))
   619                     store_function_docstring (sym, pos + end + 1 - buf);
   620                 }
   621               else if (p[1] == 'S')
   622                 ; /* Just a source file name boundary marker.  Ignore it.  */
   623 
   624               else
   625                 error ("DOC file invalid at position %"pI"d", pos);
   626             }
   627         }
   628       pos += end - buf;
   629       filled -= end - buf;
   630       memmove (buf, end, filled);
   631     }
   632 
   633   return SAFE_FREE_UNBIND_TO (count, Qnil);
   634 }

   636 /* Return true if text quoting style should default to quote `like this'.  */
   637 static bool
   638 default_to_grave_quoting_style (void)
   639 {
   640   if (!text_quoting_flag)
   641     return true;
   642   if (! DISP_TABLE_P (Vstandard_display_table))
   643     return false;
   644   Lisp_Object dv = DISP_CHAR_VECTOR (XCHAR_TABLE (Vstandard_display_table),
   645                                      LEFT_SINGLE_QUOTATION_MARK);
   646   return (VECTORP (dv) && ASIZE (dv) == 1
   647           && BASE_EQ (AREF (dv, 0), make_fixnum ('`')));
   648 }
   649 
   650 DEFUN ("text-quoting-style", Ftext_quoting_style,
   651        Stext_quoting_style, 0, 0, 0,
   652        doc: /* Return the current effective text quoting style.
   653 If the variable `text-quoting-style' is `grave', `straight' or
   654 `curve', just return that value.  If it is nil (the default), return
   655 `grave' if curved quotes cannot be displayed (for instance, on a
   656 terminal with no support for these characters), otherwise return
   657 `quote'.  Any other value is treated as `grave'.
   658 
   659 Note that in contrast to the variable `text-quoting-style', this
   660 function will never return nil.  */)
   661   (void)
   662 {
   663   /* Use grave accent and apostrophe `like this'.  */
   664   if (NILP (Vtext_quoting_style)
   665       ? default_to_grave_quoting_style ()
   666       : EQ (Vtext_quoting_style, Qgrave))
   667     return Qgrave;
   668 
   669   /* Use apostrophes 'like this'.  */
   670   else if (EQ (Vtext_quoting_style, Qstraight))
   671     return Qstraight;
   672 
   673   /* Use curved single quotes ‘like this’.  */
   674   else
   675     return Qcurve;
   676 }
