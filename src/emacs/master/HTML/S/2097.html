    81 /* Print a call tree containing all annotated functions.
    82 
    83    The call structure of the functions is represented using
    84    indentation and vertical lines.  Extra information is printed using
    85    horizontal lines that connect to the vertical line.
    86 
    87    The return value is represented using the arrow "->>".  For simple
    88    functions, the arrow can be printed on the same line as the
    89    function name.  If more output is printed, it is connected to the
    90    vertical line of the function.
    91 
    92    The first column contains the file name, the second the line
    93    number, and the third a number increasing for each trace line.
    94 
    95    Note, when trace output from several threads are mixed, the output
    96    can become misaligned, as all threads (currently) share one state.
    97    This is post prominent when the EVENTS part is enabled.

   184 /* printf-style trace output.  Output is aligned with contained heading.  */
   185 #define NSTRACE_MSG_NO_DASHES(...)                                          \
   186   do                                                                        \
   187     {                                                                       \
   188       if (nstrace_enabled)                                                  \
   189         {                                                                   \
   190           fprintf (stderr, "%-10s:%5d: [%5d]%.*s",                          \
   191                    __FILE__, __LINE__, nstrace_num++,                       \
   192                    2*nstrace_depth, "  | | | | | | | | | | | | | | | ..");  \
   193           fprintf (stderr, __VA_ARGS__);                                    \
   194           putc ('\n', stderr);                                              \
   195         }                                                                   \
   196     }                                                                       \
   197   while(0)

   271                          arguments.

   273    NSTRACE_WHEN (cond, fmt, ...) -- Enable trace output when COND is true.

   280 #define NSTRACE_WHEN(cond, ...)                                         \
   281   __attribute__((cleanup(nstrace_restore_global_trace_state)))          \
   282   int nstrace_saved_enabled_global = nstrace_enabled_global;            \
   283   __attribute__((cleanup(nstrace_leave)))                               \
   284   int nstrace_enabled = nstrace_enabled_global && (cond);               \
   285   if (nstrace_enabled) { ++nstrace_depth; }                             \
   286   else { nstrace_enabled_global = 0; }                                  \
   287   NSTRACE_MSG_NO_DASHES(__VA_ARGS__);
