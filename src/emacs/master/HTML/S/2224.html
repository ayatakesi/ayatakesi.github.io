   170 /* Re-define FD_ISSET to avoid a WSA call while we are not using
   171    network sockets.  */
   172 static int
   173 rpl_fd_isset (SOCKET fd, fd_set * set)
   174 {
   175   u_int i;
   176   if (set == NULL)
   177     return 0;
   178 
   179   for (i = 0; i < set->fd_count; i++)
   180     if (set->fd_array[i] == fd)
   181       return 1;
   182 
   183   return 0;
   184 }
