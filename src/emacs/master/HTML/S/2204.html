   149 file_access_p (char const *file, int amode)
   150 {
   151 #ifdef MSDOS

   173   errno = err;

   175 
   176   return false;
   177 }

   285        Sfind_file_name_handler, 2, 2, 0,
   286        doc: /* Return FILENAME's handler function for OPERATION, if it has one.
   287 Otherwise, return nil.
   288 A file name is handled if one of the regular expressions in
   289 `file-name-handler-alist' matches it.
   290 
   291 If OPERATION equals `inhibit-file-name-operation', then ignore
   292 any handlers that are members of `inhibit-file-name-handlers',
   293 but still do run any other handlers.  This lets handlers
   294 use the standard functions without calling themselves recursively.  */)
   295   (Lisp_Object filename, Lisp_Object operation)
   296 {
   297   /* This function must not munge the match data.  */
   298   Lisp_Object chain, inhibited_handlers, result;
   299   ptrdiff_t pos = -1;
   300 
   301   result = Qnil;
   302   CHECK_STRING (filename);
   303 
   304   if (EQ (operation, Vinhibit_file_name_operation))
   305     inhibited_handlers = Vinhibit_file_name_handlers;
   306   else
   307     inhibited_handlers = Qnil;
   308 
   309   for (chain = Vfile_name_handler_alist; CONSP (chain);
   310        chain = XCDR (chain))
   311     {
   312       Lisp_Object elt;
   313       elt = XCAR (chain);
   314       if (CONSP (elt))
   315         {
   316           Lisp_Object string = XCAR (elt);
   317           ptrdiff_t match_pos;
   318           Lisp_Object handler = XCDR (elt);
   319           Lisp_Object operations = Qnil;
   320 
   321           if (SYMBOLP (handler))
   322             operations = Fget (handler, Qoperations);
   323 
   324           if (STRINGP (string)
   325               && (match_pos = fast_string_match (string, filename)) > pos
   326               && (NILP (operations) || ! NILP (Fmemq (operation, operations))))
   327             {
   328               Lisp_Object tem;
   329 
   330               handler = XCDR (elt);
   331               tem = Fmemq (handler, inhibited_handlers);
   332               if (NILP (tem))
   333                 {
   334                   result = handler;
   335                   pos = match_pos;
   336                 }
   337             }
   338         }
   339 
   340       maybe_quit ();
   341     }
   342   return result;
   343 }

   370 /* Return the directory component of FILENAME, or nil if FILENAME does
   371    not contain a directory component.  */

   373 static Lisp_Object
   374 file_name_directory (Lisp_Object filename)
   375 {
   376   char *beg = SSDATA (filename);
   377   char const *p = beg + SBYTES (filename);
   378 
   379   while (p != beg && !IS_DIRECTORY_SEP (p[-1])
   380 #ifdef DOS_NT
   381          /* only recognize drive specifier at the beginning */
   382          && !(p[-1] == ':'
   383               /* handle the "/:d:foo" and "/:foo" cases correctly  */
   384               && ((p == beg + 2 && !IS_DIRECTORY_SEP (*beg))
   385                   || (p == beg + 4 && IS_DIRECTORY_SEP (*beg))))
   386 #endif
   387          ) p--;
   388 
   389   if (p == beg)
   390     return Qnil;
   391 #ifdef DOS_NT
   392   /* Expansion of "c:" to drive and default directory.  */
   393   Lisp_Object tem_fn;
   394   USE_SAFE_ALLOCA;
   395   SAFE_ALLOCA_STRING (beg, filename);
   396   p = beg + (p - SSDATA (filename));
   397 
   398   if (p[-1] == ':')
   399     {
   400       /* MAXPATHLEN+1 is guaranteed to be enough space for getdefdir.  */
   401       char *res = alloca (MAXPATHLEN + 1);
   402       char *r = res;
   403 
   404       if (p == beg + 4 && IS_DIRECTORY_SEP (*beg) && beg[1] == ':')
   405         {
   406           memcpy (res, beg, 2);
   407           beg += 2;
   408           r += 2;
   409         }
   410 
   411       if (getdefdir (c_toupper (*beg) - 'A' + 1, r))
   412         {
   413           size_t l = strlen (res);
   414 
   415           if (l > 3 || !IS_DIRECTORY_SEP (res[l - 1]))
   416             strcat (res, "/");
   417           beg = res;
   418           p = beg + strlen (beg);
   419           dostounix_filename (beg);
   420           tem_fn = make_specified_string (beg, -1, p - beg,
   421                                           STRING_MULTIBYTE (filename));
   422         }
   423       else
   424         tem_fn = make_specified_string (beg - 2, -1, p - beg + 2,
   425                                         STRING_MULTIBYTE (filename));
   426     }
   427   else if (STRING_MULTIBYTE (filename))
   428     {
   429       tem_fn = make_specified_string (beg, -1, p - beg, 1);
   430       dostounix_filename (SSDATA (tem_fn));
   431 #ifdef WINDOWSNT
   432       if (!NILP (Vw32_downcase_file_names))
   433         tem_fn = Fdowncase (tem_fn);
   434 #endif
   435     }
   436   else
   437     {
   438       dostounix_filename (beg);
   439       tem_fn = make_specified_string (beg, -1, p - beg, 0);
   440     }
   441   SAFE_FREE ();
   442   return tem_fn;
   443 #else  /* DOS_NT */
   444   return make_specified_string (beg, -1, p - beg, STRING_MULTIBYTE (filename));
   445 #endif  /* DOS_NT */
   446 }

   758        doc: /* Append COMPONENTS to DIRECTORY and return the resulting string.
   759 Elements in COMPONENTS must be a string or nil.
   760 DIRECTORY or the non-final elements in COMPONENTS may or may not end
   761 with a slash -- if they don't end with a slash, a slash will be
   762 inserted before contatenating.
   763 usage: (record DIRECTORY &rest COMPONENTS) */)
   764   (ptrdiff_t nargs, Lisp_Object *args)
   765 {
   766   ptrdiff_t chars = 0, bytes = 0, multibytes = 0, eargs = 0;
   767   Lisp_Object *elements = args;
   768   Lisp_Object result;
   769   ptrdiff_t i;
   770 
   771   /* First go through the list to check the types and see whether
   772      they're all of the same multibytedness. */
   773   for (i = 0; i < nargs; i++)
   774     {
   775       Lisp_Object arg = args[i];
   776       /* Skip empty and nil elements. */
   777       if (NILP (arg))
   778         continue;
   779       CHECK_STRING (arg);
   780       if (SCHARS (arg) == 0)
   781         continue;
   782       eargs++;
   783       /* Multibyte and non-ASCII. */
   784       if (STRING_MULTIBYTE (arg) && SCHARS (arg) != SBYTES (arg))
   785         multibytes++;

   787       if (i == nargs - 1
   788           || IS_DIRECTORY_SEP (*(SSDATA (arg) + SBYTES (arg) - 1)))
   789         {
   790           bytes += SBYTES (arg);
   791           chars += SCHARS (arg);
   792         }
   793       else

   795           bytes += SBYTES (arg) + 1;
   796           chars += SCHARS (arg) + 1;
   797         }
   798     }
   799 
   800   /* Convert if needed. */
   801   if ((multibytes != 0 && multibytes != nargs)
   802       || eargs != nargs)
   803     {
   804       int j = 0;
   805       elements = xmalloc (eargs * sizeof *elements);
   806       bytes = 0;
   807       chars = 0;
   808 
   809       /* Filter out nil/"". */
   810       for (i = 0; i < nargs; i++)
   811         {
   812           Lisp_Object arg = args[i];
   813           if (!NILP (arg) && SCHARS (arg) != 0)
   814             elements[j++] = arg;
   815         }
   816 
   817       for (i = 0; i < eargs; i++)
   818         {
   819           Lisp_Object arg = elements[i];
   820           /* Use multibyte or all-ASCII strings as is. */
   821           if (!STRING_MULTIBYTE (arg) && !string_ascii_p (arg))
   822             elements[i] = Fstring_to_multibyte (arg);
   823           arg = elements[i];
   824           /* We have to recompute the number of bytes. */
   825           if (i == eargs - 1
   826               || IS_DIRECTORY_SEP (*(SSDATA (arg) + SBYTES (arg) - 1)))
   827             {
   828               bytes += SBYTES (arg);
   829               chars += SCHARS (arg);
   830             }
   831           else
   832             {
   833               bytes += SBYTES (arg) + 1;
   834               chars += SCHARS (arg) + 1;
   835             }
   836         }
   837     }
   838 
   839   /* Allocate an empty string. */
   840   if (multibytes == 0)
   841     result = make_uninit_string (chars);
   842   else
   843     result = make_uninit_multibyte_string (chars, bytes);
   844   /* Null-terminate the string. */
   845   *(SSDATA (result) + SBYTES (result)) = 0;
   846 
   847   /* Copy over the data. */
   848   char *p = SSDATA (result);
   849   for (i = 0; i < eargs; i++)
   850     {
   851       Lisp_Object arg = elements[i];
   852       memcpy (p, SSDATA (arg), SBYTES (arg));
   853       p += SBYTES (arg);
   854       /* The last element shouldn't have a slash added at the end. */
   855       if (i < eargs - 1 && !IS_DIRECTORY_SEP (*(p - 1)))
   856         *p++ = DIRECTORY_SEP;
   857     }
   858 
   859   if (elements != args)
   860     xfree (elements);
   861 
   862   return result;
   863 }

   907 components followed by `..', along with the `..' itself; note that

   912 at the beginning of the file name when they are significant (e.g., UNC

   925 \(directory-file-name (file-name-directory dirname)) to traverse a

  1297 #ifdef DOS_NT
  1298     && !is_escaped              /* don't expand ~ in escaped file names */
  1299 #endif
  1300       )
  1301     {
  1302       if (IS_DIRECTORY_SEP (nm[1])
  1303           || nm[1] == 0)        /* ~ by itself */

  1305           Lisp_Object tem;
  1306 
  1307           newdir = get_homedir ();
  1308           nm++;

  1312              if we expect the directory to be multibyte.  */

  1314             {
  1315               hdir = DECODE_FILE (tem);
  1316               newdir = SSDATA (hdir);
  1317               newdirlim = newdir + SBYTES (hdir);
  1318             }
  1319           else if (!multibyte && STRING_MULTIBYTE (tem))
  1320             multibyte = 1;
  1321 #ifdef DOS_NT
  1322           collapse_newdir = false;
  1323 #endif
  1324         }
  1325       else                      /* ~user/filename */

  1327           char *nmhome = user_homedir (nm + 1);

  1350         }

  1388 #endif /* DOS_NT */

  1407       /* Note if special escape prefix is present, but remove for now.  */
  1408       if (newdir[0] == '/' && newdir[1] == ':')
  1409         {
  1410           is_escaped = 1;
  1411           newdir += 2;

  1413 #endif
  1414     }

  1416 #ifdef DOS_NT

  1426           && ! (IS_DIRECTORY_SEP (newdir[0]) && IS_DIRECTORY_SEP (newdir[1])

  1536       if (nm[0] == 0 || IS_DIRECTORY_SEP (nm[0]))

  1544           if (!(drive && nm[0] && IS_DIRECTORY_SEP (newdir[0])

  1553       else

  1611         else if (IS_DIRECTORY_SEP (p[1])
  1612                  && (p != target || IS_DIRECTORY_SEP (p[2])))

  1614           p++;
  1615         else
  1616           {
  1617             *o++ = *p++;
  1618           }
  1619       }
  1620 
  1621 #ifdef DOS_NT
  1622     /* At last, set drive name.  */
  1623 #ifdef WINDOWSNT
  1624     /* Except for network file name.  */
  1625     if (!(IS_DIRECTORY_SEP (target[0]) && IS_DIRECTORY_SEP (target[1])))
  1626 #endif /* WINDOWSNT */
  1627       {
  1628         if (!drive) emacs_abort ();
  1629         target -= 2;
  1630         target[0] = DRIVE_LETTER (drive);
  1631         target[1] = ':';
  1632       }
  1633     /* Reinsert the escape prefix if required.  */
  1634     if (is_escaped)
  1635       {
  1636         target -= 2;
  1637         target[0] = '/';
  1638         target[1] = ':';
  1639       }
  1640     result = make_specified_string (target, -1, o - target, multibyte);
  1641     dostounix_filename (SSDATA (result));
  1642 #ifdef WINDOWSNT
  1643     if (!NILP (Vw32_downcase_file_names))
  1644       result = Fdowncase (result);
  1645 #endif
  1646 #else  /* !DOS_NT */
  1647     result = make_specified_string (target, -1, o - target, multibyte);

  1649   }

  1652      and perhaps call the corresponding file name handler.  This is needed

  1658     {

  1660                             result, default_directory);
  1661       if (! STRINGP (handled_name))
  1662         error ("Invalid handler in `file-name-handler-alist'");
  1663       result = handled_name;
  1664     }
  1665 
  1666   SAFE_FREE ();
  1667   return result;

  1706   if (nm[0] == '/')
  1707     {
  1708       bool lose = 0;
  1709       p = nm;
  1710       while (*p)
  1711         {
  1712           if (p[0] == '/' && p[1] == '/')
  1713             nm = p + 1;
  1714           if (p[0] == '/' && p[1] == '~')
  1715             nm = p + 1, lose = 1;
  1716           if (p[0] == '/' && p[1] == '.'
  1717               && (p[2] == '/' || p[2] == 0
  1718                   || (p[2] == '.' && (p[3] == '/' || p[3] == 0))))
  1719             lose = 1;
  1720           p++;
  1721         }
  1722       if (!lose)
  1723         {
  1724           if (nm == SDATA (name))
  1725             return name;
  1726           return build_string (nm);
  1727         }

  1729 
  1730   /* Now determine directory to start with and put it in NEWDIR.  */
  1731 
  1732   newdir = 0;
  1733

  1735     if (nm[1] == '/' || nm[1] == 0)/* ~/filename */
  1736       {
  1737         if (!(newdir = (unsigned char *) egetenv ("HOME")))
  1738           newdir = (unsigned char *) "";
  1739         nm++;
  1740       }
  1741     else  /* ~user/filename */
  1742       {
  1743         /* Get past ~ to user.  */
  1744         unsigned char *user = nm + 1;
  1745         /* Find end of name.  */
  1746         unsigned char *ptr = (unsigned char *) strchr (user, '/');
  1747         ptrdiff_t len = ptr ? ptr - user : strlen (user);
  1748         /* Copy the user name into temp storage.  */
  1749         o = alloca (len + 1);
  1750         memcpy (o, user, len);
  1751         o[len] = 0;
  1752 
  1753         /* Look up the user name.  */
  1754         block_input ();
  1755         pw = (struct passwd *) getpwnam (o + 1);
  1756         unblock_input ();

  1758           error ("\"%s\" isn't a registered user", o + 1);
  1759 
  1760         newdir = (unsigned char *) pw->pw_dir;
  1761 
  1762         /* Discard the user name from NM.  */
  1763         nm += len;
  1764       }
  1765 
  1766   if (nm[0] != '/' && !newdir)
  1767     {
  1768       if (NILP (defalt))
  1769         defalt = current_buffer->directory;
  1770       CHECK_STRING (defalt);
  1771       newdir = SDATA (defalt);
  1772     }
  1773 
  1774   /* Now concatenate the directory and name to new space in the stack frame.  */
  1775 
  1776   tlen = (newdir ? strlen (newdir) + 1 : 0) + strlen (nm) + 1;
  1777   target = alloca (tlen);
  1778   *target = 0;
  1779 
  1780   if (newdir)
  1781     {
  1782       if (nm[0] == 0 || nm[0] == '/')
  1783         strcpy (target, newdir);
  1784       else
  1785       file_name_as_directory (target, newdir);
  1786     }
  1787 
  1788   strcat (target, nm);
  1789 
  1790   /* Now canonicalize by removing /. and /foo/.. if they appear.  */
  1791 
  1792   p = target;
  1793   o = target;
  1794 
  1795   while (*p)
  1796     {
  1797       if (*p != '/')
  1798         {
  1799           *o++ = *p++;
  1800         }
  1801       else if (!strncmp (p, "//", 2)
  1802                )
  1803         {
  1804           o = target;
  1805           p++;
  1806         }
  1807       else if (p[0] == '/' && p[1] == '.'
  1808                && (p[2] == '/' || p[2] == 0))
  1809         p += 2;
  1810       else if (!strncmp (p, "/..", 3)
  1811                /* `/../' is the "superroot" on certain file systems.  */
  1812                && o != target
  1813                && (p[3] == '/' || p[3] == 0))
  1814         {
  1815           while (o != target && *--o != '/')
  1816             ;
  1817           if (o == target && *o == '/')
  1818             ++o;
  1819           p += 3;
  1820         }
  1821       else
  1822         {
  1823           *o++ = *p++;
  1824         }
  1825     }

  1827   return make_string (target, o - target);

  2391     }

  2398       ts[1] = get_stat_mtime (&st);

  2400         xsignal2 (Qfile_date_error,
  2401                   build_string ("Cannot set file date"), newname);

  2405     report_file_error ("Write error", newname);

  2407   emacs_close (ifd);

  2412      get only the READ bit, which will make the copied file read-only,

  2415     chmod (SDATA (encoded_newname), st.st_mode & 07777);

  2417 #endif /* not WINDOWSNT */
  2418 
  2419   /* Discard the unwind protects.  */
  2420   specpdl_ptr = specpdl_ref_to_ptr (count);

  3069 /* Return true if FILE is a directory or a symlink to a directory.
  3070    Otherwise return false and set errno.  */
  3071 bool
  3072 file_directory_p (Lisp_Object file)
  3073 {
  3074 #ifdef DOS_NT
  3075   /* This is cheaper than 'stat'.  */
  3076   bool retval = faccessat (AT_FDCWD, SSDATA (file), D_OK, AT_EACCESS) == 0;
  3077   if (!retval && errno == EACCES)
  3078     errno = ENOTDIR;    /* like the non-DOS_NT branch below does */
  3079   return retval;
  3080 #else
  3081 # ifdef O_PATH
  3082   /* Use O_PATH if available, as it avoids races and EOVERFLOW issues.  */
  3083   int fd = emacs_openat (AT_FDCWD, SSDATA (file),
  3084                          O_PATH | O_CLOEXEC | O_DIRECTORY, 0);
  3085   if (0 <= fd)
  3086     {
  3087       emacs_close (fd);
  3088       return true;
  3089     }
  3090   if (errno != EINVAL)
  3091     return false;
  3092   /* O_PATH is defined but evidently this Linux kernel predates 2.6.39.
  3093      Fall back on generic POSIX code.  */
  3094 # endif
  3095   /* Use file_accessible_directory_p, as it avoids fstatat EOVERFLOW
  3096      problems and could be cheaper.  However, if it fails because FILE
  3097      is inaccessible, fall back on fstatat; if the latter fails with
  3098      EOVERFLOW then FILE must have been a directory unless a race
  3099      condition occurred (a problem hard to work around portably).  */
  3100   if (file_accessible_directory_p (file))
  3101     return true;
  3102   if (errno != EACCES)
  3103     return false;
  3104   struct stat st;
  3105   if (emacs_fstatat (AT_FDCWD, SSDATA (file), &st, 0) != 0)
  3106     return errno == EOVERFLOW;
  3107   if (S_ISDIR (st.st_mode))
  3108     return true;
  3109   errno = ENOTDIR;
  3110   return false;
  3111 #endif
  3112 }

  3159 /* If FILE is a searchable directory or a symlink to a
  3160    searchable directory, return true.  Otherwise return
  3161    false and set errno to an error number.  */
  3162 bool
  3163 file_accessible_directory_p (Lisp_Object file)
  3164 {
  3165 #ifdef DOS_NT
  3166 # ifdef WINDOWSNT
  3167   /* We need a special-purpose test because (a) NTFS security data is
  3168      not reflected in Posix-style mode bits, and (b) the trick with
  3169      accessing "DIR/.", used below on Posix hosts, doesn't work on
  3170      Windows, because "DIR/." is normalized to just "DIR" before
  3171      hitting the disk.  */
  3172   return (SBYTES (file) == 0
  3173           || w32_accessible_directory_p (SSDATA (file), SBYTES (file)));
  3174 # else  /* MSDOS */
  3175   return file_directory_p (file);
  3176 # endif  /* MSDOS */
  3177 #else    /* !DOS_NT */
  3178   /* On POSIXish platforms, use just one system call; this avoids a
  3179      race and is typically faster.  */
  3180   const char *data = SSDATA (file);
  3181   ptrdiff_t len = SBYTES (file);
  3182   char const *dir;
  3183   bool ok;
  3184   USE_SAFE_ALLOCA;
  3185 
  3186   /* Normally a file "FOO" is an accessible directory if "FOO/." exists.
  3187      There are three exceptions: "", "/", and "//".  Leave "" alone,
  3188      as it's invalid.  Append only "." to the other two exceptions as
  3189      "/" and "//" are distinct on some platforms, whereas "/", "///",
  3190      "////", etc. are all equivalent.  */
  3191   if (! len)
  3192     dir = data;
  3193   else
  3194     {
  3195       /* Just check for trailing '/' when deciding whether append '/'
  3196          before appending '.'.  That's simpler than testing the two
  3197          special cases "/" and "//", and it's a safe optimization
  3198          here.  After appending '.', append another '/' to work around
  3199          a macOS bug (Bug#30350).  */
  3200       static char const appended[] = "/./";
  3201       char *buf = SAFE_ALLOCA (len + sizeof appended);
  3202       memcpy (buf, data, len);
  3203       strcpy (buf + len, &appended[data[len - 1] == '/']);
  3204       dir = buf;
  3205     }
  3206 
  3207   ok = file_access_p (dir, F_OK);

  3678 /* This function is called after Lisp functions to decide a coding
  3679    system are called, or when they cause an error.  Before they are
  3680    called, the current buffer is set unibyte and it contains only a
  3681    newly inserted text (thus the buffer was empty before the
  3682    insertion).
  3683 
  3684    The functions may set markers, overlays, text properties, or even
  3685    alter the buffer contents, change the current buffer.
  3686 
  3687    Here, we reset all those changes by:
  3688         o set back the current buffer.
  3689         o move all markers and overlays to BEG.
  3690         o remove all text properties.
  3691         o set back the buffer multibyteness.  */
  3692 
  3693 static void
  3694 decide_coding_unwind (Lisp_Object unwind_data)
  3695 {
  3696   Lisp_Object multibyte = XCAR (unwind_data);
  3697   Lisp_Object tmp = XCDR (unwind_data);
  3698   Lisp_Object undo_list = XCAR (tmp);
  3699   Lisp_Object buffer = XCDR (tmp);
  3700 
  3701   set_buffer_internal (XBUFFER (buffer));
  3702 
  3703   /* We're about to "delete" the text by moving it back into the gap.
  3704      So move markers that set-auto-coding might have created to BEG,
  3705      just in case.  */
  3706   adjust_markers_for_delete (BEG, BEG_BYTE, Z, Z_BYTE);
  3707   adjust_overlays_for_delete (BEG, Z - BEG);
  3708   set_buffer_intervals (current_buffer, NULL);
  3709   TEMP_SET_PT_BOTH (BEG, BEG_BYTE);
  3710 
  3711   /* In case of a non-local exit from set_auto_coding_function, in order not
  3712      to end up with potentially invalid byte sequences in a multibyte buffer,
  3713      we have the following options:
  3714      1- decode the bytes in some arbitrary coding-system.
  3715      2- erase the buffer.
  3716      3- leave the buffer unibyte (which is actually the same as option (1)
  3717         where the coding-system is `raw-text-unix`).
  3718      Here we choose 2.  */
  3719 
  3720   /* Move the bytes back to (the beginning of) the gap.
  3721      In general this may have to move all the bytes, but here
  3722      this can't move more bytes than were moved during the execution
  3723      of Vset_auto_coding_function, which is normally 0 (because it
  3724      normally doesn't modify the buffer).  */
  3725   move_gap_both (Z, Z_BYTE);
  3726   ptrdiff_t inserted = Z_BYTE - BEG_BYTE;
  3727   GAP_SIZE += inserted;
  3728   ZV = Z = GPT = BEG;
  3729   ZV_BYTE = Z_BYTE = GPT_BYTE = BEG_BYTE;
  3730 
  3731   /* Pass the new `inserted` back.  */
  3732   XSETCAR (unwind_data, make_fixnum (inserted));
  3733 
  3734   /* Now we are safe to change the buffer's multibyteness directly.  */
  3735   bset_enable_multibyte_characters (current_buffer, multibyte);
  3736   bset_undo_list (current_buffer, undo_list);
  3737 }

  4460       maybe_move_gap (XBUFFER (conversion_buffer));
  4461       decoded = BUF_BEG_ADDR (XBUFFER (conversion_buffer));
  4462       inserted = (BUF_Z_BYTE (XBUFFER (conversion_buffer))
  4463                   - BUF_BEG_BYTE (XBUFFER (conversion_buffer)));
  4464 
  4465       /* Compare the beginning of the converted string with the buffer
  4466          text.  */
  4467 
  4468       bufpos = 0;
  4469       while (bufpos < inserted && same_at_start < same_at_end
  4470              && FETCH_BYTE (same_at_start) == decoded[bufpos])
  4471         same_at_start++, bufpos++;
  4472 
  4473       /* If the file matches the head of buffer completely,
  4474          there's no need to replace anything.  */
  4475 
  4476       if (bufpos == inserted)
  4477         {
  4478           /* Truncate the buffer to the size of the file.  */
  4479           if (same_at_start != same_at_end)
  4480             {
  4481               /* See previous specbind for the reason behind this.  */
  4482               specbind (intern ("buffer-file-name"), Qnil);
  4483               del_range_byte (same_at_start, same_at_end);
  4484             }
  4485           inserted = 0;
  4486 
  4487           unbind_to (this_count, Qnil);
  4488           goto handled;
  4489         }
  4490 
  4491       /* Extend the start of non-matching text area to the previous
  4492          multibyte character boundary.  */
  4493       if (! NILP (BVAR (current_buffer, enable_multibyte_characters)))
  4494         while (same_at_start > BEGV_BYTE
  4495                && ! CHAR_HEAD_P (FETCH_BYTE (same_at_start)))
  4496           same_at_start--;
  4497 
  4498       /* Scan this bufferful from the end, comparing with
  4499          the Emacs buffer.  */
  4500       bufpos = inserted;
  4501 
  4502       /* Compare with same_at_start to avoid counting some buffer text
  4503          as matching both at the file's beginning and at the end.  */
  4504       while (bufpos > 0 && same_at_end > same_at_start
  4505              && FETCH_BYTE (same_at_end - 1) == decoded[bufpos - 1])
  4506         same_at_end--, bufpos--;
  4507 
  4508       /* Extend the end of non-matching text area to the next
  4509          multibyte character boundary.  */
  4510       if (! NILP (BVAR (current_buffer, enable_multibyte_characters)))
  4511         while (same_at_end < ZV_BYTE
  4512                && ! CHAR_HEAD_P (FETCH_BYTE (same_at_end)))
  4513           same_at_end++;
  4514 
  4515       /* Don't try to reuse the same piece of text twice.  */
  4516       overlap = same_at_start - BEGV_BYTE - (same_at_end + inserted - ZV_BYTE);
  4517       if (overlap > 0)
  4518         same_at_end += overlap;
  4519       same_at_end_charpos = BYTE_TO_CHAR (same_at_end);
  4520 
  4521       /* If display currently starts at beginning of line,
  4522          keep it that way.  */
  4523       if (XBUFFER (XWINDOW (selected_window)->contents) == current_buffer)
  4524         XWINDOW (selected_window)->start_at_line_beg = !NILP (Fbolp ());
  4525 
  4526       /* Replace the chars that we need to replace,
  4527          and update INSERTED to equal the number of bytes
  4528          we are taking from the decoded string.  */
  4529       inserted -= (ZV_BYTE - same_at_end) + (same_at_start - BEGV_BYTE);
  4530 
  4531       /* See previous specbind for the reason behind this.  */
  4532       specbind (intern ("buffer-file-name"), Qnil);
  4533       if (same_at_end != same_at_start)
  4534         {
  4535           del_range_byte (same_at_start, same_at_end);
  4536           temp = GPT;
  4537           eassert (same_at_start == GPT_BYTE);
  4538           same_at_start = GPT_BYTE;
  4539         }
  4540       else
  4541         {
  4542           temp = same_at_end_charpos;
  4543         }
  4544       /* Insert from the file at the proper position.  */
  4545       SET_PT_BOTH (temp, same_at_start);
  4546       same_at_start_charpos
  4547         = buf_bytepos_to_charpos (XBUFFER (conversion_buffer),
  4548                                   same_at_start - BEGV_BYTE
  4549                                   + BUF_BEG_BYTE (XBUFFER (conversion_buffer)));
  4550       eassert (same_at_start_charpos == temp - (BEGV - BEG));
  4551       inserted_chars
  4552         = (buf_bytepos_to_charpos (XBUFFER (conversion_buffer),
  4553                                    same_at_start + inserted - BEGV_BYTE
  4554                                   + BUF_BEG_BYTE (XBUFFER (conversion_buffer)))
  4555            - same_at_start_charpos);
  4556       insert_from_buffer (XBUFFER (conversion_buffer),
  4557                           same_at_start_charpos, inserted_chars, 0);
  4558       /* Set `inserted' to the number of inserted characters.  */
  4559       inserted = PT - temp;
  4560       /* Set point before the inserted characters.  */
  4561       SET_PT_BOTH (temp, same_at_start);
  4562 
  4563       unbind_to (this_count, Qnil);
  4564 
  4565       goto handled;
  4566     }
  4567 
  4568   if (seekable || !NILP (end))
  4569     total = end_offset - beg_offset;
  4570   else
  4571     /* For a special file, all we can do is guess.  */
  4572     total = READ_BUF_SIZE;
  4573 
  4574   if (NILP (visit) && total > 0)
  4575     {
  4576       if (!NILP (BVAR (current_buffer, file_truename))
  4577           /* Make binding buffer-file-name to nil effective.  */
  4578           && !NILP (BVAR (current_buffer, filename))
  4579           && SAVE_MODIFF >= MODIFF)
  4580         we_locked_file = true;
  4581       prepare_to_modify_buffer (PT, PT, NULL);
  4582     }
  4583 
  4584   move_gap_both (PT, PT_BYTE);
  4585 
  4586   /* Ensure the gap is at least one byte larger than needed for the
  4587      estimated file size, so that in the usual case we read to EOF
  4588      without reallocating.  */
  4589   if (GAP_SIZE <= total)
  4590     make_gap (total - GAP_SIZE + 1);
  4591 
  4592   if (beg_offset != 0 || !NILP (replace))
  4593     {
  4594       if (lseek (fd, beg_offset, SEEK_SET) < 0)
  4595         report_file_error ("Setting file position", orig_filename);
  4596     }
  4597 
  4598   /* Total bytes inserted.  */
  4599   inserted = 0;
  4600 
  4601   /* Here, we don't do code conversion in the loop.  It is done by
  4602      decode_coding_gap after all data are read into the buffer.  */
  4603   {
  4604     ptrdiff_t gap_size = GAP_SIZE;
  4605 
  4606     while (NILP (end) || inserted < total)
  4607       {
  4608         ptrdiff_t this;
  4609 
  4610         if (gap_size == 0)
  4611           {
  4612             /* The size estimate was wrong.  Make the gap 50% larger.  */
  4613             make_gap (GAP_SIZE >> 1);
  4614             gap_size = GAP_SIZE - inserted;
  4615           }
  4616 
  4617         /* 'try' is reserved in some compilers (Microsoft C).  */
  4618         ptrdiff_t trytry = min (gap_size, READ_BUF_SIZE);
  4619         if (!NILP (end))
  4620           trytry = min (trytry, total - inserted);
  4621 
  4622         if (!seekable && NILP (end))
  4623           {
  4624             Lisp_Object nbytes;
  4625 
  4626             /* Read from the file, capturing `quit'.  When an
  4627                error occurs, end the loop, and arrange for a quit
  4628                to be signaled after decoding the text we read.  */
  4629             union read_non_regular data = {{fd, inserted, trytry}};
  4630             nbytes = internal_condition_case_1
  4631               (read_non_regular, make_pointer_integer (&data),
  4632                Qerror, read_non_regular_quit);
  4633 
  4634             if (NILP (nbytes))
  4635               {
  4636                 read_quit = 1;
  4637                 break;
  4638               }
  4639 
  4640             this = XFIXNUM (nbytes);
  4641           }
  4642         else
  4643           {
  4644             /* Allow quitting out of the actual I/O.  We don't make text
  4645                part of the buffer until all the reading is done, so a C-g
  4646                here doesn't do any harm.  */
  4647             this = emacs_read_quit (fd,
  4648                                     ((char *) BEG_ADDR + PT_BYTE - BEG_BYTE
  4649                                      + inserted),
  4650                                     trytry);
  4651           }
  4652 
  4653         if (this <= 0)
  4654           {
  4655             read_quit = this;
  4656             break;
  4657           }
  4658 
  4659         gap_size -= this;
  4660         inserted += this;
  4661       }
  4662   }
  4663 
  4664   /* Now we have either read all the file data into the gap,
  4665      or stop reading on I/O error or quit.  If nothing was
  4666      read, undo marking the buffer modified.  */
  4667 
  4668   if (inserted == 0)
  4669     {
  4670       if (we_locked_file)
  4671         Funlock_file (BVAR (current_buffer, file_truename));
  4672       Vdeactivate_mark = old_Vdeactivate_mark;
  4673     }
  4674   else
  4675     Fset (Qdeactivate_mark, Qt);
  4676 
  4677   emacs_close (fd);
  4678   clear_unwind_protect (fd_index);
  4679 
  4680   if (read_quit < 0)
  4681     report_file_error ("Read error", orig_filename);
  4682 
  4683  notfound:
  4684 
  4685   if (NILP (coding_system))
  4686     {
  4687       /* The coding system is not yet decided.  Decide it by an
  4688          optimized method for handling `coding:' tag.
  4689 
  4690          Note that we can get here only if the buffer was empty
  4691          before the insertion.  */
  4692       eassert (Z == BEG);
  4693 
  4694       if (!NILP (Vcoding_system_for_read))
  4695         coding_system = Vcoding_system_for_read;
  4696       else
  4697         {
  4698           /* Since we are sure that the current buffer was empty
  4699              before the insertion, we can toggle
  4700              enable-multibyte-characters directly here without taking
  4701              care of marker adjustment.  By this way, we can run Lisp
  4702              program safely before decoding the inserted text.  */
  4703           Lisp_Object multibyte
  4704             = BVAR (current_buffer, enable_multibyte_characters);
  4705           Lisp_Object unwind_data
  4706             = Fcons (multibyte,
  4707                      Fcons (BVAR (current_buffer, undo_list),
  4708                             Fcurrent_buffer ()));
  4709           specpdl_ref count1 = SPECPDL_INDEX ();
  4710 
  4711           bset_enable_multibyte_characters (current_buffer, Qnil);
  4712           bset_undo_list (current_buffer, Qt);
  4713           record_unwind_protect (decide_coding_unwind, unwind_data);
  4714 
  4715           /* Make the text read part of the buffer.  */
  4716           insert_from_gap_1 (inserted, inserted, false);
  4717 
  4718           if (inserted > 0 && ! NILP (Vset_auto_coding_function))
  4719             {
  4720               coding_system = call2 (Vset_auto_coding_function,
  4721                                      filename, make_fixnum (inserted));
  4722             }
  4723 
  4724           if (NILP (coding_system))
  4725             {
  4726               /* If the coding system is not yet decided, check
  4727                  file-coding-system-alist.  */
  4728               coding_system = CALLN (Ffind_operation_coding_system,
  4729                                      Qinsert_file_contents, orig_filename,
  4730                                      visit, beg, end, Qnil);
  4731               if (CONSP (coding_system))
  4732                 coding_system = XCAR (coding_system);
  4733             }
  4734           /* Move the text back to the gap.  */
  4735           unbind_to (count1, Qnil);
  4736           inserted = XFIXNUM (XCAR (unwind_data));
  4737         }
  4738 
  4739       if (NILP (coding_system))
  4740         coding_system = Qundecided;
  4741       else
  4742         CHECK_CODING_SYSTEM (coding_system);
  4743 
  4744       if (NILP (BVAR (current_buffer, enable_multibyte_characters)))
  4745         /* We must suppress all character code conversion except for
  4746            end-of-line conversion.  */
  4747         coding_system = raw_text_coding_system (coding_system);
  4748       setup_coding_system (coding_system, &coding);
  4749       /* Ensure we set Vlast_coding_system_used.  */
  4750       set_coding_system = true;
  4751     }
  4752 
  4753   if (!NILP (visit))
  4754     {
  4755       /* When we visit a file by raw-text, we change the buffer to
  4756          unibyte.  */
  4757       if (CODING_FOR_UNIBYTE (&coding)
  4758           /* Can't do this if part of the buffer might be preserved.  */
  4759           && NILP (replace))
  4760         {
  4761           /* Visiting a file with these coding system makes the buffer
  4762              unibyte.  */
  4763           if (inserted > 0)
  4764             bset_enable_multibyte_characters (current_buffer, Qnil);
  4765           else
  4766             Fset_buffer_multibyte (Qnil);
  4767         }
  4768     }
  4769 
  4770   eassert (PT == GPT);
  4771 
  4772   coding.dst_multibyte
  4773     = !NILP (BVAR (current_buffer, enable_multibyte_characters));
  4774   if (CODING_MAY_REQUIRE_DECODING (&coding)
  4775       && (inserted > 0 || CODING_REQUIRE_FLUSHING (&coding)))
  4776     {
  4777       /* Now we have all the new bytes at the beginning of the gap,
  4778          but `decode_coding_gap` can't have them at the beginning of the gap,
  4779          so we need to move them.  */
  4780       memmove (GAP_END_ADDR - inserted, GPT_ADDR, inserted);
  4781       decode_coding_gap (&coding, inserted);
  4782       inserted = coding.produced_char;
  4783       coding_system = CODING_ID_NAME (coding.id);
  4784     }
  4785   else if (inserted > 0)
  4786     {
  4787       /* Make the text read part of the buffer.  */
  4788       eassert (NILP (BVAR (current_buffer, enable_multibyte_characters)));
  4789       insert_from_gap_1 (inserted, inserted, false);
  4790 
  4791       invalidate_buffer_caches (current_buffer, PT, PT + inserted);
  4792       adjust_after_insert (PT, PT_BYTE, PT + inserted, PT_BYTE + inserted,
  4793                            inserted);
  4794     }
  4795 
  4796   /* Call after-change hooks for the inserted text, aside from the case
  4797      of normal visiting (not with REPLACE), which is done in a new buffer
  4798      "before" the buffer is changed.  */
  4799   if (inserted > 0 && total > 0
  4800       && (NILP (visit) || !NILP (replace)))
  4801     {
  4802       signal_after_change (PT, 0, inserted);
  4803       update_compositions (PT, PT, CHECK_BORDER);
  4804     }

  4806   /* Now INSERTED is measured in characters.  */
  4807 
  4808  handled:
  4809 
  4810   if (inserted > 0)
  4811     restore_window_points (window_markers, inserted,
  4812                            BYTE_TO_CHAR (same_at_start),
  4813                            same_at_end_charpos);
  4814 
  4815   if (!NILP (visit))
  4816     {
  4817       if (empty_undo_list_p)
  4818         bset_undo_list (current_buffer, Qnil);
  4819 
  4820       if (NILP (handler))
  4821         {
  4822           current_buffer->modtime = mtime;
  4823           current_buffer->modtime_size = st.st_size;
  4824           bset_filename (current_buffer, orig_filename);
  4825         }
  4826 
  4827       SAVE_MODIFF = MODIFF;
  4828       BUF_AUTOSAVE_MODIFF (current_buffer) = MODIFF;
  4829       XSETFASTINT (BVAR (current_buffer, save_length), Z - BEG);
  4830       if (NILP (handler))
  4831         {
  4832           if (!NILP (BVAR (current_buffer, file_truename)))
  4833             Funlock_file (BVAR (current_buffer, file_truename));
  4834           Funlock_file (filename);
  4835         }
  4836       if (!regular)
  4837         xsignal2 (Qfile_error,
  4838                   build_string ("not a regular file"), orig_filename);
  4839     }

  4841   if (set_coding_system)
  4842     Vlast_coding_system_used = coding_system;
  4843 
  4844   if (! NILP (Ffboundp (Qafter_insert_file_set_coding)))
  4845     {
  4846       insval = call2 (Qafter_insert_file_set_coding, make_fixnum (inserted),
  4847                       visit);
  4848       if (! NILP (insval))
  4849         {

  4851             wrong_type_argument (Qinserted_chars, insval);
  4852           inserted = XFIXNAT (insval);
  4853         }
  4854     }
  4855 
  4856   /* Decode file format.  */
  4857   if (inserted > 0)
  4858     {
  4859       /* Don't run point motion or modification hooks when decoding.  */
  4860       specpdl_ref count1 = SPECPDL_INDEX ();
  4861       ptrdiff_t old_inserted = inserted;
  4862       specbind (Qinhibit_point_motion_hooks, Qt);
  4863       specbind (Qinhibit_modification_hooks, Qt);

  4868 
  4869       if (NILP (replace))
  4870         {

  4872                           Qnil, make_fixnum (inserted), visit);

  4874             wrong_type_argument (Qinserted_chars, insval);
  4875           inserted = XFIXNAT (insval);
  4876         }
  4877       else
  4878         {
  4879           /* If REPLACE is non-nil and we succeeded in not replacing the
  4880              beginning or end of the buffer text with the file's contents,

  4889           ptrdiff_t opoint_byte = PT_BYTE;
  4890           ptrdiff_t oinserted = ZV - BEGV;
  4891           modiff_count ochars_modiff = CHARS_MODIFF;

  4893           TEMP_SET_PT_BOTH (BEGV, BEGV_BYTE);

  4895                           Qnil, make_fixnum (oinserted), visit);
  4896           if (! RANGED_FIXNUMP (0, insval, ZV - PT))
  4897             wrong_type_argument (Qinserted_chars, insval);

  4899             /* format_decode didn't modify buffer's characters => move
  4900                point back to position before inserted text and leave
  4901                value of inserted alone.  */
  4902             SET_PT_BOTH (opoint, opoint_byte);
  4903           else
  4904             /* format_decode modified buffer's characters => consider
  4905                entire buffer changed and leave point at point-min.  */
  4906             inserted = XFIXNAT (insval);
  4907         }
  4908 
  4909       /* For consistency with format-decode call these now iff inserted > 0
  4910          (martin 2007-06-28).  */
  4911       p = Vafter_insert_file_functions;
  4912       FOR_EACH_TAIL (p)
  4913         {
  4914           if (NILP (replace))
  4915             {
  4916               insval = call1 (XCAR (p), make_fixnum (inserted));
  4917               if (!NILP (insval))
  4918                 {
  4919                   if (! RANGED_FIXNUMP (0, insval, ZV - PT))
  4920                     wrong_type_argument (Qinserted_chars, insval);
  4921                   inserted = XFIXNAT (insval);
  4922                 }
  4923             }
  4924           else
  4925             {
  4926               /* For the rationale of this see the comment on
  4927                  format-decode above.  */
  4928               ptrdiff_t opoint = PT;
  4929               ptrdiff_t opoint_byte = PT_BYTE;
  4930               ptrdiff_t oinserted = ZV - BEGV;
  4931               modiff_count ochars_modiff = CHARS_MODIFF;
  4932 
  4933               TEMP_SET_PT_BOTH (BEGV, BEGV_BYTE);
  4934               insval = call1 (XCAR (p), make_fixnum (oinserted));
  4935               if (!NILP (insval))
  4936                 {
  4937                   if (! RANGED_FIXNUMP (0, insval, ZV - PT))
  4938                     wrong_type_argument (Qinserted_chars, insval);
  4939                   if (ochars_modiff == CHARS_MODIFF)
  4940                     /* after_insert_file_functions didn't modify
  4941                        buffer's characters => move point back to
  4942                        position before inserted text and leave value of
  4943                        inserted alone.  */
  4944                     SET_PT_BOTH (opoint, opoint_byte);
  4945                   else
  4946                     /* after_insert_file_functions did modify buffer's
  4947                        characters => consider entire buffer changed and
  4948                        leave point at point-min.  */
  4949                     inserted = XFIXNAT (insval);
  4950                 }
  4951             }
  4952         }
  4953 
  4954       if (!empty_undo_list_p)
  4955         {
  4956           bset_undo_list (current_buffer, old_undo);
  4957           if (CONSP (old_undo) && inserted != old_inserted)
  4958             {
  4959               /* Adjust the last undo record for the size change during
  4960                  the format conversion.  */
  4961               Lisp_Object tem = XCAR (old_undo);
  4962               if (CONSP (tem) && FIXNUMP (XCAR (tem))
  4963                   && FIXNUMP (XCDR (tem))
  4964                   && XFIXNUM (XCDR (tem)) == PT + old_inserted)
  4965                 XSETCDR (tem, make_fixnum (PT + inserted));
  4966             }
  4967         }
  4968       else
  4969         /* If undo_list was Qt before, keep it that way.
  4970            Otherwise start with an empty undo_list.  */
  4971         bset_undo_list (current_buffer, EQ (old_undo, Qt) ? Qt : Qnil);
  4972 
  4973       unbind_to (count1, Qnil);
  4974     }
  4975 
  4976   if (save_errno != 0)
  4977     {
  4978       /* Signal an error if visiting a file that could not be opened.  */
  4979       eassert (!NILP (visit) && NILP (handler));
  4980       report_file_errno ("Opening input file", orig_filename, save_errno);
  4981     }
  4982 
  4983   /* We made a lot of deletions and insertions above, so invalidate
  4984      the newline cache for the entire region of the inserted
  4985      characters.  */
  4986   if (current_buffer->base_buffer && current_buffer->base_buffer->newline_cache)
  4987     invalidate_region_cache (current_buffer->base_buffer,
  4988                              current_buffer->base_buffer->newline_cache,
  4989                              PT - BEG, Z - PT - inserted);
  4990   else if (current_buffer->newline_cache)
  4991     invalidate_region_cache (current_buffer,
  4992                              current_buffer->newline_cache,
  4993                              PT - BEG, Z - PT - inserted);
  4994 
  4995   if (read_quit)
  4996     quit ();
  4997 
  4998   /* Retval needs to be dealt with in all cases consistently.  */
  4999   if (NILP (val))
  5000     val = list2 (orig_filename, make_fixnum (inserted));
  5001 
  5002   return unbind_to (count, val);

  5116          subsidiaries.  */

  5118         val = raw_text_coding_system (val);
  5119     }

  5621 /* Write to descriptor DESC the NCHARS chars starting at POS of STRING.
  5622    If STRING is nil, POS is the character position in the current buffer.
  5623    Intersperse with them the annotations from *ANNOT
  5624    which fall within the range of POS to POS + NCHARS,
  5625    each at its appropriate position.
  5626 
  5627    We modify *ANNOT by discarding elements as we use them up.
  5628 
  5629    Return true if successful.  */
  5630 
  5631 static bool
  5632 a_write (int desc, Lisp_Object string, ptrdiff_t pos,
  5633          ptrdiff_t nchars, Lisp_Object *annot,
  5634          struct coding_system *coding)
  5635 {
  5636   Lisp_Object tem;
  5637   ptrdiff_t nextpos;
  5638   ptrdiff_t lastpos = pos + nchars;
  5639 
  5640   while (NILP (*annot) || CONSP (*annot))
  5641     {
  5642       tem = Fcar_safe (Fcar (*annot));
  5643       nextpos = pos - 1;
  5644       if (FIXNUMP (tem))
  5645         nextpos = XFIXNUM (tem);
  5646 
  5647       /* If there are no more annotations in this range,
  5648          output the rest of the range all at once.  */
  5649       if (! (nextpos >= pos && nextpos <= lastpos))
  5650         return e_write (desc, string, pos, lastpos, coding);
  5651 
  5652       /* Output buffer text up to the next annotation's position.  */
  5653       if (nextpos > pos)
  5654         {
  5655           if (!e_write (desc, string, pos, nextpos, coding))
  5656             return 0;
  5657           pos = nextpos;
  5658         }
  5659       /* Output the annotation.  */
  5660       tem = Fcdr (Fcar (*annot));
  5661       if (STRINGP (tem))
  5662         {
  5663           if (!e_write (desc, tem, 0, SCHARS (tem), coding))
  5664             return 0;
  5665         }
  5666       *annot = Fcdr (*annot);
  5667     }
  5668   return 1;
  5669 }

  5671 /* Maximum number of characters that the next
  5672    function encodes per one loop iteration.  */
  5673 
  5674 enum { E_WRITE_MAX = 8 * 1024 * 1024 };
  5675 
  5676 /* Write text in the range START and END into descriptor DESC,
  5677    encoding them with coding system CODING.  If STRING is nil, START
  5678    and END are character positions of the current buffer, else they
  5679    are indexes to the string STRING.  Return true if successful.  */
  5680 
  5681 static bool
  5682 e_write (int desc, Lisp_Object string, ptrdiff_t start, ptrdiff_t end,
  5683          struct coding_system *coding)
  5684 {
  5685   if (STRINGP (string))
  5686     {
  5687       start = 0;
  5688       end = SCHARS (string);
  5689     }
  5690 
  5691   /* We used to have a code for handling selective display here.  But,
  5692      now it is handled within encode_coding.  */
  5693 
  5694   while (start < end)
  5695     {
  5696       if (STRINGP (string))
  5697         {
  5698           coding->src_multibyte = SCHARS (string) < SBYTES (string);
  5699           if (CODING_REQUIRE_ENCODING (coding))
  5700             {
  5701               ptrdiff_t nchars = min (end - start, E_WRITE_MAX);
  5702 
  5703               /* Avoid creating huge Lisp string in encode_coding_object.  */
  5704               if (nchars == E_WRITE_MAX)
  5705                 coding->raw_destination = 1;
  5706 
  5707               encode_coding_object
  5708                 (coding, string, start, string_char_to_byte (string, start),
  5709                  start + nchars, string_char_to_byte (string, start + nchars),
  5710                  Qt);
  5711             }
  5712           else
  5713             {
  5714               coding->dst_object = string;
  5715               coding->consumed_char = SCHARS (string);
  5716               coding->produced = SBYTES (string);
  5717             }
  5718         }
  5719       else
  5720         {
  5721           ptrdiff_t start_byte = CHAR_TO_BYTE (start);
  5722           ptrdiff_t end_byte = CHAR_TO_BYTE (end);
  5723 
  5724           coding->src_multibyte = (end - start) < (end_byte - start_byte);
  5725           if (CODING_REQUIRE_ENCODING (coding))
  5726             {
  5727               ptrdiff_t nchars = min (end - start, E_WRITE_MAX);
  5728 
  5729               /* Likewise.  */
  5730               if (nchars == E_WRITE_MAX)
  5731                 coding->raw_destination = 1;
  5732 
  5733               encode_coding_object
  5734                 (coding, Fcurrent_buffer (), start, start_byte,
  5735                  start + nchars, CHAR_TO_BYTE (start + nchars), Qt);
  5736             }
  5737           else
  5738             {
  5739               coding->dst_object = Qnil;
  5740               coding->dst_pos_byte = start_byte;
  5741               if (start >= GPT || end <= GPT)
  5742                 {
  5743                   coding->consumed_char = end - start;
  5744                   coding->produced = end_byte - start_byte;
  5745                 }
  5746               else
  5747                 {
  5748                   coding->consumed_char = GPT - start;
  5749                   coding->produced = GPT_BYTE - start_byte;
  5750                 }
  5751             }
  5752         }
  5753 
  5754       if (coding->produced > 0)
  5755         {
  5756           char *buf = (coding->raw_destination ? (char *) coding->destination
  5757                        : (STRINGP (coding->dst_object)
  5758                           ? SSDATA (coding->dst_object)
  5759                           : (char *) BYTE_POS_ADDR (coding->dst_pos_byte)));
  5760           coding->produced -= emacs_write_quit (desc, buf, coding->produced);
  5761 
  5762           if (coding->raw_destination)
  5763             {
  5764               /* We're responsible for freeing this, see
  5765                  encode_coding_object to check why.  */
  5766               xfree (coding->destination);
  5767               coding->raw_destination = 0;
  5768             }
  5769           if (coding->produced)
  5770             return 0;
  5771         }
  5772       start += coding->consumed_char;
  5773     }
  5774 
  5775   return 1;
  5776 }
