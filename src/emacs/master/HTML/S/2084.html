   130 static inline struct print_context
   131 print_prepare (Lisp_Object printcharfun)
   132 {
   133   struct print_context pc = {
   134     .old_printcharfun = printcharfun,
   135     .old_point = -1,
   136     .start_point = -1,
   137     .old_point_byte = -1,
   138     .start_point_byte = -1,
   139     .specpdl_count = SPECPDL_INDEX (),
   140   };
   141   bool multibyte = !NILP (BVAR (current_buffer, enable_multibyte_characters));
   142   record_unwind_current_buffer ();
   143   specbind(Qprint__unreadable_callback_buffer, Fcurrent_buffer ());
   144   if (NILP (printcharfun))
   145     printcharfun = Qt;
   146   if (BUFFERP (printcharfun))
   147     {
   148       if (XBUFFER (printcharfun) != current_buffer)
   149         Fset_buffer (printcharfun);
   150       printcharfun = Qnil;
   151     }
   152   if (MARKERP (printcharfun))
   153     {
   154       if (! XMARKER (printcharfun)->buffer)
   155         error ("Marker does not point anywhere");
   156       if (XMARKER (printcharfun)->buffer != current_buffer)
   157         set_buffer_internal (XMARKER (printcharfun)->buffer);
   158       ptrdiff_t marker_pos = marker_position (printcharfun);
   159       if (marker_pos < BEGV || marker_pos > ZV)
   160         signal_error ("Marker is outside the accessible part of the buffer",
   161                       printcharfun);
   162       pc.old_point = PT;
   163       pc.old_point_byte = PT_BYTE;
   164       SET_PT_BOTH (marker_pos, marker_byte_position (printcharfun));
   165       pc.start_point = PT;
   166       pc.start_point_byte = PT_BYTE;
   167       printcharfun = Qnil;
   168     }
   169   if (NILP (printcharfun))
   170     {
   171       if (NILP (BVAR (current_buffer, enable_multibyte_characters))
   172           && ! print_escape_multibyte)
   173         specbind (Qprint_escape_multibyte, Qt);
   174       if (! NILP (BVAR (current_buffer, enable_multibyte_characters))
   175           && ! print_escape_nonascii)
   176         specbind (Qprint_escape_nonascii, Qt);
   177       if (print_buffer.buffer != NULL)
   178         {
   179           Lisp_Object string = make_string_from_bytes (print_buffer.buffer,
   180                                                        print_buffer.pos,
   181                                                        print_buffer.pos_byte);
   182           record_unwind_protect (print_unwind, string);
   183         }
   184       else
   185         {
   186           int new_size = 1000;
   187           print_buffer.buffer = xmalloc (new_size);
   188           print_buffer.size = new_size;
   189           record_unwind_protect_void (print_free_buffer);
   190         }
   191       print_buffer.pos = 0;
   192       print_buffer.pos_byte = 0;
   193     }
   194   if (EQ (printcharfun, Qt) && ! noninteractive)
   195     setup_echo_area_for_printing (multibyte);
   196   pc.printcharfun = printcharfun;
   197   return pc;
   198 }

   200 static inline void
   201 print_finish (struct print_context *pc)
   202 {
   203   if (NILP (pc->printcharfun))
   204     {
   205       if (print_buffer.pos != print_buffer.pos_byte
   206           && NILP (BVAR (current_buffer, enable_multibyte_characters)))
   207         {
   208           USE_SAFE_ALLOCA;
   209           unsigned char *temp = SAFE_ALLOCA (print_buffer.pos + 1);
   210           copy_text ((unsigned char *) print_buffer.buffer, temp,
   211                      print_buffer.pos_byte, 1, 0);
   212           insert_1_both ((char *) temp, print_buffer.pos,
   213                          print_buffer.pos, 0, 1, 0);
   214           SAFE_FREE ();
   215         }
   216       else
   217         insert_1_both (print_buffer.buffer, print_buffer.pos,
   218                        print_buffer.pos_byte, 0, 1, 0);
   219       signal_after_change (PT - print_buffer.pos, 0, print_buffer.pos);
   220     }
   221   if (MARKERP (pc->old_printcharfun))
   222     set_marker_both (pc->old_printcharfun, Qnil, PT, PT_BYTE);
   223   if (pc->old_point >= 0)
   224     SET_PT_BOTH (pc->old_point
   225                  + (pc->old_point >= pc->start_point
   226                     ? PT - pc->start_point : 0),
   227                  pc->old_point_byte
   228                  + (pc->old_point_byte >= pc->start_point_byte
   229                     ? PT_BYTE - pc->start_point_byte : 0));
   230   unbind_to (pc->specpdl_count, Qnil);
   231 }

   235 static void
   236 printchar_to_stream (unsigned int ch, FILE *stream)
   237 {
   238   Lisp_Object dv UNINIT;
   239   ptrdiff_t i = 0, n = 1;
   240   Lisp_Object coding_system = Vlocale_coding_system;
   241   bool encode_p = false;
   242 
   243   if (!NILP (Vcoding_system_for_write))
   244     coding_system = Vcoding_system_for_write;
   245   if (!NILP (coding_system))
   246     encode_p = true;
   247 
   248   if (CHAR_VALID_P (ch) && DISP_TABLE_P (Vstandard_display_table))
   249     {
   250       dv = DISP_CHAR_VECTOR (XCHAR_TABLE (Vstandard_display_table), ch);
   251       if (VECTORP (dv))
   252         {
   253           n = ASIZE (dv);
   254           goto next_char;
   255         }
   256     }
   257 
   258   while (true)
   259     {
   260       if (ASCII_CHAR_P (ch))
   261         {
   262           putc (ch, stream);
   263 #ifdef WINDOWSNT
   264           /* Send the output to a debugger (nothing happens if there
   265              isn't one).  */
   266           if (print_output_debug_flag && stream == stderr)
   267             OutputDebugString ((char []) {ch, '\0'});
   268 #endif
   269         }
   270       else
   271         {
   272           unsigned char mbstr[MAX_MULTIBYTE_LENGTH];
   273           int len = CHAR_STRING (ch, mbstr);
   274           Lisp_Object encoded_ch =
   275             make_multibyte_string ((char *) mbstr, 1, len);
   276 
   277           if (encode_p)
   278             encoded_ch = code_convert_string_norecord (encoded_ch,
   279                                                        coding_system, true);
   280           fwrite (SSDATA (encoded_ch), 1, SBYTES (encoded_ch), stream);
   281 #ifdef WINDOWSNT
   282           if (print_output_debug_flag && stream == stderr)
   283             OutputDebugString (SSDATA (encoded_ch));
   284 #endif
   285         }
   286 
   287       i++;
   288 
   289     next_char:
   290       for (; i < n; i++)
   291         if (CHARACTERP (AREF (dv, i)))
   292           break;
   293       if (! (i < n))
   294         break;
   295       ch = XFIXNAT (AREF (dv, i));
   296     }
   297 }

   299 /* Print character CH using method FUN.  FUN nil means print to
   300    print_buffer.  FUN t means print to echo area or stdout if
   301    non-interactive.  If FUN is neither nil nor t, call FUN with CH as
   302    argument.  */

   304 static void
   305 printchar (unsigned int ch, Lisp_Object fun)
   306 {
   307   if (!NILP (fun) && !EQ (fun, Qt))
   308     call1 (fun, make_fixnum (ch));
   309   else
   310     {
   311       unsigned char str[MAX_MULTIBYTE_LENGTH];
   312       int len = CHAR_STRING (ch, str);
   313 
   314       maybe_quit ();
   315 
   316       if (NILP (fun))
   317         {
   318           ptrdiff_t incr = len - (print_buffer.size - print_buffer.pos_byte);
   319           if (incr > 0)
   320             print_buffer.buffer = xpalloc (print_buffer.buffer,
   321                                            &print_buffer.size,
   322                                            incr, -1, 1);
   323           memcpy (print_buffer.buffer + print_buffer.pos_byte, str, len);
   324           print_buffer.pos += 1;
   325           print_buffer.pos_byte += len;
   326         }
   327       else if (noninteractive)
   328         {
   329           printchar_stdout_last = ch;
   330           if (DISP_TABLE_P (Vstandard_display_table))
   331             printchar_to_stream (ch, stdout);
   332           else
   333             fwrite (str, 1, len, stdout);
   334           noninteractive_need_newline = 1;
   335         }
   336       else
   337         {
   338           bool multibyte_p
   339             = !NILP (BVAR (current_buffer, enable_multibyte_characters));
   340 
   341           setup_echo_area_for_printing (multibyte_p);
   342           insert_char (ch);
   343           message_dolog ((char *) str, len, 0, multibyte_p);
   344         }
   345     }

   348 /* Output an octal escape for C.  If C is less than '\100' consult the
   349    following character (if any) to see whether to use three octal
   350    digits to avoid misinterpretation of the next character.  The next
   351    character after C will be taken from DATA, starting at byte
   352    location I, if I is less than SIZE.  Use PRINTCHARFUN to output
   353    each character.  */
   354 
   355 static void
   356 octalout (unsigned char c, unsigned char *data, ptrdiff_t i, ptrdiff_t size,
   357           Lisp_Object printcharfun)
   358 {
   359   int digits = (c > '\77' || (i < size && '0' <= data[i] && data[i] <= '7')
   360                 ? 3
   361                 : c > '\7' ? 2 : 1);
   362   printchar ('\\', printcharfun);
   363   do
   364     printchar ('0' + ((c >> (3 * --digits)) & 7), printcharfun);
   365   while (digits != 0);
   366 }

   368 /* Output SIZE characters, SIZE_BYTE bytes from string PTR using
   369    method PRINTCHARFUN.  PRINTCHARFUN nil means output to
   370    print_buffer.  PRINTCHARFUN t means output to the echo area or to
   371    stdout if non-interactive.  If neither nil nor t, call Lisp
   372    function PRINTCHARFUN for each character printed.  MULTIBYTE
   373    non-zero means PTR contains multibyte characters.
   374 
   375    In the case where PRINTCHARFUN is nil, it is safe for PTR to point
   376    to data in a Lisp string.  Otherwise that is not safe.  */

   378 static void
   379 strout (const char *ptr, ptrdiff_t size, ptrdiff_t size_byte,
   380         Lisp_Object printcharfun)
   381 {
   382   if (NILP (printcharfun))
   383     {
   384       ptrdiff_t incr = size_byte - (print_buffer.size - print_buffer.pos_byte);
   385       if (incr > 0)
   386         print_buffer.buffer = xpalloc (print_buffer.buffer,
   387                                        &print_buffer.size, incr, -1, 1);
   388       memcpy (print_buffer.buffer + print_buffer.pos_byte, ptr, size_byte);
   389       print_buffer.pos += size;
   390       print_buffer.pos_byte += size_byte;
   391     }
   392   else if (noninteractive && EQ (printcharfun, Qt))
   393     {
   394       if (DISP_TABLE_P (Vstandard_display_table))
   395         {
   396           int len;
   397           for (ptrdiff_t i = 0; i < size_byte; i += len)
   398             {
   399               int ch = string_char_and_length ((const unsigned char *) ptr + i,
   400                                                &len);
   401               printchar_to_stream (ch, stdout);
   402             }
   403         }
   404       else
   405         fwrite (ptr, 1, size_byte, stdout);
   406 
   407       noninteractive_need_newline = 1;
   408     }
   409   else if (EQ (printcharfun, Qt))
   410     {
   411       /* Output to echo area.  We're trying to avoid a little overhead
   412          here, that's the reason we don't call printchar to do the
   413          job.  */
   414       int i;
   415       bool multibyte_p
   416         = !NILP (BVAR (current_buffer, enable_multibyte_characters));
   417 
   418       setup_echo_area_for_printing (multibyte_p);
   419       message_dolog (ptr, size_byte, 0, multibyte_p);
   420 
   421       if (size == size_byte)
   422         {
   423           for (i = 0; i < size; ++i)
   424             insert_char ((unsigned char) *ptr++);
   425         }
   426       else
   427         {
   428           int len;
   429           for (i = 0; i < size_byte; i += len)
   430             {
   431               int ch = string_char_and_length ((const unsigned char *) ptr + i,
   432                                                &len);
   433               insert_char (ch);
   434             }
   435         }
   436     }
   437   else
   438     {
   439       /* PRINTCHARFUN is a Lisp function.  */
   440       ptrdiff_t i = 0;
   441 
   442       if (size == size_byte)
   443         {
   444           while (i < size_byte)
   445             {
   446               int ch = ptr[i++];
   447               printchar (ch, printcharfun);
   448             }
   449         }
   450       else
   451         {
   452           while (i < size_byte)
   453             {
   454               /* Here, we must convert each multi-byte form to the
   455                  corresponding character code before handing it to
   456                  PRINTCHAR.  */
   457               int len, ch = (string_char_and_length
   458                              ((const unsigned char *) ptr + i, &len));
   459               printchar (ch, printcharfun);
   460               i += len;
   461             }
   462         }
   463     }
   464 }

   470 static void
   471 print_string (Lisp_Object string, Lisp_Object printcharfun)
   472 {
   473   if (EQ (printcharfun, Qt) || NILP (printcharfun))
   474     {
   475       ptrdiff_t chars;
   476 
   477       if (print_escape_nonascii)
   478         string = string_escape_byte8 (string);
   479 
   480       if (STRING_MULTIBYTE (string))
   481         chars = SCHARS (string);
   482       else if (! print_escape_nonascii
   483                && (EQ (printcharfun, Qt)
   484                    ? ! NILP (BVAR (&buffer_defaults, enable_multibyte_characters))
   485                    : ! NILP (BVAR (current_buffer, enable_multibyte_characters))))
   486         {
   487           /* If unibyte string STRING contains 8-bit codes, we must
   488              convert STRING to a multibyte string containing the same
   489              character codes.  */
   490           Lisp_Object newstr;
   491           ptrdiff_t bytes;
   492 
   493           chars = SBYTES (string);
   494           bytes = count_size_as_multibyte (SDATA (string), chars);
   495           if (chars < bytes)
   496             {
   497               newstr = make_uninit_multibyte_string (chars, bytes);
   498               str_to_multibyte (SDATA (newstr), SDATA (string), chars);
   499               string = newstr;
   500             }
   501         }
   502       else
   503         chars = SBYTES (string);
   504 
   505       if (EQ (printcharfun, Qt))
   506         {
   507           /* Output to echo area.  */
   508           ptrdiff_t nbytes = SBYTES (string);
   509 
   510           /* Copy the string contents so that relocation of STRING by
   511              GC does not cause trouble.  */
   512           USE_SAFE_ALLOCA;
   513           char *buffer = SAFE_ALLOCA (nbytes);
   514           memcpy (buffer, SDATA (string), nbytes);
   515 
   516           strout (buffer, chars, nbytes, printcharfun);
   517 
   518           SAFE_FREE ();
   519         }
   520       else
   521         /* No need to copy, since output to print_buffer can't GC.  */
   522         strout (SSDATA (string), chars, SBYTES (string), printcharfun);
   523     }
   524   else
   525     {
   526       /* Otherwise, string may be relocated by printing one char.
   527          So re-fetch the string address for each character.  */
   528       ptrdiff_t i;
   529       ptrdiff_t size = SCHARS (string);
   530       ptrdiff_t size_byte = SBYTES (string);
   531       if (size == size_byte)
   532         for (i = 0; i < size; i++)
   533           printchar (SREF (string, i), printcharfun);
   534       else
   535         for (i = 0; i < size_byte; )
   536           {
   537             /* Here, we must convert each multi-byte form to the
   538                corresponding character code before handing it to PRINTCHAR.  */
   539             int len, ch = string_char_and_length (SDATA (string) + i, &len);
   540             printchar (ch, printcharfun);
   541             i += len;
   542           }
   543     }
   544 }

  1173     {

  1175       uintmax_t hi = u.ieee_nan.mantissa0;
  1176       return sprintf (buf, &"-%"PRIuMAX".0e+NaN"[!u.ieee_nan.negative],
  1177                       (hi << 31 << 1) + u.ieee_nan.mantissa1);
  1178     }
  1179 #endif
  1180 
  1181   if (NILP (Vfloat_output_format)
  1182       || !STRINGP (Vfloat_output_format))
  1183   lose:
  1184     {
  1185       /* Generate the fewest number of digits that represent the
  1186          floating point value without losing information.  */
  1187       len = dtoastr (buf, FLOAT_TO_STRING_BUFSIZE - 2, 0, 0, data);
  1188       /* The decimal point must be printed, or the byte compiler can
  1189          get confused (Bug#8033). */
  1190       width = 1;
  1191     }
  1192   else                  /* oink oink */
  1193     {
  1194       /* Check that the spec we have is fully valid.
  1195          This means not only valid for printf,
  1196          but meant for floats, and reasonable.  */
  1197       cp = SSDATA (Vfloat_output_format);
  1198 
  1199       if (cp[0] != '%')
  1200         goto lose;
  1201       if (cp[1] != '.')
  1202         goto lose;
  1203 
  1204       cp += 2;
  1205 
  1206       /* Check the width specification.  */
  1207       width = -1;
  1208       if ('0' <= *cp && *cp <= '9')
  1209         {
  1210           width = 0;
  1211           do
  1212             {
  1213               width = (width * 10) + (*cp++ - '0');
  1214               if (DBL_DIG < width)
  1215                 goto lose;
  1216             }
  1217           while (*cp >= '0' && *cp <= '9');
  1218 
  1219           /* A precision of zero is valid only for %f.  */
  1220           if (width == 0 && *cp != 'f')
  1221             goto lose;
  1222         }
  1223 
  1224       if (*cp != 'e' && *cp != 'f' && *cp != 'g')
  1225         goto lose;
  1226 
  1227       if (cp[1] != 0)
  1228         goto lose;
  1229 
  1230       len = sprintf (buf, SSDATA (Vfloat_output_format), data);
  1231     }
  1232 
  1233   /* Make sure there is a decimal point with digit after, or an
  1234      exponent, so that the value is readable as a float.  But don't do
  1235      this with "%.0f"; it's valid for that not to produce a decimal
  1236      point.  Note that width can be 0 only for %.0f.  */
  1237   if (width != 0)
  1238     {
  1239       for (cp = buf; *cp; cp++)
  1240         if ((*cp < '0' || *cp > '9') && *cp != '-')
  1241           break;
  1242 
  1243       if (*cp == '.' && cp[1] == 0)
  1244         {
  1245           cp[1] = '0';
  1246           cp[2] = 0;
  1247           len++;
  1248         }
  1249       else if (*cp == 0)
  1250         {
  1251           *cp++ = '.';
  1252           *cp++ = '0';
  1253           *cp++ = 0;
  1254           len += 2;
  1255         }
  1256     }
  1257 
  1258   return len;
  1259 }

  1371 static inline Lisp_Object
  1372 pp_stack_pop (void)
  1373 {
  1374   eassume (!pp_stack_empty_p ());
  1375   struct print_pp_entry *e = &ppstack.stack[ppstack.sp - 1];
  1376   if (e->n == 0)                /* single value */
  1377     {
  1378       --ppstack.sp;
  1379       return e->u.value;
  1380     }
  1381   /* Array of values: pop them left to right, which seems to be slightly
  1382      faster than right to left.  */
  1383   e->n--;
  1384   if (e->n == 0)
  1385     --ppstack.sp;               /* last value consumed */
  1386   return (++e->u.values)[-1];

  1389 /* Construct Vprint_number_table for the print-circle feature
  1390    according to the structure of OBJ.  OBJ itself and all its elements
  1391    will be added to Vprint_number_table recursively if it is a list,
  1392    vector, compiled function, char-table, string (its text properties
  1393    will be traced), or a symbol that has no obarray (this is for the
  1394    print-gensym feature).  The status fields of Vprint_number_table
  1395    mean whether each object appears more than once in OBJ: Qnil at the
  1396    first time, and Qt after that.  */
  1397 static void
  1398 print_preprocess (Lisp_Object obj)
  1399 {
  1400   eassert (!NILP (Vprint_circle));
  1401   ptrdiff_t base_sp = ppstack.sp;
  1402 
  1403   for (;;)
  1404     {
  1405       if (PRINT_CIRCLE_CANDIDATE_P (obj))
  1406         {
  1407           if (!HASH_TABLE_P (Vprint_number_table))
  1408             Vprint_number_table = CALLN (Fmake_hash_table, QCtest, Qeq);
  1409 
  1410           Lisp_Object num = Fgethash (obj, Vprint_number_table, Qnil);
  1411           if (!NILP (num)
  1412               /* If Vprint_continuous_numbering is non-nil and OBJ is a gensym,
  1413                  always print the gensym with a number.  This is a special for
  1414                  the lisp function byte-compile-output-docform.  */

  1416                   && SYMBOLP (obj)
  1417                   && !SYMBOL_INTERNED_P (obj)))
  1418             { /* OBJ appears more than once.  Let's remember that.  */
  1419               if (!FIXNUMP (num))
  1420                 {
  1421                   print_number_index++;
  1422                   /* Negative number indicates it hasn't been printed yet.  */
  1423                   Fputhash (obj, make_fixnum (- print_number_index),
  1424                             Vprint_number_table);
  1425                 }
  1426             }
  1427           else
  1428             {
  1429               /* OBJ is not yet recorded.  Let's add to the table.  */
  1430               Fputhash (obj, Qt, Vprint_number_table);
  1431 
  1432               switch (XTYPE (obj))
  1433                 {
  1434                 case Lisp_String:
  1435                   /* A string may have text properties,
  1436                      which can be circular. */
  1437                   traverse_intervals_noorder (string_intervals (obj),
  1438                                               print_preprocess_string, NULL);
  1439                   break;
  1440 
  1441                 case Lisp_Cons:
  1442                   if (!NILP (XCDR (obj)))
  1443                     pp_stack_push_value (XCDR (obj));
  1444                   obj = XCAR (obj);
  1445                   continue;
  1446 
  1447                 case Lisp_Vectorlike:
  1448                   {
  1449                     struct Lisp_Vector *vec = XVECTOR (obj);
  1450                     ptrdiff_t size = ASIZE (obj);
  1451                     if (size & PSEUDOVECTOR_FLAG)
  1452                       size &= PSEUDOVECTOR_SIZE_MASK;
  1453                     ptrdiff_t start = (SUB_CHAR_TABLE_P (obj)
  1454                                        ? SUB_CHAR_TABLE_OFFSET : 0);
  1455                     pp_stack_push_values (vec->contents + start, size - start);
  1456                     if (HASH_TABLE_P (obj))
  1457                       {
  1458                         struct Lisp_Hash_Table *h = XHASH_TABLE (obj);
  1459                         obj = h->key_and_value;
  1460                         continue;
  1461                       }
  1462                     break;
  1463                   }
  1464 
  1465                 default:
  1466                   break;
  1467                 }
  1468             }
  1469         }
  1470 
  1471       if (ppstack.sp <= base_sp)
  1472         break;
  1473       obj = pp_stack_pop ();
  1474     }
  1475 }

  1603 static bool
  1604 print_vectorlike (Lisp_Object obj, Lisp_Object printcharfun, bool escapeflag,
  1605                   char *buf)
  1606 {
  1607   /* First do all the vectorlike types that have a readable syntax.  */
  1608   switch (PSEUDOVECTOR_TYPE (XVECTOR (obj)))
  1609     {
  1610     case PVEC_BIGNUM:
  1611       {
  1612         ptrdiff_t size = bignum_bufsize (obj, 10);
  1613         USE_SAFE_ALLOCA;
  1614         char *str = SAFE_ALLOCA (size);
  1615         ptrdiff_t len = bignum_to_c_string (str, size, obj, 10);
  1616         strout (str, len, len, printcharfun);
  1617         SAFE_FREE ();
  1618       }
  1619       return true;
  1620 
  1621     case PVEC_BOOL_VECTOR:
  1622       {
  1623         EMACS_INT size = bool_vector_size (obj);
  1624         ptrdiff_t size_in_bytes = bool_vector_bytes (size);
  1625         ptrdiff_t real_size_in_bytes = size_in_bytes;
  1626         unsigned char *data = bool_vector_uchar_data (obj);
  1627 
  1628         int len = sprintf (buf, "#&%"pI"d\"", size);
  1629         strout (buf, len, len, printcharfun);
  1630 
  1631         /* Don't print more bytes than the specified maximum.
  1632            Negative values of print-length are invalid.  Treat them
  1633            like a print-length of nil.  */
  1634         if (FIXNATP (Vprint_length)
  1635             && XFIXNAT (Vprint_length) < size_in_bytes)
  1636           size_in_bytes = XFIXNAT (Vprint_length);
  1637 
  1638         for (ptrdiff_t i = 0; i < size_in_bytes; i++)
  1639           {
  1640             maybe_quit ();
  1641             unsigned char c = data[i];
  1642             if (c == '\n' && print_escape_newlines)
  1643               print_c_string ("\\n", printcharfun);
  1644             else if (c == '\f' && print_escape_newlines)
  1645               print_c_string ("\\f", printcharfun);
  1646             else if (c > '\177'
  1647                      || (print_escape_control_characters && c_iscntrl (c)))
  1648               {
  1649                 /* Use octal escapes to avoid encoding issues.  */
  1650                 octalout (c, data, i + 1, size_in_bytes, printcharfun);
  1651               }
  1652             else
  1653               {
  1654                 if (c == '\"' || c == '\\')
  1655                   printchar ('\\', printcharfun);
  1656                 printchar (c, printcharfun);
  1657               }
  1658           }
  1659 
  1660         if (size_in_bytes < real_size_in_bytes)
  1661           print_c_string (" ...", printcharfun);
  1662         printchar ('\"', printcharfun);
  1663       }

  1665 
  1666     default:
  1667       break;
  1668     }
  1669

  1671      syntax.  First check whether this is handled by
  1672      `print-unreadable-function'.  */
  1673   if (!NILP (Vprint_unreadable_function)
  1674       && FUNCTIONP (Vprint_unreadable_function))
  1675     {
  1676       specpdl_ref count = SPECPDL_INDEX ();

  1678          infinite recursion in the function called.  */

  1680       specbind (Qprint_unreadable_function, Qnil);
  1681 
  1682       /* If we're being called from `prin1-to-string' or the like,
  1683          we're now in the secret " prin1" buffer.  This can lead to
  1684          problems if, for instance, the callback function switches a
  1685          window to this buffer -- this will make Emacs segfault.  */

  1687           && !NILP (Fbuffer_live_p (Vprint__unreadable_callback_buffer)))
  1688         {

  1691         }

  1693                                   escapeflag? Qt: Qnil);
  1694       unbind_to (count, Qnil);
  1695 
  1696       if (!NILP (result))
  1697         {
  1698           if (STRINGP (result))
  1699             print_string (result, printcharfun);
  1700           /* It's handled, so stop processing here.  */
  1701           return true;
  1702         }
  1703     }
  1704 
  1705   /* Not handled; print unreadable object.  */
  1706   switch (PSEUDOVECTOR_TYPE (XVECTOR (obj)))
  1707     {
  1708     case PVEC_MARKER:
  1709       print_c_string ("#<marker ", printcharfun);
  1710       /* Do you think this is necessary?  */
  1711       if (XMARKER (obj)->insertion_type != 0)
  1712         print_c_string ("(moves after insertion) ", printcharfun);
  1713       if (! XMARKER (obj)->buffer)
  1714         print_c_string ("in no buffer", printcharfun);
  1715       else
  1716         {
  1717           int len = sprintf (buf, "at %"pD"d in ", marker_position (obj));
  1718           strout (buf, len, len, printcharfun);
  1719           print_string (BVAR (XMARKER (obj)->buffer, name), printcharfun);
  1720         }
  1721       printchar ('>', printcharfun);
  1722       break;
  1723 
  1724     case PVEC_SYMBOL_WITH_POS:
  1725       {
  1726         struct Lisp_Symbol_With_Pos *sp = XSYMBOL_WITH_POS (obj);
  1727         if (print_symbols_bare)
  1728           print_object (sp->sym, printcharfun, escapeflag);
  1729         else
  1730           {
  1731             print_c_string ("#<symbol ", printcharfun);
  1732             if (BARE_SYMBOL_P (sp->sym))
  1733               print_object (sp->sym, printcharfun, escapeflag);
  1734             else
  1735               print_c_string ("NOT A SYMBOL!!", printcharfun);
  1736             if (FIXNUMP (sp->pos))
  1737               {
  1738                 print_c_string (" at ", printcharfun);
  1739                 print_object (sp->pos, printcharfun, escapeflag);
  1740               }
  1741             else
  1742               print_c_string (" NOT A POSITION!!", printcharfun);
  1743             printchar ('>', printcharfun);
  1744           }
  1745       }
  1746       break;
  1747 
  1748     case PVEC_OVERLAY:
  1749       print_c_string ("#<overlay ", printcharfun);
  1750       if (! OVERLAY_BUFFER (obj))
  1751         print_c_string ("in no buffer", printcharfun);
  1752       else
  1753         {

  2095     }

  2197 static void
  2198 print_object (Lisp_Object obj, Lisp_Object printcharfun, bool escapeflag)
  2199 {
  2200   ptrdiff_t base_depth = print_depth;
  2201   ptrdiff_t base_sp = prstack.sp;
  2202   char buf[max (sizeof "from..to..in " + 2 * INT_STRLEN_BOUND (EMACS_INT),
  2203                 max (sizeof " . #" + INT_STRLEN_BOUND (intmax_t),
  2204                      max ((sizeof " with data 0x"
  2205                            + (UINTMAX_WIDTH + 4 - 1) / 4),
  2206                           40)))];
  2207   current_thread->stack_top = NEAR_STACK_TOP (buf);
  2208 
  2209  print_obj:
  2210   maybe_quit ();
  2211 
  2212   /* Detect circularities and truncate them.  */
  2213   if (NILP (Vprint_circle))
  2214     {
  2215       /* Simple but incomplete way.  */
  2216       if (print_depth >= PRINT_CIRCLE)
  2217         error ("Apparently circular structure being printed");
  2218 
  2219       for (int i = 0; i < print_depth; i++)
  2220         if (BASE_EQ (obj, being_printed[i]))
  2221           {
  2222             int len = sprintf (buf, "#%d", i);
  2223             strout (buf, len, len, printcharfun);
  2224             goto next_obj;
  2225           }
  2226       being_printed[print_depth] = obj;
  2227     }
  2228   else if (PRINT_CIRCLE_CANDIDATE_P (obj))
  2229     {
  2230       /* With the print-circle feature.  */
  2231       Lisp_Object num = Fgethash (obj, Vprint_number_table, Qnil);
  2232       if (FIXNUMP (num))
  2233         {
  2234           EMACS_INT n = XFIXNUM (num);
  2235           if (n < 0)
  2236             { /* Add a prefix #n= if OBJ has not yet been printed;
  2237                  that is, its status field is nil.  */
  2238               int len = sprintf (buf, "#%"pI"d=", -n);
  2239               strout (buf, len, len, printcharfun);
  2240               /* OBJ is going to be printed.  Remember that fact.  */
  2241               Fputhash (obj, make_fixnum (- n), Vprint_number_table);
  2242             }
  2243           else
  2244             {
  2245               /* Just print #n# if OBJ has already been printed.  */
  2246               int len = sprintf (buf, "#%"pI"d#", n);
  2247               strout (buf, len, len, printcharfun);
  2248               goto next_obj;
  2249             }
  2250         }
  2251     }
  2252 
  2253   print_depth++;
  2254 
  2255   switch (XTYPE (obj))
  2256     {
  2257     case_Lisp_Int:
  2258       {
  2259         EMACS_INT i = XFIXNUM (obj);
  2260         char escaped_name;
  2261 
  2262         if (print_integers_as_characters && i >= 0 && i <= MAX_UNICODE_CHAR
  2263             && ((escaped_name = named_escape (i))
  2264                 || graphic_base_p (i)))
  2265           {
  2266             printchar ('?', printcharfun);
  2267             if (escaped_name)
  2268               {
  2269                 printchar ('\\', printcharfun);
  2270                 i = escaped_name;
  2271               }
  2272             else if (escapeflag
  2273                      && (i == ';' || i == '\"' || i == '\'' || i == '\\'
  2274                          || i == '(' || i == ')'
  2275                          || i == '{' || i == '}'
  2276                          || i == '[' || i == ']'))
  2277               printchar ('\\', printcharfun);
  2278             printchar (i, printcharfun);
  2279           }
  2280         else
  2281           {
  2282             char *end = buf + sizeof buf;
  2283             char *start = fixnum_to_string (i, buf, end);
  2284             ptrdiff_t len = end - start;
  2285             strout (start, len, len, printcharfun);
  2286           }
  2287       }
  2288       break;
  2289 
  2290     case Lisp_Float:
  2291       {
  2292         char pigbuf[FLOAT_TO_STRING_BUFSIZE];
  2293         int len = float_to_string (pigbuf, XFLOAT_DATA (obj));
  2294         strout (pigbuf, len, len, printcharfun);
  2295       }
  2296       break;
  2297 
  2298     case Lisp_String:
  2299       if (!escapeflag)
  2300         print_string (obj, printcharfun);
  2301       else
  2302         {
  2303           ptrdiff_t i, i_byte;
  2304           ptrdiff_t size_byte;
  2305           /* True means we must ensure that the next character we output
  2306              cannot be taken as part of a hex character escape.  */
  2307           bool need_nonhex = false;
  2308           bool multibyte = STRING_MULTIBYTE (obj);
  2309 
  2310           if (! EQ (Vprint_charset_text_property, Qt))
  2311             obj = print_prune_string_charset (obj);
  2312 
  2313           if (string_intervals (obj))
  2314             print_c_string ("#(", printcharfun);
  2315 
  2316           printchar ('\"', printcharfun);
  2317           size_byte = SBYTES (obj);
  2318 
  2319           for (i = 0, i_byte = 0; i_byte < size_byte;)
  2320             {
  2321               /* Here, we must convert each multi-byte form to the
  2322                  corresponding character code before handing it to
  2323                  printchar.  */
  2324               int c = fetch_string_char_advance (obj, &i, &i_byte);
  2325 
  2326               maybe_quit ();
  2327 
  2328               if (multibyte
  2329                   ? (CHAR_BYTE8_P (c) && (c = CHAR_TO_BYTE8 (c), true))
  2330                   : (SINGLE_BYTE_CHAR_P (c) && ! ASCII_CHAR_P (c)
  2331                      && print_escape_nonascii))
  2332                 {
  2333                   /* When printing a raw 8-bit byte in a multibyte buffer, or
  2334                      (when requested) a non-ASCII character in a unibyte buffer,
  2335                      print single-byte non-ASCII string chars
  2336                      using octal escapes.  */
  2337                   octalout (c, SDATA (obj), i_byte, size_byte, printcharfun);
  2338                   need_nonhex = false;
  2339                 }
  2340               else if (multibyte
  2341                        && ! ASCII_CHAR_P (c) && print_escape_multibyte)
  2342                 {
  2343                   /* When requested, print multibyte chars using
  2344                      hex escapes.  */
  2345                   char outbuf[sizeof "\\x" + INT_STRLEN_BOUND (c)];
  2346                   int len = sprintf (outbuf, "\\x%04x", c + 0u);
  2347                   strout (outbuf, len, len, printcharfun);
  2348                   need_nonhex = true;
  2349                 }
  2350               else
  2351                 {
  2352                   /* If we just had a hex escape, and this character
  2353                      could be taken as part of it,
  2354                      output `\ ' to prevent that.  */
  2355                   if (c_isxdigit (c))
  2356                     {
  2357                       if (need_nonhex)
  2358                         print_c_string ("\\ ", printcharfun);
  2359                       printchar (c, printcharfun);
  2360                     }
  2361                   else if (c == '\n' && print_escape_newlines
  2362                            ? (c = 'n', true)
  2363                            : c == '\f' && print_escape_newlines
  2364                            ? (c = 'f', true)
  2365                            : c == '\"' || c == '\\')
  2366                     {
  2367                       printchar ('\\', printcharfun);
  2368                       printchar (c, printcharfun);
  2369                     }
  2370                   else if (print_escape_control_characters && c_iscntrl (c))
  2371                     octalout (c, SDATA (obj), i_byte, size_byte, printcharfun);
  2372                   else if (!multibyte
  2373                            && SINGLE_BYTE_CHAR_P (c)
  2374                            && !ASCII_CHAR_P (c))
  2375                     printchar (BYTE8_TO_CHAR (c), printcharfun);
  2376                   else
  2377                     printchar (c, printcharfun);
  2378                   need_nonhex = false;
  2379                 }
  2380             }
  2381           printchar ('\"', printcharfun);
  2382 
  2383           if (string_intervals (obj))
  2384             {
  2385               traverse_intervals (string_intervals (obj),
  2386                                   0, print_interval, printcharfun);
  2387               printchar (')', printcharfun);
  2388             }
  2389         }
  2390       break;
  2391 
  2392     case Lisp_Symbol:
  2393       {
  2394         Lisp_Object name = SYMBOL_NAME (obj);
  2395         ptrdiff_t size_byte = SBYTES (name);
  2396 
  2397         char *p = SSDATA (name);
  2398         bool signedp = *p == '-' || *p == '+';
  2399         ptrdiff_t len;
  2400         bool confusing =
  2401           /* Set CONFUSING if NAME looks like a number, calling
  2402              string_to_number for non-obvious cases.  */
  2403           ((c_isdigit (p[signedp]) || p[signedp] == '.')
  2404            && !NILP (string_to_number (p, 10, &len))
  2405            && len == size_byte)
  2406           /* We don't escape "." or "?" (unless they're the first
  2407              character in the symbol name).  */
  2408           || *p == '?'
  2409           || *p == '.';
  2410 
  2411         if (! NILP (Vprint_gensym)
  2412             && !SYMBOL_INTERNED_IN_INITIAL_OBARRAY_P (obj))
  2413           print_c_string ("#:", printcharfun);
  2414         else if (size_byte == 0)
  2415           {
  2416             print_c_string ("##", printcharfun);
  2417             break;
  2418           }
  2419 
  2420         ptrdiff_t i = 0;
  2421         for (ptrdiff_t i_byte = 0; i_byte < size_byte; )
  2422           {
  2423             /* Here, we must convert each multi-byte form to the
  2424                corresponding character code before handing it to PRINTCHAR.  */
  2425             int c = fetch_string_char_advance (name, &i, &i_byte);
  2426             maybe_quit ();
  2427 
  2428             if (escapeflag)
  2429               {
  2430                 if (c == '\"' || c == '\\' || c == '\''
  2431                     || c == ';' || c == '#' || c == '(' || c == ')'
  2432                     || c == ',' || c == '`'
  2433                     || c == '[' || c == ']' || c <= 040
  2434                     || c == NO_BREAK_SPACE
  2435                     || confusing)
  2436                   {
  2437                     printchar ('\\', printcharfun);
  2438                     confusing = false;
  2439                   }
  2440               }
  2441             printchar (c, printcharfun);
  2442           }
  2443       }
  2444       break;
  2445 
  2446     case Lisp_Cons:
  2447       /* If deeper than spec'd depth, print placeholder.  */
  2448       if (FIXNUMP (Vprint_level)
  2449           && print_depth > XFIXNUM (Vprint_level))
  2450         print_c_string ("...", printcharfun);
  2451       else if (print_quoted && CONSP (XCDR (obj)) && NILP (XCDR (XCDR (obj)))
  2452                && EQ (XCAR (obj), Qquote))
  2453         {
  2454           printchar ('\'', printcharfun);
  2455           obj = XCAR (XCDR (obj));
  2456           --print_depth;        /* tail recursion */
  2457           goto print_obj;
  2458         }
  2459       else if (print_quoted && CONSP (XCDR (obj)) && NILP (XCDR (XCDR (obj)))
  2460                && EQ (XCAR (obj), Qfunction))
  2461         {
  2462           print_c_string ("#'", printcharfun);
  2463           obj = XCAR (XCDR (obj));
  2464           --print_depth;        /* tail recursion */
  2465           goto print_obj;
  2466         }
  2467       /* FIXME: Do we really need the new_backquote_output gating of
  2468          special syntax for comma and comma-at?  There is basically no
  2469          benefit from it at all, and it would be nice to get rid of
  2470          the recursion here without additional complexity.  */
  2471       else if (print_quoted && CONSP (XCDR (obj)) && NILP (XCDR (XCDR (obj)))
  2472                && EQ (XCAR (obj), Qbackquote))
  2473         {
  2474           printchar ('`', printcharfun);
  2475           new_backquote_output++;
  2476           print_object (XCAR (XCDR (obj)), printcharfun, escapeflag);
  2477           new_backquote_output--;
  2478         }
  2479       else if (print_quoted && CONSP (XCDR (obj)) && NILP (XCDR (XCDR (obj)))
  2480                && (EQ (XCAR (obj), Qcomma)
  2481                    || EQ (XCAR (obj), Qcomma_at))
  2482                && new_backquote_output)
  2483         {
  2484           print_object (XCAR (obj), printcharfun, false);
  2485           new_backquote_output--;
  2486           print_object (XCAR (XCDR (obj)), printcharfun, escapeflag);
  2487           new_backquote_output++;
  2488         }
  2489       else
  2490         {
  2491           printchar ('(', printcharfun);
  2492           /* Negative values of print-length are invalid in CL.
  2493              Treat them like nil, as CMUCL does.  */
  2494           intmax_t print_length = (FIXNATP (Vprint_length)
  2495                                    ? XFIXNAT (Vprint_length)
  2496                                    : INTMAX_MAX);
  2497           if (print_length == 0)
  2498             print_c_string ("...)", printcharfun);
  2499           else
  2500             {
  2501               print_stack_push ((struct print_stack_entry){
  2502                   .type = PE_list,
  2503                   .u.list.last = obj,
  2504                   .u.list.maxlen = print_length,
  2505                   .u.list.tortoise = obj,
  2506                   .u.list.n = 2,
  2507                   .u.list.m = 2,
  2508                   .u.list.tortoise_idx = 0,
  2509                 });
  2510               /* print the car */
  2511               obj = XCAR (obj);
  2512               goto print_obj;
  2513             }
  2514         }
  2515       break;
  2516 
  2517     case Lisp_Vectorlike:
  2518       /* First do all the vectorlike types that have a readable syntax.  */
  2519       switch (PSEUDOVECTOR_TYPE (XVECTOR (obj)))
  2520         {
  2521         case PVEC_NORMAL_VECTOR:
  2522           {
  2523             print_stack_push_vector ("[", "]", obj, 0, ASIZE (obj),
  2524                                      printcharfun);
  2525             goto next_obj;
  2526           }
  2527         case PVEC_RECORD:
  2528           {
  2529             print_stack_push_vector ("#s(", ")", obj, 0, PVSIZE (obj),
  2530                                      printcharfun);
  2531             goto next_obj;
  2532           }
  2533         case PVEC_COMPILED:
  2534           {
  2535             print_stack_push_vector ("#[", "]", obj, 0, PVSIZE (obj),
  2536                                      printcharfun);
  2537             goto next_obj;
  2538           }
  2539         case PVEC_CHAR_TABLE:
  2540           {
  2541             print_stack_push_vector ("#^[", "]", obj, 0, PVSIZE (obj),
  2542                                      printcharfun);
  2543             goto next_obj;
  2544           }
  2545         case PVEC_SUB_CHAR_TABLE:
  2546           {
  2547             /* Make each lowest sub_char_table start a new line.
  2548                Otherwise we'll make a line extremely long, which
  2549                results in slow redisplay.  */
  2550             if (XSUB_CHAR_TABLE (obj)->depth == 3)
  2551               printchar ('\n', printcharfun);
  2552             print_c_string ("#^^[", printcharfun);
  2553             int n = sprintf (buf, "%d %d",
  2554                              XSUB_CHAR_TABLE (obj)->depth,
  2555                              XSUB_CHAR_TABLE (obj)->min_char);
  2556             strout (buf, n, n, printcharfun);
  2557             print_stack_push_vector ("", "]", obj,
  2558                                      SUB_CHAR_TABLE_OFFSET, PVSIZE (obj),
  2559                                      printcharfun);
  2560             goto next_obj;
  2561           }
  2562         case PVEC_HASH_TABLE:
  2563           {
  2564             struct Lisp_Hash_Table *h = XHASH_TABLE (obj);
  2565             /* Implement a readable output, e.g.:
  2566                #s(hash-table size 2 test equal data (k1 v1 k2 v2)) */
  2567             /* Always print the size.  */
  2568             int len = sprintf (buf, "#s(hash-table size %"pD"d",
  2569                                HASH_TABLE_SIZE (h));
  2570             strout (buf, len, len, printcharfun);
  2571 
  2572             if (!NILP (h->test.name))
  2573               {
  2574                 print_c_string (" test ", printcharfun);
  2575                 print_object (h->test.name, printcharfun, escapeflag);
  2576               }
  2577 
  2578             if (!NILP (h->weak))
  2579               {
  2580                 print_c_string (" weakness ", printcharfun);
  2581                 print_object (h->weak, printcharfun, escapeflag);
  2582               }
  2583 
  2584             print_c_string (" rehash-size ", printcharfun);
  2585             print_object (Fhash_table_rehash_size (obj),
  2586                           printcharfun, escapeflag);
  2587 
  2588             print_c_string (" rehash-threshold ", printcharfun);
  2589             print_object (Fhash_table_rehash_threshold (obj),
  2590                           printcharfun, escapeflag);
  2591 
  2592             if (h->purecopy)
  2593               print_c_string (" purecopy t", printcharfun);
  2594 
  2595             print_c_string (" data (", printcharfun);
  2596 
  2597             ptrdiff_t size = h->count;
  2598             /* Don't print more elements than the specified maximum.  */
  2599             if (FIXNATP (Vprint_length) && XFIXNAT (Vprint_length) < size)
  2600               size = XFIXNAT (Vprint_length);
  2601 
  2602             print_stack_push ((struct print_stack_entry){
  2603                 .type = PE_hash,
  2604                 .u.hash.obj = obj,
  2605                 .u.hash.nobjs = size * 2,
  2606                 .u.hash.idx = 0,
  2607                 .u.hash.printed = 0,
  2608                 .u.hash.truncated = (size < h->count),
  2609               });
  2610             goto next_obj;
  2611           }
  2612 
  2613         default:
  2614           break;
  2615         }
  2616 
  2617       if (print_vectorlike (obj, printcharfun, escapeflag, buf))
  2618         break;
  2619       FALLTHROUGH;
  2620 
  2621     default:
  2622       {
  2623         int len;
  2624         /* We're in trouble if this happens!
  2625            Probably should just emacs_abort ().  */
  2626         print_c_string ("#<EMACS BUG: INVALID DATATYPE ", printcharfun);
  2627         if (VECTORLIKEP (obj))
  2628           len = sprintf (buf, "(PVEC 0x%08zx)", (size_t) ASIZE (obj));
  2629         else
  2630           len = sprintf (buf, "(0x%02x)", (unsigned) XTYPE (obj));
  2631         strout (buf, len, len, printcharfun);
  2632         print_c_string ((" Save your buffers immediately"
  2633                          " and please report this bug>"),
  2634                         printcharfun);
  2635         break;
  2636       }
  2637     }
  2638   print_depth--;
  2639 
  2640  next_obj:
  2641   if (prstack.sp > base_sp)
  2642     {
  2643       /* Handle a continuation on the print stack.  */
  2644       struct print_stack_entry *e = &prstack.stack[prstack.sp - 1];
  2645       switch (e->type)
  2646         {
  2647         case PE_list:
  2648           {
  2649             /* after "(" ELEM (* " " ELEM) */
  2650             Lisp_Object next = XCDR (e->u.list.last);
  2651             if (NILP (next))
  2652               {
  2653                 /* end of list: print ")" */
  2654                 printchar (')', printcharfun);
  2655                 --prstack.sp;
  2656                 --print_depth;
  2657                 goto next_obj;
  2658               }
  2659             else if (CONSP (next))
  2660               {
  2661                 if (!NILP (Vprint_circle))
  2662                   {
  2663                     /* With the print-circle feature.  */
  2664                     Lisp_Object num = Fgethash (next, Vprint_number_table,
  2665                                                 Qnil);
  2666                     if (FIXNUMP (num))
  2667                       {
  2668                         print_c_string (" . ", printcharfun);
  2669                         obj = next;
  2670                         e->type = PE_rbrac;
  2671                         goto print_obj;
  2672                       }
  2673                   }
  2674 
  2675                 /* list continues: print " " ELEM ... */
  2676 
  2677                 printchar (' ', printcharfun);
  2678 
  2679                 --e->u.list.maxlen;
  2680                 if (e->u.list.maxlen <= 0)
  2681                   {
  2682                     print_c_string ("...)", printcharfun);
  2683                     --prstack.sp;
  2684                     --print_depth;
  2685                     goto next_obj;
  2686                   }
  2687 
  2688                 e->u.list.last = next;
  2689                 e->u.list.n--;
  2690                 if (e->u.list.n == 0)
  2691                   {
  2692                     /* Double tortoise update period and teleport it.  */
  2693                     e->u.list.tortoise_idx += e->u.list.m;
  2694                     e->u.list.m <<= 1;
  2695                     e->u.list.n = e->u.list.m;
  2696                     e->u.list.tortoise = next;
  2697                   }
  2698                 else if (BASE_EQ (next, e->u.list.tortoise))
  2699                   {
  2700                     /* FIXME: This #N tail index is somewhat ambiguous;
  2701                        see bug#55395.  */
  2702                     int len = sprintf (buf, ". #%" PRIdMAX ")",
  2703                                        e->u.list.tortoise_idx);
  2704                     strout (buf, len, len, printcharfun);
  2705                     --prstack.sp;
  2706                     --print_depth;
  2707                     goto next_obj;
  2708                   }
  2709                 obj = XCAR (next);
  2710               }
  2711             else
  2712               {
  2713                 /* non-nil ending: print " . " ELEM ")" */
  2714                 print_c_string (" . ", printcharfun);
  2715                 obj = next;
  2716                 e->type = PE_rbrac;
  2717               }
  2718             break;
  2719           }
  2720 
  2721         case PE_rbrac:
  2722           printchar (')', printcharfun);
  2723           --prstack.sp;
  2724           --print_depth;
  2725           goto next_obj;
  2726 
  2727         case PE_vector:
  2728           if (e->u.vector.idx >= e->u.vector.size)
  2729             {
  2730               if (e->u.vector.truncated)
  2731                 {
  2732                   if (e->u.vector.idx > 0)
  2733                     printchar (' ', printcharfun);
  2734                   print_c_string ("...", printcharfun);
  2735                 }
  2736               print_c_string (e->u.vector.end, printcharfun);
  2737               --prstack.sp;
  2738               --print_depth;
  2739               goto next_obj;
  2740             }
  2741           if (e->u.vector.idx > 0)
  2742             printchar (' ', printcharfun);
  2743           obj = AREF (e->u.vector.obj, e->u.vector.idx);
  2744           e->u.vector.idx++;
  2745           break;
  2746 
  2747         case PE_hash:
  2748           if (e->u.hash.printed >= e->u.hash.nobjs)
  2749             {
  2750               if (e->u.hash.truncated)
  2751                 {
  2752                   if (e->u.hash.printed)
  2753                     printchar (' ', printcharfun);
  2754                   print_c_string ("...", printcharfun);
  2755                 }
  2756               print_c_string ("))", printcharfun);
  2757               --prstack.sp;
  2758               --print_depth;
  2759               goto next_obj;
  2760             }
  2761 
  2762           if (e->u.hash.printed)
  2763             printchar (' ', printcharfun);
  2764 
  2765           struct Lisp_Hash_Table *h = XHASH_TABLE (e->u.hash.obj);
  2766           if ((e->u.hash.printed & 1) == 0)
  2767             {
  2768               Lisp_Object key;
  2769               ptrdiff_t idx = e->u.hash.idx;
  2770               while (BASE_EQ ((key = HASH_KEY (h, idx)), Qunbound))
  2771                 idx++;
  2772               e->u.hash.idx = idx;
  2773               obj = key;
  2774             }
  2775           else
  2776             {
  2777               obj = HASH_VALUE (h, e->u.hash.idx);
  2778               e->u.hash.idx++;
  2779             }
  2780           e->u.hash.printed++;
  2781           break;
  2782         }
  2783       goto print_obj;
  2784     }
  2785   eassert (print_depth == base_depth);
  2786 }
