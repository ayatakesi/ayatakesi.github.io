   263 the current buffer), it means insert output in that buffer before point.

   265  and don't wait for the program to terminate.

   347      t means use same as standard output.  */
   348   Lisp_Object error_file;
   349   Lisp_Object output_file = Qnil;
   350 #ifdef MSDOS    /* Demacs 1.1.1 91/10/16 HIRANO Satoshi */
   351   char *tempfile = NULL;
   352 #else
   353   sigset_t oldset;
   354   pid_t pid = -1;
   355 #endif
   356   int child_errno;
   357   int fd_output, fd_error;
   358   struct coding_system process_coding; /* coding-system of process output */
   359   struct coding_system argument_coding; /* coding-system of arguments */
   360   /* Set to the return value of Ffind_operation_coding_system.  */
   361   Lisp_Object coding_systems;
   362   bool discard_output;
   363 
   364   if (synch_process_pid)
   365     error ("call-process invoked recursively");

   371

   383     Lisp_Object val, *args2;

   385     /* If arguments are supplied, we may have to encode them.  */

   387       {
   388         bool must_encode = 0;
   389         Lisp_Object coding_attrs;
   390 
   391         for (i = 4; i < nargs; i++)
   392           CHECK_STRING (args[i]);
   393

   401           val = Qraw_text;

   403           {
   404             SAFE_NALLOCA (args2, 1, nargs + 1);
   405             args2[0] = Qcall_process;
   406             for (i = 0; i < nargs; i++) args2[i + 1] = args[i];

   408             val = CONSP (coding_systems) ? XCDR (coding_systems) : Qnil;

   884                  detection.  In that case, set it back to 1 because

   897     }

   903 #endif
   904

  1142 add_env (char **env, char **new_env, char *string)
  1143 {

  1198 /* This is the last thing run in a newly forked inferior
  1199    either synchronous or asynchronous.
  1200    Copy descriptors IN, OUT and ERR as descriptors 0, 1 and 2.
  1201    Initialize inferior's priority, pgrp, connected dir and environment.
  1202    then exec another program based on new_argv.
  1203 
  1204    CURRENT_DIR is an elisp string giving the path of the current
  1205    directory the subprocess should have.  Since we can't really signal
  1206    a decent error from within the child, this should be verified as an
  1207    executable directory by the parent.
  1208 
  1209    On GNUish hosts, either exec or return an error number.
  1210    On MS-Windows, either return a pid or return -1 and set errno.
  1211    On MS-DOS, either return an exit status or signal an error.  */

  1213 static CHILD_SETUP_TYPE
  1214 child_setup (int in, int out, int err, char **new_argv, char **env,
  1215              const char *current_dir)
  1216 {
  1217 #ifdef MSDOS
  1218   char *pwd_var;
  1219   char *temp;
  1220   ptrdiff_t i;
  1221 #endif
  1222 #ifdef WINDOWSNT
  1223   int cpid;
  1224   HANDLE handles[3];
  1225 #else
  1226   pid_t pid = getpid ();
  1227 #endif /* WINDOWSNT */
  1228 
  1229   /* Note that use of alloca is always safe here.  It's obvious for systems
  1230      that do not have true vfork or that have true (stack) alloca.
  1231      If using vfork and C_ALLOCA (when Emacs used to include
  1232      src/alloca.c) it is safe because that changes the superior's
  1233      static variables as if the superior had done alloca and will be
  1234      cleaned up in the usual way. */
  1235 
  1236 #ifndef DOS_NT
  1237     /* We can't signal an Elisp error here; we're in a vfork.  Since
  1238        the callers check the current directory before forking, this
  1239        should only return an error if the directory's permissions
  1240        are changed between the check and this chdir, but we should
  1241        at least check.  */
  1242     if (chdir (current_dir) < 0)
  1243       _exit (EXIT_CANCELED);
  1244 #endif
  1245 
  1246 #ifdef WINDOWSNT
  1247   prepare_standard_handles (in, out, err, handles);
  1248   set_process_dir (current_dir);
  1249   /* Spawn the child.  (See w32proc.c:sys_spawnve).  */
  1250   cpid = spawnve (_P_NOWAIT, new_argv[0], new_argv, env);
  1251   reset_standard_handles (in, out, err, handles);
  1252   return cpid;
  1253 
  1254 #else  /* not WINDOWSNT */
  1255 
  1256 #ifndef MSDOS
  1257 
  1258   restore_nofile_limit ();
  1259 
  1260   /* Redirect file descriptors and clear the close-on-exec flag on the
  1261      redirected ones.  IN, OUT, and ERR are close-on-exec so they
  1262      need not be closed explicitly.  */
  1263   dup2 (in, STDIN_FILENO);
  1264   dup2 (out, STDOUT_FILENO);
  1265   dup2 (err, STDERR_FILENO);
  1266 
  1267   setpgid (0, 0);
  1268   tcsetpgrp (0, pid);
  1269 
  1270   int errnum = emacs_exec_file (new_argv[0], new_argv, env);
  1271   exec_failed (new_argv[0], errnum);
  1272 
  1273 #else /* MSDOS */
  1274   i = strlen (current_dir);
  1275   pwd_var = xmalloc (i + 5);
  1276   temp = pwd_var + 4;
  1277   memcpy (pwd_var, "PWD=", 4);
  1278   stpcpy (temp, current_dir);
  1279 
  1280   if (i > 2 && IS_DEVICE_SEP (temp[1]) && IS_DIRECTORY_SEP (temp[2]))
  1281     {
  1282       temp += 2;
  1283       i -= 2;
  1284     }
  1285 
  1286   /* Strip trailing slashes for PWD, but leave "/" and "//" alone.  */
  1287   while (i > 2 && IS_DIRECTORY_SEP (temp[i - 1]))
  1288     temp[--i] = 0;
  1289 
  1290   pid = run_msdos_command (new_argv, pwd_var + 4, in, out, err, env);
  1291   xfree (pwd_var);
  1292   if (pid == -1)
  1293     /* An error occurred while trying to run the subprocess.  */
  1294     report_file_error ("Spawning child process", Qnil);
  1295   return pid;
  1296 #endif  /* MSDOS */
  1297 #endif  /* not WINDOWSNT */
  1298 }

  1462       if (error != 0)

  1466         }

  1471           errno = error;

  1473         }

  1476     }
  1477 #endif
  1478 
  1479 #ifndef WINDOWSNT
  1480   /* vfork, and prevent local vars from being clobbered by the vfork.  */

  1482   const char *volatile cwd_volatile = cwd;
  1483   const char *volatile ptyname_volatile = pty_name;

  1485   bool volatile ptyout_volatile = pty_out;

  1585           SETUP_SLAVE_PTY;
  1586         }
  1587 #endif /* SETUP_SLAVE_PTY */
  1588 #endif /* HAVE_PTYS */

  1590 #ifdef DARWIN_OS
  1591       /* Work around a macOS bug, where SIGCHLD is apparently

  1639     }
  1640 
  1641   eassert (0 < pid);
  1642   *newpid = pid;
  1643   return 0;
  1644 }

  1646 static bool
  1647 getenv_internal_1 (const char *var, ptrdiff_t varlen, char **value,
  1648                    ptrdiff_t *valuelen, Lisp_Object env)
  1649 {
  1650   for (; CONSP (env); env = XCDR (env))
  1651     {
  1652       Lisp_Object entry = XCAR (env);
  1653       if (STRINGP (entry)
  1654           && SBYTES (entry) >= varlen
  1655 #ifdef WINDOWSNT
  1656           /* NT environment variables are case insensitive.  */
  1657           && ! strnicmp (SSDATA (entry), var, varlen)
  1658 #else  /* not WINDOWSNT */
  1659           && ! memcmp (SDATA (entry), var, varlen)
  1660 #endif /* not WINDOWSNT */
  1661           )
  1662         {
  1663           if (SBYTES (entry) > varlen && SREF (entry, varlen) == '=')
  1664             {
  1665               *value = SSDATA (entry) + (varlen + 1);
  1666               *valuelen = SBYTES (entry) - (varlen + 1);
  1667               return 1;
  1668             }
  1669           else if (SBYTES (entry) == varlen)
  1670             {
  1671               /* Lone variable names in Vprocess_environment mean that
  1672                  variable should be removed from the environment. */
  1673               *value = NULL;
  1674               return 1;
  1675             }
  1676         }
  1677     }
  1678   return 0;
  1679 }

  1681 static bool
  1682 getenv_internal (const char *var, ptrdiff_t varlen, char **value,
  1683                  ptrdiff_t *valuelen, Lisp_Object frame)
  1684 {
  1685   /* Try to find VAR in Vprocess_environment first.  */
  1686   if (getenv_internal_1 (var, varlen, value, valuelen,
  1687                          Vprocess_environment))
  1688     return *value ? 1 : 0;
  1689 
  1690   /* On Windows we make some modifications to Emacs' environment
  1691      without recording them in Vprocess_environment.  */
  1692 #ifdef WINDOWSNT
  1693   {
  1694     char *tmpval = getenv (var);
  1695     if (tmpval)
  1696       {
  1697         *value = tmpval;
  1698         *valuelen = strlen (tmpval);
  1699         return 1;
  1700       }
  1701   }
  1702 #endif
  1703 
  1704   /* For DISPLAY try to get the values from the frame or the initial env.  */
  1705   if (strcmp (var, "DISPLAY") == 0)
  1706     {
  1707 #ifndef HAVE_PGTK
  1708       Lisp_Object display
  1709         = Fframe_parameter (NILP (frame) ? selected_frame : frame, Qdisplay);
  1710       if (STRINGP (display))
  1711         {
  1712           *value    = SSDATA (display);
  1713           *valuelen = SBYTES (display);
  1714           return 1;
  1715         }
  1716 #endif
  1717       /* If still not found, Look for DISPLAY in Vinitial_environment.  */
  1718       if (getenv_internal_1 (var, varlen, value, valuelen,
  1719                              Vinitial_environment))
  1720         return *value ? 1 : 0;
  1721     }
  1722 
  1723   return 0;
  1724 }

  1785 
  1786     i = SBYTES (current_dir);
  1787     pwd_var = xmalloc (i + 5);
  1788     record_unwind_protect_ptr (xfree, pwd_var);
  1789     temp = pwd_var + 4;
  1790     memcpy (pwd_var, "PWD=", 4);
  1791     lispstpcpy (temp, current_dir);
  1792 
  1793 #ifdef DOS_NT
  1794     /* Get past the drive letter, so that d:/ is left alone.  */
  1795     if (i > 2 && IS_DEVICE_SEP (temp[1]) && IS_DIRECTORY_SEP (temp[2]))
  1796       {
  1797         temp += 2;
  1798         i -= 2;
  1799       }
  1800 #endif /* DOS_NT */
  1801 
  1802     /* Strip trailing slashes for PWD, but leave "/" and "//" alone.  */
  1803     while (i > 2 && IS_DIRECTORY_SEP (temp[i - 1]))
  1804       temp[--i] = 0;
  1805   }
  1806 
  1807   /* Set `env' to a vector of the strings in the environment.  */
  1808 
  1809   {
  1810     register Lisp_Object tem;
  1811     register char **new_env;
  1812     char **p, **q;
  1813     register int new_length;
  1814     Lisp_Object display = Qnil;
  1815 
  1816     new_length = 0;
  1817 
  1818     for (tem = Vprocess_environment;
  1819          CONSP (tem) && STRINGP (XCAR (tem));
  1820          tem = XCDR (tem))
  1821       {
  1822         if (strncmp (SSDATA (XCAR (tem)), "DISPLAY", 7) == 0
  1823             && (SDATA (XCAR (tem)) [7] == '\0'
  1824                 || SDATA (XCAR (tem)) [7] == '='))
  1825           /* DISPLAY is specified in process-environment.  */
  1826           display = Qt;
  1827         new_length++;
  1828       }
  1829 
  1830     /* If not provided yet, use the frame's DISPLAY.  */
  1831     if (NILP (display))
  1832       {
  1833         Lisp_Object tmp = Fframe_parameter (selected_frame, Qdisplay);
  1834 
  1835 #ifdef HAVE_PGTK
  1836         /* The only time GDK actually returns correct information is
  1837            when it's running under X Windows.  DISPLAY shouldn't be
  1838            set to a Wayland display either, since that's an X specific
  1839            variable.  */
  1840         if (FRAME_WINDOW_P (SELECTED_FRAME ())
  1841             && strcmp (G_OBJECT_TYPE_NAME (FRAME_X_DISPLAY (SELECTED_FRAME ())),
  1842                        "GdkX11Display"))
  1843           tmp = Qnil;
  1844 #endif
  1845 
  1846         if (!STRINGP (tmp) && CONSP (Vinitial_environment))
  1847           /* If still not found, Look for DISPLAY in Vinitial_environment.  */
  1848           tmp = Fgetenv_internal (build_string ("DISPLAY"),
  1849                                   Vinitial_environment);
  1850         if (STRINGP (tmp))
  1851           {
  1852             display = tmp;
  1853             new_length++;
  1854           }
  1855       }
  1856 
  1857     /* new_length + 2 to include PWD and terminating 0.  */
  1858     env = new_env = xnmalloc (new_length + 2, sizeof *env);
  1859     record_unwind_protect_ptr (xfree, env);
  1860     /* If we have a PWD envvar, pass one down,
  1861        but with corrected value.  */
  1862     if (egetenv ("PWD"))
  1863       *new_env++ = pwd_var;
  1864 
  1865     if (STRINGP (display))
  1866       {
  1867         char *vdata = xmalloc (sizeof "DISPLAY=" + SBYTES (display));
  1868         record_unwind_protect_ptr (xfree, vdata);
  1869         lispstpcpy (stpcpy (vdata, "DISPLAY="), display);
  1870         new_env = add_env (env, new_env, vdata);
  1871       }
  1872 
  1873     /* Overrides.  */
  1874     for (tem = Vprocess_environment;
  1875          CONSP (tem) && STRINGP (XCAR (tem));
  1876          tem = XCDR (tem))
  1877       new_env = add_env (env, new_env, SSDATA (XCAR (tem)));
  1878 
  1879     *new_env = 0;
  1880 
  1881     /* Remove variable names without values.  */
  1882     p = q = env;
  1883     while (*p != 0)
  1884       {
  1885         while (*q != 0 && strchr (*q, '=') == NULL)
  1886           q++;
  1887         *p = *q++;
  1888         if (*p != 0)
  1889           p++;
  1890       }
  1891   }
  1892 
  1893   return env;
  1894 }

  1918 void
  1919 init_callproc (void)
  1920 {
  1921   bool data_dir = egetenv ("EMACSDATA") != 0;
  1922 
  1923   char *sh;
  1924   Lisp_Object tempdir;
  1925 
  1926   if (!NILP (Vinstallation_directory))
  1927     {
  1928       /* Add to the path the lib-src subdir of the installation dir.  */
  1929       Lisp_Object tem;
  1930       tem = Fexpand_file_name (build_string ("lib-src"),
  1931                                Vinstallation_directory);
  1932 #ifndef MSDOS
  1933           /* MSDOS uses wrapped binaries, so don't do this.  */
  1934       if (NILP (Fmember (tem, Vexec_path)))
  1935         {
  1936           /* Running uninstalled, so default to tem rather than PATH_EXEC.  */
  1937           Vexec_path = decode_env_path ("EMACSPATH", SSDATA (tem), 0);
  1938           Vexec_path = nconc2 (decode_env_path ("PATH", "", 0), Vexec_path);
  1939         }
  1940 
  1941       Vexec_directory = Ffile_name_as_directory (tem);
  1942 #endif /* not MSDOS */
  1943 
  1944       /* Maybe use ../etc as well as ../lib-src.  */
  1945       if (data_dir == 0)
  1946         {
  1947           tem = Fexpand_file_name (build_string ("etc"),
  1948                                    Vinstallation_directory);
  1949           Vdoc_directory = Ffile_name_as_directory (tem);
  1950         }
  1951     }
  1952 
  1953   /* Look for the files that should be in etc.  We don't use
  1954      Vinstallation_directory, because these files are never installed
  1955      near the executable, and they are never in the build
  1956      directory when that's different from the source directory.
  1957 
  1958      Instead, if these files are not in the nominal place, we try the
  1959      source directory.  */
  1960   if (data_dir == 0)
  1961     {
  1962       Lisp_Object tem, srcdir;
  1963       Lisp_Object lispdir = Fcar (decode_env_path (0, PATH_DUMPLOADSEARCH, 0));
  1964 
  1965       srcdir = Fexpand_file_name (build_string ("../src/"), lispdir);
  1966 
  1967       tem = Fexpand_file_name (build_string ("NEWS"), Vdata_directory);
  1968       if (!NILP (Fequal (srcdir, Vinvocation_directory))
  1969           || NILP (Ffile_exists_p (tem)) || !NILP (Vinstallation_directory))
  1970         {
  1971           Lisp_Object newdir;
  1972           newdir = Fexpand_file_name (build_string ("../etc/"), lispdir);
  1973           tem = Fexpand_file_name (build_string ("NEWS"), newdir);
  1974           if (!NILP (Ffile_exists_p (tem)))
  1975             Vdata_directory = newdir;
  1976         }
  1977     }
  1978 
  1979   if (!will_dump_p ())
  1980     {
  1981       tempdir = Fdirectory_file_name (Vexec_directory);
  1982       if (! file_accessible_directory_p (tempdir))
  1983         dir_warning ("arch-dependent data dir", Vexec_directory);
  1984     }
  1985 
  1986   tempdir = Fdirectory_file_name (Vdata_directory);
  1987   if (! file_accessible_directory_p (tempdir))
  1988     dir_warning ("arch-independent data dir", Vdata_directory);
  1989 
  1990   sh = getenv ("SHELL");
  1991   Vshell_file_name = build_string (sh ? sh : "/bin/sh");
  1992 
  1993   Lisp_Object gamedir = Qnil;
  1994   if (PATH_GAME)
  1995     {
  1996       const char *cpath_game = PATH_GAME;
  1997 #ifdef WINDOWSNT
  1998       /* On MS-Windows, PATH_GAME normally starts with a literal
  1999          "%emacs_dir%", so it will never work without some tweaking.  */
  2000       cpath_game = w32_relocate (cpath_game);
  2001 #endif
  2002       Lisp_Object path_game = build_unibyte_string (cpath_game);
  2003       if (file_accessible_directory_p (path_game))
  2004         gamedir = path_game;
  2005       else if (errno != ENOENT && errno != ENOTDIR
  2006 #ifdef DOS_NT
  2007                /* DOS/Windows sometimes return EACCES for bad file names  */
  2008                && errno != EACCES
  2009 #endif
  2010                )
  2011         dir_warning ("game dir", path_game);
  2012     }
  2013   Vshared_game_score_directory = gamedir;
  2014 }
