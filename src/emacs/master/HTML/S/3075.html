  1616    be hidden in these macros, if that's possible.
  1617 
  1618    Return the upper/left pixel position of the character cell on frame F
  1619    at ROW/COL.  */
  1620 #define FRAME_LINE_TO_PIXEL_Y(f, row)                                   \
  1621   (((row) < FRAME_TOP_MARGIN (f) ? 0 : FRAME_INTERNAL_BORDER_WIDTH (f)) \
  1622    + (row) * FRAME_LINE_HEIGHT (f))
  1623 
  1624 #define FRAME_COL_TO_PIXEL_X(f, col)            \
  1625   (FRAME_INTERNAL_BORDER_WIDTH (f)              \
  1626    + (col) * FRAME_COLUMN_WIDTH (f))
  1627 
  1628 /* Return the pixel width/height of frame F if it has
  1629    COLS columns/LINES rows.  */
  1630 #define FRAME_TEXT_COLS_TO_PIXEL_WIDTH(f, cols) \
  1631   ((cols) * FRAME_COLUMN_WIDTH (f)              \
  1632    + FRAME_SCROLL_BAR_AREA_WIDTH (f)            \
  1633    + FRAME_TOTAL_FRINGE_WIDTH (f)               \
  1634    + 2 * FRAME_INTERNAL_BORDER_WIDTH (f))
  1635 
  1636 #define FRAME_TEXT_LINES_TO_PIXEL_HEIGHT(f, lines) \
  1637   ((lines) * FRAME_LINE_HEIGHT (f)                 \
  1638    + FRAME_TOP_MARGIN_HEIGHT (f)                   \
  1639    + FRAME_SCROLL_BAR_AREA_HEIGHT (f)              \
  1640    + 2 * FRAME_INTERNAL_BORDER_WIDTH (f))
  1641 
  1642 /* Return the row/column (zero-based) of the character cell containing
  1643    the pixel on FRAME at Y/X.  */
  1644 #define FRAME_PIXEL_Y_TO_LINE(f, y)                                     \
  1645   (((y) < FRAME_TOP_MARGIN_HEIGHT (f)                                   \
  1646     ? (y)                                                               \
  1647     : ((y) < (FRAME_TOP_MARGIN_HEIGHT (f)                               \
  1648               + FRAME_INTERNAL_BORDER_WIDTH (f))                        \
  1649        ? (y) - (FRAME_TOP_MARGIN_HEIGHT (f)                             \
  1650                 + FRAME_INTERNAL_BORDER_WIDTH (f)                       \
  1651                 /* Arrange for the division to round down.  */          \
  1652                 + FRAME_LINE_HEIGHT (f) - 1)                            \
  1653        : (y) - FRAME_INTERNAL_BORDER_WIDTH (f)))                        \
  1654    / FRAME_LINE_HEIGHT (f))
  1655 
  1656 #define FRAME_PIXEL_X_TO_COL(f, x)              \
  1657   (((x) - FRAME_INTERNAL_BORDER_WIDTH (f))      \
  1658    / FRAME_COLUMN_WIDTH (f))
  1659 
  1660 /* How many columns/rows of text can we fit in WIDTH/HEIGHT pixels on
  1661    frame F (so we round down)?  */
  1662 #define FRAME_PIXEL_WIDTH_TO_TEXT_COLS(f, width)                        \
  1663   (((width)                                                             \
  1664     - FRAME_TOTAL_FRINGE_WIDTH (f)                                      \
  1665     - FRAME_SCROLL_BAR_AREA_WIDTH (f)                                   \
  1666     - 2 * FRAME_INTERNAL_BORDER_WIDTH (f))                              \
  1667    / FRAME_COLUMN_WIDTH (f))
  1668 
  1669 #define FRAME_PIXEL_HEIGHT_TO_TEXT_LINES(f, height)                     \
  1670   (((height)                                                            \
  1671     - FRAME_MARGIN_HEIGHT (f)                                           \
  1672     - FRAME_SCROLL_BAR_AREA_HEIGHT (f)                                  \
  1673     - 2 * FRAME_INTERNAL_BORDER_WIDTH (f))                              \
  1674    / FRAME_LINE_HEIGHT (f))
  1675 
  1676 /* Return the pixel width/height of frame F with a text size of
  1677    width/height.  */
  1678 #define FRAME_TEXT_TO_PIXEL_WIDTH(f, width)       \
  1679   ((width)                                        \
  1680    + FRAME_SCROLL_BAR_AREA_WIDTH (f)              \
  1681    + FRAME_TOTAL_FRINGE_WIDTH (f)                 \
  1682    + 2 * FRAME_INTERNAL_BORDER_WIDTH (f))
  1683 
  1684 #define FRAME_TEXT_TO_PIXEL_HEIGHT(f, height)        \
  1685   ((height)                                          \
  1686    + FRAME_MARGIN_HEIGHT (f)                         \
  1687    + FRAME_SCROLL_BAR_AREA_HEIGHT (f)                \
  1688    + 2 * FRAME_INTERNAL_BORDER_WIDTH (f))
  1689 
  1690 /* Return the text width/height of frame F with a pixel size of
  1691    width/height.  */
  1692 #define FRAME_PIXEL_TO_TEXT_WIDTH(f, width)       \
  1693   ((width)                                        \
  1694    - FRAME_SCROLL_BAR_AREA_WIDTH (f)              \
  1695    - FRAME_TOTAL_FRINGE_WIDTH (f)                 \
  1696    - 2 * FRAME_INTERNAL_BORDER_WIDTH (f))
  1697 
  1698 #define FRAME_PIXEL_TO_TEXT_HEIGHT(f, height)           \
  1699   ((height)                                             \
  1700    - FRAME_MARGIN_HEIGHT (f)                            \
  1701    - FRAME_SCROLL_BAR_AREA_HEIGHT (f)                   \
  1702    - 2 * FRAME_INTERNAL_BORDER_WIDTH (f))
  1703 
  1704 #define FRAME_INNER_WIDTH(f)                    \
  1705   (FRAME_PIXEL_WIDTH (f)                        \
  1706    - 2 * FRAME_INTERNAL_BORDER_WIDTH (f))
  1707 
  1708 #define FRAME_INNER_HEIGHT(f)                   \
  1709   (FRAME_PIXEL_HEIGHT (f)                       \
  1710    - FRAME_MARGIN_HEIGHT (f)                    \
  1711    - 2 * FRAME_INTERNAL_BORDER_WIDTH (f))
  1712 
  1713 /* Value is the smallest width of any character in any font on frame F.  */
  1714 #define FRAME_SMALLEST_CHAR_WIDTH(f)            \
  1715   FRAME_DISPLAY_INFO (f)->smallest_char_width
  1716 
  1717 /* Value is the smallest height of any font on frame F.  */
