    22 /* This class makes the Emacs server work reasonably on Android.
    23 
    24    There is no way to make the Unix socket publicly available on
    25    Android.
    26 
    27    Instead, this activity tries to connect to the Emacs server, to
    28    make it open files the system asks Emacs to open, and to emulate
    29    some reasonable behavior when Emacs has not yet started.
    30 
    31    First, Emacs registers itself as an application that can open text
    32    and image files.
    33 
    34    Then, when the user is asked to open a file and selects ``Emacs''
    35    as the application that will open the file, the system pops up a
    36    window, this activity, and calls the `onCreate' function.
    37 
    38    `onCreate' then tries very to find the file name of the file that
    39    was selected, and give it to emacsclient.
    40 
    41    If emacsclient successfully opens the file, then this activity
    42    starts EmacsActivity (to bring it on to the screen); otherwise, it
    43    displays the output of emacsclient or any error message that occurs
    44    and exits.  */

   104       try
   105         {
   106           /* Start emacsclient.  */
   107           process = builder.start ();

   142   public String
   143   readEmacsClientLog ()
   144   {
   145     File file, cache;
   146     FileReader reader;
   147     char[] buffer;
   148     int rc;
   149     StringBuilder builder;
   150 
   151     /* Because the ProcessBuilder functions necessary to redirect
   152        process output are not implemented on Android 7 and earlier,
   153        print a generic error message.  */
   154 
   155     if (Build.VERSION.SDK_INT < Build.VERSION_CODES.O)
   156       return ("This is likely because the Emacs server"
   157               + " is not running, or because you did"
   158               + " not grant Emacs permission to access"
   159               + " external storage.");
   160 
   161     cache = getCacheDir ();
   162     file = new File (cache, "emacsclient.log");
   163     builder = new StringBuilder ();
   164     reader = null;
   165 
   166     try
   167       {
   168         reader = new FileReader (file);
   169         buffer = new char[2048];
   170 
   171         while ((rc = reader.read (buffer, 0, 2048)) != -1)
   172           builder.append (buffer, 0, rc);
   173 
   174         reader.close ();
   175         return builder.toString ();
   176       }
   177     catch (IOException exception)
   178       {
   179         /* Close the reader if it's already been opened.  */
   180 
   181         try
   182           {
   183             if (reader != null)
   184               reader.close ();
   185           }
   186         catch (IOException e)
   187           {
   188             /* Not sure what to do here.  */
   189           }
   190 
   191         return ("Couldn't read emacsclient.log: "
   192                 + exception.toString ());
   193       }
   194   }

   216   /* Check that the specified FILE is readable.  If Android 4.4 or
   217      later is being used, return URI formatted into a `/content/' file
   218      name.
   219 
   220      If it is not, then copy the file in FD to a location in the
   221      system cache directory and return the name of that file.  */
   222 
   223   private String
   224   checkReadableOrCopy (String file, ParcelFileDescriptor fd,
   225                        Uri uri)
   226     throws IOException, FileNotFoundException
   227   {
   228     File inFile;
   229     FileOutputStream outStream;
   230     InputStream stream;
   231     byte buffer[];
   232     int read;
   233     String content;
   234 
   235     Log.d (TAG, "checkReadableOrCopy: " + file);
   236 
   237     inFile = new File (file);
   238 
   239     if (inFile.canRead ())
   240       return file;
   241 
   242     Log.d (TAG, "checkReadableOrCopy: NO");
   243 
   244     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.KITKAT)
   245       {
   246         content = "/content/" + uri.getEncodedAuthority ();
   247 
   248         for (String segment : uri.getPathSegments ())
   249           content += "/" + Uri.encode (segment);
   250 
   251         /* Append the URI query.  */
   252 
   253         if (uri.getEncodedQuery () != null)
   254           content += "?" + uri.getEncodedQuery ();
   255 
   256         Log.d (TAG, "checkReadableOrCopy: " + content);
   257 
   258         return content;
   259       }
   260 
   261     /* inFile is now the file being written to.  */
   262     inFile = new File (getCacheDir (), inFile.getName ());
   263     buffer = new byte[4098];
   264 
   265     /* Initialize both streams to NULL.  */
   266     outStream = null;
   267     stream = null;
   268 
   269     try
   270       {
   271         outStream = new FileOutputStream (inFile);
   272         stream = new FileInputStream (fd.getFileDescriptor ());
   273 
   274         while ((read = stream.read (buffer)) >= 0)
   275           outStream.write (buffer, 0, read);
   276       }
   277     finally
   278       {
   279         /* Note that this does not close FD.
   280 
   281            Keep in mind that execution is transferred to ``finally''
   282            even if an exception happens inside the while loop
   283            above.  */
   284 
   285         if (stream != null)
   286           stream.close ();
   287 
   288         if (outStream != null)
   289           outStream.close ();
   290       }
   291 
   292     return inFile.getCanonicalPath ();
   293   }

   344   public void
   345   startEmacsClient (String fileName)
   346   {
   347     String libDir;
   348     ProcessBuilder builder;
   349     Process process;
   350     EmacsClientThread thread;
   351     File file;
   352     Intent intent;
   353 
   354     /* If the Emacs service is not running, then start Emacs and make
   355        it open this file.  */
   356 
   357     if (EmacsService.SERVICE == null)
   358       {
   359         fileToOpen = fileName;
   360         intent = new Intent (EmacsOpenActivity.this,
   361                              EmacsActivity.class);
   362         finish ();
   363         startActivity (intent);
   364         return;
   365       }
   366 
   367     libDir = EmacsService.getLibraryDirectory (this);
   368     builder = new ProcessBuilder (libDir + "/libemacsclient.so",
   369                                   fileName, "--reuse-frame",
   370                                   "--timeout=10", "--no-wait");
   371 
   372     /* Redirection is unfortunately not possible in Android 7 and
   373        earlier.  */
   374 
   375     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O)
   376       {
   377         file = new File (getCacheDir (), "emacsclient.log");
   378 
   379         /* Redirect standard error to a file so that errors can be
   380            meaningfully reported.  */
   381 
   382         if (file.exists ())
   383           file.delete ();
   384 
   385         builder.redirectError (file);
   386       }
   387 
   388     /* Track process output in a new thread, since this is the UI
   389        thread and doing so here can cause deadlocks when EmacsService
   390        decides to wait for something.  */
   391 
   392     thread = new EmacsClientThread (builder);
   393     thread.start ();
   394   }

   405   @Override
   406   public void
   407   onCreate (Bundle savedInstanceState)
   408   {
   409     String action, fileName;
   410     Intent intent;
   411     Uri uri;
   412     ContentResolver resolver;
   413     ParcelFileDescriptor fd;
   414     byte[] names;
   415     String errorBlurb;
   416 
   417     super.onCreate (savedInstanceState);
   418 
   419     /* Obtain the intent that started Emacs.  */
   420     intent = getIntent ();
   421     action = intent.getAction ();
   422 
   423     if (action == null)
   424       {
   425         finish ();
   426         return;
   427       }
   428 
   429     /* Now see if the action specified is supported by Emacs.  */
   430 
   431     if (action.equals ("android.intent.action.VIEW")
   432         || action.equals ("android.intent.action.EDIT")
   433         || action.equals ("android.intent.action.PICK"))
   434       {
   435         /* Obtain the URI of the action.  */
   436         uri = intent.getData ();
   437 
   438         if (uri == null)
   439           {
   440             finish ();
   441             return;
   442           }
   443 
   444         /* Now, try to get the file name.  */
   445 
   446         if (uri.getScheme ().equals ("file"))
   447           fileName = uri.getPath ();
   448         else
   449           {
   450             fileName = null;
   451 
   452             if (uri.getScheme ().equals ("content"))
   453               {
   454                 /* This is one of the annoying Android ``content''
   455                    URIs.  Most of the time, there is actually an
   456                    underlying file, but it cannot be found without
   457                    opening the file and doing readlink on its file
   458                    descriptor in /proc/self/fd.  */
   459                 resolver = getContentResolver ();
   460                 fd = null;
   461 
   462                 try
   463                   {
   464                     fd = resolver.openFileDescriptor (uri, "r");
   465                     names = EmacsNative.getProcName (fd.getFd ());
   466 
   467                     /* What is the right encoding here? */
   468 
   469                     if (names != null)
   470                       fileName = new String (names, "UTF-8");
   471 
   472                     fileName = checkReadableOrCopy (fileName, fd, uri);
   473                   }
   474                 catch (FileNotFoundException exception)
   475                   {
   476                     /* Do nothing.  */
   477                   }
   478                 catch (IOException exception)
   479                   {
   480                     /* Do nothing.  */
   481                   }
   482 
   483                 if (fd != null)
   484                   {
   485                     try
   486                       {
   487                         fd.close ();
   488                       }
   489                     catch (IOException exception)
   490                       {
   491                         /* Do nothing.  */
   492                       }
   493                   }
   494               }
   495 
   496             if (fileName == null)
   497               {
   498                 errorBlurb = ("The URI: " + uri + " could not be opened"
   499                               + ", as it does not encode file name inform"
   500                               + "ation.");
   501                 displayFailureDialog ("Error opening file", errorBlurb);
   502                 return;
   503               }
   504           }
   505 
   506         /* And start emacsclient.  Set `currentActivity' to this now.
   507            Presumably, it will shortly become capable of displaying
   508            dialogs.  */
   509         currentActivity = this;
   510         startEmacsClient (fileName);
   511       }
   512     else
   513       finish ();
   514   }
