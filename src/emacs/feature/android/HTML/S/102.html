   205   @Override
   206   public void
   207   onCreate ()
   208   {
   209     final AssetManager manager;
   210     Context app_context;
   211     final String filesDir, libDir, cacheDir, classPath;
   212     final double pixelDensityX;
   213     final double pixelDensityY;
   214     final double scaledDensity;
   215     double tempScaledDensity;
   216 
   217     SERVICE = this;
   218     handler = new Handler (Looper.getMainLooper ());
   219     manager = getAssets ();
   220     app_context = getApplicationContext ();
   221     metrics = getResources ().getDisplayMetrics ();
   222     pixelDensityX = metrics.xdpi;
   223     pixelDensityY = metrics.ydpi;
   224     tempScaledDensity = ((metrics.scaledDensity
   225                           / metrics.density)
   226                          * pixelDensityX);
   227     resolver = getContentResolver ();
   228 
   229     /* If the density used to compute the text size is lesser than
   230        160, there's likely a bug with display density computation.
   231        Reset it to 160 in that case.
   232 
   233        Note that Android uses 160 ``dpi'' as the density where 1 point
   234        corresponds to 1 pixel, not 72 or 96 as used elsewhere.  This
   235        difference is codified in PT_PER_INCH defined in font.h.  */
   236 
   237     if (tempScaledDensity < 160)
   238       tempScaledDensity = 160;
   239 
   240     /* scaledDensity is const as required to refer to it from within
   241        the nested function below.  */
   242     scaledDensity = tempScaledDensity;
   243 
   244     try
   245       {
   246         /* Configure Emacs with the asset manager and other necessary
   247            parameters.  */
   248         filesDir = app_context.getFilesDir ().getCanonicalPath ();
   249         libDir = getLibraryDirectory (this);
   250         cacheDir = app_context.getCacheDir ().getCanonicalPath ();
   251 
   252         /* Now provide this application's apk file, so a recursive
   253            invocation of app_process (through android-emacs) can
   254            find EmacsNoninteractive.  */
   255         classPath = getApkFile ();
   256 
   257         Log.d (TAG, "Initializing Emacs, where filesDir = " + filesDir
   258                + ", libDir = " + libDir + ", and classPath = " + classPath
   259                + "; fileToOpen = " + EmacsOpenActivity.fileToOpen
   260                + "; display density: " + pixelDensityX + " by "
   261                + pixelDensityY + " scaled to " + scaledDensity);
   262 
   263         /* Start the thread that runs Emacs.  */
   264         thread = new EmacsThread (this, new Runnable () {
   265             @Override
   266             public void
   267             run ()
   268             {
   269               EmacsNative.setEmacsParams (manager, filesDir, libDir,
   270                                           cacheDir, (float) pixelDensityX,
   271                                           (float) pixelDensityY,
   272                                           (float) scaledDensity,
   273                                           classPath, EmacsService.this);
   274             }
   275           }, extraStartupArgument,
   276           /* If any file needs to be opened, open it now.  */
   277           EmacsOpenActivity.fileToOpen);
   278         thread.start ();
   279       }
   280     catch (IOException exception)
   281       {
   282         EmacsNative.emacsAbort ();
   283         return;
   284       }
   285   }

   508   public boolean
   509   detectMouse ()
   510   {
   511     InputManager manager;
   512     InputDevice device;
   513     int[] ids;
   514     int i;
   515 
   516     if (Build.VERSION.SDK_INT
   517         /* Android 4.0 and earlier don't support mouse input events at
   518            all.  */
   519         < Build.VERSION_CODES.JELLY_BEAN)
   520       return false;
   521 
   522     manager = (InputManager) getSystemService (Context.INPUT_SERVICE);
   523     ids = manager.getInputDeviceIds ();
   524 
   525     for (i = 0; i < ids.length; ++i)
   526       {
   527         device = manager.getInputDevice (ids[i]);
   528 
   529         if (device == null)
   530           continue;
   531 
   532         if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP)
   533           {
   534             if (device.supportsSource (InputDevice.SOURCE_MOUSE))
   535               return true;
   536           }
   537         else
   538           {
   539             /* `supportsSource' is only present on API level 21 and
   540                later, but earlier versions provide a bit mask
   541                containing each supported source.  */
   542 
   543             if ((device.getSources () & InputDevice.SOURCE_MOUSE) != 0)
   544               return true;
   545           }
   546       }
   547 
   548     return false;
   549   }

   586   /* Ask the system to open the specified URL in an application that
   587      understands how to open it.
   588 
   589      If SEND, tell the system to also open applications that can
   590      ``send'' the URL (through mail, for example), instead of only
   591      those that can view the URL.
   592 
   593      Value is NULL upon success, or a string describing the error
   594      upon failure.  */
   595 
   596   public String
   597   browseUrl (String url, boolean send)
   598   {
   599     Intent intent;
   600     Uri uri;
   601 
   602     try
   603       {
   604         /* Parse the URI.  */
   605         if (!send)
   606           {
   607             uri = Uri.parse (url);
   608 
   609             if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.KITKAT)
   610               {
   611                 /* On Android 4.4 and later, check if URI is actually
   612                    a file name.  If so, rewrite it into a content
   613                    provider URI, so that it can be accessed by other
   614                    programs.  */
   615 
   616                 if (uri.getScheme ().equals ("file")
   617                     && uri.getPath () != null)
   618                   uri
   619                     = DocumentsContract.buildDocumentUri ("org.gnu.emacs",
   620                                                           uri.getPath ());
   621               }
   622 
   623             Log.d (TAG, ("browseUri: browsing " + url
   624                          + " --> " + uri.getPath ()
   625                          + " --> " + uri));
   626 
   627             intent = new Intent (Intent.ACTION_VIEW, uri);
   628             intent.setFlags (Intent.FLAG_ACTIVITY_NEW_TASK
   629                              | Intent.FLAG_GRANT_READ_URI_PERMISSION);
   630           }
   631         else
   632           {
   633             intent = new Intent (Intent.ACTION_SEND);
   634             intent.setType ("text/plain");
   635             intent.putExtra (Intent.EXTRA_SUBJECT, "Sharing link");
   636             intent.putExtra (Intent.EXTRA_TEXT, url);
   637 
   638             /* Display a list of programs able to send this URL.  */
   639             intent = Intent.createChooser (intent, "Send");
   640 
   641             /* Apparently flags need to be set after a choser is
   642                created.  */
   643             intent.addFlags (Intent.FLAG_ACTIVITY_NEW_TASK);
   644           }
   645 
   646         startActivity (intent);
   647       }
   648     catch (Exception e)
   649       {
   650         return e.toString ();
   651       }
   652 
   653     return null;
   654   }

   703   public static void
   704   syncRunnable (Runnable runnable)
   705   {
   706     EmacsNative.beginSynchronous ();
   707 
   708     synchronized (runnable)
   709       {
   710         SERVICE.runOnUiThread (runnable);
   711 
   712         while (true)

   714             try

   716                 runnable.wait ();

   718               }

   722               }
   723           }
   724       }

   733      EmacsView.java) is called; when that then asks the UI thread for
   734      the current selection, a dead lock results.  To remedy this,
   735      reply to any synchronous queries now -- and prohibit more queries
   736      for the duration of `updateSelection' -- if EmacsView may have
   737      been asking for the value of the region.  */
   738 
   739   public static void
   740   icBeginSynchronous ()
   741   {
   742     /* Set servicingQuery to 2, so viewGetSelection knows it shouldn't
   743        proceed.  */
   744 
   745     if (servicingQuery.getAndSet (2) == 1)
   746       /* But if viewGetSelection is already in progress, answer it
   747          first.  */
   748       EmacsNative.answerQuerySpin ();
   749   }
   750 
   751   public static void
   752   icEndSynchronous ()
   753   {
   754     if (servicingQuery.getAndSet (0) != 2)
   755       throw new RuntimeException ("incorrect value of `servicingQuery': "
   756                                   + "likely 1");
   757   }
   758 
   759   public static int[]
   760   viewGetSelection (short window)
   761   {
   762     int[] selection;
   763 
   764     /* See if a query is already in progress from the other
   765        direction.  */
   766     if (!servicingQuery.compareAndSet (0, 1))
   767       return null;
   768 
   769     /* Now call the regular getSelection.  Note that this can't race
   770        with answerQuerySpin, as `android_servicing_query' can never be
   771        2 when icBeginSynchronous is called, so a query will always be
   772        started.  */
   773     selection = EmacsNative.getSelection (window);
   774 
   775     /* Finally, clear servicingQuery if its value is still 1.  If a
   776        query has started from the other side, it ought to be 2.  */
   777 
   778     servicingQuery.compareAndSet (1, 0);
   779     return selection;
   780   }

   806   {
   807     int oldMode;
   808 
   809     if (DEBUG_IC)
   810       Log.d (TAG, "resetIC: " + window + ", " + icMode);
   811 
   812     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU
   813         && (oldMode = window.view.getICMode ()) == icMode
   814         /* Don't do this if there is currently no input
   815            connection.  */
   816         && oldMode != IC_MODE_NULL)
   817       {
   818         if (DEBUG_IC)
   819           Log.d (TAG, "resetIC: calling invalidateInput");
   820 
   821         /* Android 33 and later allow the IM reset to be optimized out
   822            and replaced by a call to `invalidateInput', which is much
   823            faster, as it does not involve resetting the input
   824            connection.  */
   825 
   826         icBeginSynchronous ();
   827         window.view.imManager.invalidateInput (window.view);
   828         icEndSynchronous ();
   829 
   830         return;
   831       }
   832 
   833     window.view.setICMode (icMode);
   834 
   835     icBeginSynchronous ();
   836     window.view.icGeneration++;
   837     window.view.imManager.restartInput (window.view);
   838     icEndSynchronous ();
   839   }

   875   /* Open a content URI described by the bytes BYTES, a non-terminated
   876      string; make it writable if WRITABLE, and readable if READABLE.
   877      Truncate the file if TRUNCATE.
   878 
   879      Value is the resulting file descriptor or -1 upon failure.  */
   880 
   881   public int
   882   openContentUri (byte[] bytes, boolean writable, boolean readable,
   883                   boolean truncate)
   884   {
   885     String name, mode;
   886     ParcelFileDescriptor fd;
   887     int i;
   888 
   889     /* Figure out the file access mode.  */
   890 
   891     mode = "";
   892 
   893     if (readable)
   894       mode += "r";
   895 
   896     if (writable)
   897       mode += "w";
   898 
   899     if (truncate)
   900       mode += "t";
   901 
   902     /* Try to open an associated ParcelFileDescriptor.  */
   903 
   904     try
   905       {
   906         /* The usual file name encoding question rears its ugly head
   907            again.  */
   908         name = new String (bytes, "UTF-8");
   909         Log.d (TAG, "openContentUri: " + Uri.parse (name));
   910 
   911         fd = resolver.openFileDescriptor (Uri.parse (name), mode);
   912 
   913         /* Use detachFd on newer versions of Android or plain old
   914            dup.  */
   915 
   916         if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.HONEYCOMB_MR1)
   917           {
   918             i = fd.detachFd ();
   919             fd.close ();
   920 
   921             return i;
   922           }
   923         else
   924           {
   925             i = EmacsNative.dup (fd.getFd ());
   926             fd.close ();
   927 
   928             return i;
   929           }
   930       }
   931     catch (Exception exception)
   932       {
   933         return -1;
   934       }
   935   }

   939   {
   940     String mode, name;
   941     ParcelFileDescriptor fd;
   942

   944

   982   private long[]
   983   queryBattery19 ()
   984   {
   985     IntentFilter filter;
   986     Intent battery;
   987     long capacity, chargeCounter, currentAvg, currentNow;
   988     long status, remaining, plugged, temp;
   989 
   990     filter = new IntentFilter (Intent.ACTION_BATTERY_CHANGED);
   991     battery = registerReceiver (null, filter);
   992 
   993     if (battery == null)
   994       return null;
   995 
   996     capacity = battery.getIntExtra (BatteryManager.EXTRA_LEVEL, 0);
   997     chargeCounter
   998       = (battery.getIntExtra (BatteryManager.EXTRA_SCALE, 0)
   999          / battery.getIntExtra (BatteryManager.EXTRA_LEVEL, 100) * 100);
  1000     currentAvg = 0;
  1001     currentNow = 0;
  1002     status = battery.getIntExtra (BatteryManager.EXTRA_STATUS, 0);
  1003     remaining = -1;
  1004     plugged = battery.getIntExtra (BatteryManager.EXTRA_PLUGGED, 0);
  1005     temp = battery.getIntExtra (BatteryManager.EXTRA_TEMPERATURE, 0);
  1006 
  1007     return new long[] { capacity, chargeCounter, currentAvg,
  1008                         currentNow, remaining, status, plugged,
  1009                         temp, };

  1012   /* Return the status of the battery.  See struct
  1013      android_battery_status for the order of the elements
  1014      returned.
  1015 
  1016      Value may be null upon failure.  */
  1017 
  1018   public long[]
  1019   queryBattery ()
  1020   {
  1021     Object tem;
  1022     BatteryManager manager;
  1023     long capacity, chargeCounter, currentAvg, currentNow;
  1024     long status, remaining, plugged, temp;
  1025     int prop;
  1026     IntentFilter filter;
  1027     Intent battery;
  1028 
  1029     /* Android 4.4 or earlier require applications to use a different
  1030        API to query the battery status.  */
  1031 
  1032     if (Build.VERSION.SDK_INT < Build.VERSION_CODES.LOLLIPOP)
  1033       return queryBattery19 ();
  1034 
  1035     tem = getSystemService (Context.BATTERY_SERVICE);
  1036     manager = (BatteryManager) tem;
  1037     remaining = -1;
  1038 
  1039     prop = BatteryManager.BATTERY_PROPERTY_CAPACITY;
  1040     capacity = manager.getLongProperty (prop);
  1041     prop = BatteryManager.BATTERY_PROPERTY_CHARGE_COUNTER;
  1042     chargeCounter = manager.getLongProperty (prop);
  1043     prop = BatteryManager.BATTERY_PROPERTY_CURRENT_AVERAGE;
  1044     currentAvg = manager.getLongProperty (prop);
  1045     prop = BatteryManager.BATTERY_PROPERTY_CURRENT_NOW;
  1046     currentNow = manager.getLongProperty (prop);
  1047 
  1048     /* Return the battery status.  N.B. that Android 7.1 and earlier
  1049        only return ``charging'' or ``discharging''.  */
  1050 
  1051     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O)
  1052       status
  1053         = manager.getIntProperty (BatteryManager.BATTERY_PROPERTY_STATUS);
  1054     else if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M)
  1055       status = (manager.isCharging ()
  1056                 ? BatteryManager.BATTERY_STATUS_CHARGING
  1057                 : BatteryManager.BATTERY_STATUS_DISCHARGING);
  1058     else
  1059       status = (currentNow > 0
  1060                 ? BatteryManager.BATTERY_STATUS_CHARGING
  1061                 : BatteryManager.BATTERY_STATUS_DISCHARGING);
  1062 
  1063     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P)
  1064       remaining = manager.computeChargeTimeRemaining ();
  1065 
  1066     plugged = -1;
  1067     temp = -1;
  1068 
  1069     /* Now obtain additional information from the battery manager.  */
  1070 
  1071     filter = new IntentFilter (Intent.ACTION_BATTERY_CHANGED);
  1072     battery = registerReceiver (null, filter);
  1073 
  1074     if (battery != null)
  1075       {
  1076         plugged = battery.getIntExtra (BatteryManager.EXTRA_PLUGGED, 0);
  1077         temp = battery.getIntExtra (BatteryManager.EXTRA_TEMPERATURE, 0);
  1078 
  1079         /* Make status more reliable.  */
  1080         if (Build.VERSION.SDK_INT < Build.VERSION_CODES.M)
  1081           status = battery.getIntExtra (BatteryManager.EXTRA_STATUS, 0);
  1082       }
  1083 
  1084     return new long[] { capacity, chargeCounter, currentAvg,
  1085                         currentNow, remaining, status, plugged,
  1086                         temp, };
  1087   }
