   490   public boolean
   491   detectMouse ()
   492   {
   493     InputManager manager;
   494     InputDevice device;
   495     int[] ids;
   496     int i;
   497 
   498     if (Build.VERSION.SDK_INT
   499         /* Android 4.0 and earlier don't support mouse input events at
   500            all.  */
   501         < Build.VERSION_CODES.JELLY_BEAN)
   502       return false;
   503 
   504     manager = (InputManager) getSystemService (Context.INPUT_SERVICE);
   505     ids = manager.getInputDeviceIds ();
   506 
   507     for (i = 0; i < ids.length; ++i)
   508       {
   509         device = manager.getInputDevice (ids[i]);
   510 
   511         if (device == null)
   512           continue;
   513 
   514         if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP)
   515           {
   516             if (device.supportsSource (InputDevice.SOURCE_MOUSE))
   517               return true;
   518           }
   519         else
   520           {
   521             /* `supportsSource' is only present on API level 21 and
   522                later, but earlier versions provide a bit mask
   523                containing each supported source.  */
   524 
   525             if ((device.getSources () & InputDevice.SOURCE_MOUSE) != 0)
   526               return true;
   527           }
   528       }
   529 
   530     return false;
   531   }

   688   /* IMM functions such as `updateSelection' holds an internal lock
   689      that is also taken before `onCreateInputConnection' (in
   690      EmacsView.java) is called; when that then asks the UI thread for
   691      the current selection, a dead lock results.  To remedy this,
   692      reply to any synchronous queries now -- and prohibit more queries
   693      for the duration of `updateSelection' -- if EmacsView may have
   694      been asking for the value of the region.  */
   695 
   696   public static void
   697   icBeginSynchronous ()
   698   {
   699     /* Set servicingQuery to 2, so viewGetSelection knows it shouldn't
   700        proceed.  */
   701 
   702     if (servicingQuery.getAndSet (2) == 1)
   703       /* But if viewGetSelection is already in progress, answer it
   704          first.  */
   705       EmacsNative.answerQuerySpin ();
   706   }
   707 
   708   public static void
   709   icEndSynchronous ()
   710   {
   711     if (servicingQuery.getAndSet (0) != 2)
   712       throw new RuntimeException ("incorrect value of `servicingQuery': "
   713                                   + "likely 1");
   714   }

   716   public static int[]
   717   viewGetSelection (short window)
   718   {
   719     int[] selection;
   720 
   721     /* See if a query is already in progress from the other
   722        direction.  */
   723     if (!servicingQuery.compareAndSet (0, 1))
   724       return null;
   725 
   726     /* Now call the regular getSelection.  Note that this can't race
   727        with answerQuerySpin, as `android_servicing_query' can never be
   728        2 when icBeginSynchronous is called, so a query will always be
   729        started.  */
   730     selection = EmacsNative.getSelection (window);
   731 
   732     /* Finally, clear servicingQuery if its value is still 1.  If a
   733        query has started from the other side, it ought to be 2.  */
   734 
   735     servicingQuery.compareAndSet (1, 0);

   832   /* Open a content URI described by the bytes BYTES, a non-terminated
   833      string; make it writable if WRITABLE, and readable if READABLE.
   834      Truncate the file if TRUNCATE.
   835 
   836      Value is the resulting file descriptor or -1 upon failure.  */
   837 
   838   public int
   839   openContentUri (byte[] bytes, boolean writable, boolean readable,
   840                   boolean truncate)
   841   {
   842     String name, mode;
   843     ParcelFileDescriptor fd;
   844     int i;
   845 
   846     /* Figure out the file access mode.  */
   847 
   848     mode = "";
   849 
   850     if (readable)
   851       mode += "r";
   852 
   853     if (writable)
   854       mode += "w";
   855 
   856     if (truncate)
   857       mode += "t";
   858 
   859     /* Try to open an associated ParcelFileDescriptor.  */
   860 
   861     try
   862       {
   863         /* The usual file name encoding question rears its ugly head
   864            again.  */
   865         name = new String (bytes, "UTF-8");
   866         Log.d (TAG, "openContentUri: " + Uri.parse (name));
   867 
   868         fd = resolver.openFileDescriptor (Uri.parse (name), mode);
   869 
   870         /* Use detachFd on newer versions of Android or plain old
   871            dup.  */
   872 
   873         if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.HONEYCOMB_MR1)
   874           {
   875             i = fd.detachFd ();
   876             fd.close ();
   877 
   878             return i;
   879           }
   880         else
   881           {
   882             i = EmacsNative.dup (fd.getFd ());
   883             fd.close ();
   884 
   885             return i;
   886           }
   887       }
   888     catch (Exception exception)
   889       {
   890         return -1;
   891       }
   892   }

   969   /* Return the status of the battery.  See struct
   970      android_battery_status for the order of the elements
   971      returned.
   972 
   973      Value may be null upon failure.  */
   974 
   975   public long[]
   976   queryBattery ()
   977   {
   978     Object tem;
   979     BatteryManager manager;
   980     long capacity, chargeCounter, currentAvg, currentNow;
   981     long status, remaining, plugged, temp;
   982     int prop;
   983     IntentFilter filter;
   984     Intent battery;
   985 
   986     /* Android 4.4 or earlier require applications to use a different
   987        API to query the battery status.  */
   988 
   989     if (Build.VERSION.SDK_INT < Build.VERSION_CODES.LOLLIPOP)
   990       return queryBattery19 ();
   991 
   992     tem = getSystemService (Context.BATTERY_SERVICE);
   993     manager = (BatteryManager) tem;
   994     remaining = -1;
   995 
   996     prop = BatteryManager.BATTERY_PROPERTY_CAPACITY;
   997     capacity = manager.getLongProperty (prop);
   998     prop = BatteryManager.BATTERY_PROPERTY_CHARGE_COUNTER;
   999     chargeCounter = manager.getLongProperty (prop);
  1000     prop = BatteryManager.BATTERY_PROPERTY_CURRENT_AVERAGE;
  1001     currentAvg = manager.getLongProperty (prop);
  1002     prop = BatteryManager.BATTERY_PROPERTY_CURRENT_NOW;
  1003     currentNow = manager.getLongProperty (prop);
  1004 
  1005     /* Return the battery status.  N.B. that Android 7.1 and earlier
  1006        only return ``charging'' or ``discharging''.  */
  1007 
  1008     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O)
  1009       status
  1010         = manager.getIntProperty (BatteryManager.BATTERY_PROPERTY_STATUS);
  1011     else if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M)
  1012       status = (manager.isCharging ()
  1013                 ? BatteryManager.BATTERY_STATUS_CHARGING
  1014                 : BatteryManager.BATTERY_STATUS_DISCHARGING);
  1015     else
  1016       status = (currentNow > 0
  1017                 ? BatteryManager.BATTERY_STATUS_CHARGING
  1018                 : BatteryManager.BATTERY_STATUS_DISCHARGING);
  1019 
  1020     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P)
  1021       remaining = manager.computeChargeTimeRemaining ();
  1022 
  1023     plugged = -1;
  1024     temp = -1;
  1025 
  1026     /* Now obtain additional information from the battery manager.  */
  1027 
  1028     filter = new IntentFilter (Intent.ACTION_BATTERY_CHANGED);
  1029     battery = registerReceiver (null, filter);
  1030 
  1031     if (battery != null)
  1032       {
  1033         plugged = battery.getIntExtra (BatteryManager.EXTRA_PLUGGED, 0);
  1034         temp = battery.getIntExtra (BatteryManager.EXTRA_TEMPERATURE, 0);
  1035 
  1036         /* Make status more reliable.  */
  1037         if (Build.VERSION.SDK_INT < Build.VERSION_CODES.M)
  1038           status = battery.getIntExtra (BatteryManager.EXTRA_STATUS, 0);
  1039       }
  1040 
  1041     return new long[] { capacity, chargeCounter, currentAvg,
  1042                         currentNow, remaining, status, plugged,
  1043                         temp, };
  1044   }
