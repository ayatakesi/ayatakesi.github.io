   283 static int
   284 readchar (Lisp_Object readcharfun, bool *multibyte)
   285 {
   286   Lisp_Object tem;
   287   register int c;
   288   int (*readbyte) (int, Lisp_Object);
   289   unsigned char buf[MAX_MULTIBYTE_LENGTH];
   290   int i, len;
   291   bool emacs_mule_encoding = 0;
   292 
   293   if (multibyte)
   294     *multibyte = 0;
   295 
   296   readchar_offset++;
   297 
   298   if (BUFFERP (readcharfun))
   299     {
   300       register struct buffer *inbuffer = XBUFFER (readcharfun);
   301 
   302       ptrdiff_t pt_byte = BUF_PT_BYTE (inbuffer);
   303 
   304       if (! BUFFER_LIVE_P (inbuffer))
   305         return -1;
   306 
   307       if (pt_byte >= BUF_ZV_BYTE (inbuffer))
   308         return -1;
   309 
   310       if (! NILP (BVAR (inbuffer, enable_multibyte_characters)))
   311         {
   312           /* Fetch the character code from the buffer.  */
   313           unsigned char *p = BUF_BYTE_ADDRESS (inbuffer, pt_byte);
   314           int clen;
   315           c = string_char_and_length (p, &clen);
   316           pt_byte += clen;
   317           if (multibyte)
   318             *multibyte = 1;
   319         }
   320       else
   321         {
   322           c = BUF_FETCH_BYTE (inbuffer, pt_byte);
   323           if (! ASCII_CHAR_P (c))
   324             c = BYTE8_TO_CHAR (c);
   325           pt_byte++;
   326         }
   327       SET_BUF_PT_BOTH (inbuffer, BUF_PT (inbuffer) + 1, pt_byte);
   328 
   329       return c;
   330     }
   331   if (MARKERP (readcharfun))
   332     {
   333       register struct buffer *inbuffer = XMARKER (readcharfun)->buffer;
   334 
   335       ptrdiff_t bytepos = marker_byte_position (readcharfun);
   336 
   337       if (bytepos >= BUF_ZV_BYTE (inbuffer))
   338         return -1;
   339 
   340       if (! NILP (BVAR (inbuffer, enable_multibyte_characters)))
   341         {
   342           /* Fetch the character code from the buffer.  */
   343           unsigned char *p = BUF_BYTE_ADDRESS (inbuffer, bytepos);
   344           int clen;
   345           c = string_char_and_length (p, &clen);
   346           bytepos += clen;
   347           if (multibyte)
   348             *multibyte = 1;
   349         }
   350       else
   351         {
   352           c = BUF_FETCH_BYTE (inbuffer, bytepos);
   353           if (! ASCII_CHAR_P (c))
   354             c = BYTE8_TO_CHAR (c);
   355           bytepos++;
   356         }
   357 
   358       XMARKER (readcharfun)->bytepos = bytepos;
   359       XMARKER (readcharfun)->charpos++;
   360 
   361       return c;
   362     }
   363 
   364   if (EQ (readcharfun, Qlambda))
   365     {
   366       readbyte = readbyte_for_lambda;
   367       goto read_multibyte;
   368     }
   369 
   370   if (EQ (readcharfun, Qget_file_char))
   371     {
   372       eassert (infile);
   373       readbyte = readbyte_from_file;
   374       goto read_multibyte;
   375     }
   376 
   377   if (STRINGP (readcharfun))
   378     {
   379       if (read_from_string_index >= read_from_string_limit)
   380         c = -1;
   381       else if (STRING_MULTIBYTE (readcharfun))
   382         {
   383           if (multibyte)
   384             *multibyte = 1;
   385           c = (fetch_string_char_advance_no_check
   386                (readcharfun,
   387                 &read_from_string_index,
   388                 &read_from_string_index_byte));
   389         }
   390       else
   391         {
   392           c = SREF (readcharfun, read_from_string_index_byte);
   393           read_from_string_index++;
   394           read_from_string_index_byte++;
   395         }
   396       return c;
   397     }
   398 
   399   if (CONSP (readcharfun) && STRINGP (XCAR (readcharfun)))
   400     {
   401       /* This is the case that read_vector is reading from a unibyte
   402          string that contains a byte sequence previously skipped
   403          because of #@NUMBER.  The car part of readcharfun is that
   404          string, and the cdr part is a value of readcharfun given to
   405          read_vector.  */
   406       readbyte = readbyte_from_string;
   407       eassert (infile);
   408       if (EQ (XCDR (readcharfun), Qget_emacs_mule_file_char))
   409         emacs_mule_encoding = 1;
   410       goto read_multibyte;
   411     }
   412 
   413   if (EQ (readcharfun, Qget_emacs_mule_file_char))
   414     {
   415       readbyte = readbyte_from_file;
   416       eassert (infile);
   417       emacs_mule_encoding = 1;
   418       goto read_multibyte;
   419     }
   420 
   421   tem = call0 (readcharfun);
   422 
   423   if (NILP (tem))
   424     return -1;
   425   return XFIXNUM (tem);
   426 
   427  read_multibyte:
   428   if (unread_char >= 0)
   429     {
   430       c = unread_char;
   431       unread_char = -1;
   432       return c;
   433     }
   434   c = (*readbyte) (-1, readcharfun);
   435   if (c < 0)
   436     return c;
   437   if (multibyte)
   438     *multibyte = 1;
   439   if (ASCII_CHAR_P (c))
   440     return c;
   441   if (emacs_mule_encoding)
   442     return read_emacs_mule_char (c, readbyte, readcharfun);
   443   i = 0;
   444   buf[i++] = c;
   445   len = BYTES_BY_CHAR_HEAD (c);
   446   while (i < len)
   447     {
   448       buf[i++] = c = (*readbyte) (-1, readcharfun);
   449       if (c < 0 || ! TRAILING_CODE_P (c))
   450         {
   451           for (i -= c < 0; 0 < --i; )
   452             (*readbyte) (buf[i], readcharfun);
   453           return BYTE8_TO_CHAR (buf[0]);
   454         }
   455     }
   456   return STRING_CHAR (buf);
   457 }

   463 static void
   464 skip_dyn_bytes (Lisp_Object readcharfun, ptrdiff_t n)
   465 {
   466   if (FROM_FILE_P (readcharfun))
   467     {
   468       block_input ();           /* FIXME: Not sure if it's needed.  */
   469       file_seek (infile->stream, n - infile->lookahead, SEEK_CUR);
   470       unblock_input ();
   471       infile->lookahead = 0;
   472     }
   473   else
   474     { /* We're not reading directly from a file.  In that case, it's difficult
   475          to reliably count bytes, since these are usually meant for the file's
   476          encoding, whereas we're now typically in the internal encoding.
   477          But luckily, skip_dyn_bytes is used to skip over a single
   478          dynamic-docstring (or dynamic byte-code) which is always quoted such
   479          that \037 is the final char.  */
   480       int c;
   481       do {
   482         c = READCHAR;
   483       } while (c >= 0 && c != '\037');
   484     }
   485 }

   504 static void
   505 unreadchar (Lisp_Object readcharfun, int c)
   506 {
   507   readchar_offset--;
   508   if (c == -1)
   509     /* Don't back up the pointer if we're unreading the end-of-input mark,
   510        since readchar didn't advance it when we read it.  */
   511     ;
   512   else if (BUFFERP (readcharfun))
   513     {
   514       struct buffer *b = XBUFFER (readcharfun);
   515       ptrdiff_t charpos = BUF_PT (b);
   516       ptrdiff_t bytepos = BUF_PT_BYTE (b);
   517 
   518       if (! NILP (BVAR (b, enable_multibyte_characters)))
   519         bytepos -= buf_prev_char_len (b, bytepos);
   520       else
   521         bytepos--;
   522 
   523       SET_BUF_PT_BOTH (b, charpos - 1, bytepos);
   524     }
   525   else if (MARKERP (readcharfun))
   526     {
   527       struct buffer *b = XMARKER (readcharfun)->buffer;
   528       ptrdiff_t bytepos = XMARKER (readcharfun)->bytepos;
   529 
   530       XMARKER (readcharfun)->charpos--;
   531       if (! NILP (BVAR (b, enable_multibyte_characters)))
   532         bytepos -= buf_prev_char_len (b, bytepos);
   533       else
   534         bytepos--;
   535 
   536       XMARKER (readcharfun)->bytepos = bytepos;
   537     }
   538   else if (STRINGP (readcharfun))
   539     {
   540       read_from_string_index--;
   541       read_from_string_index_byte
   542         = string_char_to_byte (readcharfun, read_from_string_index);
   543     }
   544   else if (CONSP (readcharfun) && STRINGP (XCAR (readcharfun)))
   545     {
   546       unread_char = c;
   547     }
   548   else if (EQ (readcharfun, Qlambda))
   549     {
   550       unread_char = c;
   551     }
   552   else if (FROM_FILE_P (readcharfun))
   553     {
   554       unread_char = c;
   555     }
   556   else
   557     call1 (readcharfun, make_fixnum (c));
   558 }

   692 /* Read one non-ASCII character from INFILE.  The character is
   693    encoded in `emacs-mule' and the first byte is already read in
   694    C.  */
   695 
   696 static int
   697 read_emacs_mule_char (int c, int (*readbyte) (int, Lisp_Object), Lisp_Object readcharfun)
   698 {
   699   /* Emacs-mule coding uses at most 4-byte for one character.  */
   700   unsigned char buf[4];
   701   int len = emacs_mule_bytes[c];
   702   struct charset *charset;
   703   int i;
   704   unsigned code;
   705 
   706   if (len == 1)
   707     /* C is not a valid leading-code of `emacs-mule'.  */
   708     return BYTE8_TO_CHAR (c);
   709 
   710   i = 0;
   711   buf[i++] = c;
   712   while (i < len)
   713     {
   714       buf[i++] = c = (*readbyte) (-1, readcharfun);
   715       if (c < 0xA0)
   716         {
   717           for (i -= c < 0; 0 < --i; )
   718             (*readbyte) (buf[i], readcharfun);
   719           return BYTE8_TO_CHAR (buf[0]);
   720         }
   721     }
   722 
   723   if (len == 2)
   724     {
   725       charset = CHARSET_FROM_ID (emacs_mule_charset[buf[0]]);
   726       code = buf[1] & 0x7F;
   727     }
   728   else if (len == 3)
   729     {
   730       if (buf[0] == EMACS_MULE_LEADING_CODE_PRIVATE_11
   731           || buf[0] == EMACS_MULE_LEADING_CODE_PRIVATE_12)
   732         {
   733           charset = CHARSET_FROM_ID (emacs_mule_charset[buf[1]]);
   734           code = buf[2] & 0x7F;
   735         }
   736       else
   737         {
   738           charset = CHARSET_FROM_ID (emacs_mule_charset[buf[0]]);
   739           code = ((buf[1] << 8) | buf[2]) & 0x7F7F;
   740         }
   741     }
   742   else
   743     {
   744       charset = CHARSET_FROM_ID (emacs_mule_charset[buf[1]]);
   745       code = ((buf[2] << 8) | buf[3]) & 0x7F7F;
   746     }
   747   c = DECODE_CHAR (charset, code);
   748   if (c < 0)
   749     invalid_syntax ("invalid multibyte form", readcharfun);
   750   return c;
   751 }

   778 /* Read input events until we get one that's acceptable for our purposes.
   779 
   780    If NO_SWITCH_FRAME, switch-frame events are stashed
   781    until we get a character we like, and then stuffed into
   782    unread_switch_frame.
   783 
   784    If ASCII_REQUIRED, check function key events to see
   785    if the unmodified version of the symbol has a Qascii_character
   786    property, and use that character, if present.
   787 
   788    If ERROR_NONASCII, signal an error if the input we
   789    get isn't an ASCII character with modifiers.  If it's false but
   790    ASCII_REQUIRED is true, just re-read until we get an ASCII
   791    character.
   792 
   793    If INPUT_METHOD, invoke the current input method
   794    if the character warrants that.
   795 
   796    If SECONDS is a number, wait that many seconds for input, and
   797    return Qnil if no input arrives within that time.
   798 
   799    If text conversion is enabled and ASCII_REQUIRED && ERROR_NONASCII,
   800    temporarily disable any input method which wants to perform
   801    edits, unless `disable-inhibit-text-conversion'.  */

   803 static Lisp_Object
   804 read_filtered_event (bool no_switch_frame, bool ascii_required,
   805                      bool error_nonascii, bool input_method, Lisp_Object seconds)
   806 {
   807   Lisp_Object val, delayed_switch_frame;
   808   struct timespec end_time;
   809 #ifdef HAVE_TEXT_CONVERSION
   810   specpdl_ref count;
   811 #endif
   812 
   813 #ifdef HAVE_WINDOW_SYSTEM
   814   if (display_hourglass_p)
   815     cancel_hourglass ();
   816 #endif
   817 
   818 #ifdef HAVE_TEXT_CONVERSION
   819   count = SPECPDL_INDEX ();
   820 
   821   /* Don't use text conversion when trying to just read a
   822      character.  */
   823 
   824   if (ascii_required && error_nonascii
   825       && !disable_inhibit_text_conversion)
   826     {
   827       disable_text_conversion ();
   828       record_unwind_protect_void (resume_text_conversion);
   829     }
   830 #endif
   831 
   832   delayed_switch_frame = Qnil;
   833 
   834   /* Compute timeout.  */
   835   if (NUMBERP (seconds))
   836     {
   837       double duration = XFLOATINT (seconds);
   838       struct timespec wait_time = dtotimespec (duration);
   839       end_time = timespec_add (current_timespec (), wait_time);
   840     }
   841 
   842   /* Read until we get an acceptable event.  */
   843  retry:
   844   do
   845     val = read_char (0, Qnil, (input_method ? Qnil : Qt), 0,
   846                      NUMBERP (seconds) ? &end_time : NULL);
   847   while (FIXNUMP (val) && XFIXNUM (val) == -2); /* wrong_kboard_jmpbuf */
   848 
   849   if (BUFFERP (val))
   850     goto retry;

   852   /* `switch-frame' events are put off until after the next ASCII
   853      character.  This is better than signaling an error just because
   854      the last characters were typed to a separate minibuffer frame,
   855      for example.  Eventually, some code which can deal with
   856      switch-frame events will read it and process it.  */
   857   if (no_switch_frame
   858       && EVENT_HAS_PARAMETERS (val)
   859       && EQ (EVENT_HEAD_KIND (EVENT_HEAD (val)), Qswitch_frame))
   860     {
   861       delayed_switch_frame = val;
   862       goto retry;
   863     }
   864 
   865   if (ascii_required && !(NUMBERP (seconds) && NILP (val)))
   866     {
   867       /* Convert certain symbols to their ASCII equivalents.  */
   868       if (SYMBOLP (val))
   869         {
   870           Lisp_Object tem, tem1;
   871           tem = Fget (val, Qevent_symbol_element_mask);
   872           if (!NILP (tem))
   873             {
   874               tem1 = Fget (Fcar (tem), Qascii_character);
   875               /* Merge this symbol's modifier bits
   876                  with the ASCII equivalent of its basic code.  */
   877               if (!NILP (tem1))
   878                 XSETFASTINT (val, XFIXNUM (tem1) | XFIXNUM (Fcar (Fcdr (tem))));
   879             }
   880         }
   881 
   882       /* If we don't have a character now, deal with it appropriately.  */
   883       if (!FIXNUMP (val))
   884         {
   885           if (error_nonascii)
   886             {
   887               Vunread_command_events = list1 (val);
   888               error ("Non-character input-event");
   889             }
   890           else
   891             goto retry;
   892         }
   893     }
   894 
   895   if (! NILP (delayed_switch_frame))
   896     unread_switch_frame = delayed_switch_frame;
   897 
   898 #if 0
   899 
   900 #ifdef HAVE_WINDOW_SYSTEM
   901   if (display_hourglass_p)
   902     start_hourglass ();
   903 #endif
   904 
   905 #endif
   906 
   907 #ifdef HAVE_TEXT_CONVERSION
   908   return unbind_to (count, val);
   909 #else
   910   return val;
   911 #endif
   912 }

  1220   return version;

  1336 DEFUN ("load", Fload, Sload, 1, 5, 0,
  1337        doc: /* Execute a file of Lisp code named FILE.
  1338 First try FILE with `.elc' appended, then try with `.el', then try
  1339 with a system-dependent suffix of dynamic modules (see `load-suffixes'),
  1340 then try FILE unmodified (the exact suffixes in the exact order are
  1341 determined by `load-suffixes').  Environment variable references in
  1342 FILE are replaced with their values by calling `substitute-in-file-name'.
  1343 This function searches the directories in `load-path'.
  1344 
  1345 If optional second arg NOERROR is non-nil,
  1346 report no error if FILE doesn't exist.
  1347 Print messages at start and end of loading unless
  1348 optional third arg NOMESSAGE is non-nil (but `force-load-messages'
  1349 overrides that).
  1350 If optional fourth arg NOSUFFIX is non-nil, don't try adding
  1351 suffixes to the specified name FILE.
  1352 If optional fifth arg MUST-SUFFIX is non-nil, insist on
  1353 the suffix `.elc' or `.el' or the module suffix; don't accept just
  1354 FILE unless it ends in one of those suffixes or includes a directory name.
  1355 
  1356 If NOSUFFIX is nil, then if a file could not be found, try looking for
  1357 a different representation of the file by adding non-empty suffixes to
  1358 its name, before trying another file.  Emacs uses this feature to find
  1359 compressed versions of files when Auto Compression mode is enabled.
  1360 If NOSUFFIX is non-nil, disable this feature.
  1361 
  1362 The suffixes that this function tries out, when NOSUFFIX is nil, are
  1363 given by the return value of `get-load-suffixes' and the values listed
  1364 in `load-file-rep-suffixes'.  If MUST-SUFFIX is non-nil, only the
  1365 return value of `get-load-suffixes' is used, i.e. the file name is
  1366 required to have a non-empty suffix.
  1367 
  1368 When searching suffixes, this function normally stops at the first
  1369 one that exists.  If the option `load-prefer-newer' is non-nil,
  1370 however, it tries all suffixes, and uses whichever file is the newest.
  1371 
  1372 Loading a file records its definitions, and its `provide' and
  1373 `require' calls, in an element of `load-history' whose
  1374 car is the file name loaded.  See `load-history'.
  1375 
  1376 While the file is in the process of being loaded, the variable
  1377 `load-in-progress' is non-nil and the variable `load-file-name'
  1378 is bound to the file's name.
  1379 
  1380 Return t if the file exists and loads successfully.  */)
  1381   (Lisp_Object file, Lisp_Object noerror, Lisp_Object nomessage,
  1382    Lisp_Object nosuffix, Lisp_Object must_suffix)
  1383 {
  1384   file_stream stream UNINIT;
  1385   lread_fd fd;
  1386 #ifdef USE_ANDROID_ASSETS
  1387   int rc;
  1388   void *asset;
  1389 #endif
  1390   specpdl_ref fd_index UNINIT;
  1391   specpdl_ref count = SPECPDL_INDEX ();
  1392   Lisp_Object found, efound, hist_file_name;
  1393   /* True means we printed the ".el is newer" message.  */
  1394   bool newer = 0;
  1395   /* True means we are loading a compiled file.  */
  1396   bool compiled = 0;
  1397   Lisp_Object handler;
  1398   const char *fmode = "r" FOPEN_TEXT;
  1399   int version;
  1400 
  1401   CHECK_STRING (file);
  1402 
  1403   /* If file name is magic, call the handler.  */
  1404   handler = Ffind_file_name_handler (file, Qload);
  1405   if (!NILP (handler))
  1406     return
  1407       call6 (handler, Qload, file, noerror, nomessage, nosuffix, must_suffix);
  1408 
  1409   /* The presence of this call is the result of a historical accident:
  1410      it used to be in every file-operation and when it got removed
  1411      everywhere, it accidentally stayed here.  Since then, enough people
  1412      supposedly have things like (load "$PROJECT/foo.el") in their .emacs
  1413      that it seemed risky to remove.  */
  1414   if (! NILP (noerror))
  1415     {
  1416       file = internal_condition_case_1 (Fsubstitute_in_file_name, file,
  1417                                         Qt, load_error_handler);
  1418       if (NILP (file))
  1419         return Qnil;
  1420     }
  1421   else
  1422     file = Fsubstitute_in_file_name (file);
  1423 
  1424   bool no_native = suffix_p (file, ".elc");
  1425 
  1426   /* Avoid weird lossage with null string as arg,
  1427      since it would try to load a directory as a Lisp file.  */
  1428   if (SCHARS (file) == 0)
  1429     {
  1430 #if !defined USE_ANDROID_ASSETS
  1431       fd = -1;
  1432 #else
  1433       fd.asset = NULL;
  1434       fd.fd = -1;
  1435 #endif
  1436       errno = ENOENT;
  1437     }
  1438   else
  1439     {
  1440       Lisp_Object suffixes;
  1441       found = Qnil;
  1442 
  1443       if (! NILP (must_suffix))
  1444         {
  1445           /* Don't insist on adding a suffix if FILE already ends with one.  */
  1446           if (suffix_p (file, ".el")
  1447               || suffix_p (file, ".elc")
  1448 #ifdef HAVE_MODULES
  1449               || suffix_p (file, MODULES_SUFFIX)
  1450 #ifdef MODULES_SECONDARY_SUFFIX
  1451               || suffix_p (file, MODULES_SECONDARY_SUFFIX)
  1452 #endif
  1453 #endif
  1454 #ifdef HAVE_NATIVE_COMP
  1455               || suffix_p (file, NATIVE_ELISP_SUFFIX)
  1456 #endif
  1457               )
  1458             must_suffix = Qnil;
  1459           /* Don't insist on adding a suffix
  1460              if the argument includes a directory name.  */
  1461           else if (! NILP (Ffile_name_directory (file)))
  1462             must_suffix = Qnil;
  1463         }
  1464 
  1465       if (!NILP (nosuffix))
  1466         suffixes = Qnil;
  1467       else
  1468         {
  1469           suffixes = Fget_load_suffixes ();
  1470           if (NILP (must_suffix))
  1471             suffixes = CALLN (Fappend, suffixes, Vload_file_rep_suffixes);
  1472         }
  1473 
  1474 #if !defined USE_ANDROID_ASSETS
  1475       fd = openp (Vload_path, file, suffixes, &found, Qnil,
  1476                   load_prefer_newer, no_native, NULL);
  1477 #else
  1478       asset = NULL;
  1479       rc = openp (Vload_path, file, suffixes, &found, Qnil,
  1480                   load_prefer_newer, no_native, &asset);
  1481       fd.fd = rc;
  1482       fd.asset = asset;
  1483 
  1484       /* fd.asset will be non-NULL if this is actually an asset
  1485          file.  */
  1486 #endif
  1487     }
  1488 
  1489   if (lread_fd_cmp (-1))
  1490     {
  1491       if (NILP (noerror))
  1492         report_file_error ("Cannot open load file", file);
  1493       return Qnil;
  1494     }
  1495 
  1496   /* Tell startup.el whether or not we found the user's init file.  */
  1497   if (EQ (Qt, Vuser_init_file))
  1498     Vuser_init_file = found;
  1499 
  1500   /* If FD is -2, that means openp found a magic file.  */
  1501   if (lread_fd_cmp (-2))
  1502     {
  1503       if (NILP (Fequal (found, file)))
  1504         /* If FOUND is a different file name from FILE,
  1505            find its handler even if we have already inhibited
  1506            the `load' operation on FILE.  */
  1507         handler = Ffind_file_name_handler (found, Qt);
  1508       else
  1509         handler = Ffind_file_name_handler (found, Qload);
  1510       if (! NILP (handler))
  1511         return call5 (handler, Qload, found, noerror, nomessage, Qt);
  1512 #ifdef DOS_NT
  1513       /* Tramp has to deal with semi-broken packages that prepend
  1514          drive letters to remote files.  For that reason, Tramp
  1515          catches file operations that test for file existence, which
  1516          makes openp think X:/foo.elc files are remote.  However,
  1517          Tramp does not catch `load' operations for such files, so we
  1518          end up with a nil as the `load' handler above.  If we would
  1519          continue with fd = -2, we will behave wrongly, and in
  1520          particular try reading a .elc file in the "rt" mode instead
  1521          of "rb".  See bug #9311 for the results.  To work around
  1522          this, we try to open the file locally, and go with that if it
  1523          succeeds.  */
  1524       fd = emacs_open (SSDATA (ENCODE_FILE (found)), O_RDONLY, 0);
  1525       if (fd == -1)
  1526         fd = -2;
  1527 #endif
  1528     }
  1529 
  1530 #if !defined USE_ANDROID_ASSETS
  1531   if (0 <= fd)
  1532     {
  1533       fd_index = SPECPDL_INDEX ();
  1534       record_unwind_protect_int (close_file_unwind, fd);
  1535     }
  1536 #else
  1537   if (fd.asset || fd.fd >= 0)
  1538     {
  1539       /* Use a different kind of unwind_protect here.  */
  1540       fd_index = SPECPDL_INDEX ();
  1541       record_unwind_protect_ptr (close_file_unwind_android_fd,
  1542                                  &fd);
  1543     }
  1544 #endif
  1545 
  1546 #ifdef HAVE_MODULES
  1547   bool is_module =
  1548     suffix_p (found, MODULES_SUFFIX)
  1549 #ifdef MODULES_SECONDARY_SUFFIX
  1550     || suffix_p (found, MODULES_SECONDARY_SUFFIX)
  1551 #endif
  1552     ;
  1553 #else
  1554   bool is_module = false;
  1555 #endif
  1556 
  1557 #ifdef HAVE_NATIVE_COMP
  1558   bool is_native_elisp = suffix_p (found, NATIVE_ELISP_SUFFIX);
  1559 #else
  1560   bool is_native_elisp = false;
  1561 #endif
  1562 
  1563   /* Check if we're stuck in a recursive load cycle.
  1564 
  1565      2000-09-21: It's not possible to just check for the file loaded
  1566      being a member of Vloads_in_progress.  This fails because of the
  1567      way the byte compiler currently works; `provide's are not
  1568      evaluated, see font-lock.el/jit-lock.el as an example.  This
  1569      leads to a certain amount of ``normal'' recursion.
  1570 
  1571      Also, just loading a file recursively is not always an error in
  1572      the general case; the second load may do something different.  */
  1573   {
  1574     int load_count = 0;
  1575     Lisp_Object tem = Vloads_in_progress;
  1576     FOR_EACH_TAIL_SAFE (tem)
  1577       if (!NILP (Fequal (found, XCAR (tem))) && (++load_count > 3))
  1578         signal_error ("Recursive load", Fcons (found, Vloads_in_progress));
  1579     record_unwind_protect (record_load_unwind, Vloads_in_progress);
  1580     Vloads_in_progress = Fcons (found, Vloads_in_progress);
  1581   }
  1582 
  1583   /* All loads are by default dynamic, unless the file itself specifies
  1584      otherwise using a file-variable in the first line.  This is bound here
  1585      so that it takes effect whether or not we use
  1586      Vload_source_file_function.  */
  1587   specbind (Qlexical_binding, Qnil);
  1588 
  1589   Lisp_Object found_eff =
  1590     is_native_elisp
  1591     ? compute_found_effective (found)
  1592     : found;
  1593 
  1594   hist_file_name = (! NILP (Vpurify_flag)
  1595                     ? concat2 (Ffile_name_directory (file),
  1596                                Ffile_name_nondirectory (found_eff))
  1597                     : found_eff);
  1598 
  1599   version = -1;
  1600 
  1601   /* Check for the presence of unescaped character literals and warn
  1602      about them. */
  1603   specbind (Qlread_unescaped_character_literals, Qnil);
  1604   record_unwind_protect (load_warn_unescaped_character_literals, file);
  1605 
  1606   bool is_elc = suffix_p (found, ".elc");
  1607   if (is_elc
  1608       /* version = 1 means the file is empty, in which case we can
  1609          treat it as not byte-compiled.  */
  1610       || (lread_fd_p
  1611           && (version = safe_to_load_version (file, fd)) > 1))
  1612     /* Load .elc files directly, but not when they are
  1613        remote and have no handler!  */
  1614     {
  1615       if (!lread_fd_cmp (-2))
  1616         {
  1617           struct stat s1, s2;
  1618           int result;
  1619 
  1620           struct timespec epoch_timespec = {(time_t)0, 0}; /* 1970-01-01T00:00 UTC */
  1621           if (version < 0 && !(version = safe_to_load_version (file, fd)))
  1622             error ("File `%s' was not compiled in Emacs", SDATA (found));
  1623 
  1624           compiled = 1;
  1625 
  1626           efound = ENCODE_FILE (found);
  1627           fmode = "r" FOPEN_BINARY;
  1628 
  1629           /* openp already checked for newness, no point doing it again.
  1630              FIXME would be nice to get a message when openp
  1631              ignores suffix order due to load_prefer_newer.  */
  1632           if (!load_prefer_newer && is_elc)
  1633             {
  1634               result = emacs_fstatat (AT_FDCWD, SSDATA (efound), &s1, 0);
  1635               if (result == 0)
  1636                 {
  1637                   SSET (efound, SBYTES (efound) - 1, 0);
  1638                   result = emacs_fstatat (AT_FDCWD, SSDATA (efound), &s2, 0);
  1639                   SSET (efound, SBYTES (efound) - 1, 'c');
  1640                 }
  1641 
  1642               if (result == 0
  1643                   && timespec_cmp (get_stat_mtime (&s1), get_stat_mtime (&s2)) < 0)
  1644                 {
  1645                   /* Make the progress messages mention that source is newer.  */
  1646                   newer = 1;
  1647 
  1648                   /* If we won't print another message, mention this anyway.  */
  1649                   if (!NILP (nomessage) && !force_load_messages
  1650                       /* We don't want this message during
  1651                          bootstrapping for the "compile-first" .elc
  1652                          files, which have had their timestamps set to
  1653                          the epoch.  See bug #58224.  */
  1654                       && timespec_cmp (get_stat_mtime (&s1), epoch_timespec))
  1655                     {
  1656                       Lisp_Object msg_file;
  1657                       msg_file = Fsubstring (found, make_fixnum (0), make_fixnum (-1));
  1658                       message_with_string ("Source file `%s' newer than byte-compiled file; using older file",
  1659                                            msg_file, 1);
  1660                     }
  1661                 }
  1662             } /* !load_prefer_newer */
  1663         }
  1664     }
  1665   else if (!is_module && !is_native_elisp)
  1666     {
  1667       /* We are loading a source file (*.el).  */
  1668       if (!NILP (Vload_source_file_function))
  1669         {
  1670           Lisp_Object val;
  1671 
  1672           if (lread_fd_p)
  1673             {
  1674               lread_close (fd);
  1675               clear_unwind_protect (fd_index);
  1676             }
  1677           val = call4 (Vload_source_file_function, found, hist_file_name,
  1678                        NILP (noerror) ? Qnil : Qt,
  1679                        (NILP (nomessage) || force_load_messages) ? Qnil : Qt);
  1680           return unbind_to (count, val);
  1681         }
  1682     }
  1683 
  1684   if (!lread_fd_p)
  1685     {
  1686       /* We somehow got here with fd == -2, meaning the file is deemed
  1687          to be remote.  Don't even try to reopen the file locally;
  1688          just force a failure.  */
  1689       stream = file_stream_invalid;
  1690       errno = EINVAL;
  1691     }
  1692   else if (!is_module && !is_native_elisp)
  1693     {
  1694 #ifdef WINDOWSNT
  1695       emacs_close (fd);
  1696       clear_unwind_protect (fd_index);
  1697       efound = ENCODE_FILE (found);
  1698       stream = emacs_fopen (SSDATA (efound), fmode);
  1699 #else
  1700 #if !defined USE_ANDROID_ASSETS
  1701       stream = fdopen (fd, fmode);
  1702 #else
  1703       /* Android systems use special file descriptors which can point
  1704          into compressed data and double as file streams.  FMODE is
  1705          unused.  */
  1706       ((void) fmode);
  1707       stream = fd;
  1708 #endif
  1709 #endif
  1710     }
  1711 
  1712   /* Declare here rather than inside the else-part because the storage
  1713      might be accessed by the unbind_to call below.  */
  1714   struct infile input;
  1715 
  1716   if (is_module || is_native_elisp)
  1717     {
  1718       /* `module-load' uses the file name, so we can close the stream
  1719          now.  */
  1720       if (lread_fd_p)
  1721         {
  1722           lread_close (fd);
  1723           clear_unwind_protect (fd_index);
  1724         }
  1725     }
  1726   else
  1727     {
  1728       if (!file_stream_valid_p (stream))
  1729         report_file_error ("Opening stdio stream", file);
  1730       set_unwind_protect_ptr (fd_index, close_infile_unwind, infile);
  1731       input.stream = stream;
  1732       input.lookahead = 0;
  1733       infile = &input;
  1734       unread_char = -1;
  1735     }
  1736 
  1737   if (! NILP (Vpurify_flag))
  1738     Vpreloaded_file_list = Fcons (Fpurecopy (file), Vpreloaded_file_list);
  1739 
  1740   if (NILP (nomessage) || force_load_messages)
  1741     {
  1742       if (is_module)
  1743         message_with_string ("Loading %s (module)...", file, 1);
  1744       else if (is_native_elisp)
  1745         message_with_string ("Loading %s (native compiled elisp)...", file, 1);
  1746       else if (!compiled)
  1747         message_with_string ("Loading %s (source)...", file, 1);
  1748       else if (newer)
  1749         message_with_string ("Loading %s (compiled; note, source file is newer)...",
  1750                  file, 1);
  1751       else /* The typical case; compiled file newer than source file.  */
  1752         message_with_string ("Loading %s...", file, 1);
  1753     }
  1754 
  1755   specbind (Qload_file_name, hist_file_name);
  1756   specbind (Qload_true_file_name, found);
  1757   specbind (Qinhibit_file_name_operation, Qnil);
  1758   specbind (Qload_in_progress, Qt);
  1759 
  1760   if (is_module)
  1761     {
  1762 #ifdef HAVE_MODULES
  1763       loadhist_initialize (found);
  1764       Fmodule_load (found);
  1765       build_load_history (found, true);
  1766 #else
  1767       /* This cannot happen.  */
  1768       emacs_abort ();
  1769 #endif
  1770     }
  1771   else if (is_native_elisp)
  1772     {
  1773 #ifdef HAVE_NATIVE_COMP
  1774       loadhist_initialize (hist_file_name);
  1775       Fnative_elisp_load (found, Qnil);
  1776       build_load_history (hist_file_name, true);
  1777 #else

  1802   /* Run any eval-after-load forms for this file.  */

  1806   for (int i = 0; i < ARRAYELTS (saved_strings); i++)

  1809       saved_strings[i].string = NULL;
  1810       saved_strings[i].size = 0;

  1825         message_with_string ("Loading %s...done", file, 1);
  1826     }
  1827 
  1828   return Qt;
  1829 }

  1982 /* Search for a file whose name is STR, looking in directories
  1983    in the Lisp list PATH, and trying suffixes from SUFFIX.
  1984    On success, return a file descriptor (or 1 or -2 as described below).
  1985    On failure, return -1 and set errno.
  1986 
  1987    SUFFIXES is a list of strings containing possible suffixes.
  1988    The empty suffix is automatically added if the list is empty.
  1989 
  1990    PREDICATE t means the files are binary.
  1991    PREDICATE non-nil and non-t means don't open the files,
  1992    just look for one that satisfies the predicate.  In this case,
  1993    return -2 on success.  The predicate can be a lisp function or
  1994    an integer to pass to `access' (in which case file-name-handlers
  1995    are ignored).
  1996 
  1997    If STOREPTR is nonzero, it points to a slot where the name of
  1998    the file actually found should be stored as a Lisp string.
  1999    nil is stored there on failure.
  2000 
  2001    If the file we find is remote, return -2
  2002    but store the found remote file name in *STOREPTR.
  2003 
  2004    If NEWER is true, try all SUFFIXes and return the result for the
  2005    newest file that exists.  Does not apply to remote files,
  2006    platform-specific files, or if a non-nil and non-t PREDICATE is
  2007    specified.
  2008 
  2009    If NO_NATIVE is true do not try to load native code.
  2010 
  2011    If PLATFORM is non-NULL and the file being loaded lies in a special
  2012    directory, such as the Android `/assets' directory, return a handle
  2013    to that directory in *PLATFORM instead of a file descriptor; in
  2014    that case, value is -3.  */

  2016 int
  2017 openp (Lisp_Object path, Lisp_Object str, Lisp_Object suffixes,
  2018        Lisp_Object *storeptr, Lisp_Object predicate, bool newer,
  2019        bool no_native, void **platform)
  2020 {
  2021   ptrdiff_t fn_size = 100;
  2022   char buf[100];
  2023   char *fn = buf;
  2024   bool absolute;
  2025   ptrdiff_t want_length;
  2026   Lisp_Object filename;
  2027   Lisp_Object string, tail, encoded_fn, save_string;
  2028   ptrdiff_t max_suffix_len = 0;
  2029   int last_errno = ENOENT;
  2030   int save_fd = -1;
  2031 #ifdef USE_ANDROID_ASSETS
  2032   struct android_fd_or_asset platform_fd;
  2033 #endif
  2034   USE_SAFE_ALLOCA;
  2035 
  2036   /* The last-modified time of the newest matching file found.
  2037      Initialize it to something less than all valid timestamps.  */
  2038   struct timespec save_mtime = make_timespec (TYPE_MINIMUM (time_t), -1);
  2039 
  2040   CHECK_STRING (str);
  2041 
  2042   tail = suffixes;
  2043   FOR_EACH_TAIL_SAFE (tail)
  2044     {
  2045       CHECK_STRING_CAR (tail);
  2046       max_suffix_len = max (max_suffix_len,
  2047                             SBYTES (XCAR (tail)));
  2048     }
  2049 
  2050   string = filename = encoded_fn = save_string = Qnil;
  2051 
  2052   if (storeptr)
  2053     *storeptr = Qnil;
  2054 
  2055   absolute = complete_filename_p (str);
  2056 
  2057   AUTO_LIST1 (just_use_str, Qnil);
  2058   if (NILP (path))
  2059     path = just_use_str;
  2060 
  2061   /* Go through all entries in the path and see whether we find the
  2062      executable. */
  2063   FOR_EACH_TAIL_SAFE (path)
  2064    {
  2065     ptrdiff_t baselen, prefixlen;
  2066 
  2067     if (EQ (path, just_use_str))
  2068       filename = str;
  2069     else
  2070       filename = Fexpand_file_name (str, XCAR (path));
  2071     if (!complete_filename_p (filename))
  2072       /* If there are non-absolute elts in PATH (eg ".").  */
  2073       /* Of course, this could conceivably lose if luser sets
  2074          default-directory to be something non-absolute...  */
  2075       {
  2076         filename = Fexpand_file_name (filename, BVAR (current_buffer, directory));
  2077         if (!complete_filename_p (filename))
  2078           /* Give up on this path element!  */
  2079           continue;
  2080       }
  2081 
  2082     /* Calculate maximum length of any filename made from
  2083        this path element/specified file name and any possible suffix.  */
  2084     want_length = max_suffix_len + SBYTES (filename);
  2085     if (fn_size <= want_length)
  2086       {
  2087         fn_size = 100 + want_length;
  2088         fn = SAFE_ALLOCA (fn_size);
  2089       }
  2090 
  2091     /* Copy FILENAME's data to FN but remove starting /: if any.  */
  2092     prefixlen = ((SCHARS (filename) > 2
  2093                   && SREF (filename, 0) == '/'
  2094                   && SREF (filename, 1) == ':')
  2095                  ? 2 : 0);
  2096     baselen = SBYTES (filename) - prefixlen;
  2097     memcpy (fn, SDATA (filename) + prefixlen, baselen);
  2098 
  2099     /* Loop over suffixes.  */
  2100     AUTO_LIST1 (empty_string_only, empty_unibyte_string);
  2101     tail = NILP (suffixes) ? empty_string_only : suffixes;
  2102     FOR_EACH_TAIL_SAFE (tail)
  2103       {
  2104         Lisp_Object suffix = XCAR (tail);
  2105         ptrdiff_t fnlen, lsuffix = SBYTES (suffix);
  2106         Lisp_Object handler;
  2107 
  2108         /* Make complete filename by appending SUFFIX.  */
  2109         memcpy (fn + baselen, SDATA (suffix), lsuffix + 1);
  2110         fnlen = baselen + lsuffix;
  2111 
  2112         /* Check that the file exists and is not a directory.  */
  2113         /* We used to only check for handlers on non-absolute file names:
  2114            if (absolute)
  2115            handler = Qnil;
  2116            else
  2117            handler = Ffind_file_name_handler (filename, Qfile_exists_p);
  2118            It's not clear why that was the case and it breaks things like
  2119            (load "/bar.el") where the file is actually "/bar.el.gz".  */
  2120         /* make_string has its own ideas on when to return a unibyte
  2121            string and when a multibyte string, but we know better.
  2122            We must have a unibyte string when dumping, since
  2123            file-name encoding is shaky at best at that time, and in
  2124            particular default-file-name-coding-system is reset
  2125            several times during loadup.  We therefore don't want to
  2126            encode the file before passing it to file I/O library
  2127            functions.  */
  2128         if (!STRING_MULTIBYTE (filename) && !STRING_MULTIBYTE (suffix))
  2129           string = make_unibyte_string (fn, fnlen);
  2130         else
  2131           string = make_string (fn, fnlen);
  2132         handler = Ffind_file_name_handler (string, Qfile_exists_p);
  2133         if ((!NILP (handler) || (!NILP (predicate) && !EQ (predicate, Qt)))
  2134             && !FIXNATP (predicate))
  2135           {
  2136             bool exists;
  2137             if (NILP (predicate) || EQ (predicate, Qt))
  2138               exists = !NILP (Ffile_readable_p (string));
  2139             else
  2140               {
  2141                 Lisp_Object tmp = call1 (predicate, string);
  2142                 if (NILP (tmp))
  2143                   exists = false;
  2144                 else if (EQ (tmp, Qdir_ok)
  2145                          || NILP (Ffile_directory_p (string)))
  2146                   exists = true;
  2147                 else
  2148                   {
  2149                     exists = false;
  2150                     last_errno = EISDIR;
  2151                   }
  2152               }
  2153 
  2154             if (exists)
  2155               {
  2156                 /* We succeeded; return this descriptor and filename.  */
  2157                 if (storeptr)
  2158                   *storeptr = string;
  2159                 SAFE_FREE ();
  2160                 return -2;
  2161               }
  2162           }
  2163         else
  2164           {
  2165             int fd;
  2166             const char *pfn;
  2167             struct stat st;
  2168 
  2169             encoded_fn = ENCODE_FILE (string);
  2170             pfn = SSDATA (encoded_fn);
  2171 
  2172             /* Check that we can access or open it.  */
  2173             if (FIXNATP (predicate))
  2174               {
  2175                 fd = -1;
  2176                 if (INT_MAX < XFIXNAT (predicate))
  2177                   last_errno = EINVAL;
  2178                 else if (sys_faccessat (AT_FDCWD, pfn, XFIXNAT (predicate),
  2179                                         AT_EACCESS)
  2180                          == 0)
  2181                   {
  2182                     if (file_directory_p (encoded_fn))
  2183                       last_errno = EISDIR;
  2184                     else if (errno == ENOENT || errno == ENOTDIR)
  2185                       fd = 1;
  2186                     else
  2187                       last_errno = errno;
  2188                   }
  2189                 else if (! (errno == ENOENT || errno == ENOTDIR))
  2190                   last_errno = errno;
  2191               }
  2192             else
  2193               {
  2194                 /*  In some systems (like Windows) finding out if a
  2195                     file exists is cheaper to do than actually opening
  2196                     it.  Only open the file when we are sure that it
  2197                     exists.  */
  2198 #ifdef WINDOWSNT
  2199                 if (sys_faccessat (AT_FDCWD, pfn, R_OK, AT_EACCESS))
  2200                   fd = -1;
  2201                 else
  2202 #endif
  2203                   {
  2204 #if !defined USE_ANDROID_ASSETS
  2205                     fd = emacs_open (pfn, O_RDONLY, 0);
  2206 #else
  2207                     if (platform)
  2208                       {
  2209                         platform_fd = android_open_asset (pfn, O_RDONLY, 0);
  2210 
  2211                         if (platform_fd.asset
  2212                             && platform_fd.asset != (void *) -1)
  2213                           {
  2214                             *storeptr = string;
  2215                             goto handle_platform_fd;
  2216                           }
  2217 
  2218                         if (platform_fd.asset == (void *) -1)
  2219                           fd = -1;
  2220                         else
  2221                           fd = platform_fd.fd;
  2222                       }
  2223                     else
  2224                       fd = emacs_open (pfn, O_RDONLY, 0);
  2225 #endif
  2226                   }
  2227 
  2228                 if (fd < 0)
  2229                   {
  2230                     if (! (errno == ENOENT || errno == ENOTDIR))
  2231                       last_errno = errno;
  2232                   }
  2233                 else
  2234                   {
  2235                     int err = (sys_fstat (fd, &st) != 0 ? errno
  2236                                : S_ISDIR (st.st_mode) ? EISDIR : 0);
  2237                     if (err)
  2238                       {
  2239                         last_errno = err;
  2240                         emacs_close (fd);
  2241                         fd = -1;
  2242                       }
  2243                   }
  2244               }
  2245 
  2246             if (fd >= 0)
  2247               {
  2248                 if (newer && !FIXNATP (predicate))
  2249                   {
  2250                     struct timespec mtime = get_stat_mtime (&st);
  2251 
  2252                     if (timespec_cmp (mtime, save_mtime) <= 0)
  2253                       emacs_close (fd);
  2254                     else
  2255                       {
  2256                         if (0 <= save_fd)
  2257                           emacs_close (save_fd);
  2258                         save_fd = fd;
  2259                         save_mtime = mtime;
  2260                         save_string = string;
  2261                       }
  2262                   }
  2263                 else
  2264                   {
  2265                     maybe_swap_for_eln (no_native, &string, &fd,
  2266                                         get_stat_mtime (&st));
  2267                     /* We succeeded; return this descriptor and filename.  */
  2268                     if (storeptr)
  2269                       *storeptr = string;
  2270                     SAFE_FREE ();
  2271                     return fd;
  2272                   }
  2273               }
  2274 
  2275             /* No more suffixes.  Return the newest.  */
  2276             if (0 <= save_fd && ! CONSP (XCDR (tail)))
  2277               {
  2278                 maybe_swap_for_eln (no_native, &save_string, &save_fd,
  2279                                     save_mtime);
  2280                 if (storeptr)
  2281                   *storeptr = save_string;
  2282                 SAFE_FREE ();
  2283                 return save_fd;
  2284               }
  2285           }
  2286       }
  2287     if (absolute)
  2288       break;
  2289    }
  2290 
  2291   SAFE_FREE ();
  2292   errno = last_errno;
  2293   return -1;
  2294 
  2295 #ifdef USE_ANDROID_ASSETS
  2296  handle_platform_fd:
  2297 
  2298   /* Here, openp found a platform specific file descriptor.  It can't
  2299      be a directory under Android, so return it in *PLATFORM and then
  2300      -3 as the file descriptor.  */
  2301   *platform = platform_fd.asset;
  2302   return -3;
  2303 #endif

  2332         {
  2333           foundit = 1;

  2335           /* If we're loading the entire file, remove old data.  */
  2336           if (entire)
  2337             {
  2338               if (NILP (prev))
  2339                 Vload_history = XCDR (tail);
  2340               else
  2341                 Fsetcdr (prev, XCDR (tail));
  2342             }
  2343           /* Otherwise, cons on new symbols that are not already
  2344              members.  */
  2345           else
  2346             {
  2347               tem2 = Vcurrent_load_list;
  2348 
  2349               FOR_EACH_TAIL (tem2)

  2351                   newelt = XCAR (tem2);

  2354                     Fsetcar (tail, Fcons (XCAR (tem),
  2355                                           Fcons (newelt, XCDR (tem))));

  2361         prev = tail;

  2366      front of load-history, the most-recently-loaded position.  Also

  2368   if (entire || !foundit)
  2369     Vload_history = Fcons (Fnreverse (Vcurrent_load_list),
  2370                            Vload_history);

  2419 static void
  2420 readevalloop (Lisp_Object readcharfun,
  2421               struct infile *infile0,
  2422               Lisp_Object sourcename,
  2423               bool printflag,
  2424               Lisp_Object unibyte, Lisp_Object readfun,
  2425               Lisp_Object start, Lisp_Object end)
  2426 {
  2427   int c;
  2428   Lisp_Object val;
  2429   specpdl_ref count = SPECPDL_INDEX ();
  2430   struct buffer *b = 0;
  2431   bool continue_reading_p;
  2432   Lisp_Object lex_bound;
  2433   /* True if reading an entire buffer.  */
  2434   bool whole_buffer = 0;
  2435   /* True on the first time around.  */
  2436   bool first_sexp = 1;
  2437   Lisp_Object macroexpand = intern ("internal-macroexpand-for-load");
  2438 
  2439   if (!NILP (sourcename))
  2440     CHECK_STRING (sourcename);
  2441 
  2442   if (NILP (Ffboundp (macroexpand))
  2443       || (STRINGP (sourcename) && suffix_p (sourcename, ".elc")))
  2444     /* Don't macroexpand before the corresponding function is defined
  2445        and don't bother macroexpanding in .elc files, since it should have
  2446        been done already.  */
  2447     macroexpand = Qnil;
  2448 
  2449   if (MARKERP (readcharfun))
  2450     {
  2451       if (NILP (start))
  2452         start = readcharfun;
  2453     }
  2454 
  2455   if (BUFFERP (readcharfun))
  2456     b = XBUFFER (readcharfun);
  2457   else if (MARKERP (readcharfun))
  2458     b = XMARKER (readcharfun)->buffer;
  2459 
  2460   /* We assume START is nil when input is not from a buffer.  */
  2461   if (! NILP (start) && !b)
  2462     emacs_abort ();
  2463 
  2464   specbind (Qstandard_input, readcharfun);
  2465   record_unwind_protect_int (readevalloop_1, load_convert_to_unibyte);
  2466   load_convert_to_unibyte = !NILP (unibyte);
  2467 
  2468   /* If lexical binding is active (either because it was specified in
  2469      the file's header, or via a buffer-local variable), create an empty
  2470      lexical environment, otherwise, turn off lexical binding.  */
  2471   lex_bound = find_symbol_value (Qlexical_binding);
  2472   specbind (Qinternal_interpreter_environment,
  2473             (NILP (lex_bound) || BASE_EQ (lex_bound, Qunbound)
  2474              ? Qnil : list1 (Qt)));
  2475   specbind (Qmacroexp__dynvars, Vmacroexp__dynvars);
  2476 
  2477   /* Ensure sourcename is absolute, except whilst preloading.  */
  2478   if (!will_dump_p ()
  2479       && !NILP (sourcename) && !NILP (Ffile_name_absolute_p (sourcename)))
  2480     sourcename = Fexpand_file_name (sourcename, Qnil);
  2481 
  2482   loadhist_initialize (sourcename);
  2483 
  2484   continue_reading_p = 1;
  2485   while (continue_reading_p)
  2486     {
  2487       specpdl_ref count1 = SPECPDL_INDEX ();
  2488 
  2489       if (b != 0 && !BUFFER_LIVE_P (b))
  2490         error ("Reading from killed buffer");
  2491 
  2492       if (!NILP (start))
  2493         {
  2494           /* Switch to the buffer we are reading from.  */
  2495           record_unwind_protect_excursion ();
  2496           set_buffer_internal (b);
  2497 
  2498           /* Save point in it.  */
  2499           record_unwind_protect_excursion ();
  2500           /* Save ZV in it.  */
  2501           record_unwind_protect (save_restriction_restore, save_restriction_save ());
  2502           labeled_restrictions_remove_in_current_buffer ();
  2503           /* Those get unbound after we read one expression.  */
  2504 
  2505           /* Set point and ZV around stuff to be read.  */
  2506           Fgoto_char (start);
  2507           if (!NILP (end))
  2508             Fnarrow_to_region (make_fixnum (BEGV), end);
  2509 
  2510           /* Just for cleanliness, convert END to a marker
  2511              if it is an integer.  */
  2512           if (FIXNUMP (end))
  2513             end = Fpoint_max_marker ();
  2514         }
  2515 
  2516       /* On the first cycle, we can easily test here
  2517          whether we are reading the whole buffer.  */
  2518       if (b && first_sexp)
  2519         whole_buffer = (BUF_PT (b) == BUF_BEG (b) && BUF_ZV (b) == BUF_Z (b));
  2520 
  2521       eassert (!infile0 || infile == infile0);
  2522     read_next:
  2523       c = READCHAR;
  2524       if (c == ';')
  2525         {
  2526           while ((c = READCHAR) != '\n' && c != -1);
  2527           goto read_next;
  2528         }
  2529       if (c < 0)
  2530         {
  2531           unbind_to (count1, Qnil);
  2532           break;
  2533         }
  2534 
  2535       /* Ignore whitespace here, so we can detect eof.  */
  2536       if (c == ' ' || c == '\t' || c == '\n' || c == '\f' || c == '\r'
  2537           || c == NO_BREAK_SPACE)
  2538         goto read_next;
  2539       UNREAD (c);
  2540 
  2541       if (! HASH_TABLE_P (read_objects_map)
  2542           || XHASH_TABLE (read_objects_map)->count)
  2543         read_objects_map
  2544           = make_hash_table (hashtest_eq, DEFAULT_HASH_SIZE,
  2545                              DEFAULT_REHASH_SIZE, DEFAULT_REHASH_THRESHOLD,
  2546                              Qnil, false);
  2547       if (! HASH_TABLE_P (read_objects_completed)
  2548           || XHASH_TABLE (read_objects_completed)->count)
  2549         read_objects_completed
  2550           = make_hash_table (hashtest_eq, DEFAULT_HASH_SIZE,
  2551                              DEFAULT_REHASH_SIZE, DEFAULT_REHASH_THRESHOLD,
  2552                              Qnil, false);
  2553       if (!NILP (Vpurify_flag) && c == '(')
  2554         val = read0 (readcharfun, false);
  2555       else
  2556         {
  2557           if (!NILP (readfun))
  2558             {
  2559               val = call1 (readfun, readcharfun);
  2560 
  2561               /* If READCHARFUN has set point to ZV, we should
  2562                  stop reading, even if the form read sets point
  2563                  to a different value when evaluated.  */
  2564               if (BUFFERP (readcharfun))
  2565                 {
  2566                   struct buffer *buf = XBUFFER (readcharfun);
  2567                   if (BUF_PT (buf) == BUF_ZV (buf))
  2568                     continue_reading_p = 0;
  2569                 }
  2570             }
  2571           else if (! NILP (Vload_read_function))
  2572             val = call1 (Vload_read_function, readcharfun);
  2573           else
  2574             val = read_internal_start (readcharfun, Qnil, Qnil, false);
  2575         }
  2576       /* Empty hashes can be reused; otherwise, reset on next call.  */
  2577       if (HASH_TABLE_P (read_objects_map)
  2578           && XHASH_TABLE (read_objects_map)->count > 0)
  2579         read_objects_map = Qnil;
  2580       if (HASH_TABLE_P (read_objects_completed)
  2581           && XHASH_TABLE (read_objects_completed)->count > 0)
  2582         read_objects_completed = Qnil;
  2583 
  2584       if (!NILP (start) && continue_reading_p)
  2585         start = Fpoint_marker ();
  2586 
  2587       /* Restore saved point and BEGV.  */
  2588       unbind_to (count1, Qnil);
  2589 
  2590       /* Now eval what we just read.  */
  2591       if (!NILP (macroexpand))
  2592         val = readevalloop_eager_expand_eval (val, macroexpand);
  2593       else
  2594         val = eval_sub (val);
  2595 
  2596       if (printflag)
  2597         {
  2598           Vvalues = Fcons (val, Vvalues);
  2599           if (EQ (Vstandard_output, Qt))
  2600             Fprin1 (val, Qnil, Qnil);
  2601           else
  2602             Fprint (val, Qnil);
  2603         }
  2604 
  2605       first_sexp = 0;
  2606     }
  2607 
  2608   build_load_history (sourcename,
  2609                       infile0 || whole_buffer);
  2610 
  2611   unbind_to (count, Qnil);
  2612 }

  3512       if (!CONSP (pair))
  3513         invalid_syntax ("Invalid byte-code object", readcharfun);

  3516       vec[COMPILED_CONSTANTS] = XCDR (pair);
  3517     }
  3518 
  3519   if (!((STRINGP (vec[COMPILED_BYTECODE])
  3520          && VECTORP (vec[COMPILED_CONSTANTS]))
  3521         || CONSP (vec[COMPILED_BYTECODE])))
  3522     invalid_syntax ("Invalid byte-code object", readcharfun);

  3529              because it produced a raw 8-bit string for byte-code and
  3530              now such a byte-code string is loaded as multibyte with
  3531              raw 8-bit characters converted to multibyte form.
  3532              Convert them back to the original unibyte form.  */
  3533           vec[COMPILED_BYTECODE] = Fstring_as_unibyte (vec[COMPILED_BYTECODE]);

  3536       pin_string (vec[COMPILED_BYTECODE]);

  3610 /* Read a bool vector (preceded by "#&").  */
  3611 static Lisp_Object
  3612 read_bool_vector (Lisp_Object readcharfun)
  3613 {
  3614   ptrdiff_t length = 0;
  3615   for (;;)
  3616     {
  3617       int c = READCHAR;
  3618       if (c < '0' || c > '9')
  3619         {
  3620           if (c != '"')
  3621             invalid_syntax ("#&", readcharfun);
  3622           break;
  3623         }
  3624       if (ckd_mul (&length, length, 10)
  3625           || ckd_add (&length, length, c - '0'))
  3626         invalid_syntax ("#&", readcharfun);
  3627     }
  3628 
  3629   ptrdiff_t size_in_chars = bool_vector_bytes (length);
  3630   Lisp_Object str = read_string_literal (readcharfun);
  3631   if (STRING_MULTIBYTE (str)
  3632       || !(size_in_chars == SCHARS (str)
  3633            /* We used to print 1 char too many when the number of bits
  3634               was a multiple of 8.  Accept such input in case it came
  3635               from an old version.  */
  3636            || length == (SCHARS (str) - 1) * BOOL_VECTOR_BITS_PER_CHAR))
  3637     invalid_syntax ("#&...", readcharfun);
  3638 
  3639   Lisp_Object obj = make_uninit_bool_vector (length);

  3641   memcpy (data, SDATA (str), size_in_chars);
  3642   /* Clear the extraneous bits in the last byte.  */
  3643   if (length != size_in_chars * BOOL_VECTOR_BITS_PER_CHAR)
  3644     data[size_in_chars - 1] &= (1 << (length % BOOL_VECTOR_BITS_PER_CHAR)) - 1;

  3648 /* Skip (and optionally remember) a lazily-loaded string
  3649    preceded by "#@".  Return true if this was a normal skip,
  3650    false if we read #@00 (which skips to EOB/EOF).  */
  3651 static bool
  3652 skip_lazy_string (Lisp_Object readcharfun)
  3653 {
  3654   ptrdiff_t nskip = 0;
  3655   ptrdiff_t digits = 0;
  3656   for (;;)
  3657     {
  3658       int c = READCHAR;
  3659       if (c < '0' || c > '9')
  3660         {
  3661           if (nskip > 0)
  3662             /* We can't use UNREAD here, because in the code below we side-step
  3663                READCHAR.  Instead, assume the first char after #@NNN occupies
  3664                a single byte, which is the case normally since it's just
  3665                a space.  */
  3666             nskip--;
  3667           else
  3668             UNREAD (c);
  3669           break;
  3670         }
  3671       if (ckd_mul (&nskip, nskip, 10)
  3672           || ckd_add (&nskip, nskip, c - '0'))
  3673         invalid_syntax ("#@", readcharfun);
  3674       digits++;
  3675       if (digits == 2 && nskip == 0)
  3676         {
  3677           /* #@00 means "read nil and skip to end" */
  3678           skip_dyn_eof (readcharfun);
  3679           return false;
  3680         }
  3681     }
  3682 
  3683   if (load_force_doc_strings && FROM_FILE_P (readcharfun))
  3684     {
  3685       /* If we are supposed to force doc strings into core right now,
  3686          record the last string that we skipped,
  3687          and record where in the file it comes from.  */
  3688 
  3689       /* First exchange the two saved_strings.  */
  3690       verify (ARRAYELTS (saved_strings) == 2);
  3691       struct saved_string t = saved_strings[0];
  3692       saved_strings[0] = saved_strings[1];
  3693       saved_strings[1] = t;
  3694 
  3695       enum { extra = 100 };
  3696       struct saved_string *ss = &saved_strings[0];
  3697       if (ss->size == 0)
  3698         {
  3699           ss->size = nskip + extra;
  3700           ss->string = xmalloc (ss->size);
  3701         }
  3702       else if (nskip > ss->size)
  3703         {
  3704           ss->size = nskip + extra;
  3705           ss->string = xrealloc (ss->string, ss->size);
  3706         }
  3707 
  3708       file_stream instream = infile->stream;
  3709       ss->position = (file_tell (instream) - infile->lookahead);
  3710 
  3711       /* Copy that many bytes into the saved string.  */
  3712       ptrdiff_t i = 0;
  3713       int c = 0;
  3714       for (int n = min (nskip, infile->lookahead); n > 0; n--)
  3715         ss->string[i++] = c = infile->buf[--infile->lookahead];
  3716       block_input ();
  3717       for (; i < nskip && c >= 0; i++)
  3718         ss->string[i] = c = file_get_char (instream);
  3719       unblock_input ();
  3720 
  3721       ss->length = i;
  3722     }
  3723   else
  3724     /* Skip that many bytes.  */
  3725     skip_dyn_bytes (readcharfun, nskip);
  3726 
  3727   return true;
  3728 }

  3730 /* Given a lazy-loaded string designator VAL, return the actual string.
  3731    VAL is (FILENAME . POS).  */
  3732 static Lisp_Object
  3733 get_lazy_string (Lisp_Object val)
  3734 {
  3735   /* Get a doc string from the file we are loading.
  3736      If it's in a saved string, get it from there.
  3737 
  3738      Here, we don't know if the string is a bytecode string or a doc
  3739      string.  As a bytecode string must be unibyte, we always return a
  3740      unibyte string.  If it is actually a doc string, caller must make
  3741      it multibyte.  */
  3742 
  3743   /* We used to emit negative positions for 'user variables' (whose doc
  3744      strings started with an asterisk); take the absolute value for
  3745      compatibility.  */
  3746   EMACS_INT pos = eabs (XFIXNUM (XCDR (val)));
  3747   struct saved_string *ss = &saved_strings[0];
  3748   struct saved_string *ssend = ss + ARRAYELTS (saved_strings);
  3749   while (ss < ssend
  3750          && !(pos >= ss->position && pos < ss->position + ss->length))
  3751     ss++;
  3752   if (ss >= ssend)
  3753     return get_doc_string (val, 1, 0);
  3754 
  3755   ptrdiff_t start = pos - ss->position;
  3756   char *str = ss->string;
  3757   ptrdiff_t from = start;
  3758   ptrdiff_t to = start;
  3759 
  3760   /* Process quoting with ^A, and find the end of the string,
  3761      which is marked with ^_ (037).  */
  3762   while (str[from] != 037)
  3763     {
  3764       int c = str[from++];
  3765       if (c == 1)
  3766         {
  3767           c = str[from++];
  3768           str[to++] = (c == 1 ? c
  3769                        : c == '0' ? 0
  3770                        : c == '_' ? 037
  3771                        : c);
  3772         }
  3773       else
  3774         str[to++] = c;
  3775     }
  3776 
  3777   return make_unibyte_string (str + start, to - start);
  3778 }

  3950 /* Read a Lisp object.
  3951    If LOCATE_SYMS is true, symbols are read with position.  */
  3952 static Lisp_Object
  3953 read0 (Lisp_Object readcharfun, bool locate_syms)
  3954 {
  3955   char stackbuf[64];
  3956   char *read_buffer = stackbuf;
  3957   ptrdiff_t read_buffer_size = sizeof stackbuf;
  3958   char *heapbuf = NULL;
  3959 
  3960   specpdl_ref base_pdl = SPECPDL_INDEX ();
  3961   ptrdiff_t base_sp = rdstack.sp;
  3962   record_unwind_protect_intmax (read_stack_reset, base_sp);
  3963 
  3964   specpdl_ref count = SPECPDL_INDEX ();
  3965 
  3966   bool uninterned_symbol;
  3967   bool skip_shorthand;
  3968 
  3969   /* Read an object into `obj'.  */
  3970  read_obj: ;
  3971   Lisp_Object obj;
  3972   bool multibyte;
  3973   int c = READCHAR_REPORT_MULTIBYTE (&multibyte);
  3974   if (c < 0)
  3975     end_of_file_error ();

  3977   switch (c)
  3978     {
  3979     case '(':
  3980       read_stack_push ((struct read_stack_entry) {.type = RE_list_start});
  3981       goto read_obj;
  3982 
  3983     case ')':

  3985         invalid_syntax (")", readcharfun);

  3987         {
  3988         case RE_list_start:
  3989           read_stack_pop ();

  3991           break;
  3992         case RE_list:
  3993           obj = read_stack_pop ()->u.list.head;
  3994           break;
  3995         case RE_record:
  3996           {
  3997             locate_syms = read_stack_top ()->u.vector.old_locate_syms;
  3998             Lisp_Object elems = Fnreverse (read_stack_pop ()->u.vector.elems);
  3999             if (NILP (elems))
  4000               invalid_syntax ("#s", readcharfun);
  4001 
  4002             if (BASE_EQ (XCAR (elems), Qhash_table))
  4003               obj = hash_table_from_plist (XCDR (elems));
  4004             else
  4005               obj = record_from_list (elems);
  4006             break;
  4007           }
  4008         case RE_string_props:
  4009           locate_syms = read_stack_top ()->u.vector.old_locate_syms;
  4010           obj = string_props_from_rev_list (read_stack_pop () ->u.vector.elems,
  4011                                             readcharfun);
  4012           break;
  4013         default:
  4014           invalid_syntax (")", readcharfun);
  4015         }
  4016       break;
  4017 
  4018     case '[':
  4019       read_stack_push ((struct read_stack_entry) {
  4020           .type = RE_vector,
  4021           .u.vector.elems = Qnil,
  4022           .u.vector.old_locate_syms = locate_syms,
  4023         });
  4024       /* FIXME: should vectors be read with locate_syms=false?  */
  4025       goto read_obj;
  4026 
  4027     case ']':
  4028       if (read_stack_empty_p (base_sp))
  4029         invalid_syntax ("]", readcharfun);
  4030       switch (read_stack_top ()->type)
  4031         {
  4032         case RE_vector:
  4033           locate_syms = read_stack_top ()->u.vector.old_locate_syms;
  4034           obj = vector_from_rev_list (read_stack_pop ()->u.vector.elems);
  4035           break;
  4036         case RE_byte_code:
  4037           locate_syms = read_stack_top ()->u.vector.old_locate_syms;
  4038           obj = bytecode_from_rev_list (read_stack_pop ()->u.vector.elems,
  4039                                         readcharfun);
  4040           break;
  4041         case RE_char_table:
  4042           locate_syms = read_stack_top ()->u.vector.old_locate_syms;
  4043           obj = char_table_from_rev_list (read_stack_pop ()->u.vector.elems,
  4044                                           readcharfun);
  4045           break;
  4046         case RE_sub_char_table:
  4047           locate_syms = read_stack_top ()->u.vector.old_locate_syms;
  4048           obj = sub_char_table_from_rev_list (read_stack_pop ()->u.vector.elems,
  4049                                               readcharfun);
  4050           break;
  4051         default:
  4052           invalid_syntax ("]", readcharfun);
  4053           break;
  4054         }
  4055       break;
  4056 
  4057     case '#':
  4058       {
  4059         int ch = READCHAR;
  4060         switch (ch)
  4061           {
  4062           case '\'':
  4063             /* #'X -- special syntax for (function X) */
  4064             read_stack_push ((struct read_stack_entry) {
  4065                 .type = RE_special,
  4066                 .u.special.symbol = Qfunction,
  4067               });
  4068             goto read_obj;
  4069 
  4070           case '#':
  4071             /* ## -- the empty symbol */
  4072             obj = Fintern (empty_unibyte_string, Qnil);
  4073             break;
  4074 
  4075           case 's':
  4076             /* #s(...) -- a record or hash-table */
  4077             ch = READCHAR;
  4078             if (ch != '(')
  4079               {
  4080                 UNREAD (ch);
  4081                 invalid_syntax ("#s", readcharfun);
  4082               }
  4083             read_stack_push ((struct read_stack_entry) {
  4084                 .type = RE_record,
  4085                 .u.vector.elems = Qnil,
  4086                 .u.vector.old_locate_syms = locate_syms,
  4087               });
  4088             locate_syms = false;
  4089             goto read_obj;
  4090 
  4091           case '^':
  4092             /* #^[...]  -- char-table
  4093                #^^[...] -- sub-char-table */
  4094             ch = READCHAR;
  4095             if (ch == '^')
  4096               {
  4097                 ch = READCHAR;

  4637 static Lisp_Object
  4638 substitute_object_recurse (struct subst *subst, Lisp_Object subtree)
  4639 {
  4640   /* If we find the placeholder, return the target object.  */
  4641   if (EQ (subst->placeholder, subtree))
  4642     return subst->object;
  4643 
  4644   /* For common object types that can't contain other objects, don't
  4645      bother looking them up; we're done.  */
  4646   if (SYMBOLP (subtree)
  4647       || (STRINGP (subtree) && !string_intervals (subtree))
  4648       || NUMBERP (subtree))
  4649     return subtree;
  4650 
  4651   /* If we've been to this node before, don't explore it again.  */
  4652   if (!NILP (Fmemq (subtree, subst->seen)))
  4653     return subtree;
  4654 
  4655   /* If this node can be the entry point to a cycle, remember that
  4656      we've seen it.  It can only be such an entry point if it was made
  4657      by #n=, which means that we can find it as a value in
  4658      COMPLETED.  */
  4659   if (EQ (subst->completed, Qt)
  4660       || hash_lookup (XHASH_TABLE (subst->completed), subtree, NULL) >= 0)
  4661     subst->seen = Fcons (subtree, subst->seen);
  4662 
  4663   /* Recurse according to subtree's type.
  4664      Every branch must return a Lisp_Object.  */
  4665   switch (XTYPE (subtree))
  4666     {
  4667     case Lisp_Vectorlike:
  4668       {
  4669         ptrdiff_t i = 0, length = 0;
  4670         if (BOOL_VECTOR_P (subtree))
  4671           return subtree;               /* No sub-objects anyway.  */
  4672         else if (CHAR_TABLE_P (subtree) || SUB_CHAR_TABLE_P (subtree)
  4673                  || COMPILEDP (subtree) || HASH_TABLE_P (subtree)
  4674                  || RECORDP (subtree))
  4675           length = PVSIZE (subtree);
  4676         else if (VECTORP (subtree))
  4677           length = ASIZE (subtree);
  4678         else
  4679           /* An unknown pseudovector may contain non-Lisp fields, so we
  4680              can't just blindly traverse all its fields.  We used to call
  4681              `Flength' which signaled `sequencep', so I just preserved this
  4682              behavior.  */
  4683           wrong_type_argument (Qsequencep, subtree);
  4684 
  4685         if (SUB_CHAR_TABLE_P (subtree))
  4686           i = 2;
  4687         for ( ; i < length; i++)
  4688           ASET (subtree, i,
  4689                 substitute_object_recurse (subst, AREF (subtree, i)));
  4690         return subtree;
  4691       }
  4692 
  4693     case Lisp_Cons:
  4694       XSETCAR (subtree, substitute_object_recurse (subst, XCAR (subtree)));
  4695       XSETCDR (subtree, substitute_object_recurse (subst, XCDR (subtree)));
  4696       return subtree;

  4698     case Lisp_String:
  4699       {

  4701            substitute_in_interval contains part of the logic.  */
  4702 
  4703         INTERVAL root_interval = string_intervals (subtree);
  4704         traverse_intervals_noorder (root_interval,

  4724 /* Convert the initial prefix of STRING to a number, assuming base BASE.
  4725    If the prefix has floating point syntax and BASE is 10, return a
  4726    nearest float; otherwise, if the prefix has integer syntax, return
  4727    the integer; otherwise, return nil.  If PLEN, set *PLEN to the
  4728    length of the numeric prefix if there is one, otherwise *PLEN is
  4729    unspecified.  */
  4730 
  4731 Lisp_Object
  4732 string_to_number (char const *string, int base, ptrdiff_t *plen)
  4733 {
  4734   char const *cp = string;
  4735   bool float_syntax = false;
  4736   double value = 0;
  4737 
  4738   /* Negate the value ourselves.  This treats 0, NaNs, and infinity properly on
  4739      IEEE floating point hosts, and works around a formerly-common bug where
  4740      atof ("-0.0") drops the sign.  */
  4741   bool negative = *cp == '-';
  4742   bool positive = *cp == '+';
  4743 
  4744   bool signedp = negative | positive;
  4745   cp += signedp;
  4746 
  4747   enum { INTOVERFLOW = 1, LEAD_INT = 2, TRAIL_INT = 4, E_EXP = 16 };
  4748   int state = 0;
  4749   int leading_digit = digit_to_number (*cp, base);
  4750   uintmax_t n = leading_digit;
  4751   if (leading_digit >= 0)
  4752     {
  4753       state |= LEAD_INT;
  4754       for (int digit; 0 <= (digit = digit_to_number (*++cp, base)); )
  4755         {
  4756           if (INT_MULTIPLY_OVERFLOW (n, base))
  4757             state |= INTOVERFLOW;
  4758           n *= base;
  4759           if (INT_ADD_OVERFLOW (n, digit))
  4760             state |= INTOVERFLOW;
  4761           n += digit;
  4762         }
  4763     }
  4764   char const *after_digits = cp;
  4765   if (*cp == '.')
  4766     {
  4767       cp++;
  4768     }
  4769 
  4770   if (base == 10)
  4771     {
  4772       if ('0' <= *cp && *cp <= '9')
  4773         {
  4774           state |= TRAIL_INT;
  4775           do
  4776             cp++;
  4777           while ('0' <= *cp && *cp <= '9');
  4778         }
  4779       if (*cp == 'e' || *cp == 'E')
  4780         {
  4781           char const *ecp = cp;
  4782           cp++;
  4783           if (*cp == '+' || *cp == '-')
  4784             cp++;
  4785           if ('0' <= *cp && *cp <= '9')
  4786             {
  4787               state |= E_EXP;
  4788               do
  4789                 cp++;
  4790               while ('0' <= *cp && *cp <= '9');
  4791             }
  4792 #if IEEE_FLOATING_POINT
  4793           else if (cp[-1] == '+'
  4794                    && cp[0] == 'I' && cp[1] == 'N' && cp[2] == 'F')
  4795             {
  4796               state |= E_EXP;
  4797               cp += 3;
  4798               value = INFINITY;
  4799             }
  4800           else if (cp[-1] == '+'
  4801                    && cp[0] == 'N' && cp[1] == 'a' && cp[2] == 'N')
  4802             {
  4803               state |= E_EXP;
  4804               cp += 3;
  4805               union ieee754_double u
  4806                 = { .ieee_nan = { .exponent = 0x7ff, .quiet_nan = 1,
  4807                                   .mantissa0 = n >> 31 >> 1, .mantissa1 = n }};
  4808               value = u.d;
  4809             }
  4810 #endif
  4811           else
  4812             cp = ecp;
  4813         }
  4814 
  4815       /* A float has digits after the dot or an exponent.
  4816          This excludes numbers like "1." which are lexed as integers. */
  4817       float_syntax = ((state & TRAIL_INT)
  4818                       || ((state & LEAD_INT) && (state & E_EXP)));
  4819     }
  4820 
  4821   if (plen)
  4822     *plen = cp - string;
  4823 
  4824   /* Return a float if the number uses float syntax.  */
  4825   if (float_syntax)
  4826     {
  4827       /* Convert to floating point, unless the value is already known
  4828          because it is infinite or a NaN.  */
  4829       if (! value)
  4830         value = atof (string + signedp);
  4831       return make_float (negative ? -value : value);
  4832     }
  4833 
  4834   /* Return nil if the number uses invalid syntax.  */
  4835   if (! (state & LEAD_INT))
  4836     return Qnil;
  4837 
  4838   /* Fast path if the integer (san sign) fits in uintmax_t.  */
  4839   if (! (state & INTOVERFLOW))
  4840     {
  4841       if (!negative)
  4842         return make_uint (n);
  4843       if (-MOST_NEGATIVE_FIXNUM < n)
  4844         return make_neg_biguint (n);
  4845       EMACS_INT signed_n = n;
  4846       return make_fixnum (-signed_n);
  4847     }
  4848 
  4849   /* Trim any leading "+" and trailing nondigits, then return a bignum.  */
  4850   string += positive;
  4851   if (!*after_digits)
  4852     return make_bignum_str (string, base);
  4853   ptrdiff_t trimmed_len = after_digits - string;
  4854   USE_SAFE_ALLOCA;
  4855   char *trimmed = SAFE_ALLOCA (trimmed_len + 1);
  4856   memcpy (trimmed, string, trimmed_len);
  4857   trimmed[trimmed_len] = '\0';
  4858   Lisp_Object result = make_bignum_str (trimmed, base);
  4859   SAFE_FREE ();
  4860   return result;
  4861 }

  5178 /* Like 'oblookup', but considers 'Vread_symbol_shorthands',
  5179    potentially recognizing that IN is shorthand for some other
  5180    longhand name, which is then placed in OUT.  In that case,
  5181    memory is malloc'ed for OUT (which the caller must free) while
  5182    SIZE_OUT and SIZE_BYTE_OUT respectively hold the character and byte
  5183    sizes of the transformed symbol name.  If IN is not recognized
  5184    shorthand for any other symbol, OUT is set to point to NULL and
  5185    'oblookup' is called.  */
  5186 
  5187 Lisp_Object
  5188 oblookup_considering_shorthand (Lisp_Object obarray, const char *in,
  5189                                 ptrdiff_t size, ptrdiff_t size_byte, char **out,
  5190                                 ptrdiff_t *size_out, ptrdiff_t *size_byte_out)
  5191 {
  5192   Lisp_Object tail = Vread_symbol_shorthands;
  5193 
  5194   /* First, assume no transformation will take place.  */
  5195   *out = NULL;
  5196   /* Then, iterate each pair in Vread_symbol_shorthands.  */
  5197   FOR_EACH_TAIL_SAFE (tail)
  5198     {
  5199       Lisp_Object pair = XCAR (tail);
  5200       /* Be lenient to 'read-symbol-shorthands': if some element isn't a
  5201          cons, or some member of that cons isn't a string, just skip
  5202          to the next element.  */
  5203       if (!CONSP (pair))
  5204         continue;
  5205       Lisp_Object sh_prefix = XCAR (pair);
  5206       Lisp_Object lh_prefix = XCDR (pair);
  5207       if (!STRINGP (sh_prefix) || !STRINGP (lh_prefix))
  5208         continue;
  5209       ptrdiff_t sh_prefix_size = SBYTES (sh_prefix);
  5210 
  5211       /* Compare the prefix of the transformation pair to the symbol
  5212          name.  If a match occurs, do the renaming and exit the loop.
  5213          In other words, only one such transformation may take place.
  5214          Calculate the amount of memory to allocate for the longhand
  5215          version of the symbol name with xrealloc.  This isn't
  5216          strictly needed, but it could later be used as a way for
  5217          multiple transformations on a single symbol name.  */
  5218       if (sh_prefix_size <= size_byte
  5219           && memcmp (SSDATA (sh_prefix), in, sh_prefix_size) == 0)
  5220         {
  5221           ptrdiff_t lh_prefix_size = SBYTES (lh_prefix);
  5222           ptrdiff_t suffix_size = size_byte - sh_prefix_size;
  5223           *out = xrealloc (*out, lh_prefix_size + suffix_size);
  5224           memcpy (*out, SSDATA(lh_prefix), lh_prefix_size);
  5225           memcpy (*out + lh_prefix_size, in + sh_prefix_size, suffix_size);
  5226           *size_out = SCHARS (lh_prefix) - SCHARS (sh_prefix) + size;
  5227           *size_byte_out = lh_prefix_size + suffix_size;
  5228           break;
  5229         }
  5230     }
  5231   /* Now, as promised, call oblookup with the "final" symbol name to
  5232      lookup.  That function remains oblivious to whether a
  5233      transformation happened here or not, but the caller of this
  5234      function can tell by inspecting the OUT parameter.  */
  5235   if (*out)
  5236     return oblookup (obarray, *out, *size_out, *size_byte_out);
  5237   else
  5238     return oblookup (obarray, in, size, size_byte);
  5239 }

  5394 static void
  5395 load_path_check (Lisp_Object lpath)
  5396 {
  5397   Lisp_Object path_tail;
  5398 
  5399   /* The only elements that might not exist are those from
  5400      PATH_LOADSEARCH, EMACSLOADPATH.  Anything else is only added if
  5401      it exists.  */
  5402   for (path_tail = lpath; !NILP (path_tail); path_tail = XCDR (path_tail))
  5403     {
  5404       Lisp_Object dirfile;
  5405       dirfile = Fcar (path_tail);
  5406       if (STRINGP (dirfile))
  5407         {
  5408           dirfile = Fdirectory_file_name (dirfile);
  5409           if (! file_accessible_directory_p (dirfile))
  5410             dir_warning ("Lisp directory", XCAR (path_tail));
  5411         }
  5412     }
  5413 }
  5414 
  5415 /* Return the default load-path, to be used if EMACSLOADPATH is unset.
  5416    This does not include the standard site-lisp directories
  5417    under the installation prefix (i.e., PATH_SITELOADSEARCH),
  5418    but it does (unless no_site_lisp is set) include site-lisp
  5419    directories in the source/build directories if those exist and we
  5420    are running uninstalled.
  5421 
  5422    Uses the following logic:
  5423    If !will_dump: Use PATH_LOADSEARCH.
  5424    The remainder is what happens when dumping is about to happen:
  5425    If dumping, just use PATH_DUMPLOADSEARCH.
  5426    Otherwise use PATH_LOADSEARCH.
  5427 
  5428    If !initialized, then just return PATH_DUMPLOADSEARCH.
  5429    If initialized:
  5430    If Vinstallation_directory is not nil (ie, running uninstalled):
  5431    If installation-dir/lisp exists and not already a member,
  5432    we must be running uninstalled.  Reset the load-path
  5433    to just installation-dir/lisp.  (The default PATH_LOADSEARCH
  5434    refers to the eventual installation directories.  Since we
  5435    are not yet installed, we should not use them, even if they exist.)
  5436    If installation-dir/lisp does not exist, just add
  5437    PATH_DUMPLOADSEARCH at the end instead.
  5438    Add installation-dir/site-lisp (if !no_site_lisp, and exists
  5439    and not already a member) at the front.
  5440    If installation-dir != source-dir (ie running an uninstalled,
  5441    out-of-tree build) AND install-dir/src/Makefile exists BUT
  5442    install-dir/src/Makefile.in does NOT exist (this is a sanity
  5443    check), then repeat the above steps for source-dir/lisp, site-lisp.  */
  5444 
  5445 static Lisp_Object
  5446 load_path_default (void)
  5447 {
  5448   if (will_dump_p ())
  5449     /* PATH_DUMPLOADSEARCH is the lisp dir in the source directory.
  5450        We used to add ../lisp (ie the lisp dir in the build
  5451        directory) at the front here, but that should not be
  5452        necessary, since in out of tree builds lisp/ is empty, save
  5453        for Makefile.  */
  5454     return decode_env_path (0, PATH_DUMPLOADSEARCH, 0);
  5455 
  5456   Lisp_Object lpath = Qnil;
  5457 
  5458   lpath = decode_env_path (0, PATH_LOADSEARCH, 0);
  5459 
  5460   if (!NILP (Vinstallation_directory))
  5461     {
  5462       Lisp_Object tem, tem1;
  5463 
  5464       /* Add to the path the lisp subdir of the installation
  5465          dir, if it is accessible.  Note: in out-of-tree builds,
  5466          this directory is empty save for Makefile.  */
  5467       tem = Fexpand_file_name (build_string ("lisp"),
  5468                                Vinstallation_directory);
  5469       tem1 = Ffile_accessible_directory_p (tem);
  5470       if (!NILP (tem1))
  5471         {
  5472           if (NILP (Fmember (tem, lpath)))
  5473             {
  5474               /* We are running uninstalled.  The default load-path
  5475                  points to the eventual installed lisp directories.
  5476                  We should not use those now, even if they exist,
  5477                  so start over from a clean slate.  */
  5478               lpath = list1 (tem);
  5479             }
  5480         }
  5481       else
  5482         /* That dir doesn't exist, so add the build-time
  5483            Lisp dirs instead.  */
  5484         {
  5485           Lisp_Object dump_path =
  5486             decode_env_path (0, PATH_DUMPLOADSEARCH, 0);
  5487           lpath = nconc2 (lpath, dump_path);
  5488         }
  5489 
  5490       /* Add site-lisp under the installation dir, if it exists.  */
  5491       if (!no_site_lisp)
  5492         {
  5493           tem = Fexpand_file_name (build_string ("site-lisp"),
  5494                                    Vinstallation_directory);
  5495           tem1 = Ffile_accessible_directory_p (tem);
  5496           if (!NILP (tem1))
  5497             {
  5498               if (NILP (Fmember (tem, lpath)))
  5499                 lpath = Fcons (tem, lpath);
  5500             }
  5501         }
  5502 
  5503       /* If Emacs was not built in the source directory,
  5504          and it is run from where it was built, add to load-path
  5505          the lisp and site-lisp dirs under that directory.  */
  5506 
  5507       if (NILP (Fequal (Vinstallation_directory, Vsource_directory)))
  5508         {
  5509           Lisp_Object tem2;
  5510 
  5511           tem = Fexpand_file_name (build_string ("src/Makefile"),
  5512                                    Vinstallation_directory);
  5513           tem1 = Ffile_exists_p (tem);
  5514 
  5515           /* Don't be fooled if they moved the entire source tree
  5516              AFTER dumping Emacs.  If the build directory is indeed
  5517              different from the source dir, src/Makefile.in and
  5518              src/Makefile will not be found together.  */
  5519           tem = Fexpand_file_name (build_string ("src/Makefile.in"),
  5520                                    Vinstallation_directory);
  5521           tem2 = Ffile_exists_p (tem);
  5522           if (!NILP (tem1) && NILP (tem2))
  5523             {
  5524               tem = Fexpand_file_name (build_string ("lisp"),
  5525                                        Vsource_directory);
  5526 
  5527               if (NILP (Fmember (tem, lpath)))
  5528                 lpath = Fcons (tem, lpath);
  5529 
  5530               if (!no_site_lisp)
  5531                 {
  5532                   tem = Fexpand_file_name (build_string ("site-lisp"),
  5533                                            Vsource_directory);
  5534                   tem1 = Ffile_accessible_directory_p (tem);
  5535                   if (!NILP (tem1))
  5536                     {
  5537                       if (NILP (Fmember (tem, lpath)))
  5538                         lpath = Fcons (tem, lpath);
  5539                     }
  5540                 }
  5541             }
  5542         } /* Vinstallation_directory != Vsource_directory */
  5543 
  5544     } /* if Vinstallation_directory */
  5545 
  5546   return lpath;
  5547 }

  5549 void
  5550 init_lread (void)
  5551 {
  5552   /* First, set Vload_path.  */
  5553 
  5554   /* Ignore EMACSLOADPATH when dumping.  */
  5555   bool use_loadpath = !will_dump_p ();
  5556 
  5557   if (use_loadpath && egetenv ("EMACSLOADPATH"))
  5558     {
  5559       Vload_path = decode_env_path ("EMACSLOADPATH", 0, 1);
  5560 
  5561       /* Check (non-nil) user-supplied elements.  */
  5562       load_path_check (Vload_path);
  5563 
  5564       /* If no nils in the environment variable, use as-is.
  5565          Otherwise, replace any nils with the default.  */
  5566       if (! NILP (Fmemq (Qnil, Vload_path)))
  5567         {
  5568           Lisp_Object elem, elpath = Vload_path;
  5569           Lisp_Object default_lpath = load_path_default ();
  5570 
  5571           /* Check defaults, before adding site-lisp.  */
  5572           load_path_check (default_lpath);
  5573 
  5574           /* Add the site-lisp directories to the front of the default.  */
  5575           if (!no_site_lisp && PATH_SITELOADSEARCH[0] != '\0')
  5576             {
  5577               Lisp_Object sitelisp;
  5578               sitelisp = decode_env_path (0, PATH_SITELOADSEARCH, 0);
  5579               if (! NILP (sitelisp))
  5580                 default_lpath = nconc2 (sitelisp, default_lpath);
  5581             }
  5582 
  5583           Vload_path = Qnil;
  5584 
  5585           /* Replace nils from EMACSLOADPATH by default.  */
  5586           while (CONSP (elpath))
  5587             {
  5588               elem = XCAR (elpath);
  5589               elpath = XCDR (elpath);
  5590               Vload_path = CALLN (Fappend, Vload_path,
  5591                                   NILP (elem) ? default_lpath : list1 (elem));
  5592             }
  5593         }                       /* Fmemq (Qnil, Vload_path) */
  5594     }
  5595   else
  5596     {
  5597       Vload_path = load_path_default ();
  5598 
  5599       /* Check before adding site-lisp directories.
  5600          The install should have created them, but they are not
  5601          required, so no need to warn if they are absent.
  5602          Or we might be running before installation.  */
  5603       load_path_check (Vload_path);
  5604 
  5605       /* Add the site-lisp directories at the front.  */
  5606       if (!will_dump_p () && !no_site_lisp && PATH_SITELOADSEARCH[0] != '\0')
  5607         {
  5608           Lisp_Object sitelisp;
  5609           sitelisp = decode_env_path (0, PATH_SITELOADSEARCH, 0);
  5610           if (! NILP (sitelisp)) Vload_path = nconc2 (sitelisp, Vload_path);
  5611         }
  5612     }
  5613 
  5614   Vvalues = Qnil;
  5615 
  5616   load_in_progress = 0;
  5617   Vload_file_name = Qnil;
  5618   Vload_true_file_name = Qnil;
  5619   Vstandard_input = Qt;
  5620   Vloads_in_progress = Qnil;
  5621 }
