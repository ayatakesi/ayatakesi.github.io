    24 /*
    25  * XMenu:       MIT Project Athena, X Window system menu package

    27  *      XMenuSetSelection - Set a menu selection to be active or inactive.
    28  *
    29  *      Author:         Tony Della Fera, DEC
    30  *                      August, 1985
    31  *
    32  */

    34 #include "XMenuInt.h"
    35 
    36 int
    37 XMenuSetSelection(register XMenu *menu, register int p_num, register int s_num, int active)
    38                                 /* Menu object to be modified. */
    39                                 /* Pane number to be modified. */
    40                                 /* Selection number to modified. */
    41                                 /* Make selection active? */
    42 {
    43     register XMPane *p_ptr;     /* XMPane pointer. */
    44     register XMSelect *s_ptr;   /* XMSelect pointer. */
    45 
    46     /*
    47      * Find the right pane.
    48      */
    49     p_ptr = _XMGetPanePtr(menu, p_num);
    50     if (p_ptr == NULL) return(XM_FAILURE);
    51 
    52     /*
    53      * Find the right selection.
    54      */
    55     s_ptr = _XMGetSelectionPtr(p_ptr, s_num);
    56     if (s_ptr == NULL) return(XM_FAILURE);
    57 
    58     /*
    59      * Set its active switch.
    60      */
    61     s_ptr->active = active;
    62 
    63     /*
    64      * Return the selection number just set.
    65      */
    66     _XMErrorCode = XME_NO_ERROR;
    67     return(s_num);
    68 }
