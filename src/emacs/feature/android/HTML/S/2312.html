   463 struct android_ime_event
   464 {
   465   /* Type of the event.  */
   466   enum android_event_type type;
   467 
   468   /* The event serial.  */
   469   unsigned long serial;
   470 
   471   /* The associated window.  */
   472   android_window window;
   473 
   474   /* What operation is being performed.  */
   475   enum android_ime_operation operation;
   476 
   477   /* The details of the operation.  START and END provide buffer
   478      indices, and may actually mean ``left'' and ``right''.  */
   479   ptrdiff_t start, end, position;
   480 
   481   /* The number of characters in TEXT.
   482 
   483      If OPERATION is ANDROID_IME_REQUEST_CURSOR_UPDATES, then this is
   484      actually the cursor update mode associated with that
   485      operation.  */
   486   size_t length;
   487 
   488   /* TEXT is either NULL, or a pointer to LENGTH bytes of malloced
   489      UTF-16 encoded text that must be decoded by Emacs.
   490 
   491      POSITION is where point should end up after the text is
   492      committed, relative to TEXT.  If POSITION is less than 0, it is
   493      relative to TEXT's start; otherwise, it is relative to its
   494      end.  */
   495   unsigned short *text;
   496 
   497   /* Value to set the counter to after the operation completes.  */
   498   unsigned long counter;
   499 };
