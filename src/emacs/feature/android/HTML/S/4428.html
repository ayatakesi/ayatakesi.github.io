    15 static int              count_words(char *str, char delim)
    16 {
    17   int                   count;
    18 
    19   count = 0;
    20   while (*str)
    21     {
    22       if (*str != delim)
    23         {
    24           count++;
    25           if (!strchr(str + 1, delim))
    26             return (count);

    28         }
    29       else
    30         str++;
    31     }
