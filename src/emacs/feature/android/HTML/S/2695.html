    32 /* Print formatted output to the stream FP.

    40   size_t len;

    51 
    52   if (fwrite (output, 1, len, fp) < len)
    53     {
    54       if (output != buf)

    57     }
    58

    60     free (output);
    61 
    62   if (len > INT_MAX)
    63     {
    64       errno = EOVERFLOW;
    65       fseterr (fp);
    66       return -1;
    67     }
