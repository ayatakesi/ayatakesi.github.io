   216 /* Test whether FILE is accessible for AMODE.
   217    Return true if successful, false (setting errno) otherwise.  */
   218 
   219 bool
   220 file_access_p (char const *file, int amode)
   221 {
   222 #ifdef MSDOS
   223   if (amode & W_OK)
   224     {
   225       /* FIXME: The MS-DOS faccessat implementation should handle this.  */
   226       struct stat st;
   227       if (stat (file, &st) != 0)
   228         return false;
   229       errno = EPERM;
   230       return st.st_mode & S_IWRITE || S_ISDIR (st.st_mode);
   231     }
   232 #endif
   233 
   234   if (sys_faccessat (AT_FDCWD, file, amode, AT_EACCESS) == 0)
   235     return true;
   236 
   237 #ifdef CYGWIN
   238   /* Return success if faccessat failed because Cygwin couldn't
   239      determine the file's UID or GID.  */
   240   int err = errno;
   241   struct stat st;
   242   if (stat (file, &st) == 0 && (st.st_uid == -1 || st.st_gid == -1))
   243     return true;
   244   errno = err;
   245 #endif
   246 
   247   return false;
   248 }

   251    NAME the file involved, and ERRORNO the errno value.
   252 
   253    If NAME is neither null nor a pair, package it up as a singleton
   254    list before reporting it; this saves report_file_errno's caller the

   257 Lisp_Object
   258 get_file_errno_data (char const *string, Lisp_Object name, int errorno)
   259 {
   260   Lisp_Object data = CONSP (name) || NILP (name) ? name : list1 (name);
   261   char *str = emacs_strerror (errorno);
   262   AUTO_STRING (unibyte_str, str);
   263   Lisp_Object errstring
   264     = code_convert_string_norecord (unibyte_str, Vlocale_coding_system, 0);
   265   Lisp_Object errdata = Fcons (errstring, data);

   267   if (errorno == EEXIST)
   268     return Fcons (Qfile_already_exists, errdata);
   269   else

   271                   ? Qfile_missing
   272                   : (errorno == EACCES
   273                      ? Qpermission_denied
   274                      : Qfile_error),
   275                   Fcons (build_string (string), errdata));

   365        Sfind_file_name_handler, 2, 2, 0,
   366        doc: /* Return FILENAME's handler function for OPERATION, if it has one.
   367 Otherwise, return nil.
   368 A file name is handled if one of the regular expressions in
   369 `file-name-handler-alist' matches it.
   370 
   371 If OPERATION equals `inhibit-file-name-operation', then ignore
   372 any handlers that are members of `inhibit-file-name-handlers',
   373 but still do run any other handlers.  This lets handlers
   374 use the standard functions without calling themselves recursively.  */)
   375   (Lisp_Object filename, Lisp_Object operation)
   376 {
   377   /* This function must not munge the match data.  */
   378   Lisp_Object chain, inhibited_handlers, result;
   379   ptrdiff_t pos = -1;
   380 
   381   result = Qnil;
   382   CHECK_STRING (filename);
   383 
   384   if (EQ (operation, Vinhibit_file_name_operation))
   385     inhibited_handlers = Vinhibit_file_name_handlers;
   386   else
   387     inhibited_handlers = Qnil;
   388 
   389   for (chain = Vfile_name_handler_alist; CONSP (chain);
   390        chain = XCDR (chain))
   391     {
   392       Lisp_Object elt;
   393       elt = XCAR (chain);
   394       if (CONSP (elt))
   395         {
   396           Lisp_Object string = XCAR (elt);
   397           ptrdiff_t match_pos;
   398           Lisp_Object handler = XCDR (elt);
   399           Lisp_Object operations = Qnil;
   400 
   401           if (SYMBOLP (handler))
   402             operations = Fget (handler, Qoperations);
   403 
   404           if (STRINGP (string)
   405               && (match_pos = fast_string_match (string, filename)) > pos
   406               && (NILP (operations) || ! NILP (Fmemq (operation, operations))))
   407             {
   408               Lisp_Object tem;
   409 
   410               handler = XCDR (elt);
   411               tem = Fmemq (handler, inhibited_handlers);
   412               if (NILP (tem))
   413                 {
   414                   result = handler;
   415                   pos = match_pos;
   416                 }
   417             }
   418         }
   419 
   420       maybe_quit ();
   421     }
   422   return result;
   423 }

   450 /* Return the directory component of FILENAME, or nil if FILENAME does
   451    not contain a directory component.  */
   452 
   453 Lisp_Object
   454 file_name_directory (Lisp_Object filename)
   455 {
   456   char *beg = SSDATA (filename);
   457   char const *p = beg + SBYTES (filename);
   458 
   459   while (p != beg && !IS_DIRECTORY_SEP (p[-1])
   460 #ifdef DOS_NT
   461          /* only recognize drive specifier at the beginning */
   462          && !(p[-1] == ':'
   463               /* handle the "/:d:foo" and "/:foo" cases correctly  */
   464               && ((p == beg + 2 && !IS_DIRECTORY_SEP (*beg))
   465                   || (p == beg + 4 && IS_DIRECTORY_SEP (*beg))))
   466 #endif
   467          ) p--;
   468 
   469   if (p == beg)
   470     return Qnil;
   471 #ifdef DOS_NT
   472   /* Expansion of "c:" to drive and default directory.  */
   473   Lisp_Object tem_fn;
   474   USE_SAFE_ALLOCA;
   475   SAFE_ALLOCA_STRING (beg, filename);
   476   p = beg + (p - SSDATA (filename));
   477 
   478   if (p[-1] == ':')
   479     {
   480       /* MAXPATHLEN+1 is guaranteed to be enough space for getdefdir.  */
   481       char *res = alloca (MAXPATHLEN + 1);
   482       char *r = res;
   483 
   484       if (p == beg + 4 && IS_DIRECTORY_SEP (*beg) && beg[1] == ':')
   485         {
   486           memcpy (res, beg, 2);
   487           beg += 2;
   488           r += 2;
   489         }
   490 
   491       if (getdefdir (c_toupper (*beg) - 'A' + 1, r))
   492         {
   493           size_t l = strlen (res);
   494 
   495           if (l > 3 || !IS_DIRECTORY_SEP (res[l - 1]))
   496             strcat (res, "/");
   497           beg = res;
   498           p = beg + strlen (beg);
   499           dostounix_filename (beg);
   500           tem_fn = make_specified_string (beg, -1, p - beg,
   501                                           STRING_MULTIBYTE (filename));
   502         }
   503       else
   504         tem_fn = make_specified_string (beg - 2, -1, p - beg + 2,
   505                                         STRING_MULTIBYTE (filename));
   506     }
   507   else if (STRING_MULTIBYTE (filename))
   508     {
   509       tem_fn = make_specified_string (beg, -1, p - beg, 1);
   510       dostounix_filename (SSDATA (tem_fn));
   511 #ifdef WINDOWSNT
   512       if (!NILP (Vw32_downcase_file_names))
   513         tem_fn = Fdowncase (tem_fn);
   514 #endif
   515     }
   516   else
   517     {
   518       dostounix_filename (beg);
   519       tem_fn = make_specified_string (beg, -1, p - beg, 0);
   520     }
   521   SAFE_FREE ();
   522   return tem_fn;
   523 #else  /* DOS_NT */
   524   return make_specified_string (beg, -1, p - beg, STRING_MULTIBYTE (filename));
   525 #endif  /* DOS_NT */
   526 }

   857       if (NILP (arg))

  1148         )
  1149       {
  1150         default_directory = Fexpand_file_name (default_directory, Qnil);
  1151 
  1152         /* The above expansion might have produced a remote file name,
  1153            so give the handlers one last chance to DTRT.  This can
  1154            happen when both NAME and DEFAULT-DIRECTORY arguments are
  1155            relative file names, and the buffer's default-directory is
  1156            remote.  */
  1157         handler = Ffind_file_name_handler (default_directory,
  1158                                            Qexpand_file_name);
  1159         if (!NILP (handler))
  1160           {
  1161             handled_name = call3 (handler, Qexpand_file_name,
  1162                                   name, default_directory);
  1163             if (STRINGP (handled_name))
  1164               return handled_name;
  1165             error ("Invalid handler in `file-name-handler-alist'");
  1166           }
  1167       }
  1168   }
  1169   multibyte = STRING_MULTIBYTE (name);
  1170   bool defdir_multibyte = STRING_MULTIBYTE (default_directory);
  1171   if (multibyte != defdir_multibyte)
  1172     {
  1173       /* We want to make both NAME and DEFAULT_DIRECTORY have the same
  1174          multibyteness.  Strategy:
  1175          . If either NAME or DEFAULT_DIRECTORY is pure-ASCII, they
  1176            can be converted to the multibyteness of the other one
  1177            while keeping the same byte sequence.
  1178          . If both are non-ASCII, the only safe conversion is to
  1179            convert the multibyte one to be unibyte, because the
  1180            reverse conversion potentially adds bytes while raw bytes
  1181            are converted to their multibyte forms, which we will be
  1182            unable to account for, since the information about the
  1183            original multibyteness is lost.  If those additional bytes
  1184            later leak to system APIs because they are not encoded or
  1185            because they are converted to unibyte strings by keeping
  1186            the data, file APIs will fail.
  1187 
  1188          Note: One could argue that if we see a multibyte string, it
  1189          is evidence that file-name decoding was already set up, and
  1190          we could convert unibyte strings to multibyte using
  1191          DECODE_FILE.  However, this is risky, because the likes of
  1192          string_to_multibyte are able of creating multibyte strings
  1193          without any decoding.  */
  1194       if (multibyte)
  1195         {
  1196           bool name_ascii_p = SCHARS (name) == SBYTES (name);
  1197           unsigned char *p = SDATA (default_directory);
  1198 
  1199           if (!name_ascii_p)
  1200             while (*p && ASCII_CHAR_P (*p))
  1201               p++;
  1202           if (name_ascii_p || *p != '\0')
  1203             {
  1204               /* DEFAULT_DIRECTORY is unibyte and possibly non-ASCII.
  1205                  Make a unibyte string out of NAME, and arrange for
  1206                  the result of this function to be a unibyte string.
  1207                  This is needed during bootstrapping and dumping, when
  1208                  Emacs cannot decode file names, because the locale
  1209                  environment is not set up.  */
  1210               name = make_unibyte_string (SSDATA (name), SBYTES (name));
  1211               multibyte = 0;
  1212             }
  1213           else
  1214             {
  1215               /* NAME is non-ASCII and multibyte, and
  1216                  DEFAULT_DIRECTORY is unibyte and pure-ASCII: make a
  1217                  multibyte string out of DEFAULT_DIRECTORY's data.  */
  1218               default_directory =
  1219                 make_multibyte_string (SSDATA (default_directory),
  1220                                        SCHARS (default_directory),
  1221                                        SCHARS (default_directory));
  1222             }
  1223         }
  1224       else
  1225         {
  1226           unsigned char *p = SDATA (name);
  1227 
  1228           while (*p && ASCII_CHAR_P (*p))
  1229             p++;
  1230           if (*p == '\0')
  1231             {
  1232               /* DEFAULT_DIRECTORY is multibyte and NAME is unibyte
  1233                  and pure-ASCII.  Make a multibyte string out of
  1234                  NAME's data.  */
  1235               name = make_multibyte_string (SSDATA (name),
  1236                                             SCHARS (name), SCHARS (name));
  1237               multibyte = 1;
  1238             }
  1239           else
  1240             default_directory = make_unibyte_string (SSDATA (default_directory),
  1241                                                      SBYTES (default_directory));
  1242         }
  1243     }
  1244 
  1245 #ifdef WINDOWSNT
  1246   if (!NILP (Vw32_downcase_file_names))
  1247     default_directory = Fdowncase (default_directory);
  1248 #endif
  1249 
  1250   /* Make a local copy of NAME to protect it from GC in DECODE_FILE below.  */
  1251   SAFE_ALLOCA_STRING (nm, name);
  1252   nmlim = nm + SBYTES (name);
  1253 
  1254 #ifdef DOS_NT
  1255   /* Note if special escape prefix is present, but remove for now.  */
  1256   if (nm[0] == '/' && nm[1] == ':')
  1257     {
  1258       is_escaped = 1;
  1259       nm += 2;
  1260     }
  1261 
  1262   /* Find and remove drive specifier if present; this makes nm absolute
  1263      even if the rest of the name appears to be relative.  Only look for
  1264      drive specifier at the beginning.  */
  1265   if (IS_DRIVE (nm[0]) && IS_DEVICE_SEP (nm[1]))
  1266     {
  1267       drive = (unsigned char) nm[0];
  1268       nm += 2;
  1269     }
  1270 
  1271 #ifdef WINDOWSNT
  1272   /* If we see "c://somedir", we want to strip the first slash after the
  1273      colon when stripping the drive letter.  Otherwise, this expands to
  1274      "//somedir".  */
  1275   if (drive && IS_DIRECTORY_SEP (nm[0]) && IS_DIRECTORY_SEP (nm[1]))
  1276     nm++;
  1277 
  1278   /* Discard any previous drive specifier if nm is now in UNC format.  */
  1279   if (IS_DIRECTORY_SEP (nm[0]) && IS_DIRECTORY_SEP (nm[1])
  1280       && !IS_DIRECTORY_SEP (nm[2]))
  1281     drive = 0;
  1282 #endif /* WINDOWSNT */
  1283 #endif /* DOS_NT */
  1284 
  1285   /* If nm is absolute, look for `/./' or `/../' or `//''sequences; if
  1286      none are found, we can probably return right away.  We will avoid
  1287      allocating a new string if name is already fully expanded.  */
  1288   if (
  1289       IS_DIRECTORY_SEP (nm[0])
  1290 #ifdef MSDOS
  1291       && drive && !is_escaped
  1292 #endif
  1293 #ifdef WINDOWSNT
  1294       && (drive || IS_DIRECTORY_SEP (nm[1])) && !is_escaped
  1295 #endif
  1296       )
  1297     {
  1298       /* If it turns out that the filename we want to return is just a
  1299          suffix of FILENAME, we don't need to go through and edit
  1300          things; we just need to construct a new string using data
  1301          starting at the middle of FILENAME.  If we set LOSE, that
  1302          means we've discovered that we can't do that cool trick.  */
  1303       bool lose = 0;
  1304       char *p = nm;
  1305 
  1306       while (*p)
  1307         {
  1308           /* Since we know the name is absolute, we can assume that each
  1309              element starts with a "/".  */
  1310 
  1311           /* "." and ".." are hairy.  */
  1312           if (IS_DIRECTORY_SEP (p[0])
  1313               && p[1] == '.'
  1314               && (IS_DIRECTORY_SEP (p[2])
  1315                   || p[2] == 0
  1316                   || (p[2] == '.' && (IS_DIRECTORY_SEP (p[3])
  1317                                       || p[3] == 0))))
  1318             lose = 1;
  1319           /* Replace multiple slashes with a single one, except
  1320              leave leading "//" alone.  */
  1321           else if (IS_DIRECTORY_SEP (p[0])
  1322                    && IS_DIRECTORY_SEP (p[1])
  1323                    && (p != nm || IS_DIRECTORY_SEP (p[2])))
  1324             lose = 1;
  1325           p++;
  1326         }
  1327       if (!lose)
  1328         {
  1329 #ifdef DOS_NT
  1330           /* Make sure directories are all separated with /, but
  1331              avoid allocation of a new string when not required. */
  1332           dostounix_filename (nm);
  1333 #ifdef WINDOWSNT
  1334           if (IS_DIRECTORY_SEP (nm[1]))
  1335             {
  1336               if (strcmp (nm, SSDATA (name)) != 0)
  1337                 name = make_specified_string (nm, -1, nmlim - nm, multibyte);
  1338             }
  1339           else
  1340 #endif
  1341           /* Drive must be set, so this is okay.  */
  1342           if (strcmp (nm - 2, SSDATA (name)) != 0)
  1343             {
  1344               name = make_specified_string (nm, -1, p - nm, multibyte);
  1345               char temp[] = { DRIVE_LETTER (drive), ':', 0 };
  1346               AUTO_STRING_WITH_LEN (drive_prefix, temp, 2);
  1347               name = concat2 (drive_prefix, name);
  1348             }
  1349 #ifdef WINDOWSNT
  1350           if (!NILP (Vw32_downcase_file_names))
  1351             name = Fdowncase (name);
  1352 #endif
  1353 #else /* not DOS_NT */
  1354           if (strcmp (nm, SSDATA (name)) != 0)
  1355             name = make_specified_string (nm, -1, nmlim - nm, multibyte);
  1356 #endif /* not DOS_NT */
  1357           SAFE_FREE ();
  1358           return name;
  1359         }
  1360     }
  1361 
  1362   /* At this point, nm might or might not be an absolute file name.  We
  1363      need to expand ~ or ~user if present, otherwise prefix nm with
  1364      default_directory if nm is not absolute, and finally collapse /./
  1365      and /foo/../ sequences.
  1366 
  1367      We set newdir to be the appropriate prefix if one is needed:
  1368        - the relevant user directory if nm starts with ~ or ~user
  1369        - the specified drive's working dir (DOS/NT only) if nm does not
  1370          start with /
  1371        - the value of default_directory.
  1372 
  1373      Note that these prefixes are not guaranteed to be absolute (except
  1374      for the working dir of a drive).  Therefore, to ensure we always
  1375      return an absolute name, if the final prefix is not absolute we
  1376      append it to the current working directory.  */
  1377 
  1378   newdir = newdirlim = 0;
  1379 
  1380   if (nm[0] == '~'              /* prefix ~ */
  1381 #ifdef DOS_NT
  1382     && !is_escaped              /* don't expand ~ in escaped file names */
  1383 #endif
  1384       )
  1385     {
  1386       if (IS_DIRECTORY_SEP (nm[1])
  1387           || nm[1] == 0)        /* ~ by itself */
  1388         {
  1389           Lisp_Object tem;
  1390 
  1391           newdir = get_homedir ();
  1392           nm++;
  1393           tem = build_string (newdir);
  1394           newdirlim = newdir + SBYTES (tem);
  1395           /* get_homedir may return a unibyte string, which will bite us
  1396              if we expect the directory to be multibyte.  */
  1397           if (multibyte && !STRING_MULTIBYTE (tem))
  1398             {
  1399               hdir = DECODE_FILE (tem);
  1400               newdir = SSDATA (hdir);
  1401               newdirlim = newdir + SBYTES (hdir);
  1402             }
  1403           else if (!multibyte && STRING_MULTIBYTE (tem))
  1404             multibyte = 1;
  1405 #ifdef DOS_NT
  1406           collapse_newdir = false;
  1407 #endif
  1408         }
  1409       else                      /* ~user/filename */
  1410         {
  1411           char *nmhome = user_homedir (nm + 1);
  1412           if (nmhome)
  1413             {
  1414               ptrdiff_t nmhomelen = strlen (nmhome);
  1415               newdir = nmhome;
  1416               newdirlim = newdir + nmhomelen;
  1417               if (multibyte)
  1418                 {
  1419                   AUTO_STRING_WITH_LEN (lisp_nmhome, nmhome, nmhomelen);
  1420                   hdir = DECODE_FILE (lisp_nmhome);
  1421                   newdir = SSDATA (hdir);
  1422                   newdirlim = newdir + SBYTES (hdir);
  1423                 }
  1424 
  1425               while (*++nm && !IS_DIRECTORY_SEP (*nm))
  1426                 continue;
  1427 #ifdef DOS_NT
  1428               collapse_newdir = false;
  1429 #endif
  1430             }
  1431 
  1432           /* If we don't find a user of that name, leave the name
  1433              unchanged.  */
  1434         }
  1435     }
  1436 
  1437 #ifdef DOS_NT
  1438   /* On DOS and Windows, nm is absolute if a drive name was specified;
  1439      use the drive's current directory as the prefix if needed.  */
  1440   if (!newdir && drive)
  1441     {
  1442       /* Get default directory if needed to make nm absolute.  */
  1443       char *adir = NULL;
  1444       if (!IS_DIRECTORY_SEP (nm[0]))
  1445         {
  1446           adir = alloca (MAXPATHLEN + 1);
  1447           if (!getdefdir (c_toupper (drive) - 'A' + 1, adir))
  1448             adir = NULL;
  1449           else if (multibyte)
  1450             {
  1451               Lisp_Object tem = build_string (adir);
  1452 
  1453               tem = DECODE_FILE (tem);
  1454               newdirlim = adir + SBYTES (tem);
  1455               memcpy (adir, SSDATA (tem), SBYTES (tem) + 1);
  1456             }
  1457           else
  1458             newdirlim = adir + strlen (adir);
  1459         }
  1460       if (!adir)
  1461         {
  1462           /* Either nm starts with /, or drive isn't mounted.  */
  1463           adir = alloca (4);
  1464           adir[0] = DRIVE_LETTER (drive);
  1465           adir[1] = ':';
  1466           adir[2] = '/';
  1467           adir[3] = 0;
  1468           newdirlim = adir + 3;
  1469         }
  1470       newdir = adir;
  1471     }
  1472 #endif /* DOS_NT */
  1473 
  1474   /* Finally, if no prefix has been specified and nm is not absolute,
  1475      then it must be expanded relative to default_directory.  */
  1476 
  1477   if (1
  1478 #ifndef DOS_NT
  1479       /* /... alone is not absolute on DOS and Windows.  */
  1480       && !IS_DIRECTORY_SEP (nm[0])
  1481 #endif
  1482 #ifdef WINDOWSNT
  1483       && !(IS_DIRECTORY_SEP (nm[0]) && IS_DIRECTORY_SEP (nm[1])
  1484            && !IS_DIRECTORY_SEP (nm[2]))
  1485 #endif
  1486       && !newdir)
  1487     {
  1488       newdir = SSDATA (default_directory);
  1489       newdirlim = newdir + SBYTES (default_directory);
  1490 #ifdef DOS_NT
  1491       /* Note if special escape prefix is present, but remove for now.  */
  1492       if (newdir[0] == '/' && newdir[1] == ':')
  1493         {
  1494           is_escaped = 1;
  1495           newdir += 2;
  1496         }
  1497 #endif
  1498     }
  1499 
  1500 #ifdef DOS_NT
  1501   if (newdir)
  1502     {
  1503       /* First ensure newdir is an absolute name.  */
  1504       if (
  1505           /* Detect MSDOS file names with drive specifiers.  */
  1506           ! (IS_DRIVE (newdir[0])
  1507              && IS_DEVICE_SEP (newdir[1]) && IS_DIRECTORY_SEP (newdir[2]))
  1508 #ifdef WINDOWSNT
  1509           /* Detect Windows file names in UNC format.  */
  1510           && ! (IS_DIRECTORY_SEP (newdir[0]) && IS_DIRECTORY_SEP (newdir[1])
  1511                 && !IS_DIRECTORY_SEP (newdir[2]))
  1512 #endif
  1513           )
  1514         {
  1515           /* Effectively, let newdir be (expand-file-name newdir cwd).
  1516              Because of the admonition against calling expand-file-name
  1517              when we have pointers into lisp strings, we accomplish this
  1518              indirectly by prepending newdir to nm if necessary, and using
  1519              cwd (or the wd of newdir's drive) as the new newdir.  */
  1520           char *adir;
  1521 #ifdef WINDOWSNT
  1522           const int adir_size = MAX_UTF8_PATH;
  1523 #else
  1524           const int adir_size = MAXPATHLEN + 1;
  1525 #endif
  1526 
  1527           if (IS_DRIVE (newdir[0]) && IS_DEVICE_SEP (newdir[1]))
  1528             {
  1529               drive = (unsigned char) newdir[0];
  1530               newdir += 2;
  1531             }
  1532           if (!IS_DIRECTORY_SEP (nm[0]))
  1533             {
  1534               ptrdiff_t nmlen = nmlim - nm;
  1535               ptrdiff_t newdirlen = newdirlim - newdir;
  1536               char *tmp = alloca (newdirlen + file_name_as_directory_slop
  1537                                   + nmlen + 1);
  1538               ptrdiff_t dlen = file_name_as_directory (tmp, newdir, newdirlen,
  1539                                                        multibyte);
  1540               memcpy (tmp + dlen, nm, nmlen + 1);
  1541               nm = tmp;
  1542               nmlim = nm + dlen + nmlen;
  1543             }
  1544           adir = alloca (adir_size);
  1545           if (drive)
  1546             {
  1547               if (!getdefdir (c_toupper (drive) - 'A' + 1, adir))
  1548                 strcpy (adir, "/");
  1549             }
  1550           else
  1551             getcwd (adir, adir_size);
  1552           if (multibyte)
  1553             {
  1554               Lisp_Object tem = build_string (adir);
  1555 
  1556               tem = DECODE_FILE (tem);
  1557               newdirlim = adir + SBYTES (tem);
  1558               memcpy (adir, SSDATA (tem), SBYTES (tem) + 1);
  1559             }
  1560           else
  1561             newdirlim = adir + strlen (adir);
  1562           newdir = adir;
  1563         }
  1564 
  1565       /* Strip off drive name from prefix, if present.  */
  1566       if (IS_DRIVE (newdir[0]) && IS_DEVICE_SEP (newdir[1]))
  1567         {
  1568           drive = newdir[0];
  1569           newdir += 2;
  1570         }
  1571 
  1572       /* Keep only a prefix from newdir if nm starts with slash
  1573          (//server/share for UNC, nothing otherwise).  */
  1574       if (IS_DIRECTORY_SEP (nm[0]) && collapse_newdir)
  1575         {
  1576 #ifdef WINDOWSNT
  1577           if (IS_DIRECTORY_SEP (newdir[0]) && IS_DIRECTORY_SEP (newdir[1])
  1578               && !IS_DIRECTORY_SEP (newdir[2]))
  1579             {
  1580               char *adir = strcpy (alloca (newdirlim - newdir + 1), newdir);
  1581               char *p = adir + 2;
  1582               while (*p && !IS_DIRECTORY_SEP (*p)) p++;
  1583               p++;
  1584               while (*p && !IS_DIRECTORY_SEP (*p)) p++;
  1585               *p = 0;
  1586               newdir = adir;
  1587               newdirlim = newdir + strlen (adir);
  1588             }
  1589           else
  1590 #endif
  1591             newdir = newdirlim = "";
  1592         }
  1593     }
  1594 #endif /* DOS_NT */
  1595 
  1596   /* Ignore any slash at the end of newdir, unless newdir is
  1597      just "/" or "//".  */
  1598   length = newdirlim - newdir;
  1599   while (length > 1 && IS_DIRECTORY_SEP (newdir[length - 1])
  1600          && ! (length == 2 && IS_DIRECTORY_SEP (newdir[0])))
  1601     length--;
  1602 
  1603   /* Now concatenate the directory and name to new space in the stack frame.  */
  1604   tlen = length + file_name_as_directory_slop + (nmlim - nm) + 1;
  1605   eassert (tlen >= file_name_as_directory_slop + 1);
  1606 #ifdef DOS_NT
  1607   /* Reserve space for drive specifier and escape prefix, since either
  1608      or both may need to be inserted.  (The Microsoft x86 compiler
  1609      produces incorrect code if the following two lines are combined.)  */
  1610   target = alloca (tlen + 4);
  1611   target += 4;
  1612 #else  /* not DOS_NT */
  1613   target = SAFE_ALLOCA (tlen);
  1614 #endif /* not DOS_NT */
  1615   *target = 0;
  1616   nbytes = 0;
  1617 
  1618   if (newdir)
  1619     {
  1620       if (nm[0] == 0 || IS_DIRECTORY_SEP (nm[0]))
  1621         {
  1622 #ifdef DOS_NT
  1623           /* If newdir is effectively "C:/", then the drive letter will have
  1624              been stripped and newdir will be "/".  Concatenating with an
  1625              absolute directory in nm produces "//", which will then be
  1626              incorrectly treated as a network share.  Ignore newdir in
  1627              this case (keeping the drive letter).  */
  1628           if (!(drive && nm[0] && IS_DIRECTORY_SEP (newdir[0])
  1629                 && newdir[1] == '\0'))
  1630 #endif
  1631             {
  1632               memcpy (target, newdir, length);
  1633               target[length] = 0;
  1634               nbytes = length;
  1635             }
  1636         }
  1637       else
  1638         nbytes = file_name_as_directory (target, newdir, length, multibyte);
  1639     }
  1640 
  1641   memcpy (target + nbytes, nm, nmlim - nm + 1);
  1642 
  1643   /* Now canonicalize by removing `//', `/.' and `/foo/..' if they
  1644      appear.  */
  1645   {
  1646     char *p = target;
  1647     char *o = target;
  1648 
  1649     while (*p)
  1650       {
  1651         if (!IS_DIRECTORY_SEP (*p))
  1652           {
  1653             *o++ = *p++;
  1654           }
  1655         else if (p[1] == '.'
  1656                  && (IS_DIRECTORY_SEP (p[2])
  1657                      || p[2] == 0))
  1658           {
  1659             /* If "/." is the entire filename, keep the "/".  Otherwise,
  1660                just delete the whole "/.".  */
  1661             if (o == target && p[2] == '\0')
  1662               *o++ = *p;
  1663             p += 2;
  1664           }
  1665         else if (p[1] == '.' && p[2] == '.'
  1666                  /* `/../' is the "superroot" on certain file systems.
  1667                     Turned off on DOS_NT systems because they have no
  1668                     "superroot" and because this causes us to produce
  1669                     file names like "d:/../foo" which fail file-related
  1670                     functions of the underlying OS.  (To reproduce, try a
  1671                     long series of "../../" in default_directory, longer
  1672                     than the number of levels from the root.)  */
  1673 #ifndef DOS_NT
  1674                  && o != target
  1675 #endif
  1676                  && (IS_DIRECTORY_SEP (p[3]) || p[3] == 0))
  1677           {
  1678 #ifdef WINDOWSNT
  1679             char *prev_o = o;
  1680 #endif
  1681             while (o != target && (--o, !IS_DIRECTORY_SEP (*o)))
  1682               continue;
  1683 #ifdef WINDOWSNT
  1684             /* Don't go below server level in UNC filenames.  */
  1685             if (o == target + 1 && IS_DIRECTORY_SEP (*o)
  1686                 && IS_DIRECTORY_SEP (*target))
  1687               o = prev_o;
  1688             else
  1689 #endif
  1690             /* Keep initial / only if this is the whole name.  */
  1691             if (o == target && IS_ANY_SEP (*o) && p[3] == 0)
  1692               ++o;
  1693             p += 3;
  1694           }
  1695         else if (IS_DIRECTORY_SEP (p[1])
  1696                  && (p != target || IS_DIRECTORY_SEP (p[2])))
  1697           /* Collapse multiple "/", except leave leading "//" alone.  */
  1698           p++;
  1699         else
  1700           {
  1701             *o++ = *p++;
  1702           }
  1703       }
  1704 
  1705 #ifdef DOS_NT
  1706     /* At last, set drive name.  */
  1707 #ifdef WINDOWSNT
  1708     /* Except for network file name.  */
  1709     if (!(IS_DIRECTORY_SEP (target[0]) && IS_DIRECTORY_SEP (target[1])))

  2296                                   FIXNUMP (ok_if_already_exists), false);
  2297 
  2298   result = w32_copy_file (SSDATA (encoded_file), SSDATA (encoded_newname),
  2299                           !NILP (keep_time), !NILP (preserve_uid_gid),
  2300                           !NILP (preserve_permissions));
  2301   switch (result)
  2302     {
  2303     case -1:
  2304       report_file_error ("Copying file", list2 (file, newname));
  2305     case -2:
  2306       report_file_error ("Copying permissions from", file);
  2307     case -3:
  2308       xsignal2 (Qfile_date_error,
  2309                 build_string ("Cannot set file date"), newname);
  2310     case -4:
  2311       report_file_error ("Copying permissions to", newname);
  2312     }
  2313 #else /* not WINDOWSNT */

  2315 
  2316   if (!emacs_fd_valid_p (ifd))
  2317     report_file_error ("Opening input file", file);
  2318 
  2319   record_unwind_protect_ptr (close_file_unwind_emacs_fd, &ifd);
  2320 
  2321   if (emacs_fd_fstat (ifd, &st) != 0)
  2322     report_file_error ("Input file status", file);
  2323 
  2324   if (!NILP (preserve_permissions))
  2325     {
  2326 #if HAVE_LIBSELINUX
  2327       if (is_selinux_enabled ()
  2328           && emacs_fd_to_int (ifd) != -1)
  2329         {
  2330           conlength = fgetfilecon (emacs_fd_to_int (ifd),
  2331                                    &con);
  2332           if (conlength == -1)
  2333             report_file_error ("Doing fgetfilecon", file);
  2334         }
  2335 #endif
  2336     }
  2337 
  2338   /* We can copy only regular files.  */
  2339   if (!S_ISREG (st.st_mode))
  2340     report_file_errno ("Non-regular file", file,
  2341                        S_ISDIR (st.st_mode) ? EISDIR : EINVAL);
  2342 
  2343 #ifndef MSDOS
  2344   new_mask = st.st_mode & (!NILP (preserve_uid_gid) ? 0700 : 0777);
  2345 #else
  2346   new_mask = S_IREAD | S_IWRITE;
  2347 #endif
  2348 
  2349   ofd = emacs_open (SSDATA (encoded_newname), O_WRONLY | O_CREAT | O_EXCL,
  2350                     new_mask);
  2351   if (ofd < 0 && errno == EEXIST)
  2352     {
  2353       if (NILP (ok_if_already_exists) || FIXNUMP (ok_if_already_exists))

  2355                                       FIXNUMP (ok_if_already_exists), false);
  2356       already_exists = true;
  2357       ofd = emacs_open (SSDATA (encoded_newname), O_WRONLY, 0);
  2358     }
  2359   if (ofd < 0)
  2360     report_file_error ("Opening output file", newname);
  2361 
  2362   record_unwind_protect_int (close_file_unwind, ofd);
  2363 
  2364   off_t oldsize = 0, newsize;
  2365 
  2366   if (already_exists)
  2367     {
  2368       struct stat out_st;
  2369       if (sys_fstat (ofd, &out_st) != 0)
  2370         report_file_error ("Output file status", newname);
  2371       if (st.st_dev == out_st.st_dev && st.st_ino == out_st.st_ino)
  2372         report_file_errno ("Input and output files are the same",
  2373                            list2 (file, newname), 0);
  2374       if (S_ISREG (out_st.st_mode))
  2375         oldsize = out_st.st_size;
  2376     }

  2383   else

  2386       ssize_t copied;

  2388 #ifndef MSDOS
  2389       newsize = 0;

  2392         {
  2393           for (; newsize < insize; newsize += copied)
  2394             {

  2396                  (PTRDIFF_MAX, SIZE_MAX) truncated to a value that is
  2397                  surely aligned well.  */

  2399               ptrdiff_t copy_max = min (ssize_max, SIZE_MAX) >> 30 << 30;
  2400               off_t intail = insize - newsize;
  2401               ptrdiff_t len = min (intail, copy_max);
  2402               copied = copy_file_range (emacs_fd_to_int (ifd), NULL,
  2403                                         ofd, NULL, len, 0);

  2408         }
  2409 #endif /* MSDOS */

  2412          input is empty and so could be a /proc file, or if ifd is an
  2413          invention of android.c.  read+write will either succeed, or
  2414          report an error more precisely than copy_file_range
  2415          would.  */

  2417         {

  2436 
  2437 #ifndef MSDOS
  2438   /* Preserve the original file permissions, and if requested, also its
  2439      owner and group.  */

  2441     mode_t preserved_permissions = st.st_mode & 07777;
  2442     mode_t default_permissions = st.st_mode & 0777 & ~realmask;
  2443     if (!NILP (preserve_uid_gid))
  2444       {
  2445         /* Attempt to change owner and group.  If that doesn't work
  2446            attempt to change just the group, as that is sometimes allowed.
  2447            Adjust the mode mask to eliminate setuid or setgid bits
  2448            or group permissions bits that are inappropriate if the
  2449            owner or group are wrong.  */
  2450         if (fchown (ofd, st.st_uid, st.st_gid) != 0)
  2451           {
  2452             if (fchown (ofd, -1, st.st_gid) == 0)
  2453               preserved_permissions &= ~04000;
  2454             else
  2455               {
  2456                 preserved_permissions &= ~06000;
  2457 
  2458                 /* Copy the other bits to the group bits, since the
  2459                    group is wrong.  */
  2460                 preserved_permissions &= ~070;
  2461                 preserved_permissions |= (preserved_permissions & 7) << 3;
  2462                 default_permissions &= ~070;
  2463                 default_permissions |= (default_permissions & 7) << 3;
  2464               }
  2465           }
  2466       }
  2467 
  2468     switch ((!NILP (preserve_permissions)
  2469              && emacs_fd_to_int (ifd) != -1)
  2470             ? qcopy_acl (SSDATA (encoded_file),
  2471                          emacs_fd_to_int (ifd),
  2472                          SSDATA (encoded_newname), ofd,
  2473                          preserved_permissions)
  2474             : (already_exists
  2475                || (new_mask & ~realmask) == default_permissions)
  2476             ? 0
  2477             : fchmod (ofd, default_permissions))
  2478       {
  2479       case -2: report_file_error ("Copying permissions from", file);
  2480       case -1: report_file_error ("Copying permissions to", newname);
  2481       }
  2482   }
  2483 #endif  /* not MSDOS */
  2484 
  2485 #if HAVE_LIBSELINUX
  2486   if (conlength > 0)
  2487     {
  2488       /* Set the modified context back to the file.  */
  2489       bool fail = fsetfilecon (ofd, con) != 0;
  2490       /* See https://debbugs.gnu.org/11245 for ENOTSUP.  */
  2491       if (fail && errno != ENOTSUP)
  2492         report_file_error ("Doing fsetfilecon", newname);
  2493 
  2494       freecon (con);
  2495     }
  2496 #endif
  2497 
  2498   if (!NILP (keep_time))
  2499     {
  2500       struct timespec ts[2];
  2501       ts[0] = get_stat_atime (&st);
  2502       ts[1] = get_stat_mtime (&st);
  2503       if (futimens (ofd, ts) != 0
  2504           /* Various versions of the Android C library are missing
  2505              futimens, which leads a gnulib fallback to be installed
  2506              that uses fdutimens instead.  However, fdutimens is not
  2507              supported on many Android kernels, so just silently fail
  2508              if errno is ENOTSUP or ENOSYS.  */
  2509 #if defined HAVE_ANDROID && !defined ANDROID_STUBIFY
  2510           && errno != ENOTSUP
  2511           && errno != ENOSYS
  2512 #endif
  2513           )
  2514         xsignal2 (Qfile_date_error,
  2515                   build_string ("Cannot set file date"), newname);
  2516     }
  2517 
  2518   if (emacs_close (ofd) < 0)
  2519     report_file_error ("Write error", newname);
  2520 
  2521   /* Note that ifd is not closed twice because unwind_protects are
  2522      discarded at the end of this function.  */
  2523   emacs_fd_close (ifd);
  2524 
  2525 #ifdef MSDOS
  2526   /* In DJGPP v2.0 and later, fstat usually returns true file mode bits,
  2527      and if it can't, it tells so.  Otherwise, under MSDOS we usually
  2528      get only the READ bit, which will make the copied file read-only,
  2529      so it's better not to chmod at all.  */
  2530   if ((_djstat_flags & _STFAIL_WRITEBIT) == 0)
  2531     chmod (SDATA (encoded_newname), st.st_mode & 07777);
  2532 #endif /* MSDOS */
  2533 #endif /* not WINDOWSNT */
  2534 
  2535   /* Discard the unwind protects.  */
  2536   specpdl_ptr = specpdl_ref_to_ptr (count);
  2537 
  2538   return Qnil;
  2539 }

  2646 /* Return -1 if FILE is a case-insensitive file name, 0 if not,
  2647    and a positive errno value if the result cannot be determined.  */
  2648 
  2649 static int
  2650 file_name_case_insensitive_err (Lisp_Object file)
  2651 {
  2652   /* Filesystems are case-sensitive on all supported systems except
  2653      MS-Windows, MS-DOS, Cygwin, and macOS.  They are always
  2654      case-insensitive on the first two, but they may or may not be
  2655      case-insensitive on Cygwin and macOS so do a runtime test on
  2656      those two systems.  If the test is not conclusive, assume
  2657      case-insensitivity on Cygwin and case-sensitivity on macOS.
  2658 
  2659      FIXME: Mounted filesystems on Posix hosts, like Samba shares or
  2660      NFS-mounted Windows volumes, might be case-insensitive.  Can we
  2661      detect this?
  2662 
  2663      Use pathconf with _PC_CASE_INSENSITIVE or _PC_CASE_SENSITIVE if
  2664      those flags are available.  As of this writing (2019-09-15),
  2665      Cygwin is the only platform known to support the former (starting
  2666      with Cygwin-2.6.1), and macOS is the only platform known to
  2667      support the latter.  */
  2668 
  2669 #if defined _PC_CASE_INSENSITIVE || defined _PC_CASE_SENSITIVE
  2670   char *filename = SSDATA (ENCODE_FILE (file));
  2671 # ifdef _PC_CASE_INSENSITIVE
  2672   long int res = pathconf (filename, _PC_CASE_INSENSITIVE);
  2673   if (res >= 0)
  2674     return - (res > 0);
  2675 # else
  2676   long int res = pathconf (filename, _PC_CASE_SENSITIVE);
  2677   if (res >= 0)
  2678     return - (res == 0);
  2679 # endif
  2680   if (errno != EINVAL)
  2681     return errno;
  2682 #endif
  2683 
  2684 #if defined CYGWIN || defined DOS_NT
  2685   return -1;
  2686 #else
  2687   return 0;
  2688 #endif
  2689 }

  2730 If file has names other than FILE, it continues to have those names.

  2796                                         "rename to it",

  2807         return Qnil;

  2809       /* Don't prompt again.  */
  2810       ok_if_already_exists = Qt;
  2811     }

  2813     ok_if_already_exists = Qt;
  2814

  2816     report_file_errno ("Renaming", list2 (file, newname), rename_errno);
  2817

  2820   if (!dirp)
  2821     {
  2822       if (emacs_fstatat (AT_FDCWD, SSDATA (encoded_file),
  2823                          &file_st, AT_SYMLINK_NOFOLLOW)
  2824           != 0)
  2825         report_file_error ("Renaming", list2 (file, newname));
  2826       dirp = S_ISDIR (file_st.st_mode) != 0;
  2827     }
  2828   if (dirp)
  2829     call4 (Qcopy_directory, file, newname, Qt, Qnil);
  2830   else if (S_ISREG (file_st.st_mode))
  2831     Fcopy_file (file, newname, ok_if_already_exists, Qt, Qt, Qt);
  2832   else if (S_ISLNK (file_st.st_mode))
  2833     {
  2834       Lisp_Object target = emacs_readlinkat (AT_FDCWD,

  3195 /* Return true if FILE is a directory or a symlink to a directory.
  3196    Otherwise return false and set errno.  */

  3199 {
  3200 #ifdef DOS_NT
  3201   /* This is cheaper than 'stat'.  */

  3203                                D_OK, AT_EACCESS) == 0;
  3204   if (!retval && errno == EACCES)
  3205     errno = ENOTDIR;    /* like the non-DOS_NT branch below does */
  3206   return retval;

  3208 # if defined O_PATH && !(defined HAVE_ANDROID && !defined ANDROID_STUBIFY)
  3209   /* Use O_PATH if available, as it avoids races and EOVERFLOW issues.  */

  3211                          O_PATH | O_CLOEXEC | O_DIRECTORY, 0);

  3214       emacs_close (fd);
  3215       return true;

  3218     return false;

  3225      EOVERFLOW then FILE must have been a directory unless a race
  3226      condition occurred (a problem hard to work around portably).  */

  3287    searchable directory, return true.  Otherwise return
  3288    false and set errno to an error number.  */
  3289 bool
  3290 file_accessible_directory_p (Lisp_Object file)
  3291 {
  3292 #ifdef DOS_NT
  3293 # ifdef WINDOWSNT
  3294   /* We need a special-purpose test because (a) NTFS security data is
  3295      not reflected in Posix-style mode bits, and (b) the trick with
  3296      accessing "DIR/.", used below on Posix hosts, doesn't work on
  3297      Windows, because "DIR/." is normalized to just "DIR" before
  3298      hitting the disk.  */
  3299   return (SBYTES (file) == 0
  3300           || w32_accessible_directory_p (SSDATA (file), SBYTES (file)));
  3301 # else  /* MSDOS */
  3302   return file_directory_p (file);
  3303 # endif  /* MSDOS */
  3304 #else    /* !DOS_NT */
  3305   /* On POSIXish platforms, use just one system call; this avoids a
  3306      race and is typically faster.  */
  3307   const char *data = SSDATA (file);
  3308   ptrdiff_t len = SBYTES (file);
  3309   char const *dir;
  3310   bool ok;
  3311   USE_SAFE_ALLOCA;
  3312 
  3313   /* Normally a file "FOO" is an accessible directory if "FOO/." exists.
  3314      There are three exceptions: "", "/", and "//".  Leave "" alone,
  3315      as it's invalid.  Append only "." to the other two exceptions as
  3316      "/" and "//" are distinct on some platforms, whereas "/", "///",
  3317      "////", etc. are all equivalent.
  3318 
  3319      Android has a special directory named "/assets".  There is no "."
  3320      directory there, but appending a "/" is sufficient to check
  3321      whether or not it is a directory.  */
  3322   if (! len)
  3323     dir = data;
  3324   else
  3325     {

  3327          before appending '.'.  That's simpler than testing the two
  3328          special cases "/" and "//", and it's a safe optimization
  3329          here.  After appending '.', append another '/' to work around
  3330          a macOS bug (Bug#30350).  */
  3331 #if defined HAVE_ANDROID && !defined ANDROID_STUBIFY
  3332       if (!strncmp ("/assets/", data,
  3333                     sizeof "/assets" - 1))
  3334         {

  3336           char *buf = SAFE_ALLOCA (len + sizeof appended);
  3337           memcpy (buf, data, len);
  3338           strcpy (buf + len, &appended[data[len - 1] == '/']);
  3339           dir = buf;
  3340         }
  3341       else
  3342         {
  3343 #endif
  3344           static char const appended[] = "/./";
  3345           char *buf = SAFE_ALLOCA (len + sizeof appended);
  3346           memcpy (buf, data, len);
  3347           strcpy (buf + len, &appended[data[len - 1] == '/']);
  3348           dir = buf;
  3349 #if defined HAVE_ANDROID && !defined ANDROID_STUBIFY
  3350         }
  3351 #endif
  3352     }
  3353 
  3354   ok = file_access_p (dir, F_OK);
  3355   SAFE_FREE ();
  3356   return ok;
  3357 #endif  /* !DOS_NT */

  3833 /* This function is called after Lisp functions to decide a coding
  3834    system are called, or when they cause an error.  Before they are
  3835    called, the current buffer is set unibyte and it contains only a
  3836    newly inserted text (thus the buffer was empty before the
  3837    insertion).
  3838 
  3839    The functions may set markers, overlays, text properties, or even
  3840    alter the buffer contents, change the current buffer.
  3841 
  3842    Here, we reset all those changes by:
  3843         o set back the current buffer.
  3844         o move all markers and overlays to BEG.
  3845         o remove all text properties.
  3846         o set back the buffer multibyteness.  */
  3847 
  3848 static void
  3849 decide_coding_unwind (Lisp_Object unwind_data)
  3850 {
  3851   Lisp_Object multibyte = XCAR (unwind_data);
  3852   Lisp_Object tmp = XCDR (unwind_data);
  3853   Lisp_Object undo_list = XCAR (tmp);
  3854   Lisp_Object buffer = XCDR (tmp);
  3855 
  3856   set_buffer_internal (XBUFFER (buffer));
  3857 
  3858   /* We're about to "delete" the text by moving it back into the gap.
  3859      So move markers that set-auto-coding might have created to BEG,
  3860      just in case.  */
  3861   adjust_markers_for_delete (BEG, BEG_BYTE, Z, Z_BYTE);
  3862   adjust_overlays_for_delete (BEG, Z - BEG);
  3863   set_buffer_intervals (current_buffer, NULL);
  3864   TEMP_SET_PT_BOTH (BEG, BEG_BYTE);
  3865 
  3866   /* In case of a non-local exit from set_auto_coding_function, in order not
  3867      to end up with potentially invalid byte sequences in a multibyte buffer,
  3868      we have the following options:
  3869      1- decode the bytes in some arbitrary coding-system.
  3870      2- erase the buffer.
  3871      3- leave the buffer unibyte (which is actually the same as option (1)
  3872         where the coding-system is `raw-text-unix`).
  3873      Here we choose 2.  */
  3874 
  3875   /* Move the bytes back to (the beginning of) the gap.
  3876      In general this may have to move all the bytes, but here
  3877      this can't move more bytes than were moved during the execution
  3878      of Vset_auto_coding_function, which is normally 0 (because it
  3879      normally doesn't modify the buffer).  */
  3880   move_gap_both (Z, Z_BYTE);
  3881   ptrdiff_t inserted = Z_BYTE - BEG_BYTE;
  3882   GAP_SIZE += inserted;
  3883   ZV = Z = GPT = BEG;
  3884   ZV_BYTE = Z_BYTE = GPT_BYTE = BEG_BYTE;
  3885 
  3886   /* Pass the new `inserted` back.  */
  3887   XSETCAR (unwind_data, make_fixnum (inserted));
  3888 
  3889   /* Now we are safe to change the buffer's multibyteness directly.  */
  3890   bset_enable_multibyte_characters (current_buffer, multibyte);
  3891   bset_undo_list (current_buffer, undo_list);
  3892 }

  5046           ptrdiff_t opoint_byte = PT_BYTE;
  5047           ptrdiff_t oinserted = ZV - BEGV;

  5061             /* format_decode modified buffer's characters => consider
  5062                entire buffer changed and leave point at point-min.  */

  5064         }

  5072             {
  5073               insval = call1 (XCAR (p), make_fixnum (inserted));

  5075                 {
  5076                   if (! RANGED_FIXNUMP (0, insval, ZV - PT))
  5077                     wrong_type_argument (Qinserted_chars, insval);
  5078                   inserted = XFIXNAT (insval);
  5079                 }
  5080             }
  5081           else
  5082             {
  5083               /* For the rationale of this see the comment on
  5084                  format-decode above.  */
  5085               ptrdiff_t opoint = PT;
  5086               ptrdiff_t opoint_byte = PT_BYTE;
  5087               ptrdiff_t oinserted = ZV - BEGV;
  5088               modiff_count ochars_modiff = CHARS_MODIFF;
  5089 
  5090               TEMP_SET_PT_BOTH (BEGV, BEGV_BYTE);
  5091               insval = call1 (XCAR (p), make_fixnum (oinserted));
  5092               if (!NILP (insval))
  5093                 {
  5094                   if (! RANGED_FIXNUMP (0, insval, ZV - PT))
  5095                     wrong_type_argument (Qinserted_chars, insval);
  5096                   if (ochars_modiff == CHARS_MODIFF)
  5097                     /* after_insert_file_functions didn't modify
  5098                        buffer's characters => move point back to
  5099                        position before inserted text and leave value of
  5100                        inserted alone.  */
  5101                     SET_PT_BOTH (opoint, opoint_byte);
  5102                   else
  5103                     /* after_insert_file_functions did modify buffer's
  5104                        characters => consider entire buffer changed and
  5105                        leave point at point-min.  */
  5106                     inserted = XFIXNAT (insval);
  5107                 }
  5108             }
  5109         }
  5110 
  5111       if (!empty_undo_list_p)
  5112         {
  5113           bset_undo_list (current_buffer, old_undo);
  5114           if (CONSP (old_undo) && inserted != old_inserted)
  5115             {
  5116               /* Adjust the last undo record for the size change during
  5117                  the format conversion.  */
  5118               Lisp_Object tem = XCAR (old_undo);
  5119               if (CONSP (tem) && FIXNUMP (XCAR (tem))
  5120                   && FIXNUMP (XCDR (tem))
  5121                   && XFIXNUM (XCDR (tem)) == PT + old_inserted)
  5122                 XSETCDR (tem, make_fixnum (PT + inserted));
  5123             }
  5124         }
  5125       else
  5126         /* If undo_list was Qt before, keep it that way.
  5127            Otherwise start with an empty undo_list.  */
  5128         bset_undo_list (current_buffer, EQ (old_undo, Qt) ? Qt : Qnil);
  5129 
  5130       unbind_to (count1, Qnil);
  5131     }
  5132 
  5133   if (save_errno != 0)
  5134     {
  5135       /* Signal an error if visiting a file that could not be opened.  */
  5136       eassert (!NILP (visit) && NILP (handler));
  5137       report_file_errno ("Opening input file", orig_filename, save_errno);
  5138     }
  5139 
  5140   /* We made a lot of deletions and insertions above, so invalidate
  5141      the newline cache for the entire region of the inserted
  5142      characters.  */
  5143   if (current_buffer->base_buffer && current_buffer->base_buffer->newline_cache)
  5144     invalidate_region_cache (current_buffer->base_buffer,
  5145                              current_buffer->base_buffer->newline_cache,
  5146                              PT - BEG, Z - PT - inserted);
  5147   else if (current_buffer->newline_cache)
  5148     invalidate_region_cache (current_buffer,
  5149                              current_buffer->newline_cache,
  5150                              PT - BEG, Z - PT - inserted);
  5151 
  5152   if (read_quit)
  5153     quit ();
  5154 
  5155   /* Retval needs to be dealt with in all cases consistently.  */
  5156   if (NILP (val))
  5157     val = list2 (orig_filename, make_fixnum (inserted));
  5158 
  5159   return unbind_to (count, val);
  5160 }

  5780 /* Write to descriptor DESC the NCHARS chars starting at POS of STRING.
  5781    If STRING is nil, POS is the character position in the current buffer.
  5782    Intersperse with them the annotations from *ANNOT
  5783    which fall within the range of POS to POS + NCHARS,
  5784    each at its appropriate position.
  5785 
  5786    We modify *ANNOT by discarding elements as we use them up.
  5787 
  5788    Return true if successful.  */
  5789 
  5790 static bool
  5791 a_write (int desc, Lisp_Object string, ptrdiff_t pos,
  5792          ptrdiff_t nchars, Lisp_Object *annot,
  5793          struct coding_system *coding)
  5794 {
  5795   Lisp_Object tem;
  5796   ptrdiff_t nextpos;
  5797   ptrdiff_t lastpos = pos + nchars;
  5798 
  5799   while (NILP (*annot) || CONSP (*annot))
  5800     {
  5801       tem = Fcar_safe (Fcar (*annot));
  5802       nextpos = pos - 1;
  5803       if (FIXNUMP (tem))
  5804         nextpos = XFIXNUM (tem);
  5805 
  5806       /* If there are no more annotations in this range,
  5807          output the rest of the range all at once.  */
  5808       if (! (nextpos >= pos && nextpos <= lastpos))
  5809         return e_write (desc, string, pos, lastpos, coding);
  5810 
  5811       /* Output buffer text up to the next annotation's position.  */
  5812       if (nextpos > pos)
  5813         {
  5814           if (!e_write (desc, string, pos, nextpos, coding))
  5815             return 0;
  5816           pos = nextpos;
  5817         }
  5818       /* Output the annotation.  */
  5819       tem = Fcdr (Fcar (*annot));
  5820       if (STRINGP (tem))
  5821         {
  5822           if (!e_write (desc, tem, 0, SCHARS (tem), coding))
  5823             return 0;
  5824         }
  5825       *annot = Fcdr (*annot);
  5826     }
  5827   return 1;
  5828 }

  5830 /* Maximum number of characters that the next
  5831    function encodes per one loop iteration.  */
  5832 
  5833 enum { E_WRITE_MAX = 8 * 1024 * 1024 };
  5834 
  5835 /* Write text in the range START and END into descriptor DESC,
  5836    encoding them with coding system CODING.  If STRING is nil, START
  5837    and END are character positions of the current buffer, else they
  5838    are indexes to the string STRING.  Return true if successful.  */
  5839 
  5840 static bool
  5841 e_write (int desc, Lisp_Object string, ptrdiff_t start, ptrdiff_t end,
  5842          struct coding_system *coding)
  5843 {
  5844   if (STRINGP (string))
  5845     {
  5846       start = 0;
  5847       end = SCHARS (string);
  5848     }
  5849 
  5850   /* We used to have a code for handling selective display here.  But,
  5851      now it is handled within encode_coding.  */
  5852 
  5853   while (start < end)
  5854     {
  5855       if (STRINGP (string))
  5856         {
  5857           coding->src_multibyte = SCHARS (string) < SBYTES (string);
  5858           if (CODING_REQUIRE_ENCODING (coding))
  5859             {
  5860               ptrdiff_t nchars = min (end - start, E_WRITE_MAX);
  5861 
  5862               /* Avoid creating huge Lisp string in encode_coding_object.  */
  5863               if (nchars == E_WRITE_MAX)
  5864                 coding->raw_destination = 1;
  5865 
  5866               encode_coding_object
  5867                 (coding, string, start, string_char_to_byte (string, start),
  5868                  start + nchars, string_char_to_byte (string, start + nchars),
  5869                  Qt);
  5870             }
  5871           else
  5872             {
  5873               coding->dst_object = string;
  5874               coding->consumed_char = SCHARS (string);
  5875               coding->produced = SBYTES (string);
  5876             }
  5877         }
  5878       else
  5879         {
  5880           ptrdiff_t start_byte = CHAR_TO_BYTE (start);
  5881           ptrdiff_t end_byte = CHAR_TO_BYTE (end);
  5882 
  5883           coding->src_multibyte = (end - start) < (end_byte - start_byte);
  5884           if (CODING_REQUIRE_ENCODING (coding))
  5885             {
  5886               ptrdiff_t nchars = min (end - start, E_WRITE_MAX);
  5887 
  5888               /* Likewise.  */
  5889               if (nchars == E_WRITE_MAX)
  5890                 coding->raw_destination = 1;
  5891 
  5892               encode_coding_object
  5893                 (coding, Fcurrent_buffer (), start, start_byte,
  5894                  start + nchars, CHAR_TO_BYTE (start + nchars), Qt);
  5895             }
  5896           else
  5897             {
  5898               coding->dst_object = Qnil;
  5899               coding->dst_pos_byte = start_byte;
  5900               if (start >= GPT || end <= GPT)
  5901                 {
  5902                   coding->consumed_char = end - start;
  5903                   coding->produced = end_byte - start_byte;
  5904                 }
  5905               else
  5906                 {
  5907                   coding->consumed_char = GPT - start;
  5908                   coding->produced = GPT_BYTE - start_byte;
  5909                 }
  5910             }
  5911         }
  5912 
  5913       if (coding->produced > 0)
  5914         {
  5915           char *buf = (coding->raw_destination ? (char *) coding->destination
  5916                        : (STRINGP (coding->dst_object)
  5917                           ? SSDATA (coding->dst_object)
  5918                           : (char *) BYTE_POS_ADDR (coding->dst_pos_byte)));
  5919           coding->produced -= emacs_write_quit (desc, buf, coding->produced);
  5920 
  5921           if (coding->raw_destination)
  5922             {
  5923               /* We're responsible for freeing this, see
  5924                  encode_coding_object to check why.  */
  5925               xfree (coding->destination);
  5926               coding->raw_destination = 0;
  5927             }
  5928           if (coding->produced)
  5929             return 0;
  5930         }
  5931       start += coding->consumed_char;
  5932     }
  5933 
  5934   return 1;
  5935 }

  6164   if (NILP (no_message))

  6181 
  6182       listfile = Fexpand_file_name (Vauto_save_list_file_name, Qnil);
  6183 
  6184       /* Don't try to create the directory when shutting down Emacs,
  6185          because creating the directory might signal an error, and

  6244           continue;
  6245

  6247            and file changed since last auto save
  6248            and file changed since last real save.  */
  6249         if (STRINGP (BVAR (b, auto_save_file_name))
  6250             && BUF_SAVE_MODIFF (b) < BUF_MODIFF (b)
  6251             && BUF_AUTOSAVE_MODIFF (b) < BUF_MODIFF (b)
  6252             /* -1 means we've turned off autosaving for a while--see below.  */
  6253             && FIXNUMP (BVAR (b, save_length))
  6254             && XFIXNUM (BVAR (b, save_length)) >= 0

  6264                 && before_time.tv_sec - b->auto_save_failure_time < 1200)
  6265               continue;

  6267             enum { growth_factor = 4 };
  6268             verify (BUF_BYTES_MAX <= EMACS_INT_MAX / growth_factor);
  6269 
  6270             set_buffer_internal (b);
  6271             if (NILP (Vauto_save_include_big_deletions)
  6272                 && FIXNUMP (BVAR (b, save_length))
  6273                 /* A short file is likely to change a large fraction;
  6274                    spare the user annoying messages.  */
  6275                 && XFIXNUM (BVAR (b, save_length)) > 5000
  6276                 && (growth_factor * (BUF_Z (b) - BUF_BEG (b))
  6277                     < (growth_factor - 1) * XFIXNUM (BVAR (b, save_length)))
  6278                 /* These messages are frequent and annoying for `*mail*'.  */

  6280                 && NILP (no_message))
  6281               {
  6282                 /* It has shrunk too much; turn off auto-saving here.  */
  6283                 minibuffer_auto_raise = orig_minibuffer_auto_raise;
  6284                 message_with_string ("Buffer %s has shrunk a lot; auto save disabled in that buffer until next real save",
  6285                                      BVAR (b, name), 1);
  6286                 minibuffer_auto_raise = 0;
  6287                 /* Turn off auto-saving until there's a real save,
  6288                    and prevent any more warnings.  */
  6289                 XSETINT (BVAR (b, save_length), -1);
  6290                 Fsleep_for (make_fixnum (1), Qnil);
  6291                 continue;
  6292               }
  6293             if (!auto_saved && NILP (no_message))
  6294               message1 ("Auto-saving...");
  6295             internal_condition_case (auto_save_1, Qt, auto_save_error);
  6296             auto_saved = 1;
  6297             BUF_AUTOSAVE_MODIFF (b) = BUF_MODIFF (b);
  6298             XSETFASTINT (BVAR (current_buffer, save_length), Z - BEG);
  6299             set_buffer_internal (old);
  6300 
  6301             after_time = current_timespec ();
  6302 
  6303             /* If auto-save took more than 60 seconds,
  6304                assume it was an NFS failure that got a timeout.  */
  6305             if (after_time.tv_sec - before_time.tv_sec > 60)
  6306               b->auto_save_failure_time = after_time.tv_sec;
  6307           }
  6308       }
  6309 
  6310   /* Prevent another auto save till enough input events come in.  */
  6311   record_auto_save ();
  6312 
  6313   if (auto_saved && NILP (no_message))
  6314     {
  6315       if (old_message_p)
  6316         {
  6317           /* If we are going to restore an old message,
  6318              give time to read ours.  */
  6319           sit_for (make_fixnum (1), 0, 0);
  6320           restore_message ();
  6321         }
  6322       else if (!auto_save_error_occurred)
  6323         /* Don't overwrite the error message if an error occurred.
  6324            If we displayed a message and then restored a state
  6325            with no message, leave a "done" message on the screen.  */
  6326         message1 ("Auto-saving...done");
  6327     }
  6328 
  6329   Vquit_flag = oquit;
  6330 
  6331   /* This restores the message-stack status.  */
  6332   return unbind_to (count, Qnil);
  6333 }
