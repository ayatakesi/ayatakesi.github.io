  1812 INLINE unsigned long
  1813 x_make_truecolor_pixel (struct x_display_info *dpyinfo, int r, int g, int b)
  1814 {
  1815   unsigned long pr, pg, pb, pa = dpyinfo->alpha_mask;
  1816 
  1817   /* Scale down RGB values to the visual's bits per RGB, and shift
  1818      them to the right position in the pixel color.  Note that the
  1819      original RGB values are 16-bit values, as usual in X.  */
  1820   pr = (r >> (16 - dpyinfo->red_bits))   << dpyinfo->red_offset;
  1821   pg = (g >> (16 - dpyinfo->green_bits)) << dpyinfo->green_offset;
  1822   pb = (b >> (16 - dpyinfo->blue_bits))  << dpyinfo->blue_offset;
  1823 
  1824   /* Assemble the pixel color.  */
  1825   return pr | pg | pb | pa;
  1826 }
