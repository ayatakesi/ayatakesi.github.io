   139 struct android_battery_state
   140 {
   141   /* Battery charge level in integer percentage.  */
   142   intmax_t capacity;
   143 
   144   /* Battery charge level in microampere-hours.  */
   145   intmax_t charge_counter;
   146 
   147   /* Battery current in microampere-hours.  */
   148   intmax_t current_average;
   149 
   150   /* Instantaneous battery current in microampere-hours.  */
   151   intmax_t current_now;
   152 
   153   /* Estimate as to the amount of time remaining until the battery is
   154      charged, in milliseconds.  */
   155   intmax_t remaining;
   156 
   157   /* Battery status.  The value is either:
   158 
   159        2, if the battery is charging.
   160        3, if the battery is discharging.
   161        5, if the battery is full.
   162        4, if the battery is not full or discharging,
   163           but is not charging either.
   164        1, if the battery state is unknown.  */
   165   int status;
   166 
   167   /* The power source of the battery.  Value is:
   168 
   169        0, if on battery power.
   170        1, for line power.
   171        8, for dock power.
   172        2, for USB power.
   173        4, for wireless power.  */
   174   int plugged;
   175 
   176   /* The temperature of the battery in 10 * degrees centigrade.  */
   177   int temperature;
   178 };
