    74 static struct frame *
    75 android_window_to_frame (struct android_display_info *dpyinfo,
    76                          android_window wdesc)
    77 {
    78   Lisp_Object tail, frame;
    79   struct frame *f;
    80 
    81   if (wdesc == ANDROID_NONE)
    82     return NULL;
    83 
    84   FOR_EACH_FRAME (tail, frame)
    85     {
    86       f = XFRAME (frame);
    87 
    88       if (!FRAME_ANDROID_P (f))
    89         continue;
    90 
    91       if (FRAME_ANDROID_WINDOW (f) == wdesc)
    92         return f;
    93     }
    94 
    95   return NULL;
    96 }

   146 static void
   147 android_flash (struct frame *f)
   148 {
   149   struct android_gc *gc;
   150   struct android_gc_values values;
   151   int rc;
   152   fd_set fds;
   153 
   154   block_input ();
   155 
   156   values.function = ANDROID_GC_XOR;
   157   values.foreground = (FRAME_FOREGROUND_PIXEL (f)
   158                        ^ FRAME_BACKGROUND_PIXEL (f));
   159 
   160   gc = android_create_gc ((ANDROID_GC_FUNCTION
   161                            | ANDROID_GC_FOREGROUND),
   162                           &values);
   163 
   164   /* Get the height not including a menu bar widget.  */
   165   int height = FRAME_PIXEL_HEIGHT (f);
   166   /* Height of each line to flash.  */
   167   int flash_height = FRAME_LINE_HEIGHT (f);
   168   /* These will be the left and right margins of the rectangles.  */
   169   int flash_left = FRAME_INTERNAL_BORDER_WIDTH (f);
   170   int flash_right = FRAME_PIXEL_WIDTH (f) - FRAME_INTERNAL_BORDER_WIDTH (f);
   171   int width = flash_right - flash_left;
   172 
   173   /* If window is tall, flash top and bottom line.  */
   174   if (height > 3 * FRAME_LINE_HEIGHT (f))
   175     {
   176       android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc,
   177                               flash_left,
   178                               (FRAME_INTERNAL_BORDER_WIDTH (f)
   179                                + FRAME_TOP_MARGIN_HEIGHT (f)),
   180                               width, flash_height);
   181       android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc,
   182                               flash_left,
   183                               (height - flash_height
   184                                - FRAME_INTERNAL_BORDER_WIDTH (f)
   185                                - FRAME_BOTTOM_MARGIN_HEIGHT (f)),
   186                               width, flash_height);
   187 
   188     }
   189   else
   190     /* If it is short, flash it all.  */
   191     android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc,
   192                             flash_left, FRAME_INTERNAL_BORDER_WIDTH (f),
   193                             width, (height - 2
   194                                     * FRAME_INTERNAL_BORDER_WIDTH (f)));
   195 
   196   flush_frame (f);
   197 
   198   struct timespec delay = make_timespec (0, 150 * 1000 * 1000);
   199   struct timespec wakeup = timespec_add (current_timespec (), delay);
   200 
   201   /* Keep waiting until past the time wakeup or any input gets
   202      available.  */
   203   while (! detect_input_pending ())
   204     {
   205       struct timespec current = current_timespec ();
   206       struct timespec timeout;
   207 
   208       /* Break if result would not be positive.  */
   209       if (timespec_cmp (wakeup, current) <= 0)
   210         break;
   211 
   212       /* How long `select' should wait.  */
   213       timeout = make_timespec (0, 10 * 1000 * 1000);
   214 
   215       /* Wait for some input to become available on the X
   216          connection.  */
   217       FD_ZERO (&fds);
   218 
   219       /* Try to wait that long--but we might wake up sooner.  */
   220       rc = pselect (0, &fds, NULL, NULL, &timeout, NULL);
   221 
   222       /* Some input is available, exit the visible bell.  */
   223       if (rc >= 0)
   224         break;
   225     }
   226 
   227   /* If window is tall, flash top and bottom line.  */
   228   if (height > 3 * FRAME_LINE_HEIGHT (f))
   229     {
   230       android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc,
   231                               flash_left,
   232                               (FRAME_INTERNAL_BORDER_WIDTH (f)
   233                                + FRAME_TOP_MARGIN_HEIGHT (f)),
   234                               width, flash_height);
   235       android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc,
   236                               flash_left,
   237                               (height - flash_height
   238                                - FRAME_INTERNAL_BORDER_WIDTH (f)
   239                                - FRAME_BOTTOM_MARGIN_HEIGHT (f)),
   240                               width, flash_height);
   241     }
   242   else
   243     /* If it is short, flash it all.  */
   244     android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc,
   245                             flash_left, FRAME_INTERNAL_BORDER_WIDTH (f),
   246                             width, (height - 2
   247                                     * FRAME_INTERNAL_BORDER_WIDTH (f)));
   248 
   249   android_free_gc (gc);
   250   flush_frame (f);
   251 
   252   unblock_input ();
   253 }

   421 static void
   422 android_focus_changed (int type, int state,
   423                        struct android_display_info *dpyinfo,
   424                        struct frame *frame, struct input_event *bufp)
   425 {
   426   if (type == ANDROID_FOCUS_IN)
   427     {
   428       if (dpyinfo->x_focus_event_frame != frame)
   429         {
   430           android_new_focus_frame (dpyinfo, frame);
   431           dpyinfo->x_focus_event_frame = frame;
   432           bufp->kind = FOCUS_IN_EVENT;
   433           XSETFRAME (bufp->frame_or_window, frame);
   434         }
   435 
   436       frame->output_data.android->focus_state |= state;
   437     }
   438   else if (type == ANDROID_FOCUS_OUT)
   439     {
   440       frame->output_data.android->focus_state &= ~state;
   441 
   442       if (dpyinfo->x_focus_event_frame == frame)
   443         {
   444           dpyinfo->x_focus_event_frame = 0;
   445           android_new_focus_frame (dpyinfo, 0);
   446 
   447           bufp->kind = FOCUS_OUT_EVENT;
   448           XSETFRAME (bufp->frame_or_window, frame);
   449         }
   450 
   451       if (frame->pointer_invisible)
   452         android_toggle_invisible_pointer (frame, false);
   453     }
   454 }

   478 static bool
   479 android_note_mouse_movement (struct frame *frame,
   480                              struct android_motion_event *event)
   481 {
   482   struct android_display_info *dpyinfo;
   483   Emacs_Rectangle *r;
   484 
   485   if (!FRAME_ANDROID_OUTPUT (frame))
   486     return false;
   487 
   488   dpyinfo = FRAME_DISPLAY_INFO (frame);
   489   dpyinfo->last_mouse_motion_frame = frame;
   490   dpyinfo->last_mouse_motion_x = event->x;
   491   dpyinfo->last_mouse_motion_y = event->y;
   492   dpyinfo->last_mouse_movement_time = event->time;
   493 
   494   /* Has the mouse moved off the glyph it was on at the last sighting?  */
   495   r = &dpyinfo->last_mouse_glyph;
   496   if (frame != dpyinfo->last_mouse_glyph_frame
   497       || event->x < r->x || event->x >= r->x + r->width
   498       || event->y < r->y || event->y >= r->y + r->height)
   499     {
   500       frame->mouse_moved = true;
   501       note_mouse_highlight (frame, event->x, event->y);
   502       /* Remember which glyph we're now on.  */
   503       remember_mouse_glyph (frame, event->x, event->y, r);
   504       dpyinfo->last_mouse_glyph_frame = frame;
   505       return true;
   506     }
   507 
   508   return false;
   509 }

   512 mouse_or_wdesc_frame (struct android_display_info *dpyinfo, int wdesc)

   531         return w_f;

   543   dpyinfo = FRAME_DISPLAY_INFO (f);
   544   x = event->x;
   545   y = event->y;
   546 
   547   /* Make the event type NO_EVENT; we'll change that when we decide
   548      otherwise.  */
   549   result->kind = MOUSE_CLICK_EVENT;
   550   result->code = event->button - 1;
   551   result->timestamp = event->time;
   552   result->modifiers = (android_android_to_emacs_modifiers (dpyinfo,
   553                                                            event->state)
   554                        | (event->type == ANDROID_BUTTON_RELEASE
   555                           ? up_modifier : down_modifier));
   556 
   557   XSETINT (result->x, x);
   558   XSETINT (result->y, y);
   559   XSETFRAME (result->frame_or_window, f);
   560   result->arg = Qnil;
   561   return Qnil;
   562 }

   611 /* Decode STRING, an array of N little endian UTF-16 characters, into
   612    a Lisp string.  Return Qnil if the string is too large, and the
   613    encoded string otherwise.  */
   614 
   615 static Lisp_Object
   616 android_decode_utf16 (unsigned short *utf16, size_t n)
   617 {
   618   struct coding_system coding;
   619   ptrdiff_t size;
   620 
   621   if (INT_MULTIPLY_WRAPV (n, sizeof *utf16, &size))
   622     return Qnil;
   623 
   624   /* Set up the coding system.  Decoding a UTF-16 string (with no BOM)
   625      should not signal.  */
   626 
   627   memset (&coding, 0, sizeof coding);
   628 
   629   setup_coding_system (Qutf_16le, &coding);
   630   coding.source = (const unsigned char *) utf16;
   631   decode_coding_object (&coding, Qnil, 0, 0, size,
   632                         size, Qt);
   633 
   634   return coding.dst_object;
   635 }

   637 /* Handle a cursor update request for F from the input method.
   638    MODE specifies whether or not an update should be sent immediately,
   639    and whether or not they are needed in the future.
   640 
   641    If MODE & ANDROID_CURSOR_UPDATE_IMMEDIATE, report the position of
   642    F's old selected window's phys cursor now.
   643 
   644    If MODE & ANDROID_CURSOR_UPDATE_MONITOR, set
   645    `need_cursor_updates'.  */
   646 
   647 static void
   648 android_request_cursor_updates (struct frame *f, int mode)
   649 {
   650   struct window *w;
   651 
   652   if (mode & ANDROID_CURSOR_UPDATE_IMMEDIATE
   653       && WINDOWP (WINDOW_LIVE_P (f->old_selected_window)
   654                   ? f->old_selected_window
   655                   : f->selected_window))
   656     {
   657       /* Prefer the old selected window, as its selection is what was
   658          reported to the IME previously.  */
   659 
   660       w = XWINDOW (WINDOW_LIVE_P (f->old_selected_window)
   661                    ? f->old_selected_window
   662                    : f->selected_window);
   663       android_set_preeditarea (w, w->cursor.x, w->cursor.y);
   664     }
   665 
   666   /* Now say whether or not updates are needed in the future.  */
   667   FRAME_OUTPUT_DATA (f)->need_cursor_updates
   668     = (mode & ANDROID_CURSOR_UPDATE_MONITOR);
   669 }

   671 /* Handle a single input method event EVENT, delivered to the frame
   672    F.
   673 
   674    Perform the text conversion action specified inside.  */
   675 
   676 static void
   677 android_handle_ime_event (union android_event *event, struct frame *f)
   678 {
   679   Lisp_Object text UNINIT;
   680   struct android_output *output;
   681 
   682   /* First, decode the text if necessary.  */
   683 
   684   switch (event->ime.operation)
   685     {
   686     case ANDROID_IME_COMMIT_TEXT:
   687     case ANDROID_IME_SET_COMPOSING_TEXT:
   688       text = android_decode_utf16 (event->ime.text,
   689                                    event->ime.length);
   690       xfree (event->ime.text);
   691       break;
   692 
   693     default:
   694       break;
   695     }
   696 
   697   /* Finally, perform the appropriate conversion action.  */
   698 
   699   switch (event->ime.operation)
   700     {
   701     case ANDROID_IME_COMMIT_TEXT:
   702       commit_text (f, text, event->ime.position,
   703                    event->ime.counter);
   704       break;
   705 
   706     case ANDROID_IME_DELETE_SURROUNDING_TEXT:
   707       delete_surrounding_text (f, event->ime.start,
   708                                event->ime.end,
   709                                event->ime.counter);
   710       break;
   711 
   712     case ANDROID_IME_FINISH_COMPOSING_TEXT:
   713 
   714       if (event->ime.length == 2)
   715         {
   716           output = FRAME_ANDROID_OUTPUT (f);
   717 
   718           /* A new input method has connected to Emacs.  Stop
   719              reporting changes that the previous input method has
   720              asked to monitor.  */
   721 
   722           output->extracted_text_flags = 0;
   723           output->extracted_text_token = 0;
   724           output->extracted_text_hint = 0;
   725           output->need_cursor_updates = false;
   726         }
   727 
   728       finish_composing_text (f, event->ime.counter,
   729                              event->ime.length == 1);
   730 
   731       if (event->ime.length == 2)
   732         {
   733           /* Now cancel outstanding batch edits if a new input method
   734              has connected.  */
   735 
   736           f->conversion.batch_edit_flags = 0;
   737           f->conversion.batch_edit_count = 0;
   738         }
   739 
   740       break;
   741 
   742     case ANDROID_IME_SET_COMPOSING_TEXT:
   743       set_composing_text (f, text, event->ime.position,
   744                           event->ime.counter);
   745       break;
   746 
   747     case ANDROID_IME_SET_COMPOSING_REGION:
   748       set_composing_region (f, event->ime.start,
   749                             event->ime.end,
   750                             event->ime.counter);
   751       break;
   752 
   753     case ANDROID_IME_SET_POINT:
   754       textconv_set_point_and_mark (f, event->ime.start,
   755                                    event->ime.end,
   756                                    event->ime.counter);
   757       break;
   758 
   759     case ANDROID_IME_START_BATCH_EDIT:
   760       start_batch_edit (f, event->ime.counter);
   761       break;
   762 
   763     case ANDROID_IME_END_BATCH_EDIT:
   764       end_batch_edit (f, event->ime.counter);
   765       break;
   766 
   767     case ANDROID_IME_REQUEST_SELECTION_UPDATE:
   768       request_point_update (f, event->ime.counter);
   769       break;
   770 
   771     case ANDROID_IME_REQUEST_CURSOR_UPDATES:
   772       android_request_cursor_updates (f, event->ime.length);
   773       break;
   774     }
   775 }

   777 static int
   778 handle_one_android_event (struct android_display_info *dpyinfo,
   779                           union android_event *event, int *finish,
   780                           struct input_event *hold_quit)
   781 {
   782   union android_event configureEvent;
   783   struct frame *f, *any, *mouse_frame;
   784   Mouse_HLInfo *hlinfo;
   785   union buffered_input_event inev;
   786   int modifiers, count, do_help;
   787   struct android_touch_point *touchpoint, **last;
   788   Lisp_Object window;
   789   int scroll_height;
   790   double scroll_unit;
   791   int keysym;
   792   ptrdiff_t nchars, i;
   793   struct window *w;
   794 
   795   /* It is okay for this to not resemble handle_one_xevent so much.
   796      Differences in event handling code are much less nasty than
   797      stuble differences in the graphics code.  */
   798 
   799   do_help = count = 0;
   800   hlinfo = &dpyinfo->mouse_highlight;
   801   *finish = ANDROID_EVENT_NORMAL;
   802   any = android_window_to_frame (dpyinfo, event->xany.window);
   803   nchars = 0;
   804 
   805   if (any && any->wait_event_type == event->type)
   806     any->wait_event_type = 0; /* Indicates we got it.  */
   807 
   808   EVENT_INIT (inev.ie);
   809 
   810   switch (event->type)
   811     {
   812     case ANDROID_CONFIGURE_NOTIFY:
   813       configureEvent = *event;
   814 
   815       f = android_window_to_frame (dpyinfo,
   816                                    configureEvent.xconfigure.window);
   817 
   818       if (!f)
   819         goto OTHER;
   820 
   821       if (FRAME_TOOLTIP_P (f))
   822         {
   823           if (FRAME_PIXEL_HEIGHT (f) != configureEvent.xconfigure.height
   824               || FRAME_PIXEL_WIDTH (f) != configureEvent.xconfigure.width)
   825             SET_FRAME_GARBAGED (f);
   826 
   827           FRAME_PIXEL_HEIGHT (f) = configureEvent.xconfigure.height;
   828           FRAME_PIXEL_WIDTH (f) = configureEvent.xconfigure.width;
   829         }
   830 
   831       int width = configureEvent.xconfigure.width;
   832       int height = configureEvent.xconfigure.height;
   833 
   834       if (CONSP (frame_size_history))
   835         frame_size_history_extra (f, build_string ("ConfigureNotify"),
   836                                   FRAME_PIXEL_WIDTH (f),
   837                                   FRAME_PIXEL_HEIGHT (f),
   838                                   width, height, f->new_width,
   839                                   f->new_height);
   840 
   841       /* Even if the number of character rows and columns has
   842          not changed, the font size may have changed, so we need
   843          to check the pixel dimensions as well.  */
   844 
   845       if (width != FRAME_PIXEL_WIDTH (f)
   846           || height != FRAME_PIXEL_HEIGHT (f)
   847           || (f->new_size_p
   848               && ((f->new_width >= 0 && width != f->new_width)
   849                   || (f->new_height >= 0 && height != f->new_height))))
   850         {
   851           change_frame_size (f, width, height, false, true, false);
   852           android_clear_under_internal_border (f);
   853           SET_FRAME_GARBAGED (f);
   854           cancel_mouse_face (f);
   855         }
   856 
   857       /* Now change the left and top position of this window.  */
   858 
   859       {
   860         int old_left = f->left_pos;
   861         int old_top = f->top_pos;
   862         Lisp_Object frame;
   863 
   864         XSETFRAME (frame, f);
   865 
   866         {
   867           android_window root;
   868           unsigned int dummy_uint;
   869 
   870           android_get_geometry (FRAME_ANDROID_WINDOW (f),
   871                                 &root, &f->left_pos, &f->top_pos,
   872                                 &dummy_uint, &dummy_uint,
   873                                 &dummy_uint);
   874         }
   875 
   876         if (!FRAME_TOOLTIP_P (f)
   877             && (old_left != f->left_pos || old_top != f->top_pos))
   878           {
   879             inev.ie.kind = MOVE_FRAME_EVENT;
   880             XSETFRAME (inev.ie.frame_or_window, f);
   881           }
   882 
   883       if (f && FRAME_OUTPUT_DATA (f)->need_cursor_updates)
   884         {
   885           w = XWINDOW (f->selected_window);
   886           android_set_preeditarea (w, w->cursor.x, w->cursor.y);
   887         }
   888       }
   889 
   890       goto OTHER;
   891 
   892     case ANDROID_KEY_PRESS:
   893 
   894       /* Set f to any.  There are no ``outer windows'' on Android.  */
   895       f = any;
   896 
   897       /* If mouse-highlight is an integer, input clears out
   898          mouse highlighting.  */
   899       if (!hlinfo->mouse_face_hidden && FIXNUMP (Vmouse_highlight)
   900           && (any == 0
   901               || !EQ (any->tool_bar_window, hlinfo->mouse_face_window)
   902               || !EQ (any->tab_bar_window, hlinfo->mouse_face_window)))
   903         {
   904           mouse_frame = hlinfo->mouse_face_mouse_frame;
   905 
   906           clear_mouse_face (hlinfo);
   907           hlinfo->mouse_face_hidden = true;
   908 
   909           if (mouse_frame)
   910             android_flush_dirty_back_buffer_on (mouse_frame);
   911         }
   912 
   913       if (!f)
   914         goto OTHER;
   915 
   916       if (event->xkey.counter)
   917         /* This event was generated by `performEditorAction'.  Make
   918            sure it is processed before any subsequent edits.  */
   919         textconv_barrier (f, event->xkey.counter);
   920 
   921       wchar_t copy_buffer[129];
   922       wchar_t *copy_bufptr = copy_buffer;
   923       int copy_bufsiz = 128 * sizeof (wchar_t);
   924 
   925       event->xkey.state
   926         |= android_emacs_to_android_modifiers (dpyinfo,
   927                                                extra_keyboard_modifiers);
   928       modifiers = event->xkey.state;
   929 
   930       /* Common for all keysym input events.  */
   931       XSETFRAME (inev.ie.frame_or_window, any);
   932       inev.ie.modifiers
   933         = android_android_to_emacs_modifiers (dpyinfo, modifiers);
   934       inev.ie.timestamp = event->xkey.time;
   935 
   936       keysym = event->xkey.keycode;
   937 
   938       {
   939         enum android_lookup_status status_return;
   940 
   941         nchars = android_wc_lookup_string (&event->xkey, copy_bufptr,
   942                                            copy_bufsiz, &keysym,
   943                                            &status_return);
   944 
   945         /* android_lookup_string can't be called twice, so there's no
   946            way to recover from buffer overflow.  */
   947         if (status_return == ANDROID_BUFFER_OVERFLOW)
   948           goto done_keysym;
   949         else if (status_return == ANDROID_LOOKUP_NONE)
   950           {
   951             /* Don't skip preedit text events.  */
   952             if (event->xkey.keycode != (uint32_t) -1)
   953               goto done_keysym;
   954           }
   955         else if (status_return == ANDROID_LOOKUP_CHARS)
   956           keysym = ANDROID_NO_SYMBOL;
   957         else if (status_return != ANDROID_LOOKUP_KEYSYM
   958                  && status_return != ANDROID_LOOKUP_BOTH)
   959           emacs_abort ();
   960 
   961         /* Deal with pre-edit text events.  On Android, these are
   962            simply encoded as events with associated strings and a
   963            keycode set to ``-1''.  */
   964 
   965         if (event->xkey.keycode == (uint32_t) -1)
   966           {
   967             inev.ie.kind = PREEDIT_TEXT_EVENT;
   968             inev.ie.arg = Qnil;
   969 
   970             /* If text was looked up, decode it and make it the
   971                preedit text.  */
   972 
   973             if (status_return == ANDROID_LOOKUP_CHARS && nchars)
   974               {
   975                 copy_bufptr[nchars] = 0;
   976                 inev.ie.arg = from_unicode_buffer (copy_bufptr);
   977               }
   978 
   979             goto done_keysym;
   980           }
   981       }
   982 
   983       if (nchars == 1 && copy_bufptr[0] >= 32)
   984         {
   985           /* Deal with characters.  */
   986 
   987           if (copy_bufptr[0] < 128)
   988             inev.ie.kind = ASCII_KEYSTROKE_EVENT;
   989           else
   990             inev.ie.kind = MULTIBYTE_CHAR_KEYSTROKE_EVENT;
   991 
   992           inev.ie.code = copy_bufptr[0];
   993         }
   994       else if (nchars < 2 && keysym)
   995         {
   996           /* If the key is a modifier key, just return.  */
   997           if (ANDROID_IS_MODIFIER_KEY (keysym))
   998             goto done_keysym;
   999 
  1000           /* Next, deal with special ``characters'' by giving the
  1001              keycode to keyboard.c.  */
  1002           inev.ie.kind = NON_ASCII_KEYSTROKE_EVENT;
  1003           inev.ie.code = keysym;
  1004         }
  1005       else
  1006         {
  1007           /* Finally, deal with strings.  */
  1008 
  1009           for (i = 0; i < nchars; ++i)
  1010             {
  1011               inev.ie.kind = (SINGLE_BYTE_CHAR_P (copy_bufptr[i])
  1012                               ? ASCII_KEYSTROKE_EVENT
  1013                               : MULTIBYTE_CHAR_KEYSTROKE_EVENT);
  1014               inev.ie.code = copy_bufptr[i];
  1015 
  1016               /* If the character is actually '\n', then change this
  1017                  to RET.  */
  1018 
  1019               if (copy_bufptr[i] == '\n')
  1020                 {
  1021                   inev.ie.kind = NON_ASCII_KEYSTROKE_EVENT;
  1022                   inev.ie.code = 66;
  1023                 }
  1024 
  1025               kbd_buffer_store_buffered_event (&inev, hold_quit);
  1026             }
  1027 
  1028           count += nchars;
  1029           inev.ie.kind = NO_EVENT;  /* Already stored above.  */
  1030         }
  1031 
  1032       goto done_keysym;
  1033 
  1034     done_keysym:
  1035 
  1036       /* Now proceed to tell the input method the current position of
  1037          the cursor, if required.  */
  1038 
  1039       if (f && FRAME_OUTPUT_DATA (f)->need_cursor_updates)
  1040         {
  1041           w = XWINDOW (f->selected_window);
  1042           android_set_preeditarea (w, w->cursor.x, w->cursor.y);
  1043         }
  1044 
  1045       goto OTHER;
  1046 
  1047     case ANDROID_FOCUS_IN:
  1048     case ANDROID_FOCUS_OUT:
  1049       android_detect_focus_change (dpyinfo, any, event, &inev.ie);
  1050       goto OTHER;
  1051 
  1052     case ANDROID_WINDOW_ACTION:
  1053 
  1054       /* This is a special event sent by android_run_in_emacs_thread
  1055          used to make Android run stuff.  */
  1056 
  1057       if (!event->xaction.window && !event->xaction.action)
  1058         /* Don't run queries here, as it may run inside editor
  1059            commands, which can expose an inconsistent view of buffer
  1060            contents to the input method during command execution.
  1061 
  1062            Instead, wait for Emacs to return to `android_select'.  */
  1063         goto OTHER;
  1064 
  1065       f = any;
  1066 
  1067       if (event->xaction.action == 0)
  1068         {
  1069           /* Action 0 either means that a window has been destroyed
  1070              and its associated frame should be as well.  */
  1071 
  1072           if (event->xaction.window)
  1073             {
  1074               if (!f)
  1075                 goto OTHER;
  1076 
  1077               inev.ie.kind = DELETE_WINDOW_EVENT;
  1078               XSETFRAME (inev.ie.frame_or_window, f);
  1079             }
  1080         }
  1081 
  1082     case ANDROID_ENTER_NOTIFY:
  1083       f = any;
  1084 
  1085       if (f)
  1086         android_note_mouse_movement (f, &event->xmotion);
  1087       goto OTHER;
  1088 
  1089     case ANDROID_MOTION_NOTIFY:
  1090 
  1091       previous_help_echo_string = help_echo_string;
  1092       help_echo_string = Qnil;
  1093 
  1094       if (hlinfo->mouse_face_hidden)
  1095         {
  1096           hlinfo->mouse_face_hidden = false;
  1097           clear_mouse_face (hlinfo);
  1098         }
  1099 
  1100       f = any;
  1101 
  1102       if (f)
  1103         {
  1104           /* Maybe generate a SELECT_WINDOW_EVENT for
  1105              `mouse-autoselect-window' but don't let popup menus
  1106              interfere with this (Bug#1261).  */
  1107           if (!NILP (Vmouse_autoselect_window)
  1108               && !popup_activated ()
  1109               /* Don't switch if we're currently in the minibuffer.
  1110                  This tries to work around problems where the
  1111                  minibuffer gets unselected unexpectedly, and where
  1112                  you then have to move your mouse all the way down to
  1113                  the minibuffer to select it.  */
  1114               && !MINI_WINDOW_P (XWINDOW (selected_window))
  1115               /* With `focus-follows-mouse' non-nil create an event
  1116                  also when the target window is on another frame.  */
  1117               && (f == XFRAME (selected_frame)
  1118                   || !NILP (focus_follows_mouse)))
  1119             {
  1120               static Lisp_Object last_mouse_window;
  1121               Lisp_Object window
  1122                 = window_from_coordinates (f, event->xmotion.x,
  1123                                            event->xmotion.y, 0,
  1124                                            false, false);
  1125 
  1126               /* A window will be autoselected only when it is not
  1127                  selected now and the last mouse movement event was
  1128                  not in it.  The remainder of the code is a bit vague
  1129                  wrt what a "window" is.  For immediate autoselection,
  1130                  the window is usually the entire window but for GTK
  1131                  where the scroll bars don't count.  For delayed
  1132                  autoselection the window is usually the window's text
  1133                  area including the margins.  */
  1134               if (WINDOWP (window)
  1135                   && !EQ (window, last_mouse_window)
  1136                   && !EQ (window, selected_window))
  1137                 {
  1138                   inev.ie.kind = SELECT_WINDOW_EVENT;
  1139                   inev.ie.frame_or_window = window;
  1140                 }
  1141 
  1142               /* Remember the last window where we saw the mouse.  */
  1143               last_mouse_window = window;
  1144             }
  1145 
  1146           if (!android_note_mouse_movement (f, &event->xmotion))
  1147             help_echo_string = previous_help_echo_string;
  1148         }
  1149 
  1150       /* If the contents of the global variable help_echo_string
  1151          has changed, generate a HELP_EVENT.  */
  1152       if (!NILP (help_echo_string)
  1153           || !NILP (previous_help_echo_string))
  1154         do_help = 1;
  1155 
  1156       if (f)
  1157         android_flush_dirty_back_buffer_on (f);
  1158 
  1159       goto OTHER;
  1160 
  1161     case ANDROID_LEAVE_NOTIFY:
  1162       f = any;
  1163 
  1164       if (f)
  1165         {
  1166           /* Now clear dpyinfo->last_mouse_motion_frame, or
  1167              gui_redo_mouse_highlight will end up highlighting the
  1168              last known position of the mouse if a tooltip frame is
  1169              later unmapped.  */
  1170 
  1171           if (f == dpyinfo->last_mouse_motion_frame)
  1172             dpyinfo->last_mouse_motion_frame = NULL;
  1173 
  1174           /* Something similar applies to

  1204     case ANDROID_EXPOSE:

  1206       f = any;

  1217             {

  1221             }
  1222         }
  1223 
  1224       goto OTHER;
  1225 
  1226     case ANDROID_BUTTON_PRESS:
  1227     case ANDROID_BUTTON_RELEASE:
  1228       /* If we decide we want to generate an event to be seen
  1229          by the rest of Emacs, we put it here.  */
  1230 
  1231       f = any;
  1232 
  1233       Lisp_Object tab_bar_arg = Qnil;
  1234       bool tab_bar_p = false;
  1235       bool tool_bar_p = false;
  1236 
  1237       dpyinfo->last_mouse_glyph_frame = NULL;
  1238 
  1239       f = mouse_or_wdesc_frame (dpyinfo, event->xbutton.window);
  1240 
  1241       if (f && event->xbutton.type == ANDROID_BUTTON_PRESS
  1242           && !popup_activated ()
  1243           /* && !x_window_to_scroll_bar (event->xbutton.display, */
  1244           /*                          event->xbutton.window, 2) */
  1245           && !FRAME_NO_ACCEPT_FOCUS (f))

  1254             {
  1255               android_set_input_focus (FRAME_ANDROID_WINDOW (f),
  1256                                        event->xbutton.time);
  1257 
  1258               if (FRAME_PARENT_FRAME (f))
  1259                 android_raise_window (FRAME_ANDROID_WINDOW (f));
  1260             }
  1261         }
  1262

  1264         {
  1265           /* Is this in the tab-bar?  */
  1266           if (WINDOWP (f->tab_bar_window)
  1267               && WINDOW_TOTAL_LINES (XWINDOW (f->tab_bar_window)))
  1268             {
  1269               Lisp_Object window;
  1270               int x = event->xbutton.x;
  1271               int y = event->xbutton.y;
  1272 
  1273               window = window_from_coordinates (f, x, y, 0, true, true);
  1274               tab_bar_p = EQ (window, f->tab_bar_window);
  1275 
  1276               if (tab_bar_p)
  1277                 {
  1278                   tab_bar_arg = handle_tab_bar_click
  1279                     (f, x, y, (event->xbutton.type
  1280                                == ANDROID_BUTTON_PRESS),
  1281                      android_android_to_emacs_modifiers (dpyinfo,
  1282                                                          event->xbutton.state));
  1283                   android_flush_dirty_back_buffer_on (f);
  1284                 }
  1285             }
  1286

  1288           if (WINDOWP (f->tool_bar_window)
  1289               && WINDOW_TOTAL_LINES (XWINDOW (f->tool_bar_window)))
  1290             {
  1291               Lisp_Object window;
  1292               int x = event->xbutton.x;
  1293               int y = event->xbutton.y;
  1294 
  1295               window = window_from_coordinates (f, x, y, 0, true, true);
  1296               tool_bar_p = (EQ (window, f->tool_bar_window)
  1297                             && ((event->xbutton.type
  1298                                  != ANDROID_BUTTON_RELEASE)
  1299                                 || f->last_tool_bar_item != -1));
  1300 
  1301               if (tool_bar_p && event->xbutton.button < 4)
  1302                 {
  1303                   handle_tool_bar_click
  1304                     (f, x, y, (event->xbutton.type
  1305                                == ANDROID_BUTTON_PRESS),
  1306                      android_android_to_emacs_modifiers (dpyinfo,
  1307                                                          event->xbutton.state));
  1308                   android_flush_dirty_back_buffer_on (f);
  1309                 }
  1310             }
  1311 
  1312           if (!(tab_bar_p && NILP (tab_bar_arg)) && !tool_bar_p)
  1313             {
  1314               android_construct_mouse_click (&inev.ie, &event->xbutton, f);
  1315 
  1316               if (!NILP (tab_bar_arg))
  1317                 inev.ie.arg = tab_bar_arg;
  1318             }
  1319         }
  1320 
  1321       if (event->type == ANDROID_BUTTON_PRESS)
  1322         {
  1323           dpyinfo->grabbed |= (1 << event->xbutton.button);
  1324           dpyinfo->last_mouse_frame = f;
  1325           if (f && !tab_bar_p)
  1326             f->last_tab_bar_item = -1;
  1327           if (f && !tool_bar_p)
  1328             f->last_tool_bar_item = -1;
  1329         }
  1330       else
  1331         dpyinfo->grabbed &= ~(1 << event->xbutton.button);
  1332 
  1333       /* Ignore any mouse motion that happened before this event;
  1334          any subsequent mouse-movement Emacs events should reflect
  1335          only motion after the ButtonPress/Release.  */
  1336       if (f != 0)
  1337         f->mouse_moved = false;
  1338 
  1339       goto OTHER;
  1340 
  1341       /* Touch events.  The events here don't parallel X so much.  */
  1342     case ANDROID_TOUCH_DOWN:
  1343 
  1344       if (!any)
  1345         goto OTHER;
  1346 
  1347       /* This event is sent when a tool is put on the screen.  X and Y
  1348          are the location of the finger, and pointer_id identifies the
  1349          tool for as long as it is still held down.  First, see if the
  1350          touch point already exists and can be reused (this shouldn't
  1351          happen, but be safe.)  */
  1352 
  1353       touchpoint = android_find_tool (any, event->touch.pointer_id);
  1354 
  1355       if (touchpoint)
  1356         {
  1357           /* Simply update the tool position and send an update.  */
  1358           touchpoint->x = event->touch.x;
  1359           touchpoint->y = event->touch.x;
  1360           android_update_tools (any, &inev.ie);
  1361           inev.ie.timestamp = event->touch.time;
  1362 
  1363           goto OTHER;
  1364         }
  1365 
  1366       /* Otherwise, link a new touchpoint onto the output's list of
  1367          pressed tools.  */
  1368 
  1369       touchpoint = xmalloc (sizeof *touchpoint);
  1370       touchpoint->tool_id = event->touch.pointer_id;
  1371       touchpoint->x = event->touch.x;
  1372       touchpoint->y = event->touch.x;
  1373       touchpoint->next = FRAME_OUTPUT_DATA (any)->touch_points;
  1374       touchpoint->tool_bar_p = false;
  1375       FRAME_OUTPUT_DATA (any)->touch_points = touchpoint;
  1376 
  1377       /* Figure out whether or not the tool was pressed on the tool
  1378          bar.  Note that the code which runs when it was is more or
  1379          less an abuse of the mouse highlight machinery, but it works
  1380          well enough in practice.  */
  1381 
  1382       if (WINDOWP (any->tool_bar_window)
  1383           && WINDOW_TOTAL_LINES (XWINDOW (any->tool_bar_window)))
  1384         {
  1385           Lisp_Object window;
  1386           int x = event->touch.x;
  1387           int y = event->touch.y;
  1388 
  1389           window = window_from_coordinates (any, x, y, 0, true,
  1390                                             true);
  1391 
  1392           /* If this touch has started in the tool bar, do not
  1393              send it to Lisp.  Instead, simulate a tool bar
  1394              click, releasing it once it goes away.  */
  1395 
  1396           if (EQ (window, any->tool_bar_window))
  1397             {
  1398               /* Call note_mouse_highlight on the tool bar
  1399                  item.  Otherwise, get_tool_bar_item will
  1400                  return 1.
  1401 
  1402                  This is not necessary when mouse-highlight is
  1403                  nil.  */

  1405               if (!NILP (Vmouse_highlight))
  1406                 {
  1407                   /* Clear the pointer invisible flag to always make
  1408                      note_mouse_highlight do its thing.  */
  1409                   any->pointer_invisible = false;
  1410                   note_mouse_highlight (any, x, y);
  1411 
  1412                   /* Always allow future mouse motion to
  1413                      update the mouse highlight, no matter
  1414                      where it is.  */
  1415                   memset (&dpyinfo->last_mouse_glyph, 0,
  1416                           sizeof dpyinfo->last_mouse_glyph);
  1417                   dpyinfo->last_mouse_glyph_frame = any;
  1418                 }
  1419 
  1420               handle_tool_bar_click (any, x, y, true, 0);
  1421 
  1422               /* Flush any changes made by that to the front
  1423                  buffer.  */
  1424               android_flush_dirty_back_buffer_on (any);
  1425 
  1426               /* Mark the touch point as being grabbed by the tool
  1427                  bar.  */
  1428               touchpoint->tool_bar_p = true;
  1429               goto OTHER;
  1430             }
  1431         }
  1432 
  1433       /* Now generate the Emacs event.  */
  1434       inev.ie.kind = TOUCHSCREEN_BEGIN_EVENT;
  1435       inev.ie.timestamp = event->touch.time;
  1436       XSETFRAME (inev.ie.frame_or_window, any);
  1437       XSETINT (inev.ie.x, event->touch.x);
  1438       XSETINT (inev.ie.y, event->touch.y);
  1439       XSETINT (inev.ie.arg, event->touch.pointer_id);
  1440 
  1441       goto OTHER;
  1442 
  1443     case ANDROID_TOUCH_MOVE:
  1444 
  1445       if (!any)
  1446         goto OTHER;
  1447 
  1448       /* Look for the tool that moved.  */
  1449 
  1450       touchpoint = android_find_tool (any, event->touch.pointer_id);
  1451 
  1452       /* If it doesn't exist or has been grabbed by the tool bar, skip
  1453          processing this event.  */
  1454 
  1455       if (!touchpoint || touchpoint->tool_bar_p)
  1456         goto OTHER;
  1457 
  1458       /* Otherwise, update the position and send the update event.  */
  1459 
  1460       touchpoint->x = event->touch.x;
  1461       touchpoint->y = event->touch.y;
  1462       android_update_tools (any, &inev.ie);
  1463       inev.ie.timestamp = event->touch.time;
  1464 
  1465       goto OTHER;
  1466 
  1467     case ANDROID_TOUCH_UP:
  1468 
  1469       if (!any)
  1470         goto OTHER;
  1471 
  1472       /* Now find and unlink the tool in question.  */
  1473 
  1474       last = &FRAME_OUTPUT_DATA (any)->touch_points;
  1475       while ((touchpoint = *last))
  1476         {
  1477           if (touchpoint->tool_id == event->touch.pointer_id)
  1478             {
  1479               *last = touchpoint->next;
  1480 
  1481               if (touchpoint->tool_bar_p)
  1482                 {
  1483                   xfree (touchpoint);
  1484 
  1485                   /* Do what is necessary to release the tool bar and
  1486                      possibly trigger a click.  */
  1487 
  1488                   if (any->last_tool_bar_item != -1)
  1489                     handle_tool_bar_click (any, event->touch.x,
  1490                                            event->touch.y, false,
  1491                                            0);
  1492 
  1493                   /* Cancel any outstanding mouse highlight.  */
  1494                   note_mouse_highlight (any, -1, -1);
  1495                   android_flush_dirty_back_buffer_on (any);
  1496 
  1497                   goto OTHER;
  1498                 }
  1499 
  1500               /* The tool was unlinked.  Free it and generate the
  1501                  appropriate Emacs event (assuming that it was not
  1502                  grabbed by the tool bar).  */
  1503               xfree (touchpoint);
  1504 
  1505               inev.ie.kind = TOUCHSCREEN_END_EVENT;
  1506               inev.ie.timestamp = event->touch.time;
  1507 
  1508               XSETFRAME (inev.ie.frame_or_window, any);
  1509               XSETINT (inev.ie.x, event->touch.x);
  1510               XSETINT (inev.ie.y, event->touch.y);
  1511               XSETINT (inev.ie.arg, event->touch.pointer_id);
  1512 
  1513               /* Break out of the loop.  */
  1514               goto OTHER;
  1515             }
  1516           else
  1517             last = &touchpoint->next;
  1518         }
  1519 
  1520       /* No touch point was found.  This shouldn't happen.  */
  1521       goto OTHER;
  1522 
  1523       /* Wheel motion.  The events here don't parallel X because
  1524          Android doesn't have scroll valuators.  */
  1525 
  1526     case ANDROID_WHEEL:
  1527 
  1528       if (!any)
  1529         goto OTHER;
  1530 
  1531       if (fabs (event->wheel.x_delta) > 0
  1532           || fabs (event->wheel.y_delta) > 0)
  1533         {
  1534           if (mwheel_coalesce_scroll_events)
  1535             {
  1536               if (signbit (event->wheel.x_delta)
  1537                   != signbit (wheel_event_x))
  1538                 wheel_event_x = 0.0;
  1539 
  1540               if (signbit (event->wheel.y_delta)
  1541                   != signbit (wheel_event_y))
  1542                 wheel_event_y = 0.0;
  1543 
  1544               /* Tally up deltas until one of them exceeds 1.0.  */
  1545               wheel_event_x += event->wheel.x_delta;
  1546               wheel_event_y += event->wheel.y_delta;
  1547 
  1548               if (fabs (wheel_event_x) < 1.0
  1549                   && fabs (wheel_event_y) < 1.0)
  1550                 goto OTHER;
  1551             }
  1552           else
  1553             {
  1554               /* Use the deltas in the event.  */
  1555               wheel_event_x = event->wheel.x_delta;
  1556               wheel_event_y = event->wheel.y_delta;
  1557             }
  1558 
  1559           /* Determine what kind of event to send.  */
  1560           inev.ie.kind = ((fabs (wheel_event_y)
  1561                            >= fabs (wheel_event_x))
  1562                           ? WHEEL_EVENT : HORIZ_WHEEL_EVENT);
  1563           inev.ie.timestamp = event->wheel.time;
  1564 
  1565           /* Set the event coordinates.  */
  1566           XSETINT (inev.ie.x, event->wheel.x);
  1567           XSETINT (inev.ie.y, event->wheel.y);
  1568 
  1569           /* Set the frame.  */
  1570           XSETFRAME (inev.ie.frame_or_window, any);
  1571 
  1572           /* Figure out the scroll direction.  */
  1573           inev.ie.modifiers = (signbit ((fabs (wheel_event_x)
  1574                                          >= fabs (wheel_event_y))
  1575                                         ? wheel_event_x
  1576                                         : wheel_event_y)
  1577                                ? down_modifier : up_modifier);
  1578 
  1579           /* Figure out how much to scale the deltas by.  */
  1580           window = window_from_coordinates (any, event->wheel.x,
  1581                                             event->wheel.y, NULL,
  1582                                             false, false);
  1583 
  1584           if (WINDOWP (window))
  1585             scroll_height = XWINDOW (window)->pixel_height;
  1586           else
  1587             /* EVENT_X and EVENT_Y can be outside the
  1588                frame if F holds the input grab, so fall
  1589                back to the height of the frame instead.  */
  1590             scroll_height = FRAME_PIXEL_HEIGHT (any);
  1591 
  1592           scroll_unit = pow (scroll_height, 2.0 / 3.0);
  1593 
  1594           /* Add the keyboard modifiers.  */
  1595           inev.ie.modifiers
  1596             |= android_android_to_emacs_modifiers (dpyinfo,
  1597                                                    event->wheel.state);
  1598 
  1599           /* Finally include the scroll deltas.  */
  1600           inev.ie.arg = list3 (Qnil,
  1601                                make_float (wheel_event_x
  1602                                            * scroll_unit),
  1603                                make_float (wheel_event_y
  1604                                            * scroll_unit));
  1605 
  1606           wheel_event_x = 0.0;
  1607           wheel_event_y = 0.0;
  1608         }
  1609 
  1610       goto OTHER;
  1611 
  1612       /* Iconification.  This is vastly simpler than on X.  */
  1613     case ANDROID_ICONIFIED:
  1614 
  1615       if (!any)
  1616         goto OTHER;
  1617 
  1618       if (FRAME_ICONIFIED_P (any))
  1619         goto OTHER;
  1620 
  1621       SET_FRAME_VISIBLE (any, false);
  1622       SET_FRAME_ICONIFIED (any, true);
  1623 
  1624       inev.ie.kind = ICONIFY_EVENT;
  1625       XSETFRAME (inev.ie.frame_or_window, any);
  1626       goto OTHER;
  1627 
  1628     case ANDROID_DEICONIFIED:
  1629 
  1630       if (!any)
  1631         goto OTHER;
  1632 
  1633       if (!FRAME_ICONIFIED_P (any))
  1634         goto OTHER;
  1635 
  1636       SET_FRAME_VISIBLE (any, true);
  1637       SET_FRAME_ICONIFIED (any, false);
  1638 
  1639       inev.ie.kind = DEICONIFY_EVENT;
  1640       XSETFRAME (inev.ie.frame_or_window, any);
  1641       goto OTHER;
  1642 
  1643       /* Context menu handling.  */
  1644     case ANDROID_CONTEXT_MENU:
  1645 
  1646       if (dpyinfo->menu_event_id == -1
  1647           /* Previously displayed popup menus might generate events
  1648              after dismissal, which might interfere.
  1649              `current_menu_serial' is always set to an identifier
  1650              identifying the last context menu to be displayed.  */
  1651           && event->menu.menu_event_serial == current_menu_serial)
  1652         dpyinfo->menu_event_id = event->menu.menu_event_id;
  1653 
  1654       goto OTHER;
  1655 
  1656       /* Input method events.  textconv.c functions are called here to
  1657          queue events, which are then executed in a safe context
  1658          inside keyboard.c.  */
  1659     case ANDROID_INPUT_METHOD:
  1660 
  1661       if (!any)
  1662         /* Free any text allocated for this event.  */
  1663         xfree (event->ime.text);
  1664       else
  1665         android_handle_ime_event (event, any);
  1666 
  1667       goto OTHER;
  1668 
  1669     default:
  1670       goto OTHER;
  1671     }
  1672 
  1673  OTHER:
  1674   if (inev.ie.kind != NO_EVENT)
  1675     {
  1676       kbd_buffer_store_buffered_event (&inev, hold_quit);
  1677       count++;
  1678     }
  1679 
  1680   if (do_help
  1681       && !(hold_quit && hold_quit->kind != NO_EVENT))
  1682     {
  1683       Lisp_Object frame;
  1684 
  1685       if (f)
  1686         XSETFRAME (frame, f);
  1687       else
  1688         frame = Qnil;
  1689 
  1690       if (do_help > 0)
  1691         {
  1692           any_help_event_p = true;
  1693           gen_help_event (help_echo_string, frame, help_echo_window,
  1694                           help_echo_object, help_echo_pos);
  1695         }
  1696       else
  1697         {
  1698           help_echo_string = Qnil;
  1699           gen_help_event (Qnil, frame, Qnil, Qnil, 0);
  1700         }
  1701       count++;
  1702     }
  1703 
  1704   return count;
  1705 }

  1707 static int
  1708 android_read_socket (struct terminal *terminal,
  1709                      struct input_event *hold_quit)
  1710 {
  1711   int count = 0;
  1712   struct android_display_info *dpyinfo;
  1713 
  1714   dpyinfo = terminal->display_info.android;
  1715 
  1716   block_input ();
  1717   while (android_pending ())
  1718     {
  1719       int finish;
  1720       union android_event event;
  1721 
  1722       android_next_event (&event);
  1723       count += handle_one_android_event (dpyinfo, &event, &finish,
  1724                                          hold_quit);
  1725 
  1726       if (finish == ANDROID_EVENT_GOTO_OUT)
  1727         break;
  1728     }
  1729   unblock_input ();
  1730 
  1731   /* If the focus was just given to an auto-raising frame, raise it
  1732      now.  */
  1733   if (dpyinfo->pending_autoraise_frame)
  1734     {
  1735       android_raise_frame (dpyinfo->pending_autoraise_frame);
  1736       dpyinfo->pending_autoraise_frame = NULL;
  1737     }
  1738 
  1739   return count;
  1740 }

  2030 static void
  2031 android_wait_for_event (struct frame *f, int eventtype)
  2032 {
  2033   if (!FLOATP (Vandroid_wait_for_event_timeout))
  2034     return;
  2035 
  2036   int level = interrupt_input_blocked;
  2037   struct timespec tmo, tmo_at, time_now;
  2038 
  2039   f->wait_event_type = eventtype;
  2040 
  2041   /* Default timeout is 0.1 second.  Hopefully not noticeable.  */
  2042   double timeout = XFLOAT_DATA (Vandroid_wait_for_event_timeout);
  2043   time_t timeout_seconds = (time_t) timeout;
  2044   tmo = make_timespec (timeout_seconds,
  2045                        (long int) ((timeout - timeout_seconds)
  2046                                    * 1000 * 1000 * 1000));
  2047   tmo_at = timespec_add (current_timespec (), tmo);
  2048 
  2049   while (f->wait_event_type)
  2050     {
  2051       pending_signals = true;
  2052       totally_unblock_input ();
  2053       /* XTread_socket is called after unblock.  */
  2054       block_input ();
  2055       interrupt_input_blocked = level;
  2056 
  2057       time_now = current_timespec ();
  2058       if (timespec_cmp (tmo_at, time_now) < 0)
  2059         break;
  2060 
  2061       tmo = timespec_sub (tmo_at, time_now);
  2062       if (android_select (0, NULL, NULL, NULL, &tmo) == 0)
  2063         break; /* Timeout */
  2064     }
  2065 
  2066   f->wait_event_type = 0;
  2067 }

  2525 /* Set S->gc to a suitable GC for drawing glyph string S in cursor
  2526    face.  */
  2527 
  2528 static void
  2529 android_set_cursor_gc (struct glyph_string *s)
  2530 {
  2531   if (s->font == FRAME_FONT (s->f)
  2532       && s->face->background == FRAME_BACKGROUND_PIXEL (s->f)
  2533       && s->face->foreground == FRAME_FOREGROUND_PIXEL (s->f)
  2534       && !s->cmp)
  2535     s->gc = s->f->output_data.android->cursor_gc;
  2536   else
  2537     {
  2538       /* Cursor on non-default face: must merge.  */
  2539       struct android_gc_values xgcv;
  2540       unsigned long mask;
  2541 
  2542       xgcv.background = s->f->output_data.android->cursor_pixel;
  2543       xgcv.foreground = s->face->background;
  2544 
  2545       /* If the glyph would be invisible, try a different foreground.  */
  2546       if (xgcv.foreground == xgcv.background)
  2547         xgcv.foreground = s->face->foreground;
  2548       if (xgcv.foreground == xgcv.background)
  2549         xgcv.foreground = s->f->output_data.android->cursor_foreground_pixel;
  2550       if (xgcv.foreground == xgcv.background)
  2551         xgcv.foreground = s->face->foreground;
  2552 
  2553       /* Make sure the cursor is distinct from text in this face.  */
  2554       if (xgcv.background == s->face->background
  2555           && xgcv.foreground == s->face->foreground)
  2556         {
  2557           xgcv.background = s->face->foreground;
  2558           xgcv.foreground = s->face->background;
  2559         }
  2560 
  2561       mask = (ANDROID_GC_FOREGROUND | ANDROID_GC_BACKGROUND);
  2562 
  2563       if (FRAME_DISPLAY_INFO (s->f)->scratch_cursor_gc)
  2564         android_change_gc (FRAME_DISPLAY_INFO (s->f)->scratch_cursor_gc,
  2565                            mask, &xgcv);
  2566       else
  2567         FRAME_DISPLAY_INFO (s->f)->scratch_cursor_gc
  2568           = android_create_gc (mask, &xgcv);
  2569 
  2570       s->gc = FRAME_DISPLAY_INFO (s->f)->scratch_cursor_gc;
  2571     }
  2572 }

  2829 static void
  2830 android_draw_relief_rect (struct frame *f, int left_x, int top_y, int right_x,
  2831                           int bottom_y, int hwidth, int vwidth, bool raised_p,
  2832                           bool top_p, bool bot_p, bool left_p, bool right_p,
  2833                           struct android_rectangle *clip_rect)
  2834 {
  2835   struct android_gc *gc, *white_gc, *black_gc, *normal_gc;
  2836   android_drawable drawable;
  2837 
  2838   /* This code is more complicated than it has to be, because of two
  2839      minor hacks to make the boxes look nicer: (i) if width > 1, draw
  2840      the outermost line using the black relief.  (ii) Omit the four
  2841      corner pixels.  */
  2842 
  2843   white_gc = f->output_data.android->white_relief.gc;
  2844   black_gc = f->output_data.android->black_relief.gc;
  2845   normal_gc = f->output_data.android->normal_gc;
  2846 
  2847   drawable = FRAME_ANDROID_DRAWABLE (f);
  2848 
  2849   android_set_clip_rectangles (white_gc, 0, 0, clip_rect, 1);
  2850   android_set_clip_rectangles (black_gc, 0, 0, clip_rect, 1);
  2851 
  2852   if (raised_p)
  2853     gc = white_gc;
  2854   else
  2855     gc = black_gc;
  2856 
  2857   /* Draw lines.  */
  2858 
  2859   if (top_p)
  2860     android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc, left_x, top_y,
  2861                             right_x - left_x + 1, hwidth);
  2862 
  2863   if (left_p)
  2864     android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc, left_x, top_y,
  2865                             vwidth, bottom_y - top_y + 1);
  2866 
  2867   if (raised_p)
  2868     gc = black_gc;
  2869   else
  2870     gc = white_gc;
  2871 
  2872   if (bot_p)
  2873     android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc, left_x,
  2874                             bottom_y - hwidth + 1,
  2875                             right_x - left_x + 1, hwidth);
  2876 
  2877   if (right_p)
  2878     android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc,
  2879                             right_x - vwidth + 1,
  2880                             top_y, vwidth, bottom_y - top_y + 1);
  2881 
  2882   /* Draw corners.  */
  2883 
  2884   if (bot_p && left_p)
  2885     android_fill_triangle (f, raised_p ? white_gc : black_gc,
  2886                            android_make_point (left_x, bottom_y - hwidth),
  2887                            android_make_point (left_x + vwidth,
  2888                                                bottom_y - hwidth),
  2889                            android_make_point (left_x, bottom_y));
  2890 
  2891   if (top_p && right_p)
  2892     android_fill_triangle (f, raised_p ? white_gc : black_gc,
  2893                            android_make_point (right_x - vwidth, top_y),
  2894                            android_make_point (right_x, top_y),
  2895                            android_make_point (right_x - vwidth,
  2896                                                top_y + hwidth));
  2897 
  2898   /* Draw outer line.  */
  2899 
  2900   if (top_p && left_p && bot_p && right_p
  2901       && hwidth > 1 && vwidth > 1)
  2902     android_draw_rectangle (FRAME_ANDROID_DRAWABLE (f),
  2903                             black_gc, left_x, top_y,
  2904                             right_x - left_x, bottom_y - top_y);
  2905   else
  2906     {
  2907       if (top_p && hwidth > 1)
  2908         android_draw_line (drawable, black_gc, left_x, top_y,
  2909                            right_x + 1, top_y);
  2910 
  2911       if (bot_p && hwidth > 1)
  2912         android_draw_line (drawable, black_gc, left_x, bottom_y,
  2913                            right_x + 1, bottom_y);
  2914 
  2915       if (left_p && vwidth > 1)
  2916         android_draw_line (drawable, black_gc, left_x, top_y,
  2917                            left_x, bottom_y + 1);
  2918 
  2919       if (right_p && vwidth > 1)
  2920         android_draw_line (drawable, black_gc, right_x, top_y,
  2921                            right_x, bottom_y + 1);
  2922     }
  2923 
  2924   /* Erase corners.  */
  2925 
  2926   if (hwidth > 1 && vwidth > 1)
  2927     {
  2928       if (left_p && top_p && android_inside_rect_p (clip_rect, 1,
  2929                                                     left_x, top_y))
  2930         android_clear_point (f, normal_gc, left_x, top_y);
  2931 
  2932       if (left_p && bot_p && android_inside_rect_p (clip_rect, 1,
  2933                                                     left_x, bottom_y))
  2934         android_clear_point (f, normal_gc, left_x, bottom_y);
  2935 
  2936       if (right_p && top_p && android_inside_rect_p (clip_rect, 1,
  2937                                                      right_x, top_y))
  2938         android_clear_point (f, normal_gc, right_x, top_y);

  2985 #define HIGHLIGHT_COLOR_DARK_BOOST_LIMIT 48000
  2986 
  2987 static bool
  2988 android_alloc_lighter_color (struct frame *f, unsigned long *pixel,
  2989                              double factor, int delta)
  2990 {
  2991   Emacs_Color color, new;
  2992   long bright;
  2993   bool success_p;
  2994 
  2995   /* Get RGB color values.  */
  2996   color.pixel = *pixel;
  2997   android_query_colors (f, &color, 1);
  2998 
  2999   /* Change RGB values by specified FACTOR.  Avoid overflow!  */
  3000   eassert (factor >= 0);
  3001   new.red = min (0xffff, factor * color.red);
  3002   new.green = min (0xffff, factor * color.green);
  3003   new.blue = min (0xffff, factor * color.blue);
  3004 
  3005   /* Calculate brightness of COLOR.  */
  3006   bright = (2 * color.red + 3 * color.green + color.blue) / 6;
  3007 
  3008   /* We only boost colors that are darker than
  3009      HIGHLIGHT_COLOR_DARK_BOOST_LIMIT.  */
  3010   if (bright < HIGHLIGHT_COLOR_DARK_BOOST_LIMIT)
  3011     /* Make an additive adjustment to NEW, because it's dark enough so
  3012        that scaling by FACTOR alone isn't enough.  */
  3013     {
  3014       /* How far below the limit this color is (0 - 1, 1 being darker).  */
  3015       double dimness = 1 - (double) bright / HIGHLIGHT_COLOR_DARK_BOOST_LIMIT;
  3016       /* The additive adjustment.  */
  3017       int min_delta = delta * dimness * factor / 2;
  3018 
  3019       if (factor < 1)
  3020         {
  3021           new.red =   max (0, new.red -   min_delta);
  3022           new.green = max (0, new.green - min_delta);
  3023           new.blue =  max (0, new.blue -  min_delta);
  3024         }
  3025       else
  3026         {
  3027           new.red =   min (0xffff, min_delta + new.red);
  3028           new.green = min (0xffff, min_delta + new.green);
  3029           new.blue =  min (0xffff, min_delta + new.blue);
  3030         }
  3031     }
  3032 
  3033   /* Try to allocate the color.  */
  3034   success_p = android_alloc_nearest_color (f, &new);
  3035 
  3036   if (success_p)
  3037     {
  3038       if (new.pixel == *pixel)
  3039         {
  3040           /* If we end up with the same color as before, try adding
  3041              delta to the RGB values.  */
  3042           new.red = min (0xffff, delta + color.red);
  3043           new.green = min (0xffff, delta + color.green);
  3044           new.blue = min (0xffff, delta + color.blue);
  3045           success_p = android_alloc_nearest_color (f, &new);
  3046         }
  3047       else
  3048         success_p = true;
  3049 
  3050       *pixel = new.pixel;
  3051     }
  3052 
  3053   return success_p;
  3054 }

  3560 static void
  3561 android_draw_underwave (struct glyph_string *s, int decoration_width)
  3562 {
  3563   int scale_x, scale_y;
  3564 
  3565   android_get_scale_factor (&scale_x, &scale_y);
  3566 
  3567   int wave_height = 3 * scale_y, wave_length = 2 * scale_x;
  3568 
  3569   int dx, dy, x0, y0, width, x1, y1, x2, y2, xmax;
  3570   bool odd;
  3571   struct android_rectangle wave_clip, string_clip, final_clip;
  3572 
  3573   dx = wave_length;
  3574   dy = wave_height - 1;
  3575   x0 = s->x;
  3576   y0 = s->ybase + wave_height / 2;
  3577   width = decoration_width;
  3578   xmax = x0 + width;
  3579 
  3580   /* Find and set clipping rectangle */
  3581 
  3582   wave_clip.x = x0;
  3583   wave_clip.y = y0;
  3584   wave_clip.width = width;
  3585   wave_clip.height = wave_height;
  3586   get_glyph_string_clip_rect (s, &string_clip);
  3587 
  3588   if (!gui_intersect_rectangles (&wave_clip, &string_clip, &final_clip))
  3589     return;
  3590 
  3591   android_set_clip_rectangles (s->gc, 0, 0, &final_clip, 1);
  3592 
  3593   /* Draw the waves */
  3594 
  3595   x1 = x0 - (x0 % dx);
  3596   x2 = x1 + dx;
  3597   odd = (x1 / dx) & 1;
  3598   y1 = y2 = y0;
  3599 
  3600   if (odd)
  3601     y1 += dy;
  3602   else
  3603     y2 += dy;
  3604 
  3605   if (INT_MAX - dx < xmax)
  3606     emacs_abort ();
  3607 
  3608   while (x1 <= xmax)
  3609     {
  3610       android_draw_line (FRAME_ANDROID_DRAWABLE (s->f), s->gc,
  3611                          x1, y1, x2, y2);
  3612       x1  = x2, y1 = y2;
  3613       x2 += dx, y2 = y0 + odd*dy;
  3614       odd = !odd;
  3615     }
  3616 
  3617   /* Restore previous clipping rectangle(s) */
  3618   android_set_clip_rectangles (s->gc, 0, 0, s->clip, s->num_clips);
  3619 }

  3668 static void
  3669 android_draw_composite_glyph_string_foreground (struct glyph_string *s)

  3673 
  3674   /* If first glyph of S has a left box line, start drawing the text
  3675      of S to the right of that box line.  */

  3677       && s->first_glyph->left_box_line_p)
  3678     x = s->x + max (s->face->box_vertical_line_width, 0);
  3679   else
  3680     x = s->x;

  3682   /* S is a glyph string for a composition.  S->cmp_from is the index
  3683      of the first character drawn for glyphs of this composition.
  3684      S->cmp_from == 0 means we are drawing the very first character of
  3685      this composition.  */
  3686 
  3687   /* Draw a rectangle for the composition if the font for the very
  3688      first character of the composition could not be loaded.  */
  3689   if (s->font_not_found_p)
  3690     {

  3692         android_draw_rectangle (FRAME_ANDROID_DRAWABLE (s->f),
  3693                                 s->gc, x, s->y,
  3694                                 s->width - 1, s->height - 1);
  3695     }

  3697     {

  3699

  3703         if (COMPOSITION_GLYPH (s->cmp, j) != '\t')
  3704           {
  3705             int xx = x + s->cmp->offsets[j * 2];

  3709             if (s->face->overstrike)
  3710               font->driver->draw (s, j, j + 1, xx + 1, yy, false);
  3711           }
  3712     }
  3713   else
  3714     {
  3715       Lisp_Object gstring = composition_gstring_from_id (s->cmp_id);
  3716       Lisp_Object glyph;
  3717       int y = s->ybase;
  3718       int width = 0;
  3719 
  3720       for (i = j = s->cmp_from; i < s->cmp_to; i++)
  3721         {
  3722           glyph = LGSTRING_GLYPH (gstring, i);
  3723           if (NILP (LGLYPH_ADJUSTMENT (glyph)))
  3724             width += LGLYPH_WIDTH (glyph);
  3725           else
  3726             {
  3727               int xoff, yoff, wadjust;
  3728 
  3729               if (j < i)
  3730                 {
  3731                   font->driver->draw (s, j, i, x, y, false);
  3732                   if (s->face->overstrike)
  3733                     font->driver->draw (s, j, i, x + 1, y, false);
  3734                   x += width;
  3735                 }
  3736               xoff = LGLYPH_XOFF (glyph);
  3737               yoff = LGLYPH_YOFF (glyph);
  3738               wadjust = LGLYPH_WADJUST (glyph);
  3739               font->driver->draw (s, i, i + 1, x + xoff, y + yoff, false);
  3740               if (s->face->overstrike)
  3741                 font->driver->draw (s, i, i + 1, x + xoff + 1, y + yoff,
  3742                                     false);
  3743               x += wadjust;
  3744               j = i + 1;
  3745               width = 0;
  3746             }
  3747         }
  3748       if (j < i)
  3749         {
  3750           font->driver->draw (s, j, i, x, y, false);
  3751           if (s->face->overstrike)
  3752             font->driver->draw (s, j, i, x + 1, y, false);
  3753         }
  3754     }
  3755 }

  4018                   if (font && font->underline_thickness > 0)
  4019                     thickness = font->underline_thickness;
  4020                   else
  4021                     thickness = 1;
  4022                   if (underline_at_descent_line)
  4023                     position = ((s->height - thickness)
  4024                                 - (s->ybase - s->y)
  4025                                 - s->face->underline_pixels_above_descent_line);
  4026                   else
  4027                     {
  4028                       /* Get the underline position.  This is the
  4029                          recommended vertical offset in pixels from
  4030                          the baseline to the top of the underline.
  4031                          This is a signed value according to the
  4032                          specs, and its default is
  4033 
  4034                          ROUND ((maximum descent) / 2), with
  4035                          ROUND(x) = floor (x + 0.5)  */
  4036 
  4037                       if (use_underline_position_properties
  4038                           && font && font->underline_position >= 0)
  4039                         position = font->underline_position;
  4040                       else if (font)
  4041                         position = (font->descent + 1) / 2;
  4042                       else
  4043                         position = minimum_offset;
  4044                     }
  4045 
  4046                   /* Ignore minimum_offset if the amount of pixels was
  4047                      explicitly specified.  */
  4048                   if (!s->face->underline_pixels_above_descent_line)
  4049                     position = max (position, minimum_offset);
  4050                 }
  4051               /* Check the sanity of thickness and position.  We should
  4052                  avoid drawing underline out of the current line area.  */
  4053               if (s->y + s->height <= s->ybase + position)
  4054                 position = (s->height - 1) - (s->ybase - s->y);
  4055               if (s->y + s->height < s->ybase + position + thickness)
  4056                 thickness = (s->y + s->height) - (s->ybase + position);
  4057               s->underline_thickness = thickness;
  4058               s->underline_position = position;
  4059               y = s->ybase + position;
  4060               if (s->face->underline_defaulted_p)
  4061                 android_fill_rectangle (FRAME_ANDROID_DRAWABLE (s->f), s->gc,
  4062                                         s->x, y, decoration_width, thickness);
  4063               else
  4064                 {
  4065                   struct android_gc_values xgcv;
  4066                   android_get_gc_values (s->gc, ANDROID_GC_FOREGROUND, &xgcv);
  4067                   android_set_foreground (s->gc, s->face->underline_color);
  4068                   android_fill_rectangle (FRAME_ANDROID_DRAWABLE (s->f), s->gc,
  4069                                           s->x, y, decoration_width, thickness);
  4070                   android_set_foreground (s->gc, xgcv.foreground);
  4071                 }
  4072             }
  4073         }
  4074       /* Draw overline.  */
  4075       if (s->face->overline_p)
  4076         {
  4077           unsigned long dy = 0, h = 1;
  4078 
  4079           if (s->face->overline_color_defaulted_p)
  4080             android_fill_rectangle (FRAME_ANDROID_DRAWABLE (s->f),
  4081                                     s->gc, s->x, s->y + dy,
  4082                                     decoration_width, h);
  4083           else
  4084             {
  4085               struct android_gc_values xgcv;
  4086               android_get_gc_values (s->gc, ANDROID_GC_FOREGROUND, &xgcv);
  4087               android_set_foreground (s->gc, s->face->overline_color);
  4088               android_fill_rectangle (FRAME_ANDROID_DRAWABLE (s->f), s->gc,
  4089                                       s->x, s->y + dy, decoration_width, h);
  4090               android_set_foreground (s->gc, xgcv.foreground);
  4091             }
  4092         }
  4093 
  4094       /* Draw strike-through.  */
  4095       if (s->face->strike_through_p)
  4096         {
  4097           /* Y-coordinate and height of the glyph string's first
  4098              glyph.  We cannot use s->y and s->height because those
  4099              could be larger if there are taller display elements
  4100              (e.g., characters displayed with a larger font) in the
  4101              same glyph row.  */
  4102           int glyph_y = s->ybase - s->first_glyph->ascent;
  4103           int glyph_height = s->first_glyph->ascent + s->first_glyph->descent;
  4104           /* Strike-through width and offset from the glyph string's
  4105              top edge.  */
  4106           unsigned long h = 1;
  4107           unsigned long dy = (glyph_height - h) / 2;
  4108 
  4109           if (s->face->strike_through_color_defaulted_p)
  4110             android_fill_rectangle (FRAME_ANDROID_WINDOW (s->f),
  4111                                     s->gc, s->x, glyph_y + dy,
  4112                                     s->width, h);
  4113           else
  4114             {
  4115               struct android_gc_values xgcv;
  4116               android_get_gc_values (s->gc, ANDROID_GC_FOREGROUND, &xgcv);
  4117               android_set_foreground (s->gc, s->face->strike_through_color);
  4118               android_fill_rectangle (FRAME_ANDROID_DRAWABLE (s->f), s->gc,
  4119                                       s->x, glyph_y + dy, decoration_width,
  4120                                       h);
  4121               android_set_foreground (s->gc, xgcv.foreground);
  4122             }
  4123         }
  4124 
  4125       if (s->prev)
  4126         {
  4127           struct glyph_string *prev;
  4128 
  4129           for (prev = s->prev; prev; prev = prev->prev)
  4130             if (prev->hl != s->hl
  4131                 && prev->x + prev->width + prev->right_overhang > s->x)
  4132               {
  4133                 /* As prev was drawn while clipped to its own area, we
  4134                    must draw the right_overhang part using s->hl now.  */
  4135                 enum draw_glyphs_face save = prev->hl;
  4136 
  4137                 prev->hl = s->hl;
  4138                 android_set_glyph_string_gc (prev);
  4139                 android_set_glyph_string_clipping_exactly (s, prev);
  4140                 if (prev->first_glyph->type == CHAR_GLYPH)
  4141                   android_draw_glyph_string_foreground (prev);
  4142                 else
  4143                   android_draw_composite_glyph_string_foreground (prev);
  4144                 android_reset_clip_rectangles (prev->f, prev->gc);
  4145                 prev->hl = save;
  4146                 prev->num_clips = 0;
  4147               }
  4148         }
  4149 
  4150       if (s->next)
  4151         {
  4152           struct glyph_string *next;
  4153 
  4154           for (next = s->next; next; next = next->next)
  4155             if (next->hl != s->hl
  4156                 && next->x - next->left_overhang < s->x + s->width)
  4157               {
  4158                 /* As next will be drawn while clipped to its own area,
  4159                    we must draw the left_overhang part using s->hl now.  */
  4160                 enum draw_glyphs_face save = next->hl;
  4161 
  4162                 next->hl = s->hl;
  4163                 android_set_glyph_string_gc (next);
  4164                 android_set_glyph_string_clipping_exactly (s, next);
  4165                 if (next->first_glyph->type == CHAR_GLYPH)
  4166                   android_draw_glyph_string_foreground (next);
  4167                 else
  4168                   android_draw_composite_glyph_string_foreground (next);
  4169                 android_reset_clip_rectangles (next->f, next->gc);
  4170                 next->hl = save;
  4171                 next->num_clips = 0;
  4172                 next->clip_head = s->next;
  4173               }
  4174         }
  4175     }
  4176 
  4177   /* Reset clipping.  */
  4178   android_reset_clip_rectangles (s->f, s->gc);
  4179   s->num_clips = 0;
  4180 
  4181   /* Set the stippled flag that tells redisplay whether or not a
  4182      stipple was actually draw.  */
  4183 
  4184   if (s->first_glyph->type != STRETCH_GLYPH
  4185       && s->first_glyph->type != IMAGE_GLYPH
  4186       && !s->row->stipple_p)
  4187     s->row->stipple_p = s->stippled_p;
  4188 }

  4309 static void
  4310 android_draw_bar_cursor (struct window *w, struct glyph_row *row, int width,
  4311                          enum text_cursor_kinds kind)
  4312 {
  4313   struct frame *f = XFRAME (w->frame);
  4314   struct glyph *cursor_glyph;
  4315   int cursor_start_y;
  4316 
  4317   /* If cursor is out of bounds, don't draw garbage.  This can happen
  4318      in mini-buffer windows when switching between echo area glyphs
  4319      and mini-buffer.  */
  4320   cursor_glyph = get_phys_cursor_glyph (w);
  4321   if (cursor_glyph == NULL)
  4322     return;
  4323 
  4324   /* Experimental avoidance of cursor on xwidget.  */
  4325   if (cursor_glyph->type == XWIDGET_GLYPH)
  4326     return;
  4327 
  4328   /* If on an image, draw like a normal cursor.  That's usually better
  4329      visible than drawing a bar, esp. if the image is large so that
  4330      the bar might not be in the window.  */
  4331   if (cursor_glyph->type == IMAGE_GLYPH)
  4332     {
  4333       struct glyph_row *r;
  4334       r = MATRIX_ROW (w->current_matrix, w->phys_cursor.vpos);
  4335       draw_phys_cursor_glyph (w, r, DRAW_CURSOR);
  4336     }
  4337   else
  4338     {
  4339       struct android_gc *gc = FRAME_DISPLAY_INFO (f)->scratch_cursor_gc;
  4340       unsigned long mask = ANDROID_GC_FOREGROUND | ANDROID_GC_BACKGROUND;
  4341       struct face *face = FACE_FROM_ID (f, cursor_glyph->face_id);
  4342       struct android_gc_values xgcv;
  4343 
  4344       /* If the glyph's background equals the color we normally draw
  4345          the bars cursor in, the bar cursor in its normal color is
  4346          invisible.  Use the glyph's foreground color instead in this
  4347          case, on the assumption that the glyph's colors are chosen so
  4348          that the glyph is legible.  */
  4349       if (face->background == f->output_data.android->cursor_pixel)
  4350         xgcv.background = xgcv.foreground = face->foreground;
  4351       else
  4352         xgcv.background = xgcv.foreground = f->output_data.android->cursor_pixel;
  4353 
  4354       if (gc)
  4355         android_change_gc (gc, mask, &xgcv);
  4356       else
  4357         {
  4358           gc = android_create_gc (mask, &xgcv);
  4359           FRAME_DISPLAY_INFO (f)->scratch_cursor_gc = gc;
  4360         }
  4361 
  4362       android_clip_to_row (w, row, TEXT_AREA, gc);
  4363 
  4364       if (kind == BAR_CURSOR)
  4365         {
  4366           int x = WINDOW_TEXT_TO_FRAME_PIXEL_X (w, w->phys_cursor.x);
  4367 
  4368           if (width < 0)
  4369             width = FRAME_CURSOR_WIDTH (f);
  4370           width = min (cursor_glyph->pixel_width, width);
  4371 
  4372           w->phys_cursor_width = width;
  4373 
  4374           /* If the character under cursor is R2L, draw the bar cursor
  4375              on the right of its glyph, rather than on the left.  */
  4376           if ((cursor_glyph->resolved_level & 1) != 0)
  4377             x += cursor_glyph->pixel_width - width;
  4378 
  4379           android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc, x,
  4380                                   WINDOW_TO_FRAME_PIXEL_Y (w, w->phys_cursor.y),
  4381                                   width, row->height);
  4382         }
  4383       else /* HBAR_CURSOR */
  4384         {
  4385           int dummy_x, dummy_y, dummy_h;
  4386           int x = WINDOW_TEXT_TO_FRAME_PIXEL_X (w, w->phys_cursor.x);
  4387 
  4388           if (width < 0)
  4389             width = row->height;
  4390 
  4391           width = min (row->height, width);
  4392 
  4393           get_phys_cursor_geometry (w, row, cursor_glyph, &dummy_x,
  4394                                     &dummy_y, &dummy_h);
  4395 
  4396           cursor_start_y = WINDOW_TO_FRAME_PIXEL_Y (w, w->phys_cursor.y
  4397                                                     + row->height - width);
  4398 
  4399           if ((cursor_glyph->resolved_level & 1) != 0
  4400               && cursor_glyph->pixel_width > w->phys_cursor_width - 1)
  4401             x += cursor_glyph->pixel_width - w->phys_cursor_width + 1;
  4402           android_fill_rectangle (FRAME_ANDROID_DRAWABLE (f), gc, x,
  4403                                   cursor_start_y,
  4404                                   w->phys_cursor_width - 1, width);
  4405         }
  4406 
  4407       android_reset_clip_rectangles (f, gc);
  4408     }
  4409 }

  4576 /* Try to synchronize with the UI thread, waiting a certain amount of
  4577    time for outstanding editing requests to complete.
  4578 
  4579    Every time one of the text retrieval functions is called and an
  4580    editing request is made, Emacs gives the main thread approximately
  4581    100 ms to process it, in order to mostly keep the input method in

  4584 static void
  4585 android_sync_edit (void)
  4586 {
  4587   struct timespec start, end, rem;
  4588 
  4589   if (__atomic_load_n (&last_edit_counter,
  4590                        __ATOMIC_SEQ_CST)
  4591       == edit_counter)
  4592     return;
  4593 
  4594   start = current_timespec ();
  4595   end = timespec_add (start, make_timespec (0, 100000000));
  4596 
  4597   while (true)
  4598     {
  4599       rem = timespec_sub (end, current_timespec ());
  4600 
  4601       /* Timeout.  */
  4602       if (timespec_sign (rem) < 0)
  4603         break;
  4604 
  4605       if (__atomic_load_n (&last_edit_counter,
  4606                            __ATOMIC_SEQ_CST)
  4607           == edit_counter)
  4608         break;
  4609 
  4610       sem_timedwait (&edit_sem, &end);
  4611     }

  4844 /* Convert a string BUFFERS containing N characters in Emacs's
  4845    internal multibyte encoding to a Java string utilizing the
  4846    specified JNI environment.
  4847 
  4848    If N is equal to BYTES, then BUFFER is a single byte buffer.
  4849    Otherwise, BUFFER is a multibyte buffer.
  4850 
  4851    Make sure N and BYTES are absolutely correct, or you are asking for
  4852    trouble.
  4853 
  4854    Value is the string upon success, NULL otherwise.  Any exceptions
  4855    generated are not cleared.  */
  4856 
  4857 static jstring
  4858 android_text_to_string (JNIEnv *env, char *buffer, ptrdiff_t n,
  4859                         ptrdiff_t bytes)
  4860 {
  4861   jchar *utf16;
  4862   size_t size, index;
  4863   jstring string;
  4864   int encoded;
  4865 
  4866   if (n == bytes)
  4867     {
  4868       /* This buffer holds no multibyte characters.  */
  4869 
  4870       if (INT_MULTIPLY_WRAPV (n, sizeof *utf16, &size))
  4871         return NULL;
  4872 
  4873       utf16 = malloc (size);
  4874       index = 0;
  4875 
  4876       if (!utf16)
  4877         return NULL;
  4878 
  4879       while (n--)
  4880         {
  4881           utf16[index] = buffer[index];
  4882           index++;
  4883         }
  4884 
  4885       string = (*env)->NewString (env, utf16, bytes);
  4886       free (utf16);
  4887 
  4888       return string;
  4889     }
  4890 
  4891   /* Allocate enough to hold N characters.  */
  4892 
  4893   if (INT_MULTIPLY_WRAPV (n, sizeof *utf16, &size))
  4894     return NULL;
  4895 
  4896   utf16 = malloc (size);
  4897   index = 0;
  4898 
  4899   if (!utf16)
  4900     return NULL;
  4901 
  4902   while (n--)
  4903     {
  4904       eassert (CHAR_HEAD_P (*buffer));
  4905       encoded = STRING_CHAR ((unsigned char *) buffer);
  4906 
  4907       /* Now figure out how to save ENCODED into the string.
  4908          Emacs operates on multibyte characters, not UTF-16
  4909          characters with surrogate pairs as Android does.
  4910 
  4911          However, character positions in Java are represented in 2
  4912          byte units, meaning that the text position reported to
  4913          Android can become out of sync if characters are found in a
  4914          buffer that require surrogate pairs.
  4915 
  4916          The hack used by Emacs is to simply replace each multibyte
  4917          character that doesn't fit in a jchar with the NULL
  4918          character.  */
  4919 
  4920       if (encoded >= 65536)
  4921         encoded = 0;
  4922 
  4923       utf16[index++] = encoded;
  4924       buffer += BYTES_BY_CHAR_HEAD (*buffer);
  4925     }
  4926 
  4927   /* Create the string.  */
  4928   string = (*env)->NewString (env, utf16, index);
  4929   free (utf16);
  4930   return string;
  4931 }

  5918 /* Tell the input method where the composing region and selection of
  5919    F's selected window is located.  W should be F's selected window;
  5920    if it is NULL, then F->selected_window is used in its place.  */
  5921 
  5922 static void
  5923 android_update_selection (struct frame *f, struct window *w)
  5924 {
  5925   ptrdiff_t start, end, point, mark, start_offset, end_offset;
  5926   ptrdiff_t length, bytes;
  5927   struct buffer *b;
  5928   int hint, token;
  5929   char *text;
  5930   jobject extracted;
  5931   jstring string;
  5932   bool mark_active;
  5933 
  5934   if (MARKERP (f->conversion.compose_region_start))
  5935     {
  5936       eassert (MARKERP (f->conversion.compose_region_end));
  5937 
  5938       /* Indexing in android starts from 0 instead of 1.  */
  5939       start = marker_position (f->conversion.compose_region_start) - 1;
  5940       end = marker_position (f->conversion.compose_region_end) - 1;
  5941     }
  5942   else
  5943     start = -1, end = -1;
  5944 
  5945   /* Now constrain START and END to the maximium size of a Java
  5946      integer.  */
  5947   start = min (start, TYPE_MAXIMUM (jint));
  5948   end = min (end, TYPE_MAXIMUM (jint));
  5949 
  5950   if (!w)
  5951     w = XWINDOW (f->selected_window);
  5952 
  5953   /* Figure out where the point and mark are.  If the mark is not
  5954      active, then point is set to equal mark.  */
  5955   b = XBUFFER (w->contents);
  5956   point = min (w->ephemeral_last_point,
  5957                TYPE_MAXIMUM (jint));
  5958   mark = ((!NILP (BVAR (b, mark_active))
  5959            && w->last_mark != -1)
  5960           ? min (w->last_mark, TYPE_MAXIMUM (jint))
  5961           : point);
  5962 
  5963   /* Send the update.  Android doesn't have a concept of ``point'' and
  5964      ``mark''; instead, it only has a selection, where the start of
  5965      the selection is less than or equal to the end.  Also, convert
  5966      the indices from 1-based Emacs indices to 0-based Android
  5967      ones.  */
  5968   android_update_ic (FRAME_ANDROID_WINDOW (f), min (point, mark) - 1,
  5969                      max (point, mark) - 1, start, end);
  5970 
  5971   /* Update the extracted text as well, if the input method has asked
  5972      for updates.  1 is
  5973      InputConnection.GET_EXTRACTED_TEXT_MONITOR.  */
  5974 
  5975   if (FRAME_ANDROID_OUTPUT (f)->extracted_text_flags & 1)
  5976     {
  5977       hint = FRAME_ANDROID_OUTPUT (f)->extracted_text_hint;
  5978       token = FRAME_ANDROID_OUTPUT (f)->extracted_text_token;
  5979       text = get_extracted_text (f, min (hint, 600), &start,
  5980                                  &start_offset, &end_offset,
  5981                                  &length, &bytes, &mark_active);
  5982 
  5983       if (text)
  5984         {
  5985           /* Make a string out of the extracted text.  */
  5986           string = android_text_to_string (android_java_env,
  5987                                            text, length, bytes);
  5988           xfree (text);
  5989           android_exception_check ();
  5990 
  5991           /* Make extracted text out of that string.  */
  5992           extracted = android_build_extracted_text (string, start,
  5993                                                     start_offset,
  5994                                                     end_offset,
  5995                                                     mark_active);
  5996           android_exception_check_1 (string);
  5997           ANDROID_DELETE_LOCAL_REF (string);
  5998 
  5999           if (extracted)
  6000             {
  6001               /* extracted is now an associated ExtractedText object.
  6002                  Perform the update.  */
  6003               android_update_extracted_text (FRAME_ANDROID_WINDOW (f),
  6004                                              extracted, token);
  6005               ANDROID_DELETE_LOCAL_REF (extracted);
  6006             }
  6007         }
  6008     }
  6009 }

  6024 /* Notice that the input method connection to F should be reset as a
  6025    result of a change to its contents.  */
  6026 
  6027 static void
  6028 android_reset_conversion (struct frame *f)
  6029 {
  6030   enum android_ic_mode mode;
  6031   struct window *w;
  6032   struct buffer *buffer;
  6033   Lisp_Object style;
  6034   union android_event event;
  6035 
  6036   /* Reset the input method.
  6037 
  6038      Pick an appropriate ``input mode'' based on whether or not the
  6039      minibuffer window is selected; this controls whether or not
  6040      ``RET'' inserts a newline or sends an actual key event.  */
  6041 
  6042   w = XWINDOW (f->selected_window);
  6043   buffer = XBUFFER (WINDOW_BUFFER (w));
  6044 
  6045   style = (EQ (find_symbol_value (Qoverriding_text_conversion_style),
  6046                Qlambda)
  6047            ? BVAR (buffer, text_conversion_style)
  6048            : find_symbol_value (Qoverriding_text_conversion_style));
  6049 
  6050   if (NILP (style) || conversion_disabled_p ())
  6051     mode = ANDROID_IC_MODE_NULL;
  6052   else if (EQ (style, Qaction) || EQ (f->selected_window,
  6053                                       f->minibuffer_window))
  6054     mode = ANDROID_IC_MODE_ACTION;
  6055   else
  6056     mode = ANDROID_IC_MODE_TEXT;
  6057 
  6058   /* Remove any existing input method events that apply to FRAME from
  6059      the event queue.
  6060 
  6061      There's a small window between this and the call to
  6062      android_reset_ic between which more events can be generated.  */
  6063 
  6064   while (android_check_if_event (&event, android_event_is_for_frame, f))
  6065     {
  6066       switch (event.ime.operation)
  6067         {
  6068         case ANDROID_IME_COMMIT_TEXT:
  6069         case ANDROID_IME_FINISH_COMPOSING_TEXT:
  6070         case ANDROID_IME_SET_COMPOSING_TEXT:
  6071           xfree (event.ime.text);
  6072           break;
  6073 
  6074         default:
  6075           break;
  6076         }
  6077     }
  6078 
  6079   android_reset_ic (FRAME_ANDROID_WINDOW (f), mode);
  6080 
  6081   /* Clear extracted text flags.  Since the IM has been reinitialised,
  6082      it should no longer be displaying extracted text.  */
  6083   FRAME_ANDROID_OUTPUT (f)->extracted_text_flags = 0;
  6084 
  6085   /* Move its selection to the specified position.  */
  6086   android_update_selection (f, NULL);
  6087 }
