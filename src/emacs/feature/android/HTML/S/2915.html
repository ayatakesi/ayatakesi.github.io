    26 set_binary_mode (int fd, int mode)
    27 {
    28   if (isatty (fd))
    29     /* If FD refers to a console (not a pipe, not a regular file),
    30        O_TEXT is the only reasonable mode, both on input and on output.
    31        Silently ignore the request.  If we were to return -1 here,
    32        all programs that use xset_binary_mode would fail when run
    33        with console input or console output.  */
    34     return O_TEXT;
    35   else
    36     return __gl_setmode (fd, mode);
    37 }
