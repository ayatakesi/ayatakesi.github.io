   151   public AlertDialog
   152   toAlertDialog (Context context)
   153   {
   154     AlertDialog dialog;
   155     int size, styleId;
   156     int[] attrs;
   157     EmacsButton button;
   158     EmacsDialogButtonLayout layout;
   159     Button buttonView;
   160     ViewGroup.LayoutParams layoutParams;
   161     Theme theme;
   162     TypedArray attributes;
   163 
   164     size = buttons.size ();
   165     styleId = -1;
   166 
   167     if (size <= 3)
   168       {
   169         dialog = new AlertDialog.Builder (context).create ();
   170         dialog.setMessage (text);
   171         dialog.setCancelable (true);
   172         dialog.setOnDismissListener (this);
   173 
   174         if (title != null)
   175           dialog.setTitle (title);
   176 
   177         /* There are less than 4 buttons.  Add the buttons the way
   178            Android intends them to be added.  */
   179 
   180         if (size >= 1)
   181           {
   182             button = buttons.get (0);
   183             dialog.setButton (DialogInterface.BUTTON_POSITIVE,
   184                               button.name, button);
   185           }
   186 
   187         if (size >= 2)
   188           {
   189             button = buttons.get (1);
   190             dialog.setButton (DialogInterface.BUTTON_NEGATIVE,
   191                               button.name, button);
   192           }
   193 
   194         if (size >= 3)
   195           {
   196             button = buttons.get (2);
   197             dialog.setButton (DialogInterface.BUTTON_NEUTRAL,
   198                               button.name, button);
   199           }
   200       }
   201     else
   202       {
   203         /* There are more than 3 buttons.  Add them all to a special
   204            container widget that handles wrapping.  First, create the
   205            layout.  */
   206 
   207         layout = new EmacsDialogButtonLayout (context);
   208         layoutParams
   209           = new FrameLayout.LayoutParams (ViewGroup.LayoutParams.MATCH_PARENT,
   210                                           ViewGroup.LayoutParams.WRAP_CONTENT);
   211         layout.setLayoutParams (layoutParams);
   212 
   213         /* Add that layout to the dialog's custom view.
   214 
   215            android.R.id.custom is documented to work.  But looking it
   216            up returns NULL, so setView must be used instead.  */
   217 
   218         dialog = new AlertDialog.Builder (context).setView (layout).create ();
   219         dialog.setMessage (text);
   220         dialog.setCancelable (true);
   221         dialog.setOnDismissListener (this);
   222 
   223         if (title != null)
   224           dialog.setTitle (title);
   225 
   226         /* Now that the dialog has been created, set the style of each
   227            custom button to match the usual dialog buttons found on
   228            Android 5 and later.  */
   229 
   230         if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP)
   231           {
   232             /* Obtain the Theme associated with the dialog.  */
   233             theme = dialog.getContext ().getTheme ();
   234 
   235             /* Resolve the dialog button style.  */
   236             attrs
   237               = new int [] { android.R.attr.buttonBarNeutralButtonStyle, };
   238 
   239             try
   240               {
   241                 attributes = theme.obtainStyledAttributes (attrs);
   242 
   243                 /* Look for the style ID.  Default to -1 if it could
   244                    not be found.  */
   245                 styleId = attributes.getResourceId (0, -1);
   246 
   247                 /* Now clean up the TypedAttributes object.  */
   248                 attributes.recycle ();
   249               }
   250             catch (NotFoundException e)
   251               {
   252                 /* Nothing to do here.  */
   253               }
   254           }
   255 
   256         /* Create each button and add it to the layout.  Set the style
   257            if necessary.  */
   258 
   259         for (EmacsButton emacsButton : buttons)
   260           {
   261             if (styleId == -1)
   262               /* No specific style... */
   263               buttonView = new Button (context);
   264             else
   265               /* Use the given styleId.  */
   266               buttonView = new Button (context, null, 0, styleId);
   267 
   268             /* Set the text and on click handler.  */
   269             buttonView.setText (emacsButton.name);
   270             buttonView.setOnClickListener (emacsButton);
   271             buttonView.setEnabled (emacsButton.enabled);
   272             layout.addView (buttonView);
   273           }
   274       }
   275 
   276     return dialog;
   277   }

   281   @SuppressWarnings("deprecation")
   282   private boolean
   283   display1 ()
   284   {
   285     Context context;
   286     int size, type;
   287     Button buttonView;
   288     EmacsButton button;
   289     AlertDialog dialog;
   290     Window window;
   291 
   292     if (EmacsActivity.focusedActivities.isEmpty ())
   293       {
   294         /* If focusedActivities is empty then this dialog may have
   295            been displayed immediately after another popup dialog was
   296            dismissed.  Or Emacs might legitimately be in the
   297            background, possibly displaying this popup in response to
   298            an Emacsclient request.  Try the service context if it will
   299            work, then any focused EmacsOpenActivity, and finally the
   300            last EmacsActivity to be focused.  */
   301 
   302         Log.d (TAG, "display1: no focused activities...");
   303         Log.d (TAG, ("display1: EmacsOpenActivity.currentActivity: "
   304                      + EmacsOpenActivity.currentActivity));
   305 
   306         if (Build.VERSION.SDK_INT < Build.VERSION_CODES.M
   307             || Settings.canDrawOverlays (EmacsService.SERVICE))
   308           context = EmacsService.SERVICE;
   309         else if (EmacsOpenActivity.currentActivity != null)
   310           context = EmacsOpenActivity.currentActivity;
   311         else
   312           context = EmacsActivity.lastFocusedActivity;
   313 
   314         if (context == null)
   315           return false;
   316       }
   317     else
   318       /* Display using the activity context when Emacs is in the
   319          foreground, as this allows the dialog to be dismissed more
   320          consistently.  */
   321       context = EmacsActivity.focusedActivities.get (0);
   322 
   323     Log.d (TAG, "display1: using context " + context);
   324 
   325     dialog = dismissDialog = toAlertDialog (context);
   326 
   327     try
   328       {
   329         if (context == EmacsService.SERVICE)
   330           {
   331             /* Apply the system alert window type to make sure this
   332                dialog can be displayed.  */
   333 
   334             window = dialog.getWindow ();
   335             type = (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O
   336                     ? WindowManager.LayoutParams.TYPE_APPLICATION_OVERLAY
   337                     : WindowManager.LayoutParams.TYPE_PHONE);
   338             window.setType (type);
   339           }
   340 
   341         dismissDialog.show ();
   342       }
   343     catch (Exception exception)
   344       {
   345         /* This can happen when the system decides Emacs is not in the
   346            foreground any longer.  */
   347         return false;
   348       }
   349 
   350     /* If there are less than four buttons, then they must be
   351        individually enabled or disabled after the dialog is
   352        displayed.  */
   353     size = buttons.size ();
   354 
   355     if (size <= 3)
   356       {
   357         if (size >= 1)
   358           {
   359             button = buttons.get (0);
   360             buttonView
   361               = dialog.getButton (DialogInterface.BUTTON_POSITIVE);
   362             buttonView.setEnabled (button.enabled);
   363           }
   364 
   365         if (size >= 2)
   366           {
   367             button = buttons.get (1);
   368             buttonView
   369               = dialog.getButton (DialogInterface.BUTTON_NEGATIVE);
   370             buttonView.setEnabled (button.enabled);
   371           }
   372 
   373         if (size >= 3)
   374           {
   375             button = buttons.get (2);
   376             buttonView
   377               = dialog.getButton (DialogInterface.BUTTON_NEUTRAL);
   378             buttonView.setEnabled (button.enabled);
   379           }
   380       }
   381 
   382     return true;
   383   }
