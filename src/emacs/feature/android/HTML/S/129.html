    26 public final class EmacsDrawLine
    27 {
    28   public static void
    29   perform (EmacsDrawable drawable, EmacsGC gc,
    30            int x, int y, int x2, int y2)
    31   {
    32     Rect rect;
    33     Canvas canvas;
    34     Paint paint;
    35 
    36     /* TODO implement stippling.  */
    37     if (gc.fill_style == EmacsGC.GC_FILL_OPAQUE_STIPPLED)
    38       return;
    39 
    40     paint = gc.gcPaint;
    41     rect = new Rect (Math.min (x, x2 + 1),
    42                      Math.min (y, y2 + 1),
    43                      Math.max (x2 + 1, x),
    44                      Math.max (y2 + 1, y));
    45     canvas = drawable.lockCanvas (gc);
    46 
    47     if (canvas == null)
    48       return;
    49 
    50     paint.setStyle (Paint.Style.STROKE);
    51 
    52     /* Since drawLine has PostScript style behavior, adjust the
    53        coordinates appropriately.  */
    54 
    55     if (gc.clip_mask == null)
    56       canvas.drawLine ((float) x, (float) y + 0.5f,
    57                        (float) x2 + 0.5f, (float) y2 + 0.5f,
    58                        paint);
    59 
    60     /* DrawLine with clip mask not implemented; it is not used by
    61        Emacs.  */
    62     drawable.damageRect (rect);
    63   }
    64 }
