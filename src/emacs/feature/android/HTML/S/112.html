    28 public final class EmacsThread extends Thread

    34   private String extraStartupArgument;

    37   private Runnable paramsClosure;

    39   /* Whether or not to open a file after starting Emacs.  */
    40   private String fileToOpen;
    41 
    42   public
    43   EmacsThread (EmacsService service, Runnable paramsClosure,
    44                String extraStartupArgument, String fileToOpen)
    45   {
    46     super ("Emacs main thread");
    47     this.extraStartupArgument = extraStartupArgument;
    48     this.paramsClosure = paramsClosure;
    49     this.fileToOpen = fileToOpen;
    50   }
    51 
    52   @Override
    53   public void
    54   run ()
    55   {
    56     String args[];
    57 
    58     if (fileToOpen == null)
    59       {
    60         if (extraStartupArgument == null)
    61           args = new String[] { "libandroid-emacs.so", };
    62         else
    63           args = new String[] { "libandroid-emacs.so",
    64                                 extraStartupArgument, };
    65       }
    66     else
    67       {
    68         if (extraStartupArgument == null)
    69           args = new String[] { "libandroid-emacs.so",
    70                                 fileToOpen, };
    71         else
    72           args = new String[] { "libandroid-emacs.so",
    73                                 extraStartupArgument,
    74                                 fileToOpen, };
    75       }
    76 
    77     paramsClosure.run ();
    78 
    79     /* Run the native code now.  */
    80     Log.d (TAG, "run: " + Arrays.toString (args));
    81     EmacsNative.initEmacs (args, EmacsApplication.dumpFileName,
    82                            Build.VERSION.SDK_INT);
    83   }
    84 };
