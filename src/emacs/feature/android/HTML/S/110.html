   365   public synchronized void
   366   mapWindow ()
   367   {
   368     final int width, height;
   369 
   370     if (isMapped)
   371       return;
   372 
   373     isMapped = true;
   374     width = rect.width ();
   375     height = rect.height ();
   376 
   377     if (parent == null)
   378       {
   379         EmacsService.SERVICE.runOnUiThread (new Runnable () {
   380             @Override
   381             public void
   382             run ()
   383             {
   384               EmacsWindowAttachmentManager manager;
   385               WindowManager windowManager;
   386               Context ctx;
   387               Object tem;
   388               WindowManager.LayoutParams params;
   389 
   390               /* Make the view visible, first of all.  */
   391               view.setVisibility (View.VISIBLE);
   392 
   393               if (!overrideRedirect)
   394                 {
   395                   manager = EmacsWindowAttachmentManager.MANAGER;
   396 
   397                   /* If parent is the root window, notice that there are new
   398                      children available for interested activites to pick
   399                      up.  */
   400                   manager.registerWindow (EmacsWindow.this);
   401 
   402                   if (!getDontFocusOnMap ())
   403                     /* Eventually this should check no-focus-on-map.  */
   404                     view.requestFocus ();
   405                 }
   406               else
   407                 {
   408                   /* But if the window is an override-redirect window,
   409                      then:
   410 
   411                      - Find an activity that is currently active.
   412 
   413                      - Map the window as a panel on top of that
   414                        activity using the system window manager.  */
   415 
   416                   ctx = findSuitableActivityContext ();
   417                   tem = ctx.getSystemService (Context.WINDOW_SERVICE);
   418                   windowManager = (WindowManager) tem;
   419 
   420                   /* Calculate layout parameters.  */
   421                   params = getWindowLayoutParams ();
   422                   view.setLayoutParams (params);
   423 
   424                   /* Attach the view.  */
   425                   try
   426                     {
   427                       view.prepareForLayout (width, height);
   428                       windowManager.addView (view, params);
   429 
   430                       /* Record the window manager being used in the
   431                          EmacsWindow object.  */
   432                       EmacsWindow.this.windowManager = windowManager;
   433                     }
   434                   catch (Exception e)
   435                     {
   436                       Log.w (TAG,
   437                              "failed to attach override-redirect window, " + e);
   438                     }
   439                 }
   440             }
   441           });
   442       }
   443     else
   444       {
   445         /* Do the same thing as above, but don't register this
   446            window.  */
   447         EmacsService.SERVICE.runOnUiThread (new Runnable () {
   448             @Override
   449             public void
   450             run ()
   451             {
   452               /* Prior to mapping the view, set its measuredWidth and
   453                  measuredHeight to some reasonable value, in order to
   454                  avoid excessive bitmap dirtying.  */
   455 
   456               view.prepareForLayout (width, height);
   457               view.setVisibility (View.VISIBLE);
   458 
   459               if (!getDontFocusOnMap ())
   460                 view.requestFocus ();
   461             }
   462           });
   463       }
   464   }

   566   /* Return the modifier mask associated with the specified keyboard
   567      input EVENT.  Replace bits corresponding to Left or Right keys
   568      with their corresponding general modifier bits.  */
   569 
   570   private int
   571   eventModifiers (KeyEvent event)
   572   {
   573     int state;
   574 
   575     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.HONEYCOMB_MR2)
   576       state = event.getModifiers ();
   577     else
   578       {
   579         /* Replace this with getMetaState and manual
   580            normalization.  */
   581         state = event.getMetaState ();
   582 
   583         /* Normalize the state by setting the generic modifier bit if
   584            either a left or right modifier is pressed.  */
   585 
   586         if ((state & KeyEvent.META_ALT_LEFT_ON) != 0
   587             || (state & KeyEvent.META_ALT_RIGHT_ON) != 0)
   588           state |= KeyEvent.META_ALT_MASK;
   589 
   590         if ((state & KeyEvent.META_CTRL_LEFT_ON) != 0
   591             || (state & KeyEvent.META_CTRL_RIGHT_ON) != 0)
   592           state |= KeyEvent.META_CTRL_MASK;
   593       }
   594 
   595     return state;
   596   }

   627   onKeyDown (int keyCode, KeyEvent event)
   628   {
   629     int state, state_1;

   713   /* Look through the button state to determine what button EVENT was
   714      generated from.  DOWN is true if EVENT is a button press event,
   715      false otherwise.  Value is the X number of the button.  */
   716 
   717   private int
   718   whatButtonWasIt (MotionEvent event, boolean down)
   719   {
   720     int eventState, notIn;
   721 
   722     if (Build.VERSION.SDK_INT
   723         < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   724       /* Earlier versions of Android only support one mouse
   725          button.  */
   726       return 1;
   727 
   728     eventState = event.getButtonState ();
   729     notIn = (down ? eventState & ~lastButtonState
   730              : lastButtonState & ~eventState);
   731 
   732     if ((notIn & MotionEvent.BUTTON_PRIMARY) != 0)
   733       return 1;
   734 
   735     if ((notIn & MotionEvent.BUTTON_SECONDARY) != 0)
   736       return 3;
   737 
   738     if ((notIn & MotionEvent.BUTTON_TERTIARY) != 0)
   739       return 2;
   740 
   741     /* Not a real value.  */
   742     return 4;
   743   }
   744 
   745   /* Return the ID of the pointer which changed in EVENT.  Value is -1
   746      if it could not be determined, else the pointer that changed, or
   747      -2 if -1 would have been returned, but there is also a pointer
   748      that is a mouse.  */
   749 
   750   private int
   751   figureChange (MotionEvent event)
   752   {
   753     int pointerID, i, truncatedX, truncatedY, pointerIndex;
   754     Coordinate coordinate;
   755     boolean mouseFlag;
   756 
   757     /* pointerID is always initialized but the Java compiler is too
   758        dumb to know that.  */
   759     pointerID = -1;
   760     mouseFlag = false;
   761 
   762     switch (event.getActionMasked ())
   763       {
   764       case MotionEvent.ACTION_DOWN:
   765         /* Primary pointer pressed with index 0.  */
   766 
   767         /* Detect mice.  If this is a mouse event, give it to
   768            onSomeKindOfMotionEvent.  */
   769         if ((Build.VERSION.SDK_INT
   770              >= Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   771             && event.getToolType (0) == MotionEvent.TOOL_TYPE_MOUSE)
   772           return -2;
   773 
   774         pointerID = event.getPointerId (0);
   775         pointerMap.put (pointerID,
   776                         new Coordinate ((int) event.getX (0),
   777                                         (int) event.getY (0)));
   778         break;
   779 
   780       case MotionEvent.ACTION_UP:
   781 
   782         /* Detect mice.  If this is a mouse event, give it to
   783            onSomeKindOfMotionEvent.  */
   784         if ((Build.VERSION.SDK_INT
   785              >= Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   786             && event.getToolType (0) == MotionEvent.TOOL_TYPE_MOUSE)
   787           return -2;
   788 
   789         /* Primary pointer released with index 0.  */
   790         pointerID = event.getPointerId (0);
   791         pointerMap.remove (pointerID);
   792         break;
   793 
   794       case MotionEvent.ACTION_POINTER_DOWN:
   795         /* New pointer.  Find the pointer ID from the index and place
   796            it in the map.  */
   797         pointerIndex = event.getActionIndex ();
   798         pointerID = event.getPointerId (pointerIndex);
   799         pointerMap.put (pointerID,
   800                         new Coordinate ((int) event.getX (pointerIndex),
   801                                         (int) event.getY (pointerIndex)));
   802         break;
   803 
   804       case MotionEvent.ACTION_POINTER_UP:
   805         /* Pointer removed.  Remove it from the map.  */
   806         pointerIndex = event.getActionIndex ();
   807         pointerID = event.getPointerId (pointerIndex);
   808         pointerMap.remove (pointerID);
   809         break;
   810 
   811       default:
   812 
   813         /* Loop through each pointer in the event.  */
   814         for (i = 0; i < event.getPointerCount (); ++i)
   815           {
   816             pointerID = event.getPointerId (i);
   817 
   818             /* Look up that pointer in the map.  */
   819             coordinate = pointerMap.get (pointerID);
   820 
   821             if (coordinate != null)
   822               {
   823                 /* See if coordinates have changed.  */
   824                 truncatedX = (int) event.getX (i);
   825                 truncatedY = (int) event.getY (i);
   826 
   827                 if (truncatedX != coordinate.x
   828                     || truncatedY != coordinate.y)
   829                   {
   830                     /* The pointer changed.  Update the coordinate and
   831                        break out of the loop.  */
   832                     coordinate.x = truncatedX;
   833                     coordinate.y = truncatedY;
   834 
   835                     break;
   836                   }
   837               }
   838 
   839             /* See if this is a mouse.  If so, set the mouseFlag.  */
   840             if ((Build.VERSION.SDK_INT
   841                  >= Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   842                 && event.getToolType (i) == MotionEvent.TOOL_TYPE_MOUSE)
   843               mouseFlag = true;
   844           }
   845 
   846         /* Set the pointer ID to -1 if the loop failed to find any
   847            changed pointer.  If a mouse pointer was found, set it to
   848            -2.  */
   849         if (i == event.getPointerCount ())
   850           pointerID = (mouseFlag ? -2 : -1);
   851       }
   852 
   853     /* Return the pointer ID.  */
   854     return pointerID;
   855   }

   857   public boolean
   858   onTouchEvent (MotionEvent event)
   859   {
   860     int pointerID, index;
   861 
   862     /* Extract the ``touch ID'' (or in Android, the ``pointer
   863        ID''.) */
   864     pointerID = figureChange (event);
   865 
   866     if (pointerID < 0)
   867       {
   868         /* If this is a mouse event, give it to
   869            onSomeKindOfMotionEvent.  */
   870         if (pointerID == -2)
   871           return onSomeKindOfMotionEvent (event);
   872 
   873         return false;
   874       }
   875 
   876     /* Find the pointer index corresponding to the event.  */
   877     index = event.findPointerIndex (pointerID);
   878 
   879     switch (event.getActionMasked ())
   880       {
   881       case MotionEvent.ACTION_DOWN:
   882       case MotionEvent.ACTION_POINTER_DOWN:
   883         /* Touch down event.  */
   884         EmacsNative.sendTouchDown (this.handle, (int) event.getX (index),
   885                                    (int) event.getY (index),
   886                                    event.getEventTime (), pointerID);
   887         return true;
   888 
   889       case MotionEvent.ACTION_UP:
   890       case MotionEvent.ACTION_POINTER_UP:
   891       case MotionEvent.ACTION_CANCEL:
   892         /* Touch up event.  Android documentation says ACTION_CANCEL
   893            should be treated as more or less equivalent to ACTION_UP,
   894            so that is what is done here.  */
   895         EmacsNative.sendTouchUp (this.handle, (int) event.getX (index),
   896                                  (int) event.getY (index),
   897                                  event.getEventTime (), pointerID);
   898         return true;
   899 
   900       case MotionEvent.ACTION_MOVE:
   901         /* Pointer motion event.  */
   902         EmacsNative.sendTouchMove (this.handle, (int) event.getX (index),
   903                                    (int) event.getY (index),
   904                                    event.getEventTime (), pointerID);
   905         return true;
   906       }
   907 
   908     return false;
   909   }

   911   public boolean
   912   onSomeKindOfMotionEvent (MotionEvent event)
   913   {
   914     /* isFromSource is not available until API level 18.  */
   915 
   916     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.JELLY_BEAN_MR2)
   917       {
   918         if (!event.isFromSource (InputDevice.SOURCE_CLASS_POINTER))
   919           return false;
   920       }
   921     else if (event.getSource () != InputDevice.SOURCE_CLASS_POINTER)
   922       return false;
   923 
   924     switch (event.getAction ())
   925       {
   926       case MotionEvent.ACTION_HOVER_ENTER:
   927         EmacsNative.sendEnterNotify (this.handle, (int) event.getX (),
   928                                      (int) event.getY (),
   929                                      event.getEventTime ());
   930         return true;
   931 
   932       case MotionEvent.ACTION_MOVE:
   933       case MotionEvent.ACTION_HOVER_MOVE:
   934         EmacsNative.sendMotionNotify (this.handle, (int) event.getX (),
   935                                       (int) event.getY (),
   936                                       event.getEventTime ());
   937         return true;
   938 
   939       case MotionEvent.ACTION_HOVER_EXIT:
   940 
   941         /* If the exit event comes from a button press, its button
   942            state will have extra bits compared to the last known
   943            button state.  Since the exit event will interfere with
   944            tool bar button presses, ignore such splurious events.  */
   945 
   946         if ((event.getButtonState () & ~lastButtonState) == 0)
   947           EmacsNative.sendLeaveNotify (this.handle, (int) event.getX (),
   948                                        (int) event.getY (),
   949                                        event.getEventTime ());
   950 
   951         return true;
   952 
   953       case MotionEvent.ACTION_BUTTON_PRESS:
   954         /* Find the button which was pressed.  */
   955         EmacsNative.sendButtonPress (this.handle, (int) event.getX (),
   956                                      (int) event.getY (),
   957                                      event.getEventTime (),
   958                                      motionEventModifiers (event),
   959                                      whatButtonWasIt (event, true));
   960 
   961         if (Build.VERSION.SDK_INT
   962             < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   963           return true;
   964 
   965         lastButtonState = event.getButtonState ();
   966         return true;
   967 
   968       case MotionEvent.ACTION_BUTTON_RELEASE:
   969         /* Find the button which was released.  */
   970         EmacsNative.sendButtonRelease (this.handle, (int) event.getX (),
   971                                        (int) event.getY (),
   972                                        event.getEventTime (),
   973                                        motionEventModifiers (event),
   974                                        whatButtonWasIt (event, false));
   975 
   976         if (Build.VERSION.SDK_INT
   977             < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   978           return true;
   979 
   980         lastButtonState = event.getButtonState ();
   981         return true;
   982 
   983       case MotionEvent.ACTION_DOWN:
   984         /* Emacs must return true even though touch events are not
   985            handled here, because the value of this function is used by
   986            the system to decide whether or not Emacs gets ACTION_MOVE
   987            events.  */
   988         return true;
   989 
   990       case MotionEvent.ACTION_UP:
   991         /* However, if ACTION_UP reports a different button state from
   992            the last known state, look up which button was released and
   993            send a ButtonRelease event; this is to work around a bug in
   994            the framework where real ACTION_BUTTON_RELEASE events are
   995            not delivered.  */
   996 
   997         if (Build.VERSION.SDK_INT
   998             < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   999           return true;
  1000 
  1001         if (event.getButtonState () == 0 && lastButtonState != 0)
  1002           {
  1003             EmacsNative.sendButtonRelease (this.handle, (int) event.getX (),
  1004                                            (int) event.getY (),
  1005                                            event.getEventTime (),
  1006                                            motionEventModifiers (event),
  1007                                            whatButtonWasIt (event, false));
  1008             lastButtonState = event.getButtonState ();
  1009           }
  1010 
  1011         return true;
  1012 
  1013       case MotionEvent.ACTION_SCROLL:
  1014         /* Send a scroll event with the specified deltas.  */
  1015         EmacsNative.sendWheel (this.handle, (int) event.getX (),
  1016                                (int) event.getY (),
  1017                                event.getEventTime (),
  1018                                motionEventModifiers (event),
  1019                                event.getAxisValue (MotionEvent.AXIS_HSCROLL),
  1020                                event.getAxisValue (MotionEvent.AXIS_VSCROLL));
  1021         return true;
  1022       }
  1023 
  1024     return false;
  1025   }

  1027   public synchronized void
  1028   reparentTo (final EmacsWindow otherWindow, int x, int y)
  1029   {
  1030     int width, height;
  1031 
  1032     /* Reparent this window to the other window.  */
  1033 
  1034     if (parent != null)
  1035       parent.children.remove (this);
  1036 
  1037     if (otherWindow != null)
  1038       otherWindow.children.add (this);
  1039 
  1040     parent = otherWindow;
  1041 
  1042     /* Move this window to the new location.  */
  1043     width = rect.width ();
  1044     height = rect.height ();
  1045     rect.left = x;
  1046     rect.top = y;
  1047     rect.right = x + width;
  1048     rect.bottom = y + height;
  1049 
  1050     /* Now do the work necessary on the UI thread to reparent the
  1051        window.  */
  1052     EmacsService.SERVICE.runOnUiThread (new Runnable () {
  1053         @Override
  1054         public void
  1055         run ()
  1056         {
  1057           EmacsWindowAttachmentManager manager;
  1058           ViewManager parent;
  1059 
  1060           /* First, detach this window if necessary.  */
  1061           manager = EmacsWindowAttachmentManager.MANAGER;
  1062           manager.detachWindow (EmacsWindow.this);
  1063 
  1064           /* Also unparent this view.  */
  1065 
  1066           /* If the window manager is set, use that instead.  */
  1067           if (windowManager != null)
  1068             parent = windowManager;
  1069           else
  1070             parent = (ViewManager) view.getParent ();
  1071           windowManager = null;
  1072 
  1073           if (parent != null)
  1074             parent.removeView (view);
  1075 
  1076           /* Next, either add this window as a child of the new
  1077              parent's view, or make it available again.  */
  1078           if (otherWindow != null)
  1079             otherWindow.view.addView (view);
  1080           else if (EmacsWindow.this.isMapped)
  1081             manager.registerWindow (EmacsWindow.this);
  1082 
  1083           /* Request relayout.  */
  1084           view.requestLayout ();
  1085         }
  1086       });
  1087   }
