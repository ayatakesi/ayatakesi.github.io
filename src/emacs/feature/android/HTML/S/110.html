   365   public synchronized void
   366   mapWindow ()
   367   {
   368     final int width, height;
   369 
   370     if (isMapped)
   371       return;
   372 
   373     isMapped = true;
   374     width = rect.width ();
   375     height = rect.height ();
   376 
   377     if (parent == null)
   378       {
   379         EmacsService.SERVICE.runOnUiThread (new Runnable () {
   380             @Override
   381             public void
   382             run ()
   383             {
   384               EmacsWindowAttachmentManager manager;
   385               WindowManager windowManager;
   386               Context ctx;
   387               Object tem;
   388               WindowManager.LayoutParams params;
   389 
   390               /* Make the view visible, first of all.  */
   391               view.setVisibility (View.VISIBLE);
   392 
   393               if (!overrideRedirect)
   394                 {
   395                   manager = EmacsWindowAttachmentManager.MANAGER;
   396 
   397                   /* If parent is the root window, notice that there are new
   398                      children available for interested activites to pick
   399                      up.  */
   400                   manager.registerWindow (EmacsWindow.this);
   401 
   402                   if (!getDontFocusOnMap ())
   403                     /* Eventually this should check no-focus-on-map.  */
   404                     view.requestFocus ();
   405                 }
   406               else
   407                 {
   408                   /* But if the window is an override-redirect window,
   409                      then:
   410 
   411                      - Find an activity that is currently active.
   412 
   413                      - Map the window as a panel on top of that
   414                        activity using the system window manager.  */
   415 
   416                   ctx = findSuitableActivityContext ();
   417                   tem = ctx.getSystemService (Context.WINDOW_SERVICE);
   418                   windowManager = (WindowManager) tem;
   419 
   420                   /* Calculate layout parameters.  */
   421                   params = getWindowLayoutParams ();
   422                   view.setLayoutParams (params);
   423 
   424                   /* Attach the view.  */
   425                   try
   426                     {
   427                       view.prepareForLayout (width, height);
   428                       windowManager.addView (view, params);
   429 
   430                       /* Record the window manager being used in the
   431                          EmacsWindow object.  */
   432                       EmacsWindow.this.windowManager = windowManager;
   433                     }
   434                   catch (Exception e)
   435                     {
   436                       Log.w (TAG,
   437                              "failed to attach override-redirect window, " + e);
   438                     }
   439                 }
   440             }
   441           });
   442       }
   443     else
   444       {
   445         /* Do the same thing as above, but don't register this
   446            window.  */
   447         EmacsService.SERVICE.runOnUiThread (new Runnable () {
   448             @Override
   449             public void
   450             run ()
   451             {
   452               /* Prior to mapping the view, set its measuredWidth and
   453                  measuredHeight to some reasonable value, in order to
   454                  avoid excessive bitmap dirtying.  */
   455 
   456               view.prepareForLayout (width, height);
   457               view.setVisibility (View.VISIBLE);
   458 
   459               if (!getDontFocusOnMap ())
   460                 view.requestFocus ();
   461             }
   462           });
   463       }
   464   }

   691   /* Look through the button state to determine what button EVENT was
   692      generated from.  DOWN is true if EVENT is a button press event,
   693      false otherwise.  Value is the X number of the button.  */
   694 
   695   private int
   696   whatButtonWasIt (MotionEvent event, boolean down)
   697   {
   698     int eventState, notIn;
   699 
   700     if (Build.VERSION.SDK_INT
   701         < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   702       /* Earlier versions of Android only support one mouse
   703          button.  */
   704       return 1;
   705 
   706     eventState = event.getButtonState ();
   707     notIn = (down ? eventState & ~lastButtonState
   708              : lastButtonState & ~eventState);
   709 
   710     if ((notIn & MotionEvent.BUTTON_PRIMARY) != 0)
   711       return 1;
   712 
   713     if ((notIn & MotionEvent.BUTTON_SECONDARY) != 0)
   714       return 3;
   715 
   716     if ((notIn & MotionEvent.BUTTON_TERTIARY) != 0)
   717       return 2;
   718 
   719     /* Not a real value.  */
   720     return 4;
   721   }

   734 
   735     /* pointerID is always initialized but the Java compiler is too
   736        dumb to know that.  */
   737     pointerID = -1;
   738     mouseFlag = false;
   739 
   740     switch (event.getActionMasked ())
   741       {
   742       case MotionEvent.ACTION_DOWN:
   743         /* Primary pointer pressed with index 0.  */
   744 
   745         /* Detect mice.  If this is a mouse event, give it to
   746            onSomeKindOfMotionEvent.  */
   747         if ((Build.VERSION.SDK_INT
   748              >= Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   749             && event.getToolType (0) == MotionEvent.TOOL_TYPE_MOUSE)
   750           return -2;
   751 
   752         pointerID = event.getPointerId (0);
   753         pointerMap.put (pointerID,
   754                         new Coordinate ((int) event.getX (0),
   755                                         (int) event.getY (0)));
   756         break;
   757 
   758       case MotionEvent.ACTION_UP:
   759 
   760         /* Detect mice.  If this is a mouse event, give it to
   761            onSomeKindOfMotionEvent.  */
   762         if ((Build.VERSION.SDK_INT
   763              >= Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   764             && event.getToolType (0) == MotionEvent.TOOL_TYPE_MOUSE)
   765           return -2;
   766 
   767         /* Primary pointer released with index 0.  */
   768         pointerID = event.getPointerId (0);
   769         pointerMap.remove (pointerID);
   770         break;
   771 
   772       case MotionEvent.ACTION_POINTER_DOWN:
   773         /* New pointer.  Find the pointer ID from the index and place
   774            it in the map.  */
   775         pointerIndex = event.getActionIndex ();
   776         pointerID = event.getPointerId (pointerIndex);
   777         pointerMap.put (pointerID,
   778                         new Coordinate ((int) event.getX (pointerIndex),
   779                                         (int) event.getY (pointerIndex)));
   780         break;
   781 
   782       case MotionEvent.ACTION_POINTER_UP:
   783         /* Pointer removed.  Remove it from the map.  */
   784         pointerIndex = event.getActionIndex ();
   785         pointerID = event.getPointerId (pointerIndex);
   786         pointerMap.remove (pointerID);
   787         break;
   788 
   789       default:
   790 
   791         /* Loop through each pointer in the event.  */
   792         for (i = 0; i < event.getPointerCount (); ++i)
   793           {
   794             pointerID = event.getPointerId (i);
   795 
   796             /* Look up that pointer in the map.  */
   797             coordinate = pointerMap.get (pointerID);
   798 
   799             if (coordinate != null)
   800               {
   801                 /* See if coordinates have changed.  */
   802                 truncatedX = (int) event.getX (i);
   803                 truncatedY = (int) event.getY (i);
   804 
   805                 if (truncatedX != coordinate.x
   806                     || truncatedY != coordinate.y)
   807                   {
   808                     /* The pointer changed.  Update the coordinate and
   809                        break out of the loop.  */
   810                     coordinate.x = truncatedX;
   811                     coordinate.y = truncatedY;
   812 
   813                     break;
   814                   }
   815               }
   816 
   817             /* See if this is a mouse.  If so, set the mouseFlag.  */
   818             if ((Build.VERSION.SDK_INT
   819                  >= Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   820                 && event.getToolType (i) == MotionEvent.TOOL_TYPE_MOUSE)
   821               mouseFlag = true;
   822           }
   823 
   824         /* Set the pointer ID to -1 if the loop failed to find any
   825            changed pointer.  If a mouse pointer was found, set it to
   826            -2.  */
   827         if (i == event.getPointerCount ())
   828           pointerID = (mouseFlag ? -2 : -1);
   829       }
   830 
   831     /* Return the pointer ID.  */
   832     return pointerID;
   833   }

   835   public boolean
   836   onTouchEvent (MotionEvent event)
   837   {
   838     int pointerID, index;
   839 
   840     /* Extract the ``touch ID'' (or in Android, the ``pointer
   841        ID''.) */
   842     pointerID = figureChange (event);
   843 
   844     if (pointerID < 0)
   845       {
   846         /* If this is a mouse event, give it to
   847            onSomeKindOfMotionEvent.  */
   848         if (pointerID == -2)
   849           return onSomeKindOfMotionEvent (event);
   850 
   851         return false;
   852       }
   853 
   854     /* Find the pointer index corresponding to the event.  */
   855     index = event.findPointerIndex (pointerID);
   856 
   857     switch (event.getActionMasked ())
   858       {
   859       case MotionEvent.ACTION_DOWN:
   860       case MotionEvent.ACTION_POINTER_DOWN:
   861         /* Touch down event.  */
   862         EmacsNative.sendTouchDown (this.handle, (int) event.getX (index),
   863                                    (int) event.getY (index),
   864                                    event.getEventTime (), pointerID);
   865         return true;
   866 
   867       case MotionEvent.ACTION_UP:
   868       case MotionEvent.ACTION_POINTER_UP:
   869       case MotionEvent.ACTION_CANCEL:
   870         /* Touch up event.  Android documentation says ACTION_CANCEL
   871            should be treated as more or less equivalent to ACTION_UP,
   872            so that is what is done here.  */
   873         EmacsNative.sendTouchUp (this.handle, (int) event.getX (index),
   874                                  (int) event.getY (index),
   875                                  event.getEventTime (), pointerID);
   876         return true;
   877 
   878       case MotionEvent.ACTION_MOVE:
   879         /* Pointer motion event.  */
   880         EmacsNative.sendTouchMove (this.handle, (int) event.getX (index),
   881                                    (int) event.getY (index),
   882                                    event.getEventTime (), pointerID);
   883         return true;
   884       }
   885 
   886     return false;
   887   }

   889   public boolean
   890   onSomeKindOfMotionEvent (MotionEvent event)
   891   {
   892     /* isFromSource is not available until API level 18.  */
   893 
   894     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.JELLY_BEAN_MR2)
   895       {
   896         if (!event.isFromSource (InputDevice.SOURCE_CLASS_POINTER))
   897           return false;
   898       }
   899     else if (event.getSource () != InputDevice.SOURCE_CLASS_POINTER)
   900       return false;
   901 
   902     switch (event.getAction ())
   903       {
   904       case MotionEvent.ACTION_HOVER_ENTER:
   905         EmacsNative.sendEnterNotify (this.handle, (int) event.getX (),
   906                                      (int) event.getY (),
   907                                      event.getEventTime ());
   908         return true;
   909 
   910       case MotionEvent.ACTION_MOVE:
   911       case MotionEvent.ACTION_HOVER_MOVE:
   912         EmacsNative.sendMotionNotify (this.handle, (int) event.getX (),
   913                                       (int) event.getY (),
   914                                       event.getEventTime ());
   915         return true;
   916 
   917       case MotionEvent.ACTION_HOVER_EXIT:
   918 
   919         /* If the exit event comes from a button press, its button
   920            state will have extra bits compared to the last known
   921            button state.  Since the exit event will interfere with
   922            tool bar button presses, ignore such splurious events.  */
   923 
   924         if ((event.getButtonState () & ~lastButtonState) == 0)
   925           EmacsNative.sendLeaveNotify (this.handle, (int) event.getX (),
   926                                        (int) event.getY (),
   927                                        event.getEventTime ());
   928 
   929         return true;
   930 
   931       case MotionEvent.ACTION_BUTTON_PRESS:
   932         /* Find the button which was pressed.  */
   933         EmacsNative.sendButtonPress (this.handle, (int) event.getX (),
   934                                      (int) event.getY (),
   935                                      event.getEventTime (),
   936                                      lastModifiers,
   937                                      whatButtonWasIt (event, true));
   938 
   939         if (Build.VERSION.SDK_INT
   940             < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   941           return true;
   942 
   943         lastButtonState = event.getButtonState ();
   944         return true;
   945 
   946       case MotionEvent.ACTION_BUTTON_RELEASE:
   947         /* Find the button which was released.  */
   948         EmacsNative.sendButtonRelease (this.handle, (int) event.getX (),
   949                                        (int) event.getY (),
   950                                        event.getEventTime (),
   951                                        lastModifiers,
   952                                        whatButtonWasIt (event, false));
   953 
   954         if (Build.VERSION.SDK_INT
   955             < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   956           return true;
   957 
   958         lastButtonState = event.getButtonState ();
   959         return true;
   960 
   961       case MotionEvent.ACTION_DOWN:
   962         /* Emacs must return true even though touch events are not
   963            handled here, because the value of this function is used by
   964            the system to decide whether or not Emacs gets ACTION_MOVE
   965            events.  */
   966         return true;
   967 
   968       case MotionEvent.ACTION_UP:
   969         /* However, if ACTION_UP reports a different button state from
   970            the last known state, look up which button was released and
   971            send a ButtonRelease event; this is to work around a bug in
   972            the framework where real ACTION_BUTTON_RELEASE events are
   973            not delivered.  */
   974 
   975         if (Build.VERSION.SDK_INT
   976             < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   977           return true;
   978 
   979         if (event.getButtonState () == 0 && lastButtonState != 0)
   980           {
   981             EmacsNative.sendButtonRelease (this.handle, (int) event.getX (),
   982                                            (int) event.getY (),
   983                                            event.getEventTime (),
   984                                            lastModifiers,
   985                                            whatButtonWasIt (event, false));
   986             lastButtonState = event.getButtonState ();
   987           }
   988 
   989         return true;
   990 
   991       case MotionEvent.ACTION_SCROLL:
   992         /* Send a scroll event with the specified deltas.  */
   993         EmacsNative.sendWheel (this.handle, (int) event.getX (),
   994                                (int) event.getY (),
   995                                event.getEventTime (),
   996                                lastModifiers,
   997                                event.getAxisValue (MotionEvent.AXIS_HSCROLL),
   998                                event.getAxisValue (MotionEvent.AXIS_VSCROLL));
   999         return true;
  1000       }
  1001 
  1002     return false;
  1003   }

  1005   public synchronized void
  1006   reparentTo (final EmacsWindow otherWindow, int x, int y)
  1007   {
  1008     int width, height;
  1009 
  1010     /* Reparent this window to the other window.  */
  1011 
  1012     if (parent != null)
  1013       parent.children.remove (this);
  1014 
  1015     if (otherWindow != null)
  1016       otherWindow.children.add (this);
  1017 
  1018     parent = otherWindow;
  1019 
  1020     /* Move this window to the new location.  */
  1021     width = rect.width ();
  1022     height = rect.height ();
  1023     rect.left = x;
  1024     rect.top = y;
  1025     rect.right = x + width;
  1026     rect.bottom = y + height;
  1027 
  1028     /* Now do the work necessary on the UI thread to reparent the
  1029        window.  */
  1030     EmacsService.SERVICE.runOnUiThread (new Runnable () {
  1031         @Override
  1032         public void
  1033         run ()
  1034         {
  1035           EmacsWindowAttachmentManager manager;
  1036           ViewManager parent;
  1037 
  1038           /* First, detach this window if necessary.  */
  1039           manager = EmacsWindowAttachmentManager.MANAGER;
  1040           manager.detachWindow (EmacsWindow.this);
  1041 
  1042           /* Also unparent this view.  */
  1043 
  1044           /* If the window manager is set, use that instead.  */
  1045           if (windowManager != null)
  1046             parent = windowManager;
  1047           else
  1048             parent = (ViewManager) view.getParent ();
  1049           windowManager = null;
  1050 
  1051           if (parent != null)
  1052             parent.removeView (view);
  1053 
  1054           /* Next, either add this window as a child of the new
  1055              parent's view, or make it available again.  */
  1056           if (otherWindow != null)
  1057             otherWindow.view.addView (view);
  1058           else if (EmacsWindow.this.isMapped)
  1059             manager.registerWindow (EmacsWindow.this);
  1060 
  1061           /* Request relayout.  */
  1062           view.requestLayout ();
  1063         }
  1064       });
  1065   }
