   374   public synchronized void
   375   mapWindow ()
   376   {
   377     final int width, height;
   378 
   379     if (isMapped)
   380       return;
   381 
   382     isMapped = true;
   383     width = rect.width ();
   384     height = rect.height ();
   385 
   386     if (parent == null)
   387       {
   388         EmacsService.SERVICE.runOnUiThread (new Runnable () {
   389             @Override
   390             public void
   391             run ()
   392             {
   393               EmacsWindowAttachmentManager manager;
   394               WindowManager windowManager;
   395               Context ctx;
   396               Object tem;
   397               WindowManager.LayoutParams params;
   398 
   399               /* Make the view visible, first of all.  */
   400               view.setVisibility (View.VISIBLE);
   401 
   402               if (!overrideRedirect)
   403                 {
   404                   manager = EmacsWindowAttachmentManager.MANAGER;
   405 
   406                   /* If parent is the root window, notice that there are new
   407                      children available for interested activites to pick
   408                      up.  */
   409                   manager.registerWindow (EmacsWindow.this);
   410 
   411                   if (!getDontFocusOnMap ())
   412                     /* Eventually this should check no-focus-on-map.  */
   413                     view.requestFocus ();
   414                 }
   415               else
   416                 {
   417                   /* But if the window is an override-redirect window,
   418                      then:
   419 
   420                      - Find an activity that is currently active.
   421 
   422                      - Map the window as a panel on top of that
   423                        activity using the system window manager.  */
   424 
   425                   ctx = findSuitableActivityContext ();
   426                   tem = ctx.getSystemService (Context.WINDOW_SERVICE);
   427                   windowManager = (WindowManager) tem;
   428 
   429                   /* Calculate layout parameters.  */
   430                   params = getWindowLayoutParams ();
   431                   view.setLayoutParams (params);
   432 
   433                   /* Attach the view.  */
   434                   try
   435                     {
   436                       view.prepareForLayout (width, height);
   437                       windowManager.addView (view, params);
   438 
   439                       /* Record the window manager being used in the
   440                          EmacsWindow object.  */
   441                       EmacsWindow.this.windowManager = windowManager;
   442                     }
   443                   catch (Exception e)
   444                     {
   445                       Log.w (TAG,
   446                              "failed to attach override-redirect window, " + e);
   447                     }
   448                 }
   449             }
   450           });
   451       }
   452     else
   453       {
   454         /* Do the same thing as above, but don't register this
   455            window.  */
   456         EmacsService.SERVICE.runOnUiThread (new Runnable () {
   457             @Override
   458             public void
   459             run ()
   460             {
   461               /* Prior to mapping the view, set its measuredWidth and
   462                  measuredHeight to some reasonable value, in order to
   463                  avoid excessive bitmap dirtying.  */
   464 
   465               view.prepareForLayout (width, height);
   466               view.setVisibility (View.VISIBLE);
   467 
   468               if (!getDontFocusOnMap ())
   469                 view.requestFocus ();
   470             }
   471           });
   472       }
   473   }

   575   /* Return the modifier mask associated with the specified keyboard
   576      input EVENT.  Replace bits corresponding to Left or Right keys
   577      with their corresponding general modifier bits.  */
   578 
   579   private int
   580   eventModifiers (KeyEvent event)
   581   {
   582     int state;
   583 
   584     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.HONEYCOMB_MR2)
   585       state = event.getModifiers ();
   586     else
   587       {
   588         /* Replace this with getMetaState and manual
   589            normalization.  */
   590         state = event.getMetaState ();
   591 
   592         /* Normalize the state by setting the generic modifier bit if
   593            either a left or right modifier is pressed.  */
   594 
   595         if ((state & KeyEvent.META_ALT_LEFT_ON) != 0
   596             || (state & KeyEvent.META_ALT_RIGHT_ON) != 0)
   597           state |= KeyEvent.META_ALT_MASK;
   598 
   599         if ((state & KeyEvent.META_CTRL_LEFT_ON) != 0
   600             || (state & KeyEvent.META_CTRL_RIGHT_ON) != 0)
   601           state |= KeyEvent.META_CTRL_MASK;
   602       }
   603 
   604     return state;
   605   }

   643     int state, state_1;

   645 
   646     /* Compute the event's modifier mask.  */
   647     state = eventModifiers (event);

   656                                 state, keyCode,

   699   /* Mouse and touch event handling.
   700 
   701      Android does not conceptually distinguish between mouse events
   702      (those coming from a device whose movement affects the on-screen
   703      pointer image) and touch screen events.  When a touch, click, or
   704      pointer motion takes place, several kinds of event can be sent:
   705 
   706      ACTION_DOWN or ACTION_POINTER_DOWN is sent with a new coordinate
   707      and an associated ``pointer ID'' identifying the event when a
   708      click or touch takes place.  Emacs is responsible for recording
   709      both the position of this click for the purpose of determining
   710      future changes to the position of that touch.
   711 
   712      ACTION_UP or ACTION_POINTER_UP is sent with a pointer ID when the
   713      click associated with a previous ACTION_DOWN event is released.
   714 
   715      ACTION_CANCEL (or ACTION_POINTER_UP with FLAG_CANCELED) is sent
   716      if a similar situation transpires: the window system has chosen
   717      to grab of the click, and future movement will no longer be
   718      reported to Emacs.
   719 
   720      ACTION_MOVE is sent if a coordinate tied to a click that has not
   721      been released changes.  Emacs processes this event by comparing
   722      each of the coordinates within the event with its recollection of
   723      those contained within prior ACTION_DOWN and ACTION_MOVE events;
   724      the pointer ID of the difference is then reported within a touch
   725      or pointer motion event along with its new position.
   726 
   727      The events described above are all sent for both touch and mouse
   728      click events.  Determining whether an ACTION_DOWN event is
   729      associated with a button event is performed by inspecting the
   730      mouse button state associated with that event.  If it contains
   731      any mouse buttons that were not contained in the button state at
   732      the time of the last ACTION_DOWN or ACTION_UP event, the
   733      coordinate contained within is assumed to be a mouse click,
   734      leading to it and associated motion or ACTION_UP events being
   735      reported as mouse button or motion events.  Otherwise, those
   736      events are reported as touch screen events, with the touch ID set
   737      to the pointer ID.
   738 
   739      In addition to the events illustrated above, Android also sends
   740      several other types of event upon select types of activity from a
   741      mouse device:
   742 
   743      ACTION_HOVER_MOVE is sent with the coordinate of the mouse
   744      pointer if it moves above a frame prior to any click taking
   745      place.  Emacs sends a mouse motion event containing the
   746      coordinate.
   747 
   748      ACTION_HOVER_ENTER and ACTION_HOVER_LEAVE are respectively sent
   749      when the mouse pointer enters and leaves a frame.
   750 
   751      On Android 6.0 and later, ACTION_BUTTON_PRESS is sent with the
   752      coordinate of the mouse pointer if a mouse click occurs,
   753      alongside a ACTION_DOWN event.  ACTION_BUTTON_RELEASE is sent
   754      with the same information upon a mouse click being released, also
   755      accompanying an ACTION_UP event.
   756 
   757      However, both types of button events are implemented in a buggy
   758      fashion and cannot be used to report button events.  */
   759 
   760   /* Look through the button state to determine what button EVENT was
   761      generated from.  DOWN is true if EVENT is a button press event,
   762      false otherwise.  Value is the X number of the button.  */
   763 
   764   private int
   765   whatButtonWasIt (MotionEvent event, boolean down)
   766   {
   767     int eventState, notIn;
   768 
   769     /* Obtain the new button state.  */
   770     eventState = event.getButtonState ();
   771 
   772     /* Compute which button is now set or no longer set.  */
   773 
   774     notIn = (down ? eventState & ~lastButtonState
   775              : lastButtonState & ~eventState);
   776 
   777     if ((notIn & (MotionEvent.BUTTON_PRIMARY
   778                   | MotionEvent.BUTTON_SECONDARY
   779                   | MotionEvent.BUTTON_TERTIARY)) == 0)
   780       /* No buttons have been pressed, so this is a touch event.  */
   781       return 0;
   782 
   783     if ((notIn & MotionEvent.BUTTON_PRIMARY) != 0)
   784       return 1;
   785 
   786     if ((notIn & MotionEvent.BUTTON_SECONDARY) != 0)
   787       return 3;
   788 
   789     if ((notIn & MotionEvent.BUTTON_TERTIARY) != 0)
   790       return 2;
   791 
   792     /* Not a real value.  */
   793     return 4;
   794   }
   795 
   796   /* Return the mouse button associated with the specified ACTION_DOWN
   797      or ACTION_POINTER_DOWN EVENT.
   798 
   799      Value is 0 if no mouse button was pressed, or the X number of
   800      that mouse button.  */
   801 
   802   private int
   803   buttonForEvent (MotionEvent event)
   804   {
   805     /* ICS and earlier don't support true mouse button events, so
   806        treat all down events as touch screen events.  */
   807 
   808     if (Build.VERSION.SDK_INT
   809         < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
   810       return 0;
   811 
   812     return whatButtonWasIt (event, true);
   813   }

   815   /* Return the coordinate object associated with the specified
   816      EVENT, or null if it is not known.  */
   817 
   818   private Coordinate
   819   figureChange (MotionEvent event)
   820   {
   821     int i, truncatedX, truncatedY, pointerIndex, pointerID, count;
   822     Coordinate coordinate;
   823 
   824     /* Initialize this variable now.  */
   825     coordinate = null;
   826 
   827     switch (event.getActionMasked ())
   828       {
   829       case MotionEvent.ACTION_DOWN:
   830         /* Primary pointer pressed with index 0.  */
   831 
   832         pointerID = event.getPointerId (0);
   833         coordinate = new Coordinate ((int) event.getX (0),
   834                                      (int) event.getY (0),
   835                                      buttonForEvent (event),
   836                                      pointerID);
   837         pointerMap.put (pointerID, coordinate);
   838         break;
   839 
   840       case MotionEvent.ACTION_UP:
   841       case MotionEvent.ACTION_CANCEL:
   842         /* Primary pointer released with index 0.  */
   843         pointerID = event.getPointerId (0);
   844         coordinate = pointerMap.remove (pointerID);
   845         break;
   846 
   847       case MotionEvent.ACTION_POINTER_DOWN:
   848         /* New pointer.  Find the pointer ID from the index and place
   849            it in the map.  */
   850         pointerIndex = event.getActionIndex ();
   851         pointerID = event.getPointerId (pointerIndex);
   852         coordinate = new Coordinate ((int) event.getX (0),
   853                                      (int) event.getY (0),
   854                                      buttonForEvent (event),
   855                                      pointerID);
   856         pointerMap.put (pointerID, coordinate);
   857         break;
   858 
   859       case MotionEvent.ACTION_POINTER_UP:
   860         /* Pointer removed.  Remove it from the map.  */
   861         pointerIndex = event.getActionIndex ();
   862         pointerID = event.getPointerId (pointerIndex);
   863         coordinate = pointerMap.remove (pointerID);
   864         break;
   865 
   866       default:
   867 
   868         /* Loop through each pointer in the event.  */
   869 
   870         count = event.getPointerCount ();
   871         for (i = 0; i < count; ++i)
   872           {
   873             pointerID = event.getPointerId (i);
   874 
   875             /* Look up that pointer in the map.  */
   876             coordinate = pointerMap.get (pointerID);
   877 
   878             if (coordinate != null)
   879               {
   880                 /* See if coordinates have changed.  */
   881                 truncatedX = (int) event.getX (i);
   882                 truncatedY = (int) event.getY (i);
   883 
   884                 if (truncatedX != coordinate.x
   885                     || truncatedY != coordinate.y)
   886                   {
   887                     /* The pointer changed.  Update the coordinate and
   888                        break out of the loop.  */
   889                     coordinate.x = truncatedX;
   890                     coordinate.y = truncatedY;
   891 
   892                     break;
   893                   }
   894               }
   895           }
   896 
   897         /* Set coordinate to NULL if the loop failed to find any
   898            matching pointer.  */
   899 
   900         if (i == count)
   901           coordinate = null;
   902       }
   903 
   904     /* Return the pointer ID.  */
   905     return coordinate;
   906   }

   908   /* Return the modifier mask associated with the specified motion
   909      EVENT.  Replace bits corresponding to Left or Right keys with
   910      their corresponding general modifier bits.  */
   911 
   912   private int
   913   motionEventModifiers (MotionEvent event)
   914   {
   915     int state;
   916 
   917     state = event.getMetaState ();
   918 
   919     /* Normalize the state by setting the generic modifier bit if
   920        either a left or right modifier is pressed.  */
   921 
   922     if ((state & KeyEvent.META_ALT_LEFT_ON) != 0
   923         || (state & KeyEvent.META_ALT_RIGHT_ON) != 0)
   924       state |= KeyEvent.META_ALT_MASK;
   925 
   926     if ((state & KeyEvent.META_CTRL_LEFT_ON) != 0
   927         || (state & KeyEvent.META_CTRL_RIGHT_ON) != 0)
   928       state |= KeyEvent.META_CTRL_MASK;
   929 
   930     return state;
   931   }

   933   /* Process a single ACTION_DOWN, ACTION_POINTER_DOWN, ACTION_UP,
   934      ACTION_POINTER_UP, ACTION_CANCEL, or ACTION_MOVE event.
   935 
   936      Ascertain which coordinate changed and send an appropriate mouse
   937      or touch screen event.  */
   938 
   939   private void
   940   motionEvent (MotionEvent event)
   941   {
   942     Coordinate coordinate;
   943     int modifiers;
   944     long time;
   945 
   946     /* Find data associated with this event's pointer.  Namely, its
   947        current location, whether or not a change has taken place, and
   948        whether or not it is a button event.  */
   949 
   950     coordinate = figureChange (event);
   951 
   952     if (coordinate == null)
   953       return;
   954 
   955     time = event.getEventTime ();
   956 
   957     if (coordinate.button != 0)
   958       {
   959         /* This event is tied to a mouse click, so report mouse motion
   960            and button events.  */
   961 
   962         modifiers = motionEventModifiers (event);
   963 
   964         switch (event.getAction ())
   965           {
   966           case MotionEvent.ACTION_POINTER_DOWN:
   967           case MotionEvent.ACTION_DOWN:
   968             EmacsNative.sendButtonPress (this.handle, coordinate.x,
   969                                          coordinate.y, time, modifiers,
   970                                          coordinate.button);
   971             break;
   972 
   973           case MotionEvent.ACTION_POINTER_UP:
   974           case MotionEvent.ACTION_UP:
   975           case MotionEvent.ACTION_CANCEL:
   976             EmacsNative.sendButtonRelease (this.handle, coordinate.x,
   977                                            coordinate.y, time, modifiers,
   978                                            coordinate.button);
   979             break;
   980 
   981           case MotionEvent.ACTION_MOVE:
   982             EmacsNative.sendMotionNotify (this.handle, coordinate.x,
   983                                           coordinate.y, time);
   984             break;
   985           }
   986       }
   987     else
   988       {
   989         /* This event is a touch event, and the touch ID is the
   990            pointer ID.  */
   991 
   992         switch (event.getActionMasked ())
   993           {
   994           case MotionEvent.ACTION_DOWN:
   995           case MotionEvent.ACTION_POINTER_DOWN:
   996             /* Touch down event.  */
   997             EmacsNative.sendTouchDown (this.handle, coordinate.x,
   998                                        coordinate.y, time,
   999                                        coordinate.id);
  1000             break;
  1001 
  1002           case MotionEvent.ACTION_UP:
  1003           case MotionEvent.ACTION_POINTER_UP:
  1004           case MotionEvent.ACTION_CANCEL:
  1005             /* Touch up event.  Android documentation says ACTION_CANCEL
  1006                should be treated as more or less equivalent to ACTION_UP,
  1007                so that is what is done here.  */
  1008             EmacsNative.sendTouchUp (this.handle, coordinate.x,
  1009                                      coordinate.y, time, coordinate.id);
  1010             break;
  1011 
  1012           case MotionEvent.ACTION_MOVE:
  1013             /* Pointer motion event.  */
  1014             EmacsNative.sendTouchMove (this.handle, coordinate.x,
  1015                                        coordinate.y, time, coordinate.id);
  1016             break;
  1017           }
  1018       }
  1019 
  1020     if (Build.VERSION.SDK_INT
  1021         < Build.VERSION_CODES.ICE_CREAM_SANDWICH)
  1022       return;
  1023 
  1024     /* Now update the button state.  */
  1025     lastButtonState = event.getButtonState ();
  1026     return;
  1027   }

  1029   public boolean
  1030   onTouchEvent (MotionEvent event)
  1031   {
  1032     switch (event.getActionMasked ())
  1033       {
  1034       case MotionEvent.ACTION_DOWN:
  1035       case MotionEvent.ACTION_POINTER_DOWN:
  1036       case MotionEvent.ACTION_UP:
  1037       case MotionEvent.ACTION_POINTER_UP:
  1038       case MotionEvent.ACTION_CANCEL:
  1039       case MotionEvent.ACTION_MOVE:
  1040         motionEvent (event);
  1041         return true;
  1042       }
  1043 
  1044     return false;
  1045   }

  1047   public boolean
  1048   onGenericMotionEvent (MotionEvent event)
  1049   {
  1050     switch (event.getAction ())
  1051       {
  1052       case MotionEvent.ACTION_HOVER_ENTER:
  1053         EmacsNative.sendEnterNotify (this.handle, (int) event.getX (),
  1054                                      (int) event.getY (),
  1055                                      event.getEventTime ());
  1056         return true;
  1057 
  1058       case MotionEvent.ACTION_HOVER_MOVE:
  1059         EmacsNative.sendMotionNotify (this.handle, (int) event.getX (),
  1060                                       (int) event.getY (),
  1061                                       event.getEventTime ());
  1062         return true;
  1063 
  1064       case MotionEvent.ACTION_HOVER_EXIT:
  1065 
  1066         /* If the exit event comes from a button press, its button
  1067            state will have extra bits compared to the last known
  1068            button state.  Since the exit event will interfere with
  1069            tool bar button presses, ignore such splurious events.  */
  1070 
  1071         if ((event.getButtonState () & ~lastButtonState) == 0)
  1072           EmacsNative.sendLeaveNotify (this.handle, (int) event.getX (),
  1073                                        (int) event.getY (),
  1074                                        event.getEventTime ());
  1075 
  1076         return true;
  1077 
  1078       case MotionEvent.ACTION_DOWN:
  1079       case MotionEvent.ACTION_POINTER_DOWN:
  1080       case MotionEvent.ACTION_UP:
  1081       case MotionEvent.ACTION_POINTER_UP:
  1082       case MotionEvent.ACTION_CANCEL:
  1083       case MotionEvent.ACTION_MOVE:
  1084         /* MotionEvents may either be sent to onGenericMotionEvent or
  1085            onTouchEvent depending on if Android thinks it is a mouse
  1086            event or not, but we detect them ourselves.  */
  1087         motionEvent (event);
  1088         return true;
  1089 
  1090       case MotionEvent.ACTION_SCROLL:
  1091         /* Send a scroll event with the specified deltas.  */
  1092         EmacsNative.sendWheel (this.handle, (int) event.getX (),
  1093                                (int) event.getY (),
  1094                                event.getEventTime (),
  1095                                motionEventModifiers (event),
  1096                                event.getAxisValue (MotionEvent.AXIS_HSCROLL),
  1097                                event.getAxisValue (MotionEvent.AXIS_VSCROLL));
  1098         return true;
  1099       }
  1100 
  1101     return false;
  1102   }

  1110 
  1111     /* Reparent this window to the other window.  */
  1112

  1144
