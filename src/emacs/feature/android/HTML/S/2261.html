    96 static emacs_dir *
    97 open_directory (Lisp_Object dirname, Lisp_Object encoded_dirname, int *fdp)
    98 {
    99   char *name = SSDATA (encoded_dirname);
   100   emacs_dir *d;
   101   int fd, opendir_errno;
   102 
   103 #if defined DOS_NT || (defined HAVE_ANDROID && !defined ANDROID_STUBIFY)
   104   /* On DOS_NT, directories cannot be opened.  The emulation assumes
   105      that any file descriptor other than AT_FDCWD corresponds to the
   106      most recently opened directory.  This hack is good enough for
   107      Emacs.
   108 
   109      This code is also used on Android for a different reason: a
   110      special `assets' directory outside the normal file system is used
   111      to open assets inside the Android application package, and must
   112      be listed using the opendir-like interface provided in
   113      android.h.  */
   114   fd = 0;
   115 #ifndef HAVE_ANDROID
   116   d = opendir (name);
   117 #else
   118   d = android_opendir (name);
   119 
   120   if (d)
   121     fd = android_dirfd (d);
   122 #endif
   123   opendir_errno = errno;
   124 #else
   125   fd = emacs_open (name, O_RDONLY | O_DIRECTORY, 0);
   126   if (fd < 0)
   127     {
   128       opendir_errno = errno;
   129       d = 0;
   130     }
   131   else
   132     {
   133       d = fdopendir (fd);
   134       opendir_errno = errno;
   135       if (! d)
   136         emacs_close (fd);
   137     }
   138 #endif
   139 
   140   if (!d)
   141     report_file_errno ("Opening directory", dirname, opendir_errno);
   142   *fdp = fd;
   143   return d;
   144 }

   164 static struct dirent *
   165 read_dirent (emacs_dir *dir, Lisp_Object dirname)
   166 {
   167   while (true)
   168     {
   169       errno = 0;
   170       struct dirent *dp = emacs_readdir (dir);
   171       if (dp || errno == 0)
   172         return dp;
   173       if (! (errno == EAGAIN || errno == EINTR))
   174         {
   175 #ifdef WINDOWSNT
   176           /* The MS-Windows implementation of 'opendir' doesn't
   177              actually open a directory until the first call to
   178              'readdir'.  If 'readdir' fails to open the directory, it
   179              sets errno to ENOENT or EACCES, see w32.c.  */
   180           if (errno == ENOENT || errno == EACCES)
   181             report_file_error ("Opening directory", dirname);
   182 #endif
   183           report_file_error ("Reading directory", dirname);
   184         }
   185       maybe_quit ();
   186     }
   187 }

   194 Lisp_Object
   195 directory_files_internal (Lisp_Object directory, Lisp_Object full,
   196                           Lisp_Object match, Lisp_Object nosort, bool attrs,
   197                           Lisp_Object id_format, Lisp_Object return_count)
   198 {
   199   EMACS_INT ind = 0, last = MOST_POSITIVE_FIXNUM;
   200 
   201   if (!NILP (return_count))
   202     {
   203       CHECK_FIXNAT (return_count);
   204       last = XFIXNAT (return_count);
   205     }
   206 
   207   if (!NILP (match))
   208     CHECK_STRING (match);
   209 
   210   /* Don't let the compiler optimize away all copies of DIRECTORY,
   211      which would break GC; see Bug#16986.  */
   212   Lisp_Object volatile directory_volatile = directory;
   213 
   214   Lisp_Object dirfilename = Fdirectory_file_name (directory);
   215 
   216   /* Note: ENCODE_FILE and DECODE_FILE can GC because they can run
   217      run_pre_post_conversion_on_str which calls Lisp directly and
   218      indirectly.  */
   219   Lisp_Object encoded_dirfilename = ENCODE_FILE (dirfilename);
   220 
   221   int fd;
   222 
   223   /* Keep in mind that FD is not always a real file descriptor on
   224      Android.  */
   225   emacs_dir *d = open_directory (dirfilename, encoded_dirfilename, &fd);
   226 
   227   /* Unfortunately, we can now invoke expand-file-name and
   228      file-attributes on filenames, both of which can throw, so we must
   229      do a proper unwind-protect.  */
   230   specpdl_ref count = SPECPDL_INDEX ();
   231   record_unwind_protect_ptr (directory_files_internal_unwind, d);
   232 
   233 #ifdef WINDOWSNT
   234   Lisp_Object w32_save = Qnil;
   235   if (attrs)
   236     {
   237       /* Do this only once to avoid doing it (in w32.c:stat) for each
   238          file in the directory, when we call file_attributes below.  */
   239       record_unwind_protect (directory_files_internal_w32_unwind,
   240                              Vw32_get_true_file_attributes);
   241       w32_save = Vw32_get_true_file_attributes;
   242       if (EQ (Vw32_get_true_file_attributes, Qlocal))
   243         {
   244           /* w32.c:stat will notice these bindings and avoid calling
   245              GetDriveType for each file.  */
   246           if (is_slow_fs (SSDATA (encoded_dirfilename)))
   247             Vw32_get_true_file_attributes = Qnil;
   248           else
   249             Vw32_get_true_file_attributes = Qt;
   250         }
   251     }
   252 #endif
   253 
   254   if (!NILP (full) && !STRING_MULTIBYTE (directory))
   255     { /* We will be concatenating 'directory' with local file name.
   256          We always decode local file names, so in order to safely concatenate
   257          them we need 'directory' to be decoded as well (bug#56469).  */
   258       directory = DECODE_FILE (directory);
   259     }
   260 
   261   ptrdiff_t directory_nbytes = SBYTES (directory);
   262   re_match_object = Qt;
   263 
   264   /* Decide whether we need to add a directory separator.  */
   265   bool needsep = (directory_nbytes == 0
   266                   || !IS_ANY_SEP (SREF (directory, directory_nbytes - 1)));
   267 
   268   /* Windows users want case-insensitive wildcards.  */
   269   Lisp_Object case_table = Qnil;
   270 #ifdef WINDOWSNT
   271   case_table = BVAR (&buffer_defaults, case_canon_table);
   272 #endif
   273 
   274   /* Read directory entries and accumulate them into LIST.  */
   275   Lisp_Object list = Qnil;
   276   for (struct dirent *dp; (dp = read_dirent (d, directory)); )
   277     {
   278       ptrdiff_t len = dirent_namelen (dp);
   279       Lisp_Object name = make_unibyte_string (dp->d_name, len);
   280       Lisp_Object finalname = name;
   281 
   282       /* This can GC.  */
   283       name = DECODE_FILE (name);
   284 
   285       maybe_quit ();
   286 
   287       if (!NILP (match)
   288           && fast_string_match_internal (match, name, case_table) < 0)
   289         continue;
   290 
   291       Lisp_Object fileattrs UNINIT;
   292       if (attrs)
   293         {
   294           fileattrs = file_attributes (fd, dp->d_name, directory, name,
   295                                        id_format);
   296           if (NILP (fileattrs))
   297             continue;
   298         }
   299 
   300       if (!NILP (full))
   301         {
   302           ptrdiff_t name_nbytes = SBYTES (name);
   303           ptrdiff_t nbytes = directory_nbytes + needsep + name_nbytes;
   304           ptrdiff_t nchars = SCHARS (directory) + needsep + SCHARS (name);
   305           /* DECODE_FILE may return non-ASCII unibyte strings (e.g. when
   306              file-name-coding-system is 'binary'), so we don't know for sure
   307              that the bytes we have follow our internal utf-8 representation
   308              for multibyte strings.  If nchars == nbytes we don't need to
   309              care and just return a unibyte string; and if not, that means
   310              one of 'name' or 'directory' is multibyte, in which case we
   311              presume that the other one would also be multibyte if it
   312              contained non-ASCII.
   313              FIXME: This last presumption is broken when 'directory' is
   314              multibyte (with non-ASCII), and 'name' is unibyte with non-ASCII
   315              (because file-name-coding-system is 'binary').  */
   316           finalname = (nchars == nbytes)
   317                       ? make_uninit_string (nbytes)
   318                       : make_uninit_multibyte_string (nchars, nbytes);
   319           memcpy (SDATA (finalname), SDATA (directory), directory_nbytes);
   320           if (needsep)
   321             SSET (finalname, directory_nbytes, DIRECTORY_SEP);
   322           memcpy (SDATA (finalname) + directory_nbytes + needsep,
   323                   SDATA (name), name_nbytes);
   324         }
   325       else
   326         finalname = name;
   327 
   328       if (ind == last)
   329           break;
   330       ind ++;
   331 
   332       list = Fcons (attrs ? Fcons (finalname, fileattrs) : finalname, list);
   333     }
   334 
   335   emacs_closedir (d);
   336 #ifdef WINDOWSNT
   337   if (attrs)
   338     Vw32_get_true_file_attributes = w32_save;
   339 #endif
   340 
   341   /* Discard the unwind protect.  */
   342   specpdl_ptr = specpdl_ref_to_ptr (count);
   343 
   344   if (NILP (nosort))
   345     list = Fsort (Fnreverse (list),
   346                   attrs ? Qfile_attributes_lessp : Qstring_lessp);
   347 
   348   (void) directory_volatile;
   349   return list;
   350 }

   490 static Lisp_Object
   491 file_name_completion (Lisp_Object file, Lisp_Object dirname, bool all_flag,
   492                       Lisp_Object predicate)
   493 {
   494   ptrdiff_t bestmatchsize = 0;
   495   int matchcount = 0;
   496   /* If ALL_FLAG is 1, BESTMATCH is the list of all matches, decoded.
   497      If ALL_FLAG is 0, BESTMATCH is either nil
   498      or the best match so far, not decoded.  */
   499   Lisp_Object bestmatch, tem, elt, name;
   500   Lisp_Object encoded_file;
   501   Lisp_Object encoded_dir;
   502   bool directoryp;
   503   /* If not INCLUDEALL, exclude files in completion-ignored-extensions as
   504      well as "." and "..".  Until shown otherwise, assume we can't exclude
   505      anything.  */
   506   bool includeall = 1;
   507   bool check_decoded = false;
   508   specpdl_ref count = SPECPDL_INDEX ();
   509 
   510   elt = Qnil;
   511 
   512   CHECK_STRING (file);
   513 
   514   bestmatch = Qnil;
   515   encoded_file = encoded_dir = Qnil;
   516   specbind (Qdefault_directory, dirname);
   517 
   518   /* Do completion on the encoded file name
   519      because the other names in the directory are (we presume)
   520      encoded likewise.  We decode the completed string at the end.  */
   521   /* Actually, this is not quite true any more: we do most of the completion
   522      work with decoded file names, but we still do some filtering based
   523      on the encoded file name.  */
   524   encoded_file = ENCODE_FILE (file);
   525   encoded_dir = ENCODE_FILE (Fdirectory_file_name (dirname));
   526 
   527   Lisp_Object file_encoding = Vfile_name_coding_system;
   528   if (NILP (Vfile_name_coding_system))
   529     file_encoding = Vdefault_file_name_coding_system;
   530   /* If the file-name encoding decomposes characters, as we do for
   531      HFS+ filesystems, we need to make an additional comparison of
   532      decoded names in order to filter false positives, such as "a"
   533      falsely matching "a-ring".  */
   534   if (!NILP (file_encoding)
   535       && !NILP (plist_get (Fcoding_system_plist (file_encoding),
   536                            Qdecomposed_characters)))
   537     {
   538       check_decoded = true;
   539       if (STRING_MULTIBYTE (file))
   540         {
   541           /* Recompute FILE to make sure any decomposed characters in
   542              it are re-composed by the post-read-conversion.
   543              Otherwise, any decomposed characters will be rejected by
   544              the additional check below.  */
   545           file = DECODE_FILE (encoded_file);
   546         }
   547     }
   548   int fd;
   549   emacs_dir *d = open_directory (dirname, encoded_dir, &fd);
   550   record_unwind_protect_ptr (directory_files_internal_unwind, d);
   551 
   552   /* Loop reading directory entries.  */
   553   Lisp_Object zero = make_fixnum (0);
   554   ptrdiff_t enc_file_len = SCHARS (encoded_file);
   555   Lisp_Object file_len = make_fixnum (SCHARS (file));
   556   for (struct dirent *dp; (dp = read_dirent (d, dirname)); )
   557     {
   558       ptrdiff_t len = dirent_namelen (dp);
   559       bool canexclude = 0;
   560 
   561       maybe_quit ();
   562 
   563       if (len < enc_file_len
   564           /* scmp cannot reliably compare non-ASCII strings while
   565              ignoring letter-case.  */
   566           || (!completion_ignore_case
   567               && scmp (dp->d_name, SSDATA (encoded_file), enc_file_len) >= 0))
   568         continue;
   569 
   570       name = make_unibyte_string (dp->d_name, len);
   571       name = DECODE_FILE (name);
   572       ptrdiff_t name_blen = SBYTES (name), name_len = SCHARS (name);
   573       if (completion_ignore_case
   574           && !BASE_EQ (Fcompare_strings (name, zero, file_len, file, zero,
   575                                          file_len, Qt),
   576                        Qt))
   577             continue;
   578 
   579       switch (dirent_type (dp))
   580         {
   581         case DT_DIR:
   582           directoryp = true;
   583           break;
   584 
   585         case DT_LNK: case DT_UNKNOWN:
   586           directoryp = file_name_completion_dirp (fd, dp, len);
   587           break;
   588 
   589         default:
   590           directoryp = false;
   591           break;
   592         }
   593 
   594       tem = Qnil;
   595       /* If all_flag is set, always include all.
   596          It would not actually be helpful to the user to ignore any possible
   597          completions when making a list of them.  */
   598       if (!all_flag)
   599         {
   600           ptrdiff_t skip;
   601           Lisp_Object cmp_len = make_fixnum (name_len);
   602 
   603 #if 0 /* FIXME: The `scmp' call compares an encoded and a decoded string. */
   604           /* If this entry matches the current bestmatch, the only
   605              thing it can do is increase matchcount, so don't bother
   606              investigating it any further.  */
   607           if (!completion_ignore_case
   608               /* The return result depends on whether it's the sole match.  */
   609               && matchcount > 1
   610               && !includeall /* This match may allow includeall to 0.  */
   611               && len >= bestmatchsize
   612               && 0 > scmp (dp->d_name, SSDATA (bestmatch), bestmatchsize))
   613             continue;
   614 #endif
   615 
   616           if (directoryp)
   617             {
   618 #ifndef TRIVIAL_DIRECTORY_ENTRY
   619 #define TRIVIAL_DIRECTORY_ENTRY(n) (!strcmp (n, ".") || !strcmp (n, ".."))
   620 #endif
   621               /* "." and ".." are never interesting as completions, and are
   622                  actually in the way in a directory with only one file.  */
   623               if (TRIVIAL_DIRECTORY_ENTRY (dp->d_name))
   624                 canexclude = 1;
   625               else if (len > enc_file_len)
   626                 /* Ignore directories if they match an element of
   627                    completion-ignored-extensions which ends in a slash.  */
   628                 for (tem = Vcompletion_ignored_extensions;
   629                      CONSP (tem); tem = XCDR (tem))
   630                   {
   631                     ptrdiff_t elt_len;
   632                     char *p1;
   633 
   634                     elt = XCAR (tem);
   635                     if (!STRINGP (elt))
   636                       continue;
   637                     elt_len = SBYTES (elt) - 1; /* -1 for trailing / */
   638                     if (elt_len <= 0)
   639                       continue;
   640                     p1 = SSDATA (elt);
   641                     if (p1[elt_len] != '/')
   642                       continue;
   643                     skip = name_blen - elt_len;
   644                     if (skip < 0)
   645                       continue;
   646 
   647                     if (!completion_ignore_case
   648                         && scmp (SSDATA (name) + skip, p1, elt_len) >= 0)
   649                       continue;
   650                     if (completion_ignore_case)
   651                       {
   652                         elt_len = SCHARS (elt) - 1;
   653                         skip = name_len - elt_len;
   654                         cmp_len = make_fixnum (elt_len);
   655                         if (skip < 0
   656                             || !BASE_EQ (Fcompare_strings (name,
   657                                                            make_fixnum (skip),
   658                                                            Qnil,
   659                                                            elt, zero, cmp_len,
   660                                                            Qt),
   661                                          Qt))
   662                           continue;
   663                       }
   664                     break;
   665                   }
   666             }
   667           else
   668             {
   669               /* Compare extensions-to-be-ignored against end of this file name */
   670               /* if name is not an exact match against specified string */
   671               if (len > enc_file_len)
   672                 /* and exit this for loop if a match is found */
   673                 for (tem = Vcompletion_ignored_extensions;
   674                      CONSP (tem); tem = XCDR (tem))
   675                   {
   676                     elt = XCAR (tem);
   677                     if (!STRINGP (elt)) continue;
   678                     ptrdiff_t elt_len = SBYTES (elt);
   679                     skip = len - elt_len;
   680                     if (skip < 0) continue;
   681 
   682                     if (!completion_ignore_case
   683                         && (scmp (SSDATA (name) + skip, SSDATA (elt), elt_len)
   684                             >= 0))
   685                       continue;
   686                     if (completion_ignore_case)
   687                       {
   688                         elt_len = SCHARS (elt);
   689                         skip = name_len - elt_len;
   690                         cmp_len = make_fixnum (elt_len);
   691                         if (skip < 0
   692                             || !BASE_EQ (Fcompare_strings (name,
   693                                                            make_fixnum (skip),
   694                                                            Qnil,
   695                                                            elt, zero, cmp_len,
   696                                                            Qt),
   697                                          Qt))
   698                           continue;
   699                       }
   700                     break;
   701                   }
   702             }
   703 
   704           /* If an ignored-extensions match was found,
   705              don't process this name as a completion.  */
   706           if (CONSP (tem))
   707             canexclude = 1;
   708 
   709           if (!includeall && canexclude)
   710             /* We're not including all files and this file can be excluded.  */
   711             continue;
   712 
   713           if (includeall && !canexclude)
   714             { /* If we have one non-excludable file, we want to exclude the
   715                  excludable files.  */
   716               includeall = 0;
   717               /* Throw away any previous excludable match found.  */
   718               bestmatch = Qnil;
   719               bestmatchsize = 0;
   720               matchcount = 0;
   721             }
   722         }
   723 
   724       Lisp_Object regexps, table = (completion_ignore_case
   725                                     ? Vascii_canon_table : Qnil);
   726 
   727       /* Ignore this element if it fails to match all the regexps.  */
   728       for (regexps = Vcompletion_regexp_list; CONSP (regexps);
   729            regexps = XCDR (regexps))
   730         if (fast_string_match_internal (XCAR (regexps), name, table) < 0)
   731           break;
   732 
   733       if (CONSP (regexps))
   734         continue;
   735 
   736       /* This is a possible completion */
   737       if (directoryp)
   738         /* This completion is a directory; make it end with '/'.  */
   739         name = Ffile_name_as_directory (name);
   740 
   741       /* Test the predicate, if any.  */
   742       if (!NILP (predicate) && NILP (call1 (predicate, name)))
   743         continue;
   744 
   745       /* Reject entries where the encoded strings match, but the
   746          decoded don't.  For example, "a" should not match "a-ring" on
   747          file systems that store decomposed characters. */
   748       if (check_decoded && SCHARS (file) <= SCHARS (name))
   749         {
   750           /* FIXME: This is a copy of the code below.  */
   751           ptrdiff_t compare = SCHARS (file);
   752           Lisp_Object cmp
   753             = Fcompare_strings (name, zero, make_fixnum (compare),
   754                                 file, zero, make_fixnum (compare),
   755                                 completion_ignore_case ? Qt : Qnil);
   756           if (!BASE_EQ (cmp, Qt))
   757             continue;
   758         }
   759 
   760       /* Suitably record this match.  */
   761 
   762       matchcount += matchcount <= 1;
   763 
   764       if (all_flag)
   765         bestmatch = Fcons (name, bestmatch);
   766       else if (NILP (bestmatch))
   767         {
   768           bestmatch = name;
   769           bestmatchsize = SCHARS (name);
   770         }
   771       else
   772         {
   773           /* FIXME: This is a copy of the code in Ftry_completion.  */
   774           ptrdiff_t compare = min (bestmatchsize, SCHARS (name));
   775           Lisp_Object cmp
   776             = Fcompare_strings (bestmatch, zero, make_fixnum (compare),
   777                                 name, zero, make_fixnum (compare),
   778                                 completion_ignore_case ? Qt : Qnil);
   779           ptrdiff_t matchsize = BASE_EQ (cmp, Qt)
   780                                 ? compare : eabs (XFIXNUM (cmp)) - 1;
   781 
   782           if (completion_ignore_case)
   783             {
   784               /* If this is an exact match except for case,
   785                  use it as the best match rather than one that is not
   786                  an exact match.  This way, we get the case pattern
   787                  of the actual match.  */
   788               /* This tests that the current file is an exact match
   789                  but BESTMATCH is not (it is too long).  */
   790               if ((matchsize == SCHARS (name)
   791                    && matchsize + directoryp < SCHARS (bestmatch))
   792                   ||
   793                   /* If there is no exact match ignoring case,
   794                      prefer a match that does not change the case
   795                      of the input.  */
   796                   /* If there is more than one exact match aside from
   797                      case, and one of them is exact including case,
   798                      prefer that one.  */
   799                   /* This == checks that, of current file and BESTMATCH,
   800                      either both or neither are exact.  */
   801                   (((matchsize == SCHARS (name))
   802                     ==
   803                     (matchsize + directoryp == SCHARS (bestmatch)))
   804                    && (cmp = Fcompare_strings (name, zero,
   805                                                make_fixnum (SCHARS (file)),
   806                                                file, zero,
   807                                                Qnil,
   808                                                Qnil),
   809                        BASE_EQ (Qt, cmp))
   810                    && (cmp = Fcompare_strings (bestmatch, zero,
   811                                                make_fixnum (SCHARS (file)),
   812                                                file, zero,
   813                                                Qnil,
   814                                                Qnil),
   815                        ! BASE_EQ (Qt, cmp))))
   816                 bestmatch = name;
   817             }
   818           bestmatchsize = matchsize;
   819 
   820           /* If the best completion so far is reduced to the string
   821              we're trying to complete, then we already know there's no
   822              other completion, so there's no point looking any further.  */
   823           if (matchsize <= SCHARS (file)
   824               && !includeall /* A future match may allow includeall to 0.  */
   825               /* If completion-ignore-case is non-nil, don't
   826                  short-circuit because we want to find the best
   827                  possible match *including* case differences.  */
   828               && (!completion_ignore_case || matchsize == 0)
   829               /* The return value depends on whether it's the sole match.  */
   830               && matchcount > 1)
   831             break;
   832 
   833         }
   834     }
   835 
   836   /* This closes the directory.  */
   837   bestmatch = unbind_to (count, bestmatch);
   838 
   839   if (all_flag || NILP (bestmatch))
   840     return bestmatch;
   841   /* Return t if the supplied string is an exact match (counting case);
   842      it does not require any change to be made.  */
   843   if (matchcount == 1 && !NILP (Fequal (bestmatch, file)))
   844     return Qt;
   845   bestmatch = Fsubstring (bestmatch, make_fixnum (0),
   846                           make_fixnum (bestmatchsize));
   847   return bestmatch;
   848 }

   850 /* Compare exactly LEN chars of strings at S1 and S2,
   851    ignoring case if appropriate.
   852    Return -1 if strings match,
   853    else number of chars that match at the beginning.  */
   854 
   855 static ptrdiff_t
   856 scmp (const char *s1, const char *s2, ptrdiff_t len)
   857 {
   858   register ptrdiff_t l = len;
   859 
   860   if (completion_ignore_case)
   861     {
   862       /* WARNING: This only works for pure ASCII strings, as we
   863          compare bytes, not characters!  Use Fcompare_strings for
   864          comparing non-ASCII strings case-insensitively.  */
   865       while (l
   866              && (downcase ((unsigned char) *s1++)
   867                  == downcase ((unsigned char) *s2++)))
   868         l--;
   869     }
   870   else
   871     {
   872       while (l && *s1++ == *s2++)
   873         l--;
   874     }
   875   if (l == 0)
   876     return -1;
   877   else
   878     return len - l;
   879 }

  1000 static Lisp_Object
  1001 file_attributes (int fd, char const *name,
  1002                  Lisp_Object dirname, Lisp_Object filename,
  1003                  Lisp_Object id_format)
  1004 {
  1005   specpdl_ref count = SPECPDL_INDEX ();
  1006   struct stat s;
  1007 
  1008   /* An array to hold the mode string generated by filemodestring,
  1009      including its terminating space and null byte.  */
  1010   char modes[sizeof "-rwxr-xr-x "];
  1011 
  1012   char *uname = NULL, *gname = NULL;
  1013 
  1014   int err = EINVAL;
  1015 
  1016 #if defined O_PATH && !defined HAVE_CYGWIN_O_PATH_BUG   \
  1017   && !(defined HAVE_ANDROID && !defined ANDROID_STUBIFY)
  1018   int namefd = emacs_openat (fd, name, O_PATH | O_CLOEXEC | O_NOFOLLOW, 0);
  1019   if (namefd < 0)
  1020     err = errno;
  1021   else
  1022     {
  1023       record_unwind_protect_int (close_file_unwind, namefd);
  1024       if (sys_fstat (namefd, &s) != 0)
  1025         {
  1026           err = errno;
  1027           /* The Linux kernel before version 3.6 does not support
  1028              fstat on O_PATH file descriptors.  Handle this error like
  1029              missing support for O_PATH.  */
  1030           if (err == EBADF)
  1031             err = EINVAL;
  1032         }
  1033       else
  1034         {
  1035           err = 0;
  1036           fd = namefd;
  1037           name = "";
  1038         }
  1039     }
  1040 #endif
  1041 
  1042   if (err == EINVAL)
  1043     {
  1044 #ifdef WINDOWSNT
  1045       /* We usually don't request accurate owner and group info,
  1046          because it can be expensive on Windows to get that, and most
  1047          callers of 'lstat' don't need that.  But here we do want that
  1048          information to be accurate.  */
  1049       w32_stat_get_owner_group = 1;
  1050 #endif
  1051       err = emacs_fstatat (fd, name, &s, AT_SYMLINK_NOFOLLOW) == 0 ? 0 : errno;
  1052 #ifdef WINDOWSNT
  1053       w32_stat_get_owner_group = 0;
  1054 #endif
  1055     }
  1056 
  1057   if (err != 0)
  1058     return unbind_to (count, file_attribute_errno (filename, err));
  1059 
  1060   Lisp_Object file_type;
  1061   if (S_ISLNK (s.st_mode))
  1062     {
  1063       /* On systems lacking O_PATH support there is a race if the
  1064          symlink is replaced between the call to fstatat and the call
  1065          to emacs_readlinkat.  Detect this race unless the replacement
  1066          is also a symlink.  */
  1067       file_type = check_emacs_readlinkat (fd, filename, name);
  1068       if (NILP (file_type))
  1069         return unbind_to (count, Qnil);
  1070     }
  1071   else
  1072     file_type = S_ISDIR (s.st_mode) ? Qt : Qnil;
  1073 
  1074   unbind_to (count, Qnil);
  1075 
  1076   if (!(NILP (id_format) || EQ (id_format, Qinteger)))
  1077     {
  1078       uname = stat_uname (&s);
  1079       gname = stat_gname (&s);
  1080     }
  1081 
  1082   filemodestring (&s, modes);
  1083 
  1084   return CALLN (Flist,
  1085                 file_type,
  1086                 make_fixnum (s.st_nlink),
  1087                 (uname
  1088                  ? DECODE_SYSTEM (build_unibyte_string (uname))
  1089                  : INT_TO_INTEGER (s.st_uid)),
  1090                 (gname
  1091                  ? DECODE_SYSTEM (build_unibyte_string (gname))
  1092                  : INT_TO_INTEGER (s.st_gid)),
  1093                 make_lisp_time (get_stat_atime (&s)),
  1094                 make_lisp_time (get_stat_mtime (&s)),
  1095                 make_lisp_time (get_stat_ctime (&s)),
  1096 
  1097                 /* If the file size is a 4-byte type, assume that
  1098                    files of sizes in the 2-4 GiB range wrap around to
  1099                    negative values, as this is a common bug on older
  1100                    32-bit platforms.  */
  1101                 INT_TO_INTEGER (sizeof (s.st_size) == 4
  1102                             ? s.st_size & 0xffffffffu
  1103                             : s.st_size),
  1104 
  1105                 make_string (modes, 10),
  1106                 Qt,
  1107                 INT_TO_INTEGER (s.st_ino),
  1108                 INT_TO_INTEGER (s.st_dev));
  1109 }
