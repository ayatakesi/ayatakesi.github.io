    33 /* This program takes a directory as input, and generates a
    34    ``directory-tree'' file suitable for inclusion in an Android
    35    application package.
    36 
    37    Such a file records the layout of the `assets' directory in the
    38    package.  Emacs records this information itself and uses it in the
    39    Android emulation of readdir, because the system asset manager APIs
    40    are routinely buggy, and are often unable to locate directories or
    41    files.
    42 
    43    The file is packed, with no data alignment guarantees made.  The
    44    file starts with the bytes "EMACS", following which is the name of
    45    the first file or directory, a NULL byte and an unsigned int
    46    indicating the offset from the start of the file to the start of
    47    the next sibling.  Following that is a list of subdirectories or
    48    files in the same format.  The long is stored LSB first.  */
    49

    52 struct directory_tree
    53 {
    54   /* The offset to the next sibling.  */
    55   size_t offset;
    56 
    57   /* The name of this directory or file.  */
    58   char *name;
    59 
    60   /* Subdirectories and files inside this directory.  */
    61   struct directory_tree *children, *next;
    62 };

    89 /* Recursively build a struct directory_tree structure for each
    90    subdirectory or file in DIR, in preparation for writing it out to
    91    disk.  PARENT should be the directory tree associated with the
    92    parent directory, or else PARENT->offset must be initialized to
    93    5.  */
    94 
    95 static void
    96 main_1 (DIR *dir, struct directory_tree *parent)
    97 {
    98   struct dirent *dirent;
    99   int dir_fd, fd;
   100   struct stat statb;
   101   struct directory_tree *this, **last;
   102   size_t length;
   103   DIR *otherdir;
   104 
   105   dir_fd = dirfd (dir);
   106   last = &parent->children;
   107 
   108   while ((dirent = readdir (dir)))
   109     {
   110       /* Determine what kind of file DIRENT is.  */
   111 
   112       if (fstatat (dir_fd, dirent->d_name, &statb,
   113                    AT_SYMLINK_NOFOLLOW) == -1)
   114         croak ("fstatat");
   115 
   116       /* Ignore . and ...  */
   117 
   118       if (!strcmp (dirent->d_name, ".")
   119           || !strcmp (dirent->d_name, ".."))
   120         continue;
   121 
   122       length = strlen (dirent->d_name);
   123 
   124       if (statb.st_mode & S_IFDIR)
   125         {
   126           /* This is a directory.  Write its name followed by a
   127              trailing slash, then a NULL byte, and the offset to the
   128              next sibling.  */
   129           this = xmalloc (sizeof *this);
   130           this->children = NULL;
   131           this->next = NULL;
   132           *last = this;
   133           last = &this->next;
   134           this->name = xmalloc (length + 2);
   135           strcpy (this->name, dirent->d_name);
   136 
   137           /* Now record the offset to the end of this directory.  This
   138              is length + 1, for the file name, and 5 more bytes for
   139              the trailing NULL and long.  */
   140           this->offset = parent->offset + length + 6;
   141 
   142           /* Terminate that with a slash and trailing NULL byte.  */
   143           this->name[length] = '/';
   144           this->name[length + 1] = '\0';
   145 
   146           /* Open and build that directory recursively.  */
   147 
   148           fd = openat (dir_fd, dirent->d_name, O_DIRECTORY,
   149                        O_RDONLY);
   150           if (fd < 0)
   151             croak ("openat");
   152           otherdir = fdopendir (fd);
   153           if (!otherdir)
   154             croak ("fdopendir");
   155 
   156           main_1 (otherdir, this);
   157 
   158           /* Close this directory.  */
   159           closedir (otherdir);
   160 
   161           /* Finally, set parent->offset to this->offset as well.  */
   162           parent->offset = this->offset;
   163         }
   164       else if (statb.st_mode & S_IFREG)
   165         {
   166           /* This is a regular file.  */
   167           this = xmalloc (sizeof *this);
   168           this->children = NULL;
   169           this->next = NULL;
   170           *last = this;
   171           last = &this->next;
   172           this->name = xmalloc (length + 1);
   173           strcpy (this->name, dirent->d_name);
   174 
   175           /* This is one byte shorter because there is no trailing
   176              slash.  */
   177           this->offset = parent->offset + length + 5;
   178           parent->offset = this->offset;
   179         }
   180     }
   181 }

   183 /* Write the struct directory_tree TREE and all of is children to the
   184    file descriptor FD.  OFFSET is the offset of TREE and may be
   185    modified; it is only used for checking purposes.  */
   186 
   187 static void
   188 main_2 (int fd, struct directory_tree *tree, size_t *offset)
   189 {
   190   ssize_t size;
   191   struct directory_tree *child;
   192   unsigned int output;
   193 
   194   /* Write tree->name with the trailing NULL byte.  */
   195   size = strlen (tree->name) + 1;
   196   if (write (fd, tree->name, size) < size)
   197     croak ("write");
   198 
   199   /* Write the offset.  */
   200 #ifdef WORDS_BIGENDIAN
   201   output = bswap_32 (tree->offset);
   202 #else
   203   output = tree->offset;
   204 #endif
   205   if (write (fd, &output, 4) < 1)
   206     croak ("write");
   207   size += 4;
   208 
   209   /* Now update offset.  */
   210   *offset += size;
   211 
   212   /* Write out each child.  */
   213   for (child = tree->children; child; child = child->next)
   214     main_2 (fd, child, offset);
   215 
   216   /* Verify the offset is correct.  */
   217   if (tree->offset != *offset)
   218     {
   219       fprintf (stderr,
   220                "asset-directory-tool: invalid offset: expected %tu, "
   221                "got %tu.\n"
   222                "Please report this bug to bug-gnu-emacs@gnu.org, along\n"
   223                "with an archive containing the contents of the java/inst"
   224                "all_temp directory.\n",
   225                tree->offset, *offset);
   226       abort ();
   227     }
   228 }

   230 int
   231 main (int argc, char **argv)
   232 {
   233   int fd;
   234   DIR *indir;
   235   struct directory_tree tree;
   236   size_t offset;
   237 
   238   if (argc != 3)
   239     {
   240       fprintf (stderr, "usage: %s directory output-file\n",
   241                argv[0]);
   242       return EXIT_FAILURE;
   243     }
   244 
   245   fd = open (argv[2], O_CREAT | O_TRUNC | O_RDWR,
   246              S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP);
   247 
   248   if (fd < 0)
   249     {
   250       perror ("open");
   251       return EXIT_FAILURE;
   252     }
   253 
   254   indir = opendir (argv[1]);
   255 
   256   if (!indir)
   257     {
   258       perror ("opendir");
   259       return EXIT_FAILURE;
   260     }
   261 
   262   /* Write the first 5 byte header to FD.  */
   263 
   264   if (write (fd, "EMACS", 5) < 5)
   265     {
   266       perror ("write");
   267       return EXIT_FAILURE;
   268     }
   269 
   270   /* Now iterate through children of INDIR, building the directory
   271      tree.  */
   272   tree.offset = 5;
   273   tree.children = NULL;
   274 
   275   main_1 (indir, &tree);
   276   closedir (indir);
   277 
   278   /* Finally, write the directory tree to the output file.  */
   279   offset = 5;
   280   for (; tree.children; tree.children = tree.children->next)
   281     main_2 (fd, tree.children, &offset);
   282 
   283   return 0;
   284 }
