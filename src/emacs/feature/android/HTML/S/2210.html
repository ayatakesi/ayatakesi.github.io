   343 static void *
   344 android_run_select_thread (void *data)
   345 {
   346   /* Apparently this is required too.  */
   347   JNI_STACK_ALIGNMENT_PROLOGUE;
   348 
   349   int rc;
   350 #if __ANDROID_API__ < 16
   351   int nfds;
   352   fd_set readfds;
   353   char byte;
   354 #else
   355   sigset_t signals, waitset;
   356   int sig;
   357 #endif
   358 
   359 #if __ANDROID_API__ < 16
   360   /* A completely different implementation is used when building for
   361      Android versions earlier than 16, because pselect with a signal
   362      mask does not work there.  Instead of blocking SIGUSR1 and
   363      unblocking it inside pselect, a file descriptor is used instead.
   364      Something is written to the file descriptor every time select is
   365      supposed to return.  */
   366 
   367   while (true)
   368     {
   369       /* Wait for the thread to be released.  */
   370       while (sem_wait (&android_pselect_start_sem) < 0)
   371         ;;
   372 
   373       /* Get the select lock and call pselect.  API 8 does not have
   374          working pselect in any sense.  Instead, pselect wakes up on
   375          select_pipe[0].  */
   376 
   377       pthread_mutex_lock (&event_queue.select_mutex);
   378       nfds = android_pselect_nfds;
   379 
   380       if (android_pselect_readfds)
   381         readfds = *android_pselect_readfds;
   382       else
   383         FD_ZERO (&readfds);
   384 
   385       if (nfds < select_pipe[0] + 1)
   386         nfds = select_pipe[0] + 1;
   387       FD_SET (select_pipe[0], &readfds);
   388 
   389       rc = pselect (nfds, &readfds,
   390                     android_pselect_writefds,
   391                     android_pselect_exceptfds,
   392                     android_pselect_timeout,
   393                     NULL);
   394 
   395       /* Subtract 1 from rc if readfds contains the select pipe, and
   396          also remove it from that set.  */
   397 
   398       if (rc != -1 && FD_ISSET (select_pipe[0], &readfds))
   399         {
   400           rc -= 1;
   401           FD_CLR (select_pipe[0], &readfds);
   402 
   403           /* If no file descriptors aside from the select pipe are
   404              ready, then pretend that an error has occurred.  */
   405           if (!rc)
   406             rc = -1;
   407         }
   408 
   409       /* Save the read file descriptor set back again.  */
   410 
   411       if (android_pselect_readfds)
   412         *android_pselect_readfds = readfds;
   413 
   414       android_pselect_rc = rc;
   415       pthread_mutex_unlock (&event_queue.select_mutex);
   416 
   417       /* Signal the main thread that there is now data to read.  Hold
   418          the event queue lock during this process to make sure this
   419          does not happen before the main thread begins to wait for the
   420          condition variable.  */
   421 
   422       pthread_mutex_lock (&event_queue.mutex);
   423       pthread_cond_broadcast (&event_queue.read_var);
   424       pthread_mutex_unlock (&event_queue.mutex);
   425 
   426       /* Read a single byte from the select pipe.  */
   427       read (select_pipe[0], &byte, 1);
   428 
   429       /* Signal the Emacs thread that pselect is done.  If read_var
   430          was signaled by android_write_event, event_queue.mutex could
   431          still be locked, so this must come before.  */
   432       sem_post (&android_pselect_sem);
   433     }
   434 #else
   435   if (pthread_sigmask (SIG_BLOCK, &signals, NULL))
   436     __android_log_print (ANDROID_LOG_FATAL, __func__,
   437                          "pthread_sigmask: %s",
   438                          strerror (errno));
   439 
   440   sigfillset (&signals);
   441   sigdelset (&signals, SIGUSR1);
   442   sigemptyset (&waitset);
   443   sigaddset (&waitset, SIGUSR1);
   444 
   445   while (true)
   446     {
   447       /* Wait for the thread to be released.  */
   448       while (sem_wait (&android_pselect_start_sem) < 0)
   449         ;;
   450 
   451       /* Clear the ``pselect interrupted'' flag.  This is safe because
   452          right now, SIGUSR1 is blocked.  */
   453       android_pselect_interrupted = 0;
   454 
   455       /* Get the select lock and call pselect.  */
   456       pthread_mutex_lock (&event_queue.select_mutex);
   457       rc = pselect (android_pselect_nfds,
   458                     android_pselect_readfds,
   459                     android_pselect_writefds,
   460                     android_pselect_exceptfds,
   461                     android_pselect_timeout,
   462                     &signals);
   463       android_pselect_rc = rc;
   464       pthread_mutex_unlock (&event_queue.select_mutex);
   465 
   466       /* Signal the main thread that there is now data to read.  Hold
   467          the event queue lock during this process to make sure this
   468          does not happen before the main thread begins to wait for the
   469          condition variable.  */
   470 
   471       pthread_mutex_lock (&event_queue.mutex);
   472       pthread_cond_broadcast (&event_queue.read_var);
   473       pthread_mutex_unlock (&event_queue.mutex);
   474 
   475       /* Check `android_pselect_interrupted' instead of rc and errno.
   476 
   477          This is because `pselect' does not return an rc of -1 upon
   478          being interrupted in some versions of Android, but does set
   479          signal masks correctly.  */
   480 
   481       if (!android_pselect_interrupted)
   482         /* Now, wait for SIGUSR1, unless pselect was interrupted and
   483            the signal was already delivered.  The Emacs thread will
   484            always send this signal after read_var is triggered or the
   485            UI thread has sent an event.  */
   486         sigwait (&waitset, &sig);
   487 
   488       /* Signal the Emacs thread that pselect is done.  If read_var
   489          was signaled by android_write_event, event_queue.mutex could
   490          still be locked, so this must come before.  */
   491       sem_post (&android_pselect_sem);
   492     }
   493 #endif
   494 
   495   return NULL;
   496 }

   534                          strerror (errno));
   535 
   536   if (pthread_cond_init (&event_queue.read_var, NULL))
   537     __android_log_print (ANDROID_LOG_FATAL, __func__,
   538                          "pthread_cond_init: %s",
   539                          strerror (errno));
   540 
   541   sem_init (&android_pselect_sem, 0, 0);
   542   sem_init (&android_pselect_start_sem, 0, 0);
   543   sem_init (&android_query_sem, 0, 0);
   544 
   545   event_queue.events.next = &event_queue.events;
   546   event_queue.events.last = &event_queue.events;
   547 
   548 #if __ANDROID_API__ >= 16
   549 
   550   /* Before starting the select thread, make sure the disposition for
   551      SIGUSR1 is correct.  */
   552   sigfillset (&sa.sa_mask);
   553   sa.sa_sigaction = android_handle_sigusr1;
   554   sa.sa_flags = SA_SIGINFO;
   555 
   556 #else
   557 
   558   /* Set up the file descriptor used to wake up pselect.  */
   559   if (pipe2 (select_pipe, O_CLOEXEC) < 0)
   560     __android_log_print (ANDROID_LOG_FATAL, __func__,
   561                          "pipe2: %s", strerror (errno));
   562 
   563   /* Make sure the read end will fit in fd_set.  */

   565     __android_log_print (ANDROID_LOG_FATAL, __func__,
   566                          "read end of select pipe"
   567                          " lies outside FD_SETSIZE!");
   568 
   569 #endif
   570 
   571   if (sigaction (SIGUSR1, &sa, NULL))
   572     __android_log_print (ANDROID_LOG_FATAL, __func__,
   573                          "sigaction: %s",
   574                          strerror (errno));
   575 
   576   /* Start the select thread.  */
   577   if (pthread_create (&event_queue.select_thread, NULL,
   578                       android_run_select_thread, NULL))
   579     __android_log_print (ANDROID_LOG_FATAL, __func__,
   580                          "pthread_create: %s",
   581                          strerror (errno));
   582 }

   625 void
   626 android_next_event (union android_event *event_return)
   627 {
   628   struct android_event_container *container;
   629 
   630   pthread_mutex_lock (&event_queue.mutex);
   631 
   632   /* Wait for events to appear if there are none available to
   633      read.  */
   634   if (!event_queue.num_events)
   635     pthread_cond_wait (&event_queue.read_var,
   636                        &event_queue.mutex);
   637 
   638   /* Obtain the event from the end of the queue.  */
   639   container = event_queue.events.last;
   640   eassert (container != &event_queue.events);
   641 
   642   /* Remove the event from the queue and copy it to the caller
   643      supplied buffer.  */
   644   container->last->next = container->next;
   645   container->next->last = container->last;
   646   *event_return = container->event;
   647   event_queue.num_events--;
   648 
   649   /* Free the container.  */
   650   free (container);
   651 
   652   /* Unlock the queue.  */
   653   pthread_mutex_unlock (&event_queue.mutex);
   654 }

   656 bool
   657 android_check_if_event (union android_event *event_return,
   658                         bool (*predicate) (union android_event *,
   659                                            void *),
   660                         void *arg)
   661 {
   662   struct android_event_container *container;
   663 
   664   pthread_mutex_lock (&event_queue.mutex);
   665 
   666   /* Loop over each event.  */
   667   container = event_queue.events.last;
   668   for (; container != &event_queue.events; container = container->last)
   669     {
   670       /* See if the predicate matches.  */
   671       if ((*predicate) (&container->event, arg))
   672         {
   673           /* Copy out the event and return true.  */
   674           *event_return = container->event;
   675           --event_queue.num_events;
   676 
   677           /* Unlink container.  */
   678           container->last->next = container->next;
   679           container->next->last = container->last;
   680           free (container);
   681           pthread_mutex_unlock (&event_queue.mutex);
   682           return true;
   683         }
   684     }
   685 
   686   pthread_mutex_unlock (&event_queue.mutex);
   687   return false;
   688 }

   690 void
   691 android_write_event (union android_event *event)
   692 {
   693   struct android_event_container *container;
   694 
   695   container = malloc (sizeof *container);
   696 
   697   if (!container)
   698     return;
   699 
   700   /* If the event queue hasn't been initialized yet, return false.  */
   701   if (!event_queue.events.next)
   702     return;
   703 
   704   pthread_mutex_lock (&event_queue.mutex);
   705   container->next = event_queue.events.next;
   706   container->last = &event_queue.events;
   707   container->next->last = container;
   708   container->last->next = container;
   709   container->event = *event;
   710   event_queue.num_events++;
   711   pthread_cond_broadcast (&event_queue.read_var);
   712   pthread_mutex_unlock (&event_queue.mutex);
   713 
   714   /* Now set pending_signals to true, and raise SIGIO to interrupt any
   715      ongoing reads if the event is important.  */
   716   pending_signals = true;
   717 
   718   switch (event->type)
   719     {
   720       /* Key press and window action events are considered important,
   721          as they either end up quitting or asking for responses to the
   722          IME.  */
   723     case ANDROID_KEY_PRESS:
   724     case ANDROID_WINDOW_ACTION:
   725       kill (getpid (), SIGIO);
   726       break;
   727 
   728     default:
   729       break;
   730     }
   731 }

   735 /* Whether or not the UI thread has been waiting for a significant
   736    amount of time for a function to run in the main thread, and Emacs
   737    should answer the query ASAP.  */
   738 static bool android_urgent_query;
   739 
   740 int
   741 android_select (int nfds, fd_set *readfds, fd_set *writefds,
   742                 fd_set *exceptfds, struct timespec *timeout)
   743 {
   744   int nfds_return;
   745 #if __ANDROID_API__ < 16
   746   static char byte;
   747 #endif
   748 
   749   /* Since Emacs is reading keyboard input again, signify that queries
   750      from input methods are no longer ``urgent''.  */
   751 
   752   __atomic_clear (&android_urgent_query, __ATOMIC_RELEASE);
   753 
   754   /* Check for and run anything the UI thread wants to run on the main
   755      thread.  */
   756   android_check_query ();
   757 
   758   pthread_mutex_lock (&event_queue.mutex);
   759 
   760   if (event_queue.num_events)
   761     {
   762       pthread_mutex_unlock (&event_queue.mutex);
   763       return 1;
   764     }
   765 
   766   nfds_return = 0;
   767 
   768   pthread_mutex_lock (&event_queue.select_mutex);
   769   android_pselect_nfds = nfds;
   770   android_pselect_readfds = readfds;
   771   android_pselect_writefds = writefds;
   772   android_pselect_exceptfds = exceptfds;
   773   android_pselect_timeout = timeout;
   774   pthread_mutex_unlock (&event_queue.select_mutex);
   775 
   776   /* Release the select thread.  */
   777   sem_post (&android_pselect_start_sem);
   778 
   779   /* Start waiting for the event queue condition to be set.  */
   780   pthread_cond_wait (&event_queue.read_var, &event_queue.mutex);
   781 
   782 #if __ANDROID_API__ >= 16
   783   /* Interrupt the select thread now, in case it's still in
   784      pselect.  */
   785   pthread_kill (event_queue.select_thread, SIGUSR1);
   786 #else
   787   /* Interrupt the select thread by writing to the select pipe.  */
   788   if (write (select_pipe[1], &byte, 1) != 1)
   789     __android_log_print (ANDROID_LOG_FATAL, __func__,
   790                          "write: %s", strerror (errno));
   791 #endif
   792 
   793   /* Unlock the event queue mutex.  */
   794   pthread_mutex_unlock (&event_queue.mutex);
   795 
   796   /* Wait for pselect to return in any case.  This must be done with
   797      the event queue mutex unlocked.  Otherwise, the pselect thread
   798      can hang if it tries to lock the event queue mutex to signal
   799      read_var after the UI thread has already done so.  */
   800   while (sem_wait (&android_pselect_sem) < 0)
   801     ;;
   802 
   803   /* If there are now events in the queue, return 1.  */
   804 
   805   pthread_mutex_lock (&event_queue.mutex);
   806   if (event_queue.num_events)
   807     nfds_return = 1;
   808   pthread_mutex_unlock (&event_queue.mutex);
   809 
   810   /* Add the return value of pselect if it has also found ready file
   811      descriptors.  */
   812 
   813   if (android_pselect_rc >= 0)
   814     nfds_return += android_pselect_rc;
   815   else if (!nfds_return)
   816     /* If pselect was interrupted and nfds_return is 0 (meaning that
   817        no events have been read), indicate that an error has taken
   818        place.  */
   819     nfds_return = android_pselect_rc;
   820 
   821   if ((android_pselect_rc < 0) && nfds_return >= 0)
   822     {
   823       /* Clear the file descriptor sets if events will be delivered
   824          but no file descriptors have become ready to prevent the
   825          caller from misinterpreting a non-zero return value.  */
   826 
   827       if (readfds)
   828         FD_ZERO (readfds);
   829 
   830       if (writefds)
   831         FD_ZERO (writefds);
   832 
   833       if (exceptfds)
   834         FD_ZERO (exceptfds);
   835     }
   836 
   837   /* This is to shut up process.c when pselect gets EINTR.  */
   838   if (nfds_return < 0)
   839     errno = EINTR;
   840 
   841   /* Now check for and run anything the UI thread wants to run in the
   842      main thread.  */
   843   android_check_query ();
   844 
   845   return nfds_return;
   846 }

   883 /* Asset directory handling functions.  ``directory-tree'' is a file in
   884    the root of the assets directory describing its contents.
   885 
   886    See lib-src/asset-directory-tool for more details.  */
   887 
   888 /* The Android directory tree.  */
   889 static const char *directory_tree;
   890 
   891 /* The size of the directory tree.  */
   892 static size_t directory_tree_size;
   893 
   894 /* Read an unaligned (32-bit) long from the address POINTER.  */
   895 
   896 static unsigned int
   897 android_extract_long (char *pointer)
   898 {
   899   unsigned int number;
   900 
   901   memcpy (&number, pointer, sizeof number);
   902   return number;
   903 }
   904 
   905 /* Scan to the file FILE in the asset directory tree.  Return a
   906    pointer to the end of that file (immediately before any children)
   907    in the directory tree, or NULL if that file does not exist.
   908 
   909    If returning non-NULL, also return the offset to the end of the
   910    last subdirectory or file in *LIMIT_RETURN.  LIMIT_RETURN may be
   911    NULL.
   912 
   913    FILE must have less than 11 levels of nesting.  If it ends with a
   914    trailing slash, then NULL will be returned if it is not actually a
   915    directory.  */
   916 
   917 static const char *
   918 android_scan_directory_tree (char *file, size_t *limit_return)
   919 {
   920   char *token, *saveptr, *copy, *copy1, *start, *max, *limit;
   921   size_t token_length, ntokens, i;
   922   char *tokens[10];
   923 
   924   USE_SAFE_ALLOCA;
   925 
   926   /* Skip past the 5 byte header.  */
   927   start = (char *) directory_tree + 5;
   928 
   929   /* Figure out the current limit.  */
   930   limit = (char *) directory_tree + directory_tree_size;
   931 
   932   /* Now, split `file' into tokens, with the delimiter being the file
   933      name separator.  Look for the file and seek past it.  */
   934 
   935   ntokens = 0;
   936   saveptr = NULL;
   937   copy = copy1 = xstrdup (file);
   938   memset (tokens, 0, sizeof tokens);
   939 
   940   while ((token = strtok_r (copy, "/", &saveptr)))
   941     {
   942       copy = NULL;
   943 
   944       /* Make sure ntokens is within bounds.  */
   945       if (ntokens == ARRAYELTS (tokens))
   946         {
   947           xfree (copy1);
   948           goto fail;
   949         }
   950 
   951       tokens[ntokens] = SAFE_ALLOCA (strlen (token) + 1);
   952       memcpy (tokens[ntokens], token, strlen (token) + 1);
   953       ntokens++;
   954     }
   955 
   956   /* Free the copy created for strtok_r.  */
   957   xfree (copy1);
   958 
   959   /* If there are no tokens, just return the start of the directory
   960      tree.  */
   961 
   962   if (!ntokens)
   963     {
   964       SAFE_FREE ();
   965 
   966       /* Return the size of the directory tree as the limit.
   967          Do not subtract the initial header bytes, as the limit
   968          is an offset from the start of the file.  */
   969 
   970       if (limit_return)
   971         *limit_return = directory_tree_size;
   972 
   973       return start;
   974     }
   975 
   976   /* Loop through tokens, indexing the directory tree each time.  */
   977 
   978   for (i = 0; i < ntokens; ++i)
   979     {
   980       token = tokens[i];
   981 
   982       /* Figure out how many bytes to compare.  */
   983       token_length = strlen (token);
   984 
   985     again:
   986 
   987       /* If this would be past the directory, return NULL.  */
   988       if (start + token_length > limit)
   989         goto fail;
   990 
   991       /* Now compare the file name.  */
   992       if (!memcmp (start, token, token_length))
   993         {
   994           /* They probably match.  Find the NULL byte.  It must be
   995              either one byte past start + token_length, with the last
   996              byte a trailing slash (indicating that it is a
   997              directory), or just start + token_length.  Return 4 bytes
   998              past the next NULL byte.  */
   999 
  1000           max = memchr (start, 0, limit - start);
  1001 
  1002           if (max != start + token_length
  1003               && !(max == start + token_length + 1
  1004                    && *(max - 1) == '/'))
  1005             goto false_positive;
  1006 
  1007           /* Return it if it exists and is in range, and this is the
  1008              last token.  Otherwise, set it as start and the limit as
  1009              start + the offset and continue the loop.  */
  1010 
  1011           if (max && max + 5 <= limit)
  1012             {
  1013               if (i < ntokens - 1)
  1014                 {
  1015                   start = max + 5;
  1016                   limit = ((char *) directory_tree
  1017                            + android_extract_long (max + 1));
  1018 
  1019                   /* Make sure limit is still in range.  */
  1020                   if (limit > directory_tree + directory_tree_size
  1021                       || start > directory_tree + directory_tree_size)
  1022                     goto fail;
  1023 
  1024                   continue;
  1025                 }
  1026 
  1027               /* Now see if max is not a directory and file is.  If
  1028                  file is a directory, then return NULL.  */
  1029               if (*(max - 1) != '/' && file[strlen (file) - 1] == '/')
  1030                 max = NULL;
  1031               else
  1032                 {
  1033                   /* Figure out the limit.  */
  1034                   if (limit_return)
  1035                     *limit_return = android_extract_long (max + 1);
  1036 
  1037                   /* Go to the end of this file.  */
  1038                   max += 5;
  1039                 }
  1040 
  1041               SAFE_FREE ();
  1042               return max;
  1043             }
  1044 
  1045           /* Return NULL otherwise.  */
  1046           __android_log_print (ANDROID_LOG_WARN, __func__,
  1047                                "could not scan to end of directory tree"
  1048                                ": %s", file);
  1049           goto fail;
  1050         }
  1051 
  1052     false_positive:
  1053 
  1054       /* No match was found.  Set start to the next sibling and try
  1055          again.  */
  1056 
  1057       start = memchr (start, 0, limit - start);
  1058 
  1059       if (!start || start + 5 > limit)
  1060         goto fail;
  1061 
  1062       start = ((char *) directory_tree
  1063                + android_extract_long (start + 1));
  1064 
  1065       /* Make sure start is still in bounds.  */
  1066 
  1067       if (start > limit)
  1068         goto fail;
  1069 
  1070       /* Continue the loop.  */
  1071       goto again;
  1072     }
  1073 
  1074  fail:
  1075   SAFE_FREE ();
  1076   return NULL;
  1077 }

  1079 /* Return whether or not the directory tree entry DIR is a
  1080    directory.
  1081 
  1082    DIR should be a value returned by
  1083    `android_scan_directory_tree'.  */
  1084 
  1085 static bool
  1086 android_is_directory (const char *dir)
  1087 {
  1088   /* If the directory is the directory tree, then it is a
  1089      directory.  */
  1090   if (dir == directory_tree + 5)
  1091     return true;

  1096           && *(dir - 6) == '/');
  1097 }

  1202 /* Return the content URI corresponding to a `/content' file name,
  1203    or NULL if it is not a content URI.
  1204 
  1205    This function is not reentrant.  */
  1206 
  1207 static const char *
  1208 android_get_content_name (const char *filename)
  1209 {
  1210   static char buffer[PATH_MAX + 1];
  1211   char *head, *token, *saveptr, *copy;
  1212   size_t n;
  1213 
  1214   n = PATH_MAX;
  1215 
  1216   /* Find the file name described if it starts with `/content'.  If
  1217      just the directory is described, return content://.  */
  1218 
  1219   filename = android_is_special_directory (filename, "/content");
  1220 
  1221   if (!filename)
  1222     return NULL;
  1223 
  1224   if (!*filename)
  1225     return "content://";
  1226 
  1227   /* Now copy FILENAME into a buffer and convert it into a content
  1228      URI.  */
  1229 
  1230   copy = xstrdup (filename);
  1231   token = saveptr = NULL;
  1232   head = stpcpy (buffer, "content:/");
  1233 
  1234   /* Split FILENAME by slashes.  */
  1235 
  1236   while ((token = strtok_r (!token ? copy : NULL,
  1237                             "/", &saveptr)))
  1238     {

  1250   xfree (copy);
  1251

  1312 /* Like fstatat.  However, if dirfd is AT_FDCWD and PATHNAME is an
  1313    asset, find the information for the corresponding asset, and if
  1314    dirfd is an offset into directory_tree as returned by
  1315    `android_dirfd', find the information within the corresponding

  1317 
  1318 int
  1319 android_fstatat (int dirfd, const char *restrict pathname,
  1320                  struct stat *restrict statbuf, int flags)
  1321 {
  1322   AAsset *asset_desc;
  1323   const char *asset;
  1324   const char *asset_dir;
  1325   int fd, rc;
  1326 
  1327   /* Look up whether or not DIRFD belongs to an open struct
  1328      android_dir.  */
  1329 
  1330   if (dirfd != AT_FDCWD)
  1331     dirfd
  1332       = android_lookup_asset_directory_fd (dirfd, &pathname,
  1333                                            pathname);
  1334 
  1335   if (dirfd == AT_FDCWD
  1336       && asset_manager
  1337       && (asset = android_get_asset_name (pathname)))
  1338     {
  1339       /* Look up whether or not PATHNAME happens to be a
  1340          directory.  */

  1342                                                NULL);

  1344       if (!asset_dir)
  1345         {
  1346           errno = ENOENT;
  1347           return -1;
  1348         }
  1349 
  1350       if (android_is_directory (asset_dir))
  1351         {
  1352           memset (statbuf, 0, sizeof *statbuf);
  1353 
  1354           /* Fill in the stat buffer.  */
  1355           statbuf->st_mode = S_IFDIR | S_IRUSR | S_IRGRP | S_IROTH;
  1356           return 0;
  1357         }
  1358 
  1359       /* AASSET_MODE_STREAMING is fastest here.  */
  1360       asset_desc = AAssetManager_open (asset_manager, asset,
  1361                                        AASSET_MODE_STREAMING);
  1362 
  1363       if (!asset_desc)
  1364         return ENOENT;
  1365 
  1366       memset (statbuf, 0, sizeof *statbuf);
  1367 
  1368       /* Fill in the stat buffer.  */
  1369       statbuf->st_mode = S_IFREG | S_IRUSR | S_IRGRP | S_IROTH;
  1370       statbuf->st_size = AAsset_getLength (asset_desc);

  1374       return 0;
  1375     }
  1376 
  1377   if (dirfd == AT_FDCWD
  1378       && android_init_gui

  1387         return -1;
  1388 
  1389       rc = fstat (fd, statbuf);
  1390       android_close (fd);
  1391       return rc;
  1392     }
  1393 
  1394   return fstatat (dirfd, pathname, statbuf, flags);
  1395 }

  1422 /* Do the same as android_hack_asset_fd, but use an unlinked temporary
  1423    file to cater to old Android kernels where ashmem files are not
  1424    readable.  */
  1425 
  1426 static int
  1427 android_hack_asset_fd_fallback (AAsset *asset)
  1428 {
  1429   int fd;
  1430   char filename[PATH_MAX];
  1431   size_t size;
  1432   void *mem;
  1433 
  1434   /* Assets must be small enough to fit in size_t, if off_t is
  1435      larger.  */
  1436   size = AAsset_getLength (asset);
  1437 
  1438   /* Get an unlinked file descriptor from a file in the cache
  1439      directory, which is guaranteed to only be written to by Emacs.
  1440      Creating an ashmem file descriptor and reading from it doesn't
  1441      work on these old Android versions.  */
  1442 
  1443   snprintf (filename, PATH_MAX, "%s/temp~unlinked.%d",
  1444             android_cache_dir, getpid ());
  1445   fd = open (filename, O_CREAT | O_RDWR | O_TRUNC,
  1446              S_IRUSR | S_IWUSR);
  1447 
  1448   if (fd < 0)
  1449     return -1;
  1450 
  1451   if (unlink (filename))
  1452     goto fail;
  1453 
  1454   if (ftruncate (fd, size))
  1455     goto fail;
  1456 
  1457   mem = mmap (NULL, size, PROT_WRITE, MAP_SHARED, fd, 0);
  1458   if (mem == MAP_FAILED)
  1459     {
  1460       __android_log_print (ANDROID_LOG_ERROR, __func__,
  1461                            "mmap: %s", strerror (errno));
  1462       goto fail;
  1463     }
  1464 
  1465   if (AAsset_read (asset, mem, size) != size)
  1466     {
  1467       /* Too little was read.  Close the file descriptor and
  1468          report an error.  */
  1469       __android_log_print (ANDROID_LOG_ERROR, __func__,
  1470                            "AAsset_read: %s", strerror (errno));
  1471       goto fail;
  1472     }
  1473 
  1474   munmap (mem, size);
  1475   return fd;
  1476 
  1477  fail:
  1478   close (fd);
  1479   return -1;
  1480 }

  1482 /* Pointer to the `ASharedMemory_create' function which is loaded
  1483    dynamically.  */
  1484 static int (*asharedmemory_create) (const char *, size_t);
  1485 
  1486 /* Return whether or not shared memory file descriptors can also be
  1487    read from, and are thus suitable for creating asset files.
  1488 
  1489    This does not work on some ancient Android systems running old
  1490    versions of the kernel.  */
  1491 
  1492 static bool
  1493 android_detect_ashmem (void)
  1494 {
  1495   int fd, rc;
  1496   void *mem;
  1497   char test_buffer[10];
  1498 
  1499   memcpy (test_buffer, "abcdefghi", 10);
  1500 
  1501   /* Create the file descriptor to be used for the test.  */
  1502 
  1503   /* Android 28 and earlier let Emacs access /dev/ashmem directly, so
  1504      prefer that over using ASharedMemory.  */
  1505 
  1506   if (android_api_level <= 28)
  1507     {
  1508       fd = open ("/dev/ashmem", O_RDWR);
  1509 
  1510       if (fd < 0)
  1511         return false;
  1512 
  1513       /* An empty name means the memory area will exist until the file
  1514          descriptor is closed, because no other process can
  1515          attach.  */
  1516       rc = ioctl (fd, ASHMEM_SET_NAME, "");
  1517 
  1518       if (rc < 0)
  1519         {
  1520           close (fd);
  1521           return false;
  1522         }
  1523 
  1524       rc = ioctl (fd, ASHMEM_SET_SIZE, sizeof test_buffer);
  1525 
  1526       if (rc < 0)
  1527         {
  1528           close (fd);
  1529           return false;
  1530         }
  1531     }
  1532   else
  1533     {
  1534       /* On the other hand, SELinux restrictions on Android 29 and
  1535          later require that Emacs use a system service to obtain
  1536          shared memory.  Load this dynamically, as this service is not
  1537          available on all versions of the NDK.  */
  1538 
  1539       if (!asharedmemory_create)
  1540         {
  1541           *(void **) (&asharedmemory_create)
  1542             = dlsym (RTLD_DEFAULT, "ASharedMemory_create");
  1543 
  1544           if (!asharedmemory_create)
  1545             {
  1546               __android_log_print (ANDROID_LOG_FATAL, __func__,
  1547                                    "dlsym: %s\n",
  1548                                    strerror (errno));
  1549               emacs_abort ();
  1550             }
  1551         }
  1552 
  1553       fd = asharedmemory_create ("", sizeof test_buffer);
  1554 
  1555       if (fd < 0)
  1556         return false;
  1557     }
  1558 
  1559   /* Now map the resource and write the test contents.  */
  1560 
  1561   mem = mmap (NULL, sizeof test_buffer, PROT_WRITE,
  1562               MAP_SHARED, fd, 0);
  1563   if (mem == MAP_FAILED)
  1564     {
  1565       close (fd);
  1566       return false;
  1567     }
  1568 
  1569   /* Copy over the test contents.  */
  1570   memcpy (mem, test_buffer, sizeof test_buffer);
  1571 
  1572   /* Return anyway even if munmap fails.  */
  1573   munmap (mem, sizeof test_buffer);
  1574 
  1575   /* Try to read the content back into test_buffer.  If this does not
  1576      compare equal to the original string, or the read fails, then
  1577      ashmem descriptors are not readable on this system.  */
  1578 
  1579   if ((read (fd, test_buffer, sizeof test_buffer)
  1580        != sizeof test_buffer)
  1581       || memcmp (test_buffer, "abcdefghi", sizeof test_buffer))
  1582     {
  1583       __android_log_print (ANDROID_LOG_WARN, __func__,
  1584                            "/dev/ashmem does not produce real"
  1585                            " temporary files on this system, so"
  1586                            " Emacs will fall back to creating"
  1587                            " unlinked temporary files.");
  1588       close (fd);
  1589       return false;
  1590     }
  1591 
  1592   close (fd);
  1593   return true;
  1594 }

  1769 /* `open' and such are modified even though they exist on Android,
  1770    because Emacs treats "/assets/" as a special directory that must
  1771    contain all assets in the application package.  */
  1772 
  1773 int
  1774 android_open (const char *filename, int oflag, mode_t mode)
  1775 {
  1776   const char *name;
  1777   AAsset *asset;
  1778   int fd;
  1779   size_t length;
  1780   jobject string;
  1781 
  1782   if (asset_manager && (name = android_get_asset_name (filename)))
  1783     {
  1784       /* If Emacs is trying to write to the file, return NULL.  */
  1785 
  1786       if (oflag & O_WRONLY || oflag & O_RDWR)
  1787         {
  1788           errno = EROFS;
  1789           return -1;
  1790         }
  1791 
  1792       if (oflag & O_DIRECTORY)
  1793         {
  1794           errno = ENOTSUP;
  1795           return -1;
  1796         }
  1797 
  1798       /* If given AASSET_MODE_BUFFER (which is what Emacs probably
  1799          does, given that a file descriptor is not always available),
  1800          the framework fails to uncompress the data before it returns
  1801          a file descriptor.  */
  1802       asset = AAssetManager_open (asset_manager, name,
  1803                                   AASSET_MODE_STREAMING);
  1804 
  1805       if (!asset)
  1806         {
  1807           errno = ENOENT;
  1808           return -1;
  1809         }
  1810 
  1811       /* Create a shared memory file descriptor containing the asset
  1812          contents.
  1813 
  1814          The documentation misleads people into thinking that
  1815          AAsset_openFileDescriptor does precisely this.  However, it
  1816          instead returns an offset into any uncompressed assets in the
  1817          ZIP archive.  This cannot be found in its documentation.  */
  1818 
  1819       fd = android_hack_asset_fd (asset);
  1820 
  1821       if (fd == -1)
  1822         {
  1823           AAsset_close (asset);
  1824           errno = ENXIO;
  1825           return -1;
  1826         }
  1827 
  1828       /* If O_CLOEXEC is specified, make the file descriptor close on
  1829          exec too.  */
  1830       if (oflag & O_CLOEXEC)
  1831         android_close_on_exec (fd);
  1832 
  1833       if (fd >= ANDROID_MAX_ASSET_FD || fd < 0)
  1834         {
  1835           /* Too bad.  Pretend this is an out of memory error.  */
  1836           errno = ENOMEM;
  1837 
  1838           if (fd >= 0)
  1839             close (fd);
  1840 
  1841           fd = -1;
  1842         }
  1843       else
  1844         {
  1845           assert (!(android_table[fd].flags
  1846                     & ANDROID_FD_TABLE_ENTRY_IS_VALID));
  1847           android_table[fd].flags = ANDROID_FD_TABLE_ENTRY_IS_VALID;
  1848           memset (&android_table[fd].statb, 0,
  1849                   sizeof android_table[fd].statb);
  1850 
  1851           /* Fill in some information that will be reported to
  1852              callers of android_fstat, among others.  */
  1853           android_table[fd].statb.st_mode
  1854             = S_IFREG | S_IRUSR | S_IRGRP | S_IROTH;
  1855 
  1856           /* Owned by root.  */
  1857           android_table[fd].statb.st_uid = 0;
  1858           android_table[fd].statb.st_gid = 0;
  1859 
  1860           /* Size of the file.  */
  1861           android_table[fd].statb.st_size
  1862             = AAsset_getLength (asset);
  1863         }
  1864 
  1865       AAsset_close (asset);
  1866       return fd;
  1867     }
  1868 
  1869   if (android_init_gui && android_content_name_p (filename))
  1870     {
  1871       /* This is a content:// URI.  Ask the system for a descriptor to
  1872          that file.  */
  1873 
  1874       name = android_get_content_name (filename);
  1875       length = strlen (name);
  1876 
  1877       /* Check if the mode is valid.  */
  1878 
  1879       if (oflag & O_DIRECTORY)
  1880         {
  1881           errno = ENOTSUP;
  1882           return -1;
  1883         }
  1884 
  1885       /* Allocate a buffer to hold the file name.  */
  1886       string = (*android_java_env)->NewByteArray (android_java_env,
  1887                                                   length);
  1888       if (!string)
  1889         {
  1890           (*android_java_env)->ExceptionClear (android_java_env);
  1891           errno = ENOMEM;
  1892           return -1;
  1893         }

  1977 /* Return the name of the file behind a file descriptor FD by reading
  1978    /proc/self/fd/.  Place the name in BUFFER, which should be able to
  1979    hold size bytes.  Value is 0 upon success, and 1 upon failure.  */
  1980 
  1981 static int
  1982 android_proc_name (int fd, char *buffer, size_t size)
  1983 {
  1984   char format[sizeof "/proc/self/fd/"
  1985               + INT_STRLEN_BOUND (int)];
  1986   ssize_t read;
  1987 
  1988   sprintf (format, "/proc/self/fd/%d", fd);
  1989   read = readlink (format, buffer, size - 1);
  1990 
  1991   if (read == -1)
  1992     return 1;
  1993 
  1994   buffer[read] = '\0';
  1995   return 0;
  1996 }

  2517 JNIEXPORT void JNICALL
  2518 NATIVE_NAME (initEmacs) (JNIEnv *env, jobject object, jarray argv,
  2519                          jobject dump_file_object, jint api_level)
  2520 {
  2521   /* android_emacs_init is not main, so GCC is not nice enough to add
  2522      the stack alignment prologue.
  2523 
  2524      Unfortunately for us, dalvik on Android 4.0.x calls native code
  2525      with a 4 byte aligned stack, so this prologue must be inserted
  2526      before each function exported via JNI.  */
  2527 
  2528   JNI_STACK_ALIGNMENT_PROLOGUE;
  2529 
  2530   char **c_argv;
  2531   jsize nelements, i;
  2532   jobject argument;
  2533   const char *c_argument;
  2534   char *dump_file;
  2535 
  2536   /* Set the Android API level.  */
  2537   android_api_level = api_level;
  2538 
  2539   android_java_env = env;
  2540 
  2541   nelements = (*env)->GetArrayLength (env, argv);
  2542   c_argv = alloca (sizeof *c_argv * nelements);
  2543 
  2544   for (i = 0; i < nelements; ++i)
  2545     {
  2546       argument = (*env)->GetObjectArrayElement (env, argv, i);
  2547       c_argument = (*env)->GetStringUTFChars (env, (jstring) argument,
  2548                                               NULL);
  2549 
  2550       if (!c_argument)
  2551         emacs_abort ();
  2552 
  2553       /* Note that c_argument is in ``modified UTF-8 encoding'', but
  2554          we don't care as NUL bytes are not being specified inside.  */
  2555       c_argv[i] = alloca (strlen (c_argument) + 1);
  2556       strcpy (c_argv[i], c_argument);
  2557       (*env)->ReleaseStringUTFChars (env, (jstring) argument, c_argument);
  2558     }
  2559 
  2560   android_init_emacs_service ();
  2561   android_init_emacs_pixmap ();
  2562   android_init_graphics_point ();
  2563   android_init_emacs_drawable ();
  2564   android_init_emacs_window ();
  2565   android_init_emacs_cursor ();
  2566 
  2567   /* Set HOME to the app data directory.  */
  2568   setenv ("HOME", android_files_dir, 1);
  2569 
  2570   /* Set TMPDIR to the temporary files directory.  */
  2571   setenv ("TMPDIR", android_cache_dir, 1);
  2572 
  2573   /* And finally set "SHELL" to /system/bin/sh.  Otherwise, some
  2574      programs will look for /bin/sh, which is problematic.  */
  2575   setenv ("SHELL", "/system/bin/sh", 1);
  2576 
  2577   /* Set the cwd to that directory as well.  */
  2578   if (chdir (android_files_dir))
  2579     __android_log_print (ANDROID_LOG_WARN, __func__,
  2580                          "chdir: %s", strerror (errno));
  2581 
  2582   /* Initialize the Android GUI as long as the service object was
  2583      set.  */
  2584 
  2585   if (emacs_service)
  2586     android_init_gui = true;
  2587 
  2588   /* Now see if a dump file has been specified and should be used.  */
  2589   dump_file = NULL;
  2590 
  2591   if (dump_file_object)
  2592     {
  2593       c_argument
  2594         = (*env)->GetStringUTFChars (env, (jstring) dump_file_object,
  2595                                      NULL);
  2596 
  2597       /* Copy the Java string data once.  */
  2598       dump_file = strdup (c_argument);
  2599 
  2600       /* Release the Java string data.  */
  2601       (*env)->ReleaseStringUTFChars (env, (jstring) dump_file_object,
  2602                                      c_argument);
  2603     }
  2604 
  2605   /* Delete local references to objects that are no longer needed.  */
  2606   ANDROID_DELETE_LOCAL_REF (argv);
  2607   ANDROID_DELETE_LOCAL_REF (dump_file_object);
  2608 
  2609   /* Restore the signal mask at the time of startup if it was changed
  2610      to block unwanted signals from reaching system threads.  */
  2611 
  2612   if (signal_mask_changed_p)
  2613     pthread_sigmask (SIG_SETMASK, &startup_signal_mask, NULL);
  2614 
  2615   /* Now start Emacs proper.  */
  2616   android_emacs_init (nelements, c_argv, dump_file);
  2617 
  2618   /* android_emacs_init should never return.  */
  2619   emacs_abort ();
  2620 }

  3044 JNIEXPORT void JNICALL
  3045 NATIVE_NAME (blitRect) (JNIEnv *env, jobject object,
  3046                         jobject src, jobject dest,
  3047                         jint x1, jint y1, jint x2, jint y2)
  3048 {
  3049   AndroidBitmapInfo src_info, dest_info;
  3050   unsigned char *src_data_1, *dest_data_1;
  3051   void *src_data, *dest_data;
  3052 
  3053   /* N.B. that X2 and Y2 represent the pixel past the edge of the
  3054      rectangle; thus, the width is x2 - x1 and the height is y2 -
  3055      y1.  */
  3056 
  3057   memset (&src_info, 0, sizeof src_info);
  3058   memset (&dest_info, 0, sizeof dest_info);
  3059   AndroidBitmap_getInfo (env, src, &src_info);
  3060   AndroidBitmap_getInfo (env, dest, &dest_info);
  3061 
  3062   /* If the stride is 0 after a call to `getInfo', assume it
  3063      failed.  */
  3064 
  3065   if (!src_info.stride || !dest_info.stride)
  3066     return;
  3067 
  3068   /* If formats differ, abort.  */
  3069   eassert (src_info.format == dest_info.format
  3070            && src_info.format == ANDROID_BITMAP_FORMAT_RGBA_8888);
  3071 
  3072   /* Lock the image data.  */
  3073   src_data = NULL;
  3074   AndroidBitmap_lockPixels (env, src, &src_data);
  3075 
  3076   if (!src_data)
  3077     return;
  3078 
  3079   dest_data = NULL;
  3080   AndroidBitmap_lockPixels (env, dest, &dest_data);
  3081 
  3082   if (!dest_data)
  3083     goto fail1;
  3084 
  3085   /* Now clip the rectangle to the bounds of the source and
  3086      destination bitmap.  */
  3087 
  3088   x1 = MAX (x1, 0);
  3089   y1 = MAX (y1, 0);
  3090   x2 = MAX (x2, 0);
  3091   y2 = MAX (y2, 0);
  3092 
  3093 
  3094   if (x1 >= src_info.width
  3095       || x1 >= dest_info.width)
  3096     x1 = MIN (dest_info.width - 1, src_info.width - 1);
  3097 
  3098   if (x2 > src_info.width
  3099       || x2 > dest_info.width)
  3100     x2 = MIN (src_info.width, dest_info.width);
  3101 
  3102   if (y1 >= src_info.height
  3103       || y1 >= dest_info.height)
  3104     y1 = MIN (dest_info.height - 1, src_info.height - 1);
  3105 
  3106   if (y2 > src_info.height
  3107       || y2 > dest_info.height)
  3108     y2 = MIN (src_info.height, dest_info.height);
  3109 
  3110   if (x1 >= x2 || y1 >= y2)
  3111     goto fail2;
  3112 
  3113   /* Determine the address of the first line to copy.  */
  3114 
  3115   src_data_1 = src_data;
  3116   dest_data_1 = dest_data;
  3117   src_data_1 += x1 * 4;
  3118   src_data_1 += y1 * src_info.stride;
  3119   dest_data_1 += x1 * 4;
  3120   dest_data_1 += y1 * dest_info.stride;
  3121 
  3122   /* Start copying each line.  */
  3123 
  3124   while (y1 != y2)
  3125     {
  3126       memcpy (dest_data_1, src_data_1, (x2 - x1) * 4);
  3127       src_data_1 += src_info.stride;
  3128       dest_data_1 += dest_info.stride;
  3129       y1++;
  3130     }
  3131 
  3132   /* Complete the copy and unlock the bitmap.  */
  3133 
  3134  fail2:
  3135   AndroidBitmap_unlockPixels (env, dest);
  3136  fail1:
  3137   AndroidBitmap_unlockPixels (env, src);
  3138 }

  4234 /* Copy N pixels from SRC to DST, using MASK as a depth 1 clip
  4235    mask.  */
  4236 
  4237 static void
  4238 android_neon_mask_line (unsigned int *src, unsigned int *dst,
  4239                         unsigned char *mask, int n)
  4240 {
  4241   uint32x4_t src_low, src_high, dst_low, dst_high;
  4242   int16x8_t vmask;
  4243   int32x4_t ext_mask_low, ext_mask_high, low, high;
  4244   int rem, i;
  4245 
  4246   /* Calculate the remainder.  */
  4247   rem = n & 7, n &= ~7;
  4248 
  4249   /* Process eight pixels at a time.  */
  4250 
  4251   if (n)
  4252     {
  4253     again:
  4254       /* Load the low and high four pixels from the source.  */
  4255       src_low = vld1q_u32 (src);
  4256       src_high = vld1q_u32 (src + 4);
  4257 
  4258       /* Do the same with the destination.  */
  4259       dst_low = vld1q_u32 (dst);
  4260       dst_high = vld1q_u32 (dst + 4);
  4261 
  4262       /* Load and sign extend the mask.  */
  4263       vmask = vmovl_s8 (vld1_u8 (mask));
  4264       ext_mask_low = vmovl_s16 (vget_low_s16 (vmask));
  4265       ext_mask_high = vmovl_s16 (vget_high_s16 (vmask));
  4266 
  4267       /* Reinterpret the mask.  */
  4268       low = vreinterpretq_u32_s32 (ext_mask_low);
  4269       high = vreinterpretq_u32_s32 (ext_mask_high);
  4270 
  4271       /* Apply the mask.  */
  4272       dst_low = vbicq_u32 (dst_low, low);
  4273       src_low = vandq_u32 (src_low, low);
  4274       dst_high = vbicq_u32 (dst_high, high);
  4275       src_high = vandq_u32 (src_high, high);
  4276 
  4277       /* Write the result after combining both masked vectors.  */
  4278       vst1q_u32 (dst, vorrq_u32 (dst_low, src_low));
  4279       vst1q_u32 (dst + 4, vorrq_u32 (dst_high, src_high));
  4280 
  4281       /* Adjust src, dst and mask.  */
  4282       dst += 8;
  4283       src += 8;
  4284       mask += 8;
  4285 
  4286       /* See if this loop should continue.  */
  4287       n -= 8;
  4288       if (n > 0)
  4289         goto again;
  4290     }
  4291 
  4292   /* Process the remaining pixels.  */
  4293 
  4294   for (i = 0; i < rem; ++i)
  4295     {
  4296       /* Sign extend the mask.  */
  4297       n = ((signed char *) mask)[i];
  4298 
  4299       /* Combine src and dst.  */
  4300       dst[i] = ((src[i] & n) | (dst[i] & ~n));
  4301     }
  4302 }

  5303 void
  5304 android_put_pixel (struct android_image *ximg, int x, int y,
  5305                    unsigned long pixel)
  5306 {
  5307   char *byte, *word;
  5308   unsigned int r, g, b;
  5309   unsigned int pixel_int;
  5310 
  5311   /* Ignore out-of-bounds accesses.  */
  5312 
  5313   if (x >= ximg->width || y >= ximg->height || x < 0 || y < 0)
  5314     return;
  5315 
  5316   switch (ximg->depth)
  5317     {
  5318     case 1:
  5319       byte = ximg->data + y * ximg->bytes_per_line + x / 8;
  5320 
  5321       if (pixel)
  5322         *byte |= (1 << x % 8);
  5323       else
  5324         *byte &= ~(1 << x % 8);
  5325       break;
  5326 
  5327     case 24:
  5328       /* Unaligned accesses are problematic on Android devices.  */
  5329       word = ximg->data + y * ximg->bytes_per_line + x * 4;
  5330 
  5331       /* Swizzle the pixel into ABGR format.  Android uses Skia's
  5332          ``native color type'', which is ABGR.  This is despite the
  5333          format being named ``ARGB'', and more confusingly
  5334          `ANDROID_BITMAP_FORMAT_RGBA_8888' in bitmap.h.  */
  5335       r = pixel & 0x00ff0000;
  5336       g = pixel & 0x0000ff00;
  5337       b = pixel & 0x000000ff;
  5338       pixel = (r >> 16) | g | (b << 16) | 0xff000000;
  5339 
  5340       pixel_int = pixel;
  5341       memcpy (word, &pixel_int, sizeof pixel_int);
  5342       break;
  5343     }
  5344 }

  5369       r = pixel & 0x000000ff;

  5371 
  5372       return pixel;
  5373     }
  5374 
  5375   emacs_abort ();

  5378 struct android_image *
  5379 android_get_image (android_drawable handle,
  5380                    enum android_image_format format)
  5381 {
  5382   jobject drawable, bitmap;
  5383   AndroidBitmapInfo bitmap_info;
  5384   size_t byte_size;
  5385   void *data;
  5386   struct android_image *image;
  5387   unsigned char *data1, *data2;
  5388   int i, x;
  5389 
  5390   drawable = android_resolve_handle2 (handle, ANDROID_HANDLE_WINDOW,
  5391                                       ANDROID_HANDLE_PIXMAP);
  5392 
  5393   /* Look up the drawable and get the bitmap corresponding to it.
  5394      Then, lock the bitmap's bits.  */
  5395   bitmap = (*android_java_env)->CallObjectMethod (android_java_env,
  5396                                                   drawable,
  5397                                                   drawable_class.get_bitmap);
  5398   android_exception_check ();
  5399 
  5400   /* Clear the bitmap info structure.  */
  5401   memset (&bitmap_info, 0, sizeof bitmap_info);
  5402 
  5403   /* The NDK doc seems to imply this function can fail but doesn't say
  5404      what value it gives when it does! */
  5405   AndroidBitmap_getInfo (android_java_env, bitmap, &bitmap_info);
  5406 
  5407   if (!bitmap_info.stride)
  5408     {
  5409       ANDROID_DELETE_LOCAL_REF (bitmap);
  5410       memory_full (0);
  5411     }
  5412 
  5413   /* Compute how big the image data will be.  Fail if it would be too
  5414      big.  */
  5415 
  5416   if (bitmap_info.format != ANDROID_BITMAP_FORMAT_A_8)
  5417     {
  5418       if (INT_MULTIPLY_WRAPV ((size_t) bitmap_info.stride,
  5419                               (size_t) bitmap_info.height,
  5420                               &byte_size))
  5421         {
  5422           ANDROID_DELETE_LOCAL_REF (bitmap);
  5423           memory_full (0);
  5424         }
  5425     }
  5426   else
  5427     /* This A8 image will be packed into A1 later on.  */
  5428     byte_size = (bitmap_info.width + 7) / 8;
  5429 
  5430   /* Lock the image data.  Once again, the NDK documentation says the
  5431      call can fail, but does not say how to determine whether or not
  5432      it has failed, nor how the address is aligned.  */
  5433   data = NULL;
  5434   AndroidBitmap_lockPixels (android_java_env, bitmap, &data);
  5435 
  5436   if (!data)
  5437     {
  5438       /* Take a NULL pointer to mean that AndroidBitmap_lockPixels
  5439          failed.  */
  5440       ANDROID_DELETE_LOCAL_REF (bitmap);
  5441       memory_full (0);
  5442     }
  5443 
  5444   /* Copy the data into a new struct android_image.  */
  5445   image = xmalloc (sizeof *image);
  5446   image->width = bitmap_info.width;
  5447   image->height = bitmap_info.height;
  5448   image->data = malloc (byte_size);
  5449 
  5450   if (!image->data)
  5451     {
  5452       ANDROID_DELETE_LOCAL_REF (bitmap);
  5453       xfree (image);
  5454       memory_full (byte_size);
  5455     }
  5456 
  5457   /* Use the format of the bitmap to determine the image depth.  */
  5458   switch (bitmap_info.format)
  5459     {
  5460     case ANDROID_BITMAP_FORMAT_RGBA_8888:
  5461       image->depth = 24;
  5462       image->bits_per_pixel = 32;
  5463       break;
  5464 
  5465       /* A8 images are used by Emacs to represent bitmaps.  They have
  5466          to be packed manually.  */
  5467     case ANDROID_BITMAP_FORMAT_A_8:
  5468       image->depth = 1;
  5469       image->bits_per_pixel = 1;
  5470       break;
  5471 
  5472       /* Other formats are currently not supported.  */
  5473     default:
  5474       emacs_abort ();
  5475     }
  5476 
  5477   image->format = format;
  5478 
  5479   if (image->depth == 24)
  5480     {
  5481       image->bytes_per_line = bitmap_info.stride;
  5482 
  5483       /* Copy the bitmap data over.  */
  5484       memcpy (image->data, data, byte_size);
  5485     }
  5486   else
  5487     {
  5488       /* Pack the A8 image data into bits manually.  */
  5489       image->bytes_per_line = (image->width + 7) / 8;
  5490 
  5491       data1 = (unsigned char *) image->data;
  5492       data2 = data;
  5493 
  5494       for (i = 0; i < image->height; ++i)
  5495         {
  5496           for (x = 0; x < image->width; ++x)
  5497             /* Some bits in data1 might be initialized at this point,
  5498                but they will all be set properly later.  */
  5499             data1[x / 8] = (data2[x]
  5500                             ? (data1[x / 8] | (1 << (x % 8)))
  5501                             : (data1[x / 8] & ~(1 << (x % 8))));
  5502 
  5503           data1 += image->bytes_per_line;
  5504           data2 += bitmap_info.stride;
  5505         }
  5506     }
  5507 
  5508   /* Unlock the bitmap pixels.  */
  5509   AndroidBitmap_unlockPixels (android_java_env, bitmap);
  5510 
  5511   /* Delete the bitmap reference.  */
  5512   ANDROID_DELETE_LOCAL_REF (bitmap);
  5513   return image;
  5514 }

  6371 struct dirent *
  6372 android_readdir (struct android_dir *dir)

  6375   const char *last;
  6376 
  6377   if (dir->asset_dir)
  6378     {
  6379       /* There are no more files to read.  */
  6380       if (dir->asset_dir >= dir->asset_limit)
  6381         return NULL;
  6382 
  6383       /* Otherwise, scan forward looking for the next NULL byte.  */
  6384       last = memchr (dir->asset_dir, 0,

  6392       last++;

  6399       memset (&dirent, 0, sizeof dirent);

  6401       dirent.d_off = 0;

  6417       dirent.d_name[sizeof dirent.d_name - 1] = '\0';

  6795 /* Return the interpolation of the four pixels TL, TR, BL, and BR,
  6796    according to the weights DISTX and DISTY.  */
  6797 
  6798 static unsigned int
  6799 android_four_corners_bilinear (unsigned int tl, unsigned int tr,
  6800                                unsigned int bl, unsigned int br,
  6801                                int distx, int disty)
  6802 {
  6803   int distxy, distxiy, distixy, distixiy;
  6804   uint32_t f, r;
  6805 
  6806   distxy = distx * disty;
  6807   distxiy = (distx << 8) - distxy;
  6808   distixy = (disty << 8) - distxy;
  6809   distixiy = (256 * 256 - (disty << 8)
  6810               - (distx << 8) + distxy);
  6811 
  6812   /* Red */
  6813   r = ((tl & 0x000000ff) * distixiy + (tr & 0x000000ff) * distxiy
  6814        + (bl & 0x000000ff) * distixy  + (br & 0x000000ff) * distxy);
  6815 
  6816   /* Green */
  6817   f = ((tl & 0x0000ff00) * distixiy + (tr & 0x0000ff00) * distxiy
  6818        + (bl & 0x0000ff00) * distixy  + (br & 0x0000ff00) * distxy);
  6819   r |= f & 0xff000000;
  6820 
  6821   /* Now do the upper two components.  */
  6822   tl >>= 16;
  6823   tr >>= 16;
  6824   bl >>= 16;
  6825   br >>= 16;
  6826   r >>= 16;
  6827 
  6828   /* Blue */
  6829   f = ((tl & 0x000000ff) * distixiy + (tr & 0x000000ff) * distxiy
  6830        + (bl & 0x000000ff) * distixy  + (br & 0x000000ff) * distxy);
  6831   r |= f & 0x00ff0000;
  6832 
  6833   /* Alpha */
  6834   f = ((tl & 0x0000ff00) * distixiy + (tr & 0x0000ff00) * distxiy
  6835        + (bl & 0x0000ff00) * distixy  + (br & 0x0000ff00) * distxy);
  6836   r |= f & 0xff000000;
  6837 
  6838   return r;
  6839 }

  6841 /* Return the interpolation of the four pixels closest to at X, Y in
  6842    IMAGE, according to weights in both axes computed from X and Y.
  6843    IMAGE must be depth 24, or the behavior is undefined.  */
  6844 
  6845 static unsigned int
  6846 android_fetch_pixel_bilinear (struct android_image *image,
  6847                               float x, float y)
  6848 {
  6849   int x1, y1, x2, y2;
  6850   float distx, disty;
  6851   unsigned int top_left, top_right;
  6852   unsigned int bottom_left, bottom_right;
  6853   char *word;
  6854 
  6855   /* Compute the four closest corners to X and Y.  */
  6856   x1 = (int) x;
  6857   x2 = x1 + 1;
  6858   y1 = (int) y;
  6859   y2 = y1 + 1;
  6860 
  6861   /* Make sure all four corners are within range.  */
  6862   x1 = MAX (0, MIN (image->width - 1, x1));
  6863   y1 = MAX (0, MIN (image->height - 1, y1));
  6864   x2 = MAX (0, MIN (image->width - 1, x2));
  6865   y2 = MAX (0, MIN (image->height - 1, y2));
  6866 
  6867   /* Compute the X and Y biases.  These are numbers between 0f and
  6868      1f.  */
  6869   distx = x - x1;
  6870   disty = y - y1;
  6871 
  6872   /* Fetch the four closest pixels.  */
  6873   word = image->data + y1 * image->bytes_per_line + x1 * 4;
  6874   memcpy (&top_left, word, sizeof top_left);
  6875   word = image->data + y1 * image->bytes_per_line + x2 * 4;
  6876   memcpy (&top_right, word, sizeof top_right);
  6877   word = image->data + y2 * image->bytes_per_line + x1 * 4;

  6889    matrix MATRIX utilizing a bilinear filter.  Place the result in
  6890    OUT.  The matrix maps from the coordinate space of OUT to
  6891    IMAGE.  */
  6892 
  6893 void
  6894 android_project_image_bilinear (struct android_image *image,
  6895                                 struct android_image *out,
  6896                                 struct android_transform *transform)

  7167 /* The thread from which a query against a thread is currently being
  7168    made, if any.  Value is 0 if no query is in progress, 1 if a query
  7169    is being made from the UI thread to the main thread, and 2 if a
  7170    query is being made the other way around.  */
  7171 static char android_servicing_query;
  7172 
  7173 /* Function that is waiting to be run in the Emacs thread.  */
  7174 static void (*android_query_function) (void *);
  7175 
  7176 /* Context for that function.  */
  7177 static void *android_query_context;
  7178 
  7179 /* Deadlock protection.  The UI thread and the Emacs thread must
  7180    sometimes make synchronous queries to each other, which are
  7181    normally answered inside each thread's respective event loop.
  7182    Deadlocks can happen when both threads simultaneously make such
  7183    synchronous queries and block waiting for each others responses.
  7184 
  7185    The Emacs thread can be interrupted to service any queries made by
  7186    the UI thread, but is not possible the other way around.
  7187 
  7188    To avoid such deadlocks, an atomic counter is provided.  This
  7189    counter is set to two every time a query starts from the main
  7190    thread, and is set to zero every time one ends.  If the UI thread
  7191    tries to make a query and sees that the counter is two, it simply
  7192    returns so that its event loop can proceed to perform and respond
  7193    to the query.  If the Emacs thread sees that the counter is one,
  7194    then it stops to service all queries being made by the input
  7195    method, then proceeds to make its query with the counter set to
  7196    2.
  7197 
  7198    The memory synchronization is simple: all writes to
  7199    `android_query_context' and `android_query_function' are depended
  7200    on by writes to the atomic counter.  Loads of the new value from
  7201    the counter are then guaranteed to make those writes visible.  The
  7202    separate flag `android_urgent_query' does not depend on anything
  7203    itself; however, the input signal handler executes a memory fence
  7204    to ensure that all query related writes become visible.  */
  7205 
  7206 /* Run any function that the UI thread has asked to run, and then
  7207    signal its completion.  */
  7208 
  7209 static void
  7210 android_check_query (void)
  7211 {
  7212   void (*proc) (void *);
  7213   void *closure;
  7214 
  7215   if (!__atomic_load_n (&android_servicing_query, __ATOMIC_ACQUIRE))
  7216     return;
  7217 
  7218   /* First, load the procedure and closure.  */
  7219   closure = android_query_context;
  7220   proc = android_query_function;
  7221 
  7222   if (!proc)
  7223     return;
  7224 
  7225   proc (closure);
  7226 
  7227   /* Finish the query.  */
  7228   android_query_context = NULL;
  7229   android_query_function = NULL;
  7230   __atomic_store_n (&android_servicing_query, 0, __ATOMIC_RELEASE);
  7231   __atomic_clear (&android_urgent_query, __ATOMIC_RELEASE);
  7232 
  7233   /* Signal completion.  */
  7234   sem_post (&android_query_sem);
  7235 }

  7366 /* Synchronously ask the Emacs thread to run the specified PROC with
  7367    the given CLOSURE.  Return if this fails, or once PROC is run.
  7368 
  7369    PROC may be run from inside maybe_quit.
  7370 
  7371    It is not okay to run Lisp code which signals or performs non
  7372    trivial tasks inside PROC.
  7373 
  7374    Return 1 if the Emacs thread is currently waiting for the UI thread
  7375    to respond and PROC could not be run, or 0 otherwise.  */
  7376 
  7377 int
  7378 android_run_in_emacs_thread (void (*proc) (void *), void *closure)
  7379 {
  7380   union android_event event;
  7381   char old;
  7382   int rc;
  7383   struct timespec timeout;
  7384 
  7385   event.xaction.type = ANDROID_WINDOW_ACTION;
  7386   event.xaction.serial = ++event_serial;
  7387   event.xaction.window = 0;
  7388   event.xaction.action = 0;
  7389 
  7390   /* Set android_query_function and android_query_context.  */
  7391   android_query_context = closure;
  7392   android_query_function = proc;
  7393 
  7394   /* Don't allow deadlocks to happen; make sure the Emacs thread is
  7395      not waiting for something to be done (in that case,
  7396      `android_query_context' is 2.)  */
  7397 
  7398   old = 0;
  7399   if (!__atomic_compare_exchange_n (&android_servicing_query, &old,
  7400                                     1, false, __ATOMIC_ACQ_REL,
  7401                                     __ATOMIC_ACQUIRE))
  7402     {
  7403       android_query_context = NULL;
  7404       android_query_function = NULL;
  7405 
  7406       /* The two variables above may still be non-NULL from the POV of
  7407          the main thread, as no happens-before constraint is placed on
  7408          those stores wrt a future load from `android_servicing_query'.  */
  7409 
  7410       return 1;
  7411     }
  7412 
  7413   /* Send a dummy event.  `android_check_query' will be called inside
  7414      wait_reading_process_output after the event arrives.
  7415 
  7416      Otherwise, android_select will call android_check_thread the next
  7417      time it is entered.  */
  7418   android_write_event (&event);
  7419 
  7420   /* Start waiting for the function to be executed.  First, wait two
  7421      seconds for the query to execute normally.  */
  7422 
  7423   timeout.tv_sec = 2;
  7424   timeout.tv_nsec = 0;
  7425   timeout = timespec_add (current_timespec (), timeout);
  7426 
  7427   /* See if an urgent query was recently answered without entering the
  7428      keyboard loop in between.  When that happens, raise SIGIO to
  7429      continue processing queries as soon as possible.  */
  7430 
  7431   if (__atomic_load_n (&android_urgent_query, __ATOMIC_ACQUIRE))
  7432     kill (getpid (), SIGIO);
  7433 
  7434  again:
  7435   rc = sem_timedwait (&android_query_sem, &timeout);
  7436 
  7437   if (rc < 0)
  7438     {
  7439       if (errno == EINTR)
  7440         goto again;
  7441 
  7442       eassert (errno == ETIMEDOUT);
  7443 
  7444       __android_log_print (ANDROID_LOG_VERBOSE, __func__,
  7445                            "Timed out waiting for response"
  7446                            " from main thread...");
  7447 
  7448       /* The query timed out.  At this point, set
  7449          `android_urgent_query' to true.  */
  7450       __atomic_store_n (&android_urgent_query, true,
  7451                         __ATOMIC_RELEASE);
  7452 
  7453       /* And raise SIGIO.  Now that the query is considered urgent,
  7454          the main thread will reply while reading async input.
  7455 
  7456          Normally, the main thread waits for the keyboard loop to be
  7457          entered before responding, in order to avoid responding with
  7458          inaccurate results taken during command executioon.  */
  7459       kill (getpid (), SIGIO);
  7460 
  7461       /* Wait for the query to complete.  `android_urgent_query' is
  7462          only cleared by either `android_select' or
  7463          `android_check_query', so there's no need to worry about the
  7464          flag being cleared before the query is processed.  */
  7465       while (sem_wait (&android_query_sem) < 0)
  7466         ;;
  7467     }
  7468 
  7469   /* At this point, `android_servicing_query' should either be zero if
  7470      the query was answered or two if the main thread has started a
  7471      query.  */
  7472 
  7473   eassert (!__atomic_load_n (&android_servicing_query,
  7474                              __ATOMIC_ACQUIRE)
  7475            || (__atomic_load_n (&android_servicing_query,
  7476                                 __ATOMIC_ACQUIRE) == 2));
  7477 
  7478   return 0;
  7479 }

  7642 /* External asset management interface.  By using functions here
  7643    to read and write from files, Emacs can avoid opening a
  7644    shared memory file descriptor for each ``asset'' file.  */
  7645 
  7646 /* Like android_open.  However, return a structure that can
  7647    either directly hold an AAsset or a file descriptor.
  7648 
  7649    Value is the structure upon success.  Upon failure, value
  7650    consists of an uninitialized file descriptor, but its asset
  7651    field is set to -1, and errno is set accordingly.  */
  7652 
  7653 struct android_fd_or_asset
  7654 android_open_asset (const char *filename, int oflag, mode_t mode)
  7655 {
  7656   const char *name;
  7657   struct android_fd_or_asset fd;
  7658   AAsset *asset;
  7659 
  7660   /* Initialize FD by setting its asset to an invalid
  7661      pointer.  */
  7662   fd.asset = (void *) -1;
  7663 
  7664   /* See if this is an asset.  */
  7665 
  7666   if (asset_manager && (name = android_get_asset_name (filename)))
  7667     {
  7668       /* Return failure for unsupported flags.  */
  7669 
  7670       if (oflag & O_WRONLY || oflag & O_RDWR)
  7671         {
  7672           errno = EROFS;
  7673           return fd;
  7674         }
  7675 
  7676       if (oflag & O_DIRECTORY)
  7677         {
  7678           errno = ENOTSUP;
  7679           return fd;
  7680         }
  7681 
  7682       /* Now try to open the asset.  */
  7683       asset = AAssetManager_open (asset_manager, name,
  7684                                   AASSET_MODE_STREAMING);
  7685 
  7686       if (!asset)
  7687         {
  7688           errno = ENOENT;
  7689           return fd;
  7690         }
  7691 
  7692       /* Return the asset.  */
  7693       fd.asset = asset;
  7694       return fd;
  7695     }
  7696 
  7697   /* If the file is not an asset, fall back to android_open and
  7698      get a regular file descriptor.  */
  7699 
  7700   fd.fd = android_open (filename, oflag, mode);
  7701   if (fd.fd < 0)
  7702     return fd;
  7703 
  7704   /* Set fd.asset to NULL, signifying that it is a file
  7705      descriptor.  */
  7706   fd.asset = NULL;
  7707   return fd;
  7708 }

  7865 /* Process execution.
  7866 
  7867    Newer Android systems use SELinux to restrict user programs from
  7868    executing programs installed in the application data directory for
  7869    security reasons.  Emacs uses a `loader' binary installed in the
  7870    application data directory to manually load executables and replace
  7871    the `execve' system call.  */
  7872 
  7873 enum
  7874   {
  7875     /* Maximum number of arguments available.  */
  7876     MAXARGS = 1024,
  7877   };
  7878 
  7879 /* Rewrite the command line given in *ARGV to utilize the `exec1'
  7880    bootstrap binary if necessary.
  7881 
  7882    Value is 0 upon success, else 1.  Set errno upon failure.
  7883 
  7884    ARGV holds a pointer to a NULL-terminated array of arguments given
  7885    to `emacs_spawn'.  */
  7886 
  7887 int
  7888 android_rewrite_spawn_argv (const char ***argv)
  7889 {
  7890   static const char *new_args[MAXARGS];
  7891   static char exec1_name[PATH_MAX], loader_name[PATH_MAX];
  7892   size_t i, nargs;
  7893 
  7894   /* This isn't required on Android 9 or earlier.  */
  7895 
  7896   if (android_api_level < 29 || !android_use_exec_loader)
  7897     return 0;
  7898 
  7899   /* Get argv[0]; this should never be NULL.
  7900      Then, verify that it exists and is executable.  */
  7901 
  7902   eassert (**argv);
  7903   if (access (**argv, R_OK | X_OK))
  7904     return 1;
  7905 
  7906   /* Count the number of arguments in *argv.  */
  7907 
  7908   nargs = 0;
  7909   while ((*argv)[nargs])
  7910     ++nargs;
  7911 
  7912   /* nargs now holds the number of arguments in argv.  If it's larger
  7913      than MAXARGS, return failure.  */
  7914 
  7915   if (nargs + 2 > MAXARGS)
  7916     {
  7917       errno = E2BIG;
  7918       return 1;
  7919     }
  7920 
  7921   /* Fill in the name of `libexec1.so'.  */
  7922   snprintf (exec1_name, PATH_MAX, "%s/libexec1.so",
  7923             android_lib_dir);
  7924 
  7925   /* And libloader.so.  */
  7926   snprintf (loader_name, PATH_MAX, "%s/libloader.so",
  7927             android_lib_dir);
  7928 
  7929   /* Now fill in the first two arguments.  */
  7930   new_args[0] = exec1_name;
  7931   new_args[1] = loader_name;
  7932 
  7933   /* And insert the rest, including the trailing NULL.  */
  7934   for (i = 0; i < nargs + 1; ++i)
  7935     new_args[i + 2] = (*argv)[i];
  7936 
  7937   /* Replace argv.  */
  7938   *argv = new_args;
  7939 
  7940   /* Return success.  */
  7941   return 0;
  7942 }
