    61   private static final class Item implements MenuItem.OnMenuItemClickListener
    62   {
    63     public int itemID;
    64     public String itemName, tooltip;
    65     public EmacsContextMenu subMenu;
    66     public boolean isEnabled, isCheckable, isChecked;
    67     public EmacsView inflatedView;
    68     public boolean isRadio;
    69 
    70     @Override
    71     public boolean
    72     onMenuItemClick (MenuItem item)
    73     {
    74       Log.d (TAG, "onMenuItemClick: " + itemName + " (" + itemID + ")");
    75 
    76       if (subMenu != null)
    77         {
    78           /* Android 6.0 and earlier don't support nested submenus
    79              properly, so display the submenu popup by hand.  */
    80 
    81           if (Build.VERSION.SDK_INT < Build.VERSION_CODES.N)
    82             {
    83               Log.d (TAG, "onMenuItemClick: displaying submenu " + subMenu);
    84 
    85               /* Still set wasSubmenuSelected -- if not set, the
    86                  dismissal of this context menu will result in a
    87                  context menu event being sent.  */
    88               wasSubmenuSelected = -2;
    89 
    90               /* Running a popup menu from inside a click handler
    91                  doesn't work, so make sure it is displayed
    92                  outside.  */
    93 
    94               inflatedView.post (new Runnable () {
    95                   @Override
    96                   public void
    97                   run ()
    98                   {
    99                     inflatedView.popupMenu (subMenu, 0, 0, true);
   100                   }
   101                 });
   102 
   103               return true;
   104             }
   105 
   106           /* After opening a submenu within a submenu, Android will
   107              send onContextMenuClosed for a ContextMenuBuilder.  This
   108              will normally confuse Emacs into thinking that the
   109              context menu has been dismissed.  Wrong!
   110 
   111              Setting this flag makes EmacsActivity to only handle
   112              SubMenuBuilder being closed, which always means the menu
   113              has actually been dismissed.
   114 
   115              However, these extraneous events aren't sent on devices
   116              where submenus display without dismissing their parents.
   117              Thus, only ignore the close event if it happens within
   118              300 milliseconds of the submenu being selected.  */
   119           wasSubmenuSelected = System.currentTimeMillis ();
   120           return false;
   121         }
   122 
   123       /* Send a context menu event.  */
   124       EmacsNative.sendContextMenu ((short) 0, itemID,
   125                                    lastMenuEventSerial);
   126 
   127       /* Say that an item has already been selected.  */
   128       itemAlreadySelected = true;
   129       return true;
   130     }
   131   };

   214   /* Add the contents of this menu to MENU.  Assume MENU will be
   215      displayed in INFLATEDVIEW.  */

   217   private void
   218   inflateMenuItems (Menu menu, EmacsView inflatedView)
   219   {
   220     Intent intent;
   221     MenuItem menuItem;
   222     SubMenu submenu;
   223 
   224     for (Item item : menuItems)
   225       {
   226         if (item.subMenu != null)
   227           {
   228             /* This is a submenu.  On versions of Android which
   229                support doing so, create the submenu and add the
   230                contents of the menu to it.
   231 
   232                Note that Android 4.0 and later technically supports
   233                having multiple layers of nested submenus, but if they
   234                are used, onContextMenuClosed becomes unreliable.  */
   235 
   236             if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.N)
   237               {
   238                 submenu = menu.addSubMenu (item.itemName);
   239                 item.subMenu.inflateMenuItems (submenu, inflatedView);
   240 
   241                 /* This is still needed to set wasSubmenuSelected.  */
   242                 menuItem = submenu.getItem ();
   243               }
   244             else
   245               menuItem = menu.add (item.itemName);
   246 
   247             item.inflatedView = inflatedView;
   248             menuItem.setOnMenuItemClickListener (item);
   249           }
   250         else
   251           {
   252             if (item.isRadio)
   253               menuItem = menu.add (++lastGroupId, Menu.NONE, Menu.NONE,
   254                                    item.itemName);
   255             else
   256               menuItem = menu.add (item.itemName);
   257             menuItem.setOnMenuItemClickListener (item);
   258 
   259             /* If the item ID is zero, then disable the item.  */
   260             if (item.itemID == 0 || !item.isEnabled)
   261               menuItem.setEnabled (false);
   262 
   263             /* Now make the menu item display a checkmark as
   264                appropriate.  */
   265 
   266             if (item.isCheckable)
   267               menuItem.setCheckable (true);
   268 
   269             if (item.isChecked)
   270               menuItem.setChecked (true);
   271 
   272             /* Define an exclusively checkable group if the item is a
   273                radio button.  */
   274 
   275             if (item.isRadio)
   276               menu.setGroupCheckable (lastGroupId, true, true);
   277 
   278             /* If the tooltip text is set and the system is new enough
   279                to support menu item tooltips, set it on the item.  */
   280 
   281             if (item.tooltip != null
   282                 && Build.VERSION.SDK_INT >= Build.VERSION_CODES.O)
   283               menuItem.setTooltipText (item.tooltip);
   284           }
   285       }
   286   }

   306   /* Like display, but does the actual work and runs in the main

   308 
   309   private boolean
   310   display1 (EmacsWindow window, int xPosition, int yPosition)
   311   {
   312     /* Set this flag to false.  It is used to decide whether or not to
   313        send 0 in response to the context menu being closed.  */
   314     itemAlreadySelected = false;
   315 
   316     /* No submenu has been selected yet.  */
   317     wasSubmenuSelected = -1;
   318 
   319     return window.view.popupMenu (this, xPosition, yPosition,
   320                                   false);
   321   }

   327   public boolean
   328   display (final EmacsWindow window, final int xPosition,
   329            final int yPosition, final int serial)
   330   {
   331     Runnable runnable;
   332     final EmacsHolder<Boolean> rc;
   333 
   334     rc = new EmacsHolder<Boolean> ();
   335     rc.thing = false;
   336 
   337     runnable = new Runnable () {
   338         @Override
   339         public void
   340         run ()
   341         {
   342           synchronized (this)
   343             {
   344               lastMenuEventSerial = serial;
   345               rc.thing = display1 (window, xPosition, yPosition);
   346               notify ();
   347             }
   348         }
   349       };
   350 
   351     EmacsService.syncRunnable (runnable);
   352     return rc.thing;
   353   }
