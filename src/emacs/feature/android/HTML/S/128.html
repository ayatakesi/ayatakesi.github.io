    33 /* This originally extended SurfaceView.  However, doing so proved to
    34    be too slow, and Android's surface view keeps up to three of its
    35    own back buffers, which use too much memory (up to 96 MB for a
    36    single frame.) */
    37 
    38 public final class EmacsSurfaceView extends View
    39 {
    40   private static final String TAG = "EmacsSurfaceView";
    41 
    42   /* The complete buffer contents at the time of the last draw.  */
    43   private Bitmap frontBuffer;
    44 
    45   /* Canvas representing the front buffer.  */
    46   private Canvas bitmapCanvas;
    47 
    48   /* Reference to the last bitmap copied to the front buffer.  */
    49   private WeakReference<Bitmap> bitmap;
    50 
    51   /* Paint objects used on the main and UI threads, respectively.  */
    52   private static final Paint bitmapPaint, uiThreadPaint;
    53 
    54   static
    55   {
    56     /* Create two different Paint objects; one is used on the main
    57        thread for buffer swaps, while the other is used from the UI
    58        thread in `onDraw'.  This is necessary because Paint objects
    59        are not thread-safe, even if their uses are interlocked.  */
    60 
    61     bitmapPaint = new Paint ();
    62     uiThreadPaint = new Paint ();
    63   };

    65   public
    66   EmacsSurfaceView (EmacsView view)
    67   {
    68     super (view.getContext ());
    69 
    70     this.bitmap = new WeakReference<Bitmap> (null);
    71   }

    73   private void
    74   copyToFrontBuffer (Bitmap bitmap, Rect damageRect)
    75   {
    76     EmacsService.checkEmacsThread ();
    77 
    78     if (Build.VERSION.SDK_INT != Build.VERSION_CODES.O
    79         && Build.VERSION.SDK_INT != Build.VERSION_CODES.O_MR1
    80         && Build.VERSION.SDK_INT != Build.VERSION_CODES.N_MR1
    81         && Build.VERSION.SDK_INT != Build.VERSION_CODES.N)
    82       {
    83         /* If `drawBitmap' can safely be used while a bitmap is locked
    84            by another thread, continue here... */
    85 
    86         if (damageRect != null)
    87           bitmapCanvas.drawBitmap (bitmap, damageRect, damageRect,
    88                                    bitmapPaint);
    89         else
    90           bitmapCanvas.drawBitmap (bitmap, 0f, 0f, bitmapPaint);
    91       }
    92     else
    93       {
    94         /* But if it can not, as on Android 7.0 through 8.1, then use
    95            a replacement function.  */
    96 
    97         if (damageRect != null)
    98           EmacsNative.blitRect (bitmap, frontBuffer,
    99                                 damageRect.left,
   100                                 damageRect.top,
   101                                 damageRect.right,
   102                                 damageRect.bottom);
   103         else
   104           EmacsNative.blitRect (bitmap, frontBuffer, 0, 0,
   105                                 bitmap.getWidth (),
   106                                 bitmap.getHeight ());
   107       }
   108   }
   109 
   110   private void
   111   reconfigureFrontBuffer (Bitmap bitmap)
   112   {
   113     /* First, remove the old front buffer.  */
   114 
   115     if (frontBuffer != null)
   116       {
   117         frontBuffer.recycle ();
   118         frontBuffer = null;
   119         bitmapCanvas = null;
   120       }
   121 
   122     this.bitmap = new WeakReference<Bitmap> (bitmap);
   123 
   124     /* Next, create the new front buffer if necessary.  */
   125 
   126     if (bitmap != null && frontBuffer == null)
   127       {
   128         if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O)
   129           frontBuffer = Bitmap.createBitmap (bitmap.getWidth (),
   130                                              bitmap.getHeight (),
   131                                              Bitmap.Config.ARGB_8888,
   132                                              false);
   133         else
   134           frontBuffer = Bitmap.createBitmap (bitmap.getWidth (),
   135                                              bitmap.getHeight (),
   136                                              Bitmap.Config.ARGB_8888);
   137 
   138         bitmapCanvas = new Canvas (frontBuffer);
   139 
   140         /* And copy over the bitmap contents.  */
   141         copyToFrontBuffer (bitmap, null);
   142       }
   143     else if (bitmap != null)
   144       /* Just copy over the bitmap contents.  */
   145       copyToFrontBuffer (bitmap, null);
   146   }

   148   public synchronized void
   149   setBitmap (Bitmap bitmap, Rect damageRect)
   150   {
   151     if (bitmap != this.bitmap.get ())
   152       reconfigureFrontBuffer (bitmap);
   153     else if (bitmap != null)
   154       copyToFrontBuffer (bitmap, damageRect);
   155 
   156     if (bitmap != null)
   157       {
   158         /* In newer versions of Android, the invalid rectangle is
   159            supposedly internally calculated by the system.  How that
   160            is done is unknown, but calling `invalidateRect' is now
   161            deprecated.
   162 
   163            Fortunately, nobody has deprecated the version of
   164            `postInvalidate' that accepts a dirty rectangle.  */
   165 
   166         if (damageRect != null)
   167           postInvalidate (damageRect.left, damageRect.top,
   168                           damageRect.right, damageRect.bottom);
   169         else
   170           postInvalidate ();
   171       }
   172   }
   173 
   174   @Override
   175   public synchronized void
   176   onDraw (Canvas canvas)
   177   {
   178     /* Paint the view's bitmap; the bitmap might be recycled right
   179        now.  */
   180 
   181     if (frontBuffer != null)
   182       canvas.drawBitmap (frontBuffer, 0f, 0f, uiThreadPaint);
   183   }
