   625 int
   626 gnutls_try_handshake (struct Lisp_Process *proc)
   627 {
   628   gnutls_session_t state = proc->gnutls_state;
   629   int ret;
   630   bool non_blocking = proc->is_non_blocking_client;
   631   /* Sleep for ten milliseconds when busy-looping in
   632      gnutls_handshake.  */
   633   struct timespec delay = { 0, 1000 * 1000 * 10 };
   634 
   635   if (proc->gnutls_complete_negotiation_p)
   636     non_blocking = false;
   637 
   638   if (non_blocking)
   639     proc->gnutls_p = true;
   640 
   641   while ((ret = gnutls_handshake (state)) < 0)
   642     {
   643       if (emacs_gnutls_handle_error (state, ret) == 0) /* fatal */
   644         break;
   645       maybe_quit ();
   646       if (non_blocking && ret != GNUTLS_E_INTERRUPTED)
   647         break;
   648       nanosleep (&delay, NULL);
   649     }
   650 
   651   proc->gnutls_initstage = GNUTLS_STAGE_HANDSHAKE_TRIED;
   652 
   653   if (ret == GNUTLS_E_SUCCESS)
   654     {
   655       /* Here we're finally done.  */
   656       proc->gnutls_initstage = GNUTLS_STAGE_READY;
   657     }
   658   else
   659     {
   660       /* check_memory_full (gnutls_alert_send_appropriate (state, ret));  */
   661     }
   662   return ret;
   663 }

   690 static int
   691 emacs_gnutls_handshake (struct Lisp_Process *proc)
   692 {
   693   gnutls_session_t state = proc->gnutls_state;
   694 
   695   if (proc->gnutls_initstage < GNUTLS_STAGE_HANDSHAKE_CANDO)
   696     return -1;
   697 
   698   if (proc->gnutls_initstage < GNUTLS_STAGE_TRANSPORT_POINTERS_SET)
   699     {
   700 # ifdef WINDOWSNT
   701       /* On W32 we cannot transfer socket handles between different runtime
   702          libraries, so we tell GnuTLS to use our special push/pull
   703          functions.  */
   704       gnutls_transport_set_ptr2 (state,
   705                                  (gnutls_transport_ptr_t) proc,
   706                                  (gnutls_transport_ptr_t) proc);
   707       gnutls_transport_set_push_function (state, &emacs_gnutls_push);
   708       gnutls_transport_set_pull_function (state, &emacs_gnutls_pull);
   709 # else
   710       /* This is how GnuTLS takes sockets: as file descriptors passed
   711          in.  For an Emacs process socket, infd and outfd are the
   712          same but we use this two-argument version for clarity.  */
   713       gnutls_transport_set_ptr2 (state,
   714                                  (void *) (intptr_t) proc->infd,
   715                                  (void *) (intptr_t) proc->outfd);
   716       if (proc->is_non_blocking_client)
   717         gnutls_transport_set_errno_function (state,
   718                                              emacs_gnutls_nonblock_errno);
   719 # endif
   720 
   721       proc->gnutls_initstage = GNUTLS_STAGE_TRANSPORT_POINTERS_SET;
   722     }
   723 
   724   return gnutls_try_handshake (proc);
   725 }

   741 ptrdiff_t
   742 emacs_gnutls_write (struct Lisp_Process *proc, const char *buf, ptrdiff_t nbyte)
   743 {
   744   gnutls_session_t state = proc->gnutls_state;
   745 
   746   if (proc->gnutls_initstage != GNUTLS_STAGE_READY)
   747     {
   748       errno = EAGAIN;
   749       return 0;
   750     }
   751 
   752   ptrdiff_t bytes_written = 0;
   753 
   754   while (nbyte > 0)
   755     {
   756       ssize_t rtnval;
   757       do
   758         rtnval = gnutls_record_send (state, buf, nbyte);
   759       while (rtnval == GNUTLS_E_INTERRUPTED);
   760 
   761       if (rtnval < 0)
   762         {
   763           emacs_gnutls_handle_error (state, rtnval);
   764           break;
   765         }
   766 
   767       buf += rtnval;
   768       nbyte -= rtnval;
   769       bytes_written += rtnval;
   770     }
   771 
   772   return (bytes_written);
   773 }

   775 ptrdiff_t
   776 emacs_gnutls_read (struct Lisp_Process *proc, char *buf, ptrdiff_t nbyte)
   777 {
   778   gnutls_session_t state = proc->gnutls_state;
   779 
   780   if (proc->gnutls_initstage != GNUTLS_STAGE_READY)
   781     {
   782       errno = EAGAIN;
   783       return -1;
   784     }
   785 
   786   ssize_t rtnval;
   787   do
   788     rtnval = gnutls_record_recv (state, buf, nbyte);
   789   while (rtnval == GNUTLS_E_INTERRUPTED);
   790 
   791   if (rtnval >= 0)
   792     return rtnval;
   793   else if (rtnval == GNUTLS_E_UNEXPECTED_PACKET_LENGTH)
   794     /* The peer closed the connection. */
   795     return 0;
   796   else
   797     return emacs_gnutls_handle_error (state, rtnval);
   798 }

   826   int errnum = EINVAL;
   827 
   828   if (gnutls_error_is_fatal (err))
   829     {
   830       int level = 1;
   831       /* Mostly ignore "The TLS connection was non-properly
   832          terminated" message which just means that the peer closed the
   833          connection.  */
   834 # ifdef HAVE_GNUTLS3
   835       if (err == GNUTLS_E_PREMATURE_TERMINATION)
   836         level = 3;
   837 # endif
   838 
   839       GNUTLS_LOG2 (level, max_log_level, "fatal error:", str);
   840       ret = 0;
   841     }
   842   else
   843     {
   844       ret = -1;
   845 
   846       switch (err)
   847         {
   848         case GNUTLS_E_AGAIN:
   849           GNUTLS_LOG2 (3,
   850                        max_log_level,
   851                        "retry:",
   852                        str);
   853           FALLTHROUGH;
   854         default:
   855           GNUTLS_LOG2 (1,
   856                        max_log_level,
   857                        "non-fatal error:",
   858                        str);
   859         }
   860 
   861       switch (err)
   862         {
   863         case GNUTLS_E_AGAIN:
   864           errnum = EAGAIN;
   865           break;
   866 
   867 # ifdef EMSGSIZE
   868         case GNUTLS_E_LARGE_PACKET:
   869         case GNUTLS_E_PUSH_ERROR:
   870           errnum = EMSGSIZE;
   871           break;
   872 # endif
   873 
   874 # if defined HAVE_GNUTLS3 && defined ECONNRESET
   875         case GNUTLS_E_PREMATURE_TERMINATION:
   876           errnum = ECONNRESET;
   877           break;
   878 # endif
   879         }
   880     }
   881 
   882   if (err == GNUTLS_E_WARNING_ALERT_RECEIVED
   883       || err == GNUTLS_E_FATAL_ALERT_RECEIVED)
   884     {
   885       int alert = gnutls_alert_get (session);
   886       int level = (err == GNUTLS_E_FATAL_ALERT_RECEIVED) ? 0 : 1;
   887       str = gnutls_alert_get_name (alert);
   888       if (!str)
   889         str = "unknown";
   890 
   891       GNUTLS_LOG2 (level, max_log_level, "Received alert: ", str);
   892     }
   893 
   894   errno = errnum;
   895   return ret;
   896 }

   899    known symbol like 'gnutls_e_interrupted' and 'gnutls_e_again' or
   900    simply the integer value of the error.  GNUTLS_E_SUCCESS is mapped

  1081 static Lisp_Object
  1082 gnutls_hex_string (unsigned char *buf, ptrdiff_t buf_size, const char *prefix)
  1083 {
  1084   ptrdiff_t prefix_length = strlen (prefix);
  1085   ptrdiff_t retlen;
  1086   if (ckd_mul (&retlen, buf_size, 3)
  1087       || ckd_add (&retlen, retlen, prefix_length - (buf_size != 0)))
  1088     string_overflow ();
  1089   Lisp_Object ret = make_uninit_string (retlen);
  1090   char *string = SSDATA (ret);
  1091   strcpy (string, prefix);
  1092 
  1093   for (ptrdiff_t i = 0; i < buf_size; i++)
  1094     sprintf (string + i * 3 + prefix_length,
  1095              i == buf_size - 1 ? "%02x" : "%02x:",
  1096              buf[i]);
  1097 
  1098   return ret;
  1099 }

  1129 static Lisp_Object
  1130 emacs_gnutls_certificate_details (gnutls_x509_crt_t cert)
  1131 {
  1132   Lisp_Object res = Qnil;
  1133   int err;
  1134   size_t buf_size;
  1135 
  1136   /* Version. */
  1137   {
  1138     int version = gnutls_x509_crt_get_version (cert);
  1139     check_memory_full (version);
  1140     if (version >= GNUTLS_E_SUCCESS)
  1141       res = nconc2 (res, list2 (intern (":version"),
  1142                                 make_fixnum (version)));
  1143   }
  1144 
  1145   /* Serial. */
  1146   buf_size = 0;
  1147   err = gnutls_x509_crt_get_serial (cert, NULL, &buf_size);
  1148   check_memory_full (err);
  1149   if (err == GNUTLS_E_SHORT_MEMORY_BUFFER)
  1150     {
  1151       void *serial = xmalloc (buf_size);
  1152       err = gnutls_x509_crt_get_serial (cert, serial, &buf_size);
  1153       check_memory_full (err);
  1154       if (err >= GNUTLS_E_SUCCESS)
  1155         res = nconc2 (res, list2 (intern (":serial-number"),
  1156                                   gnutls_hex_string (serial, buf_size, "")));
  1157       xfree (serial);
  1158     }
  1159 
  1160   /* Issuer. */
  1161   buf_size = 0;
  1162   err = gnutls_x509_crt_get_issuer_dn (cert, NULL, &buf_size);
  1163   check_memory_full (err);
  1164   if (err == GNUTLS_E_SHORT_MEMORY_BUFFER)
  1165     {
  1166       char *dn = xmalloc (buf_size);
  1167       err = gnutls_x509_crt_get_issuer_dn (cert, dn, &buf_size);
  1168       check_memory_full (err);
  1169       if (err >= GNUTLS_E_SUCCESS)
  1170         res = nconc2 (res, list2 (intern (":issuer"),
  1171                                   make_string (dn, buf_size)));
  1172       xfree (dn);
  1173     }
  1174 
  1175   /* Validity. */
  1176   {
  1177     /* Add 1 to the buffer size, since 1900 is added to tm_year and
  1178        that might add 1 to the year length.  */
  1179     char buf[INT_STRLEN_BOUND (int) + 1 + sizeof "-12-31"];
  1180     struct tm t;
  1181     time_t tim = gnutls_x509_crt_get_activation_time (cert);
  1182 
  1183     if (gmtime_r (&tim, &t) && strftime (buf, sizeof buf, "%Y-%m-%d", &t))
  1184       res = nconc2 (res, list2 (intern (":valid-from"), build_string (buf)));
  1185 
  1186     tim = gnutls_x509_crt_get_expiration_time (cert);
  1187     if (gmtime_r (&tim, &t) && strftime (buf, sizeof buf, "%Y-%m-%d", &t))
  1188       res = nconc2 (res, list2 (intern (":valid-to"), build_string (buf)));
  1189   }
  1190 
  1191   /* Subject. */
  1192   buf_size = 0;
  1193   err = gnutls_x509_crt_get_dn (cert, NULL, &buf_size);
  1194   check_memory_full (err);
  1195   if (err == GNUTLS_E_SHORT_MEMORY_BUFFER)
  1196     {
  1197       char *dn = xmalloc (buf_size);
  1198       err = gnutls_x509_crt_get_dn (cert, dn, &buf_size);
  1199       check_memory_full (err);
  1200       if (err >= GNUTLS_E_SUCCESS)
  1201         res = nconc2 (res, list2 (intern (":subject"),
  1202                                   make_string (dn, buf_size)));
  1203       xfree (dn);
  1204     }
  1205 
  1206   /* SubjectPublicKeyInfo. */
  1207   {
  1208     unsigned int bits;
  1209 
  1210     err = gnutls_x509_crt_get_pk_algorithm (cert, &bits);
  1211     check_memory_full (err);
  1212     if (err >= GNUTLS_E_SUCCESS)
  1213       {
  1214         const char *name = gnutls_pk_algorithm_get_name (err);
  1215         if (name)
  1216           res = nconc2 (res, list2 (intern (":public-key-algorithm"),
  1217                                     build_string (name)));
  1218 
  1219         name = gnutls_sec_param_get_name (gnutls_pk_bits_to_sec_param
  1220                                           (err, bits));
  1221         res = nconc2 (res, list2 (intern (":certificate-security-level"),
  1222                                   build_string (name)));
  1223       }
  1224   }
  1225 
  1226   /* Unique IDs. */
  1227   buf_size = 0;
  1228   err = gnutls_x509_crt_get_issuer_unique_id (cert, NULL, &buf_size);
  1229   check_memory_full (err);
  1230   if (err == GNUTLS_E_SHORT_MEMORY_BUFFER)
  1231     {
  1232       char *buf = xmalloc (buf_size);
  1233       err = gnutls_x509_crt_get_issuer_unique_id (cert, buf, &buf_size);
  1234       check_memory_full (err);
  1235       if (err >= GNUTLS_E_SUCCESS)
  1236         res = nconc2 (res, list2 (intern (":issuer-unique-id"),
  1237                                   make_string (buf, buf_size)));
  1238       xfree (buf);
  1239     }
  1240 
  1241   buf_size = 0;
  1242   err = gnutls_x509_crt_get_subject_unique_id (cert, NULL, &buf_size);
  1243   check_memory_full (err);
  1244   if (err == GNUTLS_E_SHORT_MEMORY_BUFFER)
  1245     {
  1246       char *buf = xmalloc (buf_size);
  1247       err = gnutls_x509_crt_get_subject_unique_id (cert, buf, &buf_size);
  1248       check_memory_full (err);
  1249       if (err >= GNUTLS_E_SUCCESS)
  1250         res = nconc2 (res, list2 (intern (":subject-unique-id"),
  1251                                   make_string (buf, buf_size)));
  1252       xfree (buf);
  1253     }
  1254 
  1255   /* Signature. */
  1256   err = gnutls_x509_crt_get_signature_algorithm (cert);
  1257   check_memory_full (err);
  1258   if (err >= GNUTLS_E_SUCCESS)
  1259     {
  1260       const char *name = gnutls_sign_get_name (err);
  1261       if (name)
  1262         res = nconc2 (res, list2 (intern (":signature-algorithm"),
  1263                                   build_string (name)));
  1264     }
  1265 
  1266   /* Public key ID. */
  1267   buf_size = 0;
  1268   err = gnutls_x509_crt_get_key_id (cert, 0, NULL, &buf_size);
  1269   check_memory_full (err);
  1270   if (err == GNUTLS_E_SHORT_MEMORY_BUFFER)
  1271     {
  1272       void *buf = xmalloc (buf_size);
  1273       err = gnutls_x509_crt_get_key_id (cert, 0, buf, &buf_size);
  1274       check_memory_full (err);
  1275       if (err >= GNUTLS_E_SUCCESS)
  1276         res = nconc2 (res, list2 (intern (":public-key-id"),
  1277                                   gnutls_hex_string (buf, buf_size, "sha1:")));
  1278       xfree (buf);
  1279     }
  1280 
  1281   /* Certificate fingerprint. */
  1282   buf_size = 0;
  1283   err = gnutls_x509_crt_get_fingerprint (cert, GNUTLS_DIG_SHA1,
  1284                                          NULL, &buf_size);
  1285   check_memory_full (err);
  1286   if (err == GNUTLS_E_SHORT_MEMORY_BUFFER)
  1287     {
  1288       void *buf = xmalloc (buf_size);
  1289       err = gnutls_x509_crt_get_fingerprint (cert, GNUTLS_DIG_SHA1,
  1290                                              buf, &buf_size);
  1291       check_memory_full (err);
  1292       if (err >= GNUTLS_E_SUCCESS)
  1293         res = nconc2 (res, list2 (intern (":certificate-id"),
  1294                                   gnutls_hex_string (buf, buf_size, "sha1:")));
  1295       xfree (buf);
  1296     }
  1297 
  1298   /* PEM */
  1299   res = nconc2 (res, list2 (intern (":pem"),
  1300                             emacs_gnutls_certificate_export_pem(cert)));
  1301 
  1302   return res;
  1303 }

  1636 Lisp_Object
  1637 gnutls_verify_boot (Lisp_Object proc, Lisp_Object proplist)
  1638 {
  1639   int ret;
  1640   struct Lisp_Process *p = XPROCESS (proc);
  1641   gnutls_session_t state = p->gnutls_state;
  1642   unsigned int peer_verification;
  1643   Lisp_Object warnings;
  1644   int max_log_level = p->gnutls_log_level;
  1645   Lisp_Object hostname, verify_error;
  1646   bool verify_error_all = false;
  1647   char *c_hostname;
  1648 
  1649   if (NILP (proplist))
  1650     proplist = Fcdr (plist_get (p->childp, QCtls_parameters));
  1651 
  1652   verify_error = plist_get (proplist, QCverify_error);
  1653   hostname = plist_get (proplist, QChostname);
  1654 
  1655   if (EQ (verify_error, Qt))
  1656     verify_error_all = true;
  1657   else if (NILP (Flistp (verify_error)))
  1658     {
  1659       boot_error (p,
  1660                   "gnutls-boot: invalid :verify_error parameter (not a list)");
  1661       return Qnil;
  1662     }
  1663 
  1664   if (!STRINGP (hostname))
  1665     {
  1666       boot_error (p, "gnutls-boot: invalid :hostname parameter (not a string)");
  1667       return Qnil;
  1668     }
  1669   c_hostname = SSDATA (hostname);
  1670 
  1671   /* Now verify the peer, following
  1672      https://www.gnu.org/software/gnutls/manual/html_node/Verifying-peer_0027s-certificate.html.
  1673      The peer should present at least one certificate in the chain; do a
  1674      check of the certificate's hostname with
  1675      gnutls_x509_crt_check_hostname against :hostname.  */
  1676 
  1677   ret = gnutls_certificate_verify_peers2 (state, &peer_verification);
  1678   if (ret < GNUTLS_E_SUCCESS)
  1679     return gnutls_make_error (ret);
  1680 
  1681   p->gnutls_peer_verification = peer_verification;
  1682 
  1683   warnings = plist_get (Fgnutls_peer_status (proc), intern (":warnings"));
  1684   if (!NILP (warnings))
  1685     {
  1686       for (Lisp_Object tail = warnings; CONSP (tail); tail = XCDR (tail))
  1687         {
  1688           Lisp_Object warning = XCAR (tail);
  1689           Lisp_Object message = Fgnutls_peer_status_warning_describe (warning);
  1690           if (!NILP (message))
  1691             GNUTLS_LOG2 (1, max_log_level, "verification:", SSDATA (message));
  1692         }
  1693     }
  1694 
  1695   if (peer_verification != 0)
  1696     {
  1697       if (verify_error_all
  1698           || !NILP (Fmember (QCtrustfiles, verify_error)))
  1699         {
  1700           emacs_gnutls_deinit (proc);
  1701           boot_error (p,
  1702                       "Certificate validation failed %s, verification code %x",
  1703                       c_hostname, peer_verification);
  1704           return Qnil;
  1705         }
  1706       else
  1707         {
  1708           GNUTLS_LOG2 (1, max_log_level, "certificate validation failed:",
  1709                        c_hostname);
  1710         }
  1711     }
  1712 
  1713   /* Up to here the process is the same for X.509 certificates and
  1714      OpenPGP keys.  From now on X.509 certificates are assumed.  This
  1715      can be easily extended to work with openpgp keys as well.  */
  1716   if (gnutls_certificate_type_get (state) == GNUTLS_CRT_X509)
  1717     {
  1718       const gnutls_datum_t *cert_list;
  1719       unsigned int cert_list_length;
  1720       int failed_import = 0;
  1721 
  1722       cert_list = gnutls_certificate_get_peers (state, &cert_list_length);
  1723 
  1724       if (cert_list == NULL)
  1725         {
  1726           emacs_gnutls_deinit (proc);
  1727           boot_error (p, "No x509 certificate was found\n");
  1728           return Qnil;
  1729         }
  1730 
  1731       /* Check only the first certificate in the given chain, but
  1732          store them all.  */
  1733       p->gnutls_certificates =
  1734         xmalloc (cert_list_length * sizeof (gnutls_x509_crt_t));
  1735       p->gnutls_certificates_length = cert_list_length;
  1736 
  1737       for (int i = cert_list_length - 1; i >= 0; i--)
  1738         {
  1739           gnutls_x509_crt_t cert;
  1740 
  1741           gnutls_x509_crt_init (&cert);
  1742 
  1743           if (ret < GNUTLS_E_SUCCESS)
  1744             failed_import = ret;
  1745           else
  1746             {
  1747               ret = gnutls_x509_crt_import (cert, &cert_list[i],
  1748                                             GNUTLS_X509_FMT_DER);
  1749 
  1750               if (ret < GNUTLS_E_SUCCESS)
  1751                 failed_import = ret;
  1752             }
  1753 
  1754           p->gnutls_certificates[i] = cert;
  1755         }
  1756 
  1757       if (failed_import != 0)
  1758         {
  1759           gnutls_deinit_certificates (p);
  1760           return gnutls_make_error (failed_import);
  1761         }
  1762 
  1763       int err = gnutls_x509_crt_check_hostname (p->gnutls_certificates[0],
  1764                                                 c_hostname);
  1765       check_memory_full (err);
  1766       if (!err)
  1767         {
  1768           p->gnutls_extra_peer_verification |= CERTIFICATE_NOT_MATCHING;
  1769           if (verify_error_all
  1770               || !NILP (Fmember (QChostname, verify_error)))
  1771             {
  1772               emacs_gnutls_deinit (proc);
  1773               boot_error (p, "The x509 certificate does not match \"%s\"",
  1774                           c_hostname);
  1775               return Qnil;
  1776             }
  1777           else
  1778             GNUTLS_LOG2 (1, max_log_level, "x509 certificate does not match:",
  1779                          c_hostname);
  1780         }
  1781     }
  1782 
  1783   /* Set this flag only if the whole initialization succeeded.  */
  1784   p->gnutls_p = true;
  1785 
  1786   return gnutls_make_error (ret);
  1787 }

  2448 static Lisp_Object
  2449 gnutls_symmetric (bool encrypting, Lisp_Object cipher,
  2450                   Lisp_Object key, Lisp_Object iv,
  2451                   Lisp_Object input, Lisp_Object aead_auth)
  2452 {
  2453   if (BUFFERP (key) || STRINGP (key))
  2454     key = list1 (key);
  2455 
  2456   CHECK_CONS (key);
  2457 
  2458   if (BUFFERP (input) || STRINGP (input))
  2459     input = list1 (input);
  2460 
  2461   CHECK_CONS (input);
  2462 
  2463   if (BUFFERP (iv) || STRINGP (iv))
  2464     iv = list1 (iv);
  2465 
  2466   CHECK_CONS (iv);
  2467 
  2468 
  2469   const char *desc = encrypting ? "encrypt" : "decrypt";
  2470 
  2471   gnutls_cipher_algorithm_t gca = GNUTLS_CIPHER_UNKNOWN;
  2472 
  2473   Lisp_Object info = Qnil;
  2474   if (STRINGP (cipher))
  2475     cipher = intern (SSDATA (cipher));
  2476 
  2477   if (SYMBOLP (cipher))
  2478     {
  2479       if (NILP (cipher_cache))
  2480         cipher_cache = Fgnutls_ciphers ();
  2481       info = Fassq (cipher, cipher_cache);
  2482       if (!CONSP (info))
  2483         xsignal2 (Qerror,
  2484                   build_string ("GnuTLS cipher is invalid or not found"),
  2485                   cipher);
  2486       info = XCDR (info);
  2487     }
  2488   else if (TYPE_RANGED_FIXNUMP (gnutls_cipher_algorithm_t, cipher))
  2489     gca = XFIXNUM (cipher);
  2490   else
  2491     info = cipher;
  2492 
  2493   if (!NILP (info) && CONSP (info))
  2494     {
  2495       Lisp_Object v = plist_get (info, QCcipher_id);
  2496       if (TYPE_RANGED_FIXNUMP (gnutls_cipher_algorithm_t, v))
  2497         gca = XFIXNUM (v);
  2498     }
  2499 
  2500   ptrdiff_t key_size = gnutls_cipher_get_key_size (gca);
  2501   if (key_size == 0)
  2502     xsignal2 (Qerror,
  2503               build_string ("GnuTLS cipher is invalid or not found"), cipher);
  2504 
  2505   ptrdiff_t kstart_byte, kend_byte;
  2506   const char *kdata = extract_data_from_object (key, &kstart_byte, &kend_byte);
  2507 
  2508   if (kdata == NULL)
  2509     error ("GnuTLS cipher key extraction failed");
  2510 
  2511   if (kend_byte - kstart_byte != key_size)
  2512     error (("GnuTLS cipher %s/%s key length %"pD"d is not equal to "
  2513             "the required %"pD"d"),
  2514            gnutls_cipher_get_name (gca), desc,
  2515            kend_byte - kstart_byte, key_size);
  2516 
  2517   ptrdiff_t vstart_byte, vend_byte;
  2518   char *vdata = extract_data_from_object (iv, &vstart_byte, &vend_byte);
  2519 
  2520   if (vdata == NULL)
  2521     error ("GnuTLS cipher IV extraction failed");
  2522 
  2523   ptrdiff_t iv_size = gnutls_cipher_get_iv_size (gca);
  2524   if (vend_byte - vstart_byte != iv_size)
  2525     error (("GnuTLS cipher %s/%s IV length %"pD"d is not equal to "
  2526             "the required %"pD"d"),
  2527            gnutls_cipher_get_name (gca), desc,
  2528            vend_byte - vstart_byte, iv_size);
  2529 
  2530   Lisp_Object actual_iv = make_unibyte_string (vdata, vend_byte - vstart_byte);
  2531 
  2532   ptrdiff_t istart_byte, iend_byte;
  2533   const char *idata
  2534     = extract_data_from_object (input, &istart_byte, &iend_byte);
  2535 
  2536   if (idata == NULL)
  2537     error ("GnuTLS cipher input extraction failed");
  2538 
  2539   /* Is this an AEAD cipher? */
  2540   if (gnutls_cipher_get_tag_size (gca) > 0)
  2541     {
  2542       Lisp_Object aead_output =
  2543         gnutls_symmetric_aead (encrypting, gca, cipher,
  2544                                kdata, kend_byte - kstart_byte,
  2545                                vdata, vend_byte - vstart_byte,
  2546                                idata, iend_byte - istart_byte,
  2547                                aead_auth);
  2548       if (STRINGP (XCAR (key)))
  2549         Fclear_string (XCAR (key));
  2550       return aead_output;
  2551     }
  2552 
  2553   ptrdiff_t cipher_block_size = gnutls_cipher_get_block_size (gca);
  2554   if ((iend_byte - istart_byte) % cipher_block_size != 0)
  2555     error (("GnuTLS cipher %s/%s input block length %"pD"d is not a multiple "
  2556             "of the required %"pD"d"),
  2557            gnutls_cipher_get_name (gca), desc,
  2558            iend_byte - istart_byte, cipher_block_size);
  2559 
  2560   gnutls_cipher_hd_t hcipher;
  2561   gnutls_datum_t key_datum
  2562     = { (unsigned char *) kdata, kend_byte - kstart_byte };
  2563 
  2564   int ret = gnutls_cipher_init (&hcipher, gca, &key_datum, NULL);
  2565 
  2566   if (ret < GNUTLS_E_SUCCESS)
  2567     error ("GnuTLS cipher %s/%s initialization failed: %s",
  2568            gnutls_cipher_get_name (gca), desc, emacs_gnutls_strerror (ret));
  2569 
  2570   /* Note that this will not support streaming block mode. */
  2571   gnutls_cipher_set_iv (hcipher, vdata, vend_byte - vstart_byte);
  2572 
  2573   /* GnuTLS docs: "For the supported ciphers the encrypted data length
  2574      will equal the plaintext size."  */
  2575   ptrdiff_t storage_length = iend_byte - istart_byte;
  2576   Lisp_Object storage = make_uninit_string (storage_length);
  2577 
  2578   ret = ((encrypting ? gnutls_cipher_encrypt2 : gnutls_cipher_decrypt2)
  2579          (hcipher, idata, iend_byte - istart_byte,
  2580           SSDATA (storage), storage_length));
  2581 
  2582   if (STRINGP (XCAR (key)))
  2583     Fclear_string (XCAR (key));
  2584 
  2585   if (ret < GNUTLS_E_SUCCESS)
  2586     {
  2587       gnutls_cipher_deinit (hcipher);
  2588       if (encrypting)
  2589         error ("GnuTLS cipher %s encryption failed: %s",
  2590                gnutls_cipher_get_name (gca), emacs_gnutls_strerror (ret));
  2591       else
  2592         error ("GnuTLS cipher %s decryption failed: %s",
  2593                gnutls_cipher_get_name (gca), emacs_gnutls_strerror (ret));
  2594     }
  2595 
  2596   gnutls_cipher_deinit (hcipher);
  2597 
  2598   return list2 (storage, actual_iv);
  2599 }
