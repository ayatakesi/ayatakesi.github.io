    80   public static void
    81   invalidateFocus1 (EmacsWindow window)
    82   {
    83     if (window.view.isFocused ())
    84       focusedWindow = window;
    85 
    86     for (EmacsWindow child : window.children)
    87       invalidateFocus1 (child);
    88   }

    90   public static void
    91   invalidateFocus ()
    92   {
    93     EmacsWindow oldFocus;
    94 
    95     /* Walk through each focused activity and assign the window focus
    96        to the bottom-most focused window within.  Record the old focus
    97        as well.  */
    98     oldFocus = focusedWindow;
    99     focusedWindow = null;
   100 
   101     for (EmacsActivity activity : focusedActivities)
   102       {
   103         if (activity.window != null)
   104           invalidateFocus1 (activity.window);
   105       }
   106 
   107     /* Send focus in- and out- events to the previous and current
   108        focus.  */
   109 
   110     if (oldFocus != null)
   111       EmacsNative.sendFocusOut (oldFocus.handle,
   112                                 System.currentTimeMillis ());
   113 
   114     if (focusedWindow != null)
   115       EmacsNative.sendFocusIn (focusedWindow.handle,
   116                                System.currentTimeMillis ());
   117   }

   119   @Override
   120   public final void
   121   detachWindow ()
   122   {
   123     syncFullscreenWith (null);
   124 
   125     if (window == null)
   126       Log.w (TAG, "detachWindow called, but there is no window");
   127     else
   128       {
   129         /* Clear the window's pointer to this activity and remove the
   130            window's view.  */
   131         window.setConsumer (null);
   132 
   133         /* The window can't be iconified any longer.  */
   134         window.noticeDeiconified ();
   135         layout.removeView (window.view);
   136         window = null;
   137 
   138         invalidateFocus ();
   139       }
   140   }

   142   @Override
   143   public final void
   144   attachWindow (EmacsWindow child)
   145   {
   146     Log.d (TAG, "attachWindow: " + child);
   147 
   148     if (window != null)
   149       throw new IllegalStateException ("trying to attach window when one"
   150                                        + " already exists");
   151 
   152     syncFullscreenWith (child);
   153 
   154     /* Record and attach the view.  */
   155 
   156     window = child;
   157     layout.addView (window.view);
   158     child.setConsumer (this);
   159 
   160     /* If the window isn't no-focus-on-map, focus its view.  */
   161     if (!child.getDontFocusOnMap ())
   162       window.view.requestFocus ();
   163 
   164     /* If the activity is iconified, send that to the window.  */
   165     if (isPaused)
   166       window.noticeIconified ();
   167 
   168     /* Invalidate the focus.  */
   169     invalidateFocus ();
   170   }

   243   @Override
   244   public final void
   245   onDestroy ()
   246   {
   247     EmacsWindowAttachmentManager manager;
   248     boolean isMultitask;
   249 
   250     manager = EmacsWindowAttachmentManager.MANAGER;
   251 
   252     /* The activity will die shortly hereafter.  If there is a window
   253        attached, close it now.  */
   254     Log.d (TAG, "onDestroy " + this);
   255     isMultitask = this instanceof EmacsMultitaskActivity;
   256     manager.removeWindowConsumer (this, isMultitask || isFinishing ());
   257     focusedActivities.remove (this);
   258     invalidateFocus ();
   259 
   260     /* Remove this activity from the static field, lest it leak.  */
   261     if (lastFocusedActivity == this)
   262       lastFocusedActivity = null;
   263 
   264     super.onDestroy ();
   265   }

   267   @Override
   268   public final void
   269   onWindowFocusChanged (boolean isFocused)
   270   {
   271     Log.d (TAG, ("onWindowFocusChanged: "
   272                  + (isFocused ? "YES" : "NO")));
   273 
   274     if (isFocused && !focusedActivities.contains (this))
   275       {
   276         focusedActivities.add (this);
   277         lastFocusedActivity = this;
   278 
   279         /* Update the window insets as the focus change may have
   280            changed the window insets as well, and the system does not
   281            automatically restore visibility flags.  */
   282 
   283         if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.JELLY_BEAN
   284             && Build.VERSION.SDK_INT < Build.VERSION_CODES.R
   285             && isFullscreen)
   286           syncFullscreenWith (window);
   287       }
   288     else
   289       focusedActivities.remove (this);
   290 
   291     invalidateFocus ();
   292   }

   314   @Override
   315   public final void
   316   onContextMenuClosed (Menu menu)
   317   {
   318     int serial;
   319 
   320     Log.d (TAG, "onContextMenuClosed: " + menu);
   321 
   322     /* See the comment inside onMenuItemClick.  */
   323 
   324     if (((EmacsContextMenu.wasSubmenuSelected == -2)
   325          || (EmacsContextMenu.wasSubmenuSelected >= 0
   326              && ((System.currentTimeMillis ()
   327                   - EmacsContextMenu.wasSubmenuSelected)
   328                  <= 300)))
   329         || menu == lastClosedMenu)
   330       {
   331         EmacsContextMenu.wasSubmenuSelected = -1;
   332         lastClosedMenu = menu;
   333         return;
   334       }
   335 
   336     /* lastClosedMenu is set because Android apparently calls this
   337        function twice.  */
   338 
   339     lastClosedMenu = null;
   340 
   341     /* Send a context menu event given that no menu item has already
   342        been selected.  */
   343     if (!EmacsContextMenu.itemAlreadySelected)
   344       {
   345         serial = EmacsContextMenu.lastMenuEventSerial;
   346         EmacsNative.sendContextMenu ((short) 0, 0,
   347                                      serial);
   348       }
   349 
   350     super.onContextMenuClosed (menu);
   351   }

   353   @SuppressWarnings ("deprecation")
   354   public final void
   355   syncFullscreenWith (EmacsWindow emacsWindow)
   356   {
   357     WindowInsetsController controller;
   358     Window window;
   359     int behavior, flags;
   360     View view;
   361 
   362     if (emacsWindow != null)
   363       isFullscreen = emacsWindow.fullscreen;
   364     else
   365       isFullscreen = false;
   366 
   367     /* On Android 11 or later, use the window insets controller to
   368        control whether or not the view is fullscreen.  */
   369 
   370     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R)
   371       {
   372         window = getWindow ();
   373 
   374         /* If there is no attached window, return immediately.  */
   375         if (window == null)
   376           return;
   377 
   378         behavior = WindowInsetsController.BEHAVIOR_SHOW_TRANSIENT_BARS_BY_SWIPE;
   379         controller = window.getInsetsController ();
   380         controller.setSystemBarsBehavior (behavior);
   381 
   382         if (isFullscreen)
   383           controller.hide (WindowInsets.Type.statusBars ()
   384                            | WindowInsets.Type.navigationBars ());
   385         else
   386           controller.show (WindowInsets.Type.statusBars ()
   387                            | WindowInsets.Type.navigationBars ());
   388 
   389         return;
   390       }
   391 
   392     if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.JELLY_BEAN)
   393       {
   394         /* On Android 4.1 or later, use `setSystemUiVisibility'.  */
   395 
   396         window = getWindow ();
   397 
   398         if (window == null)
   399           return;
   400 
   401         view = window.getDecorView ();
   402 
   403         if (isFullscreen)
   404           {
   405             flags = 0;
   406             flags |= View.SYSTEM_UI_FLAG_FULLSCREEN;
   407 
   408             if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.KITKAT)
   409               {
   410                 /* These flags means that Emacs will be full screen as
   411                    long as the state flag is set.  */
   412                 flags |= View.SYSTEM_UI_FLAG_HIDE_NAVIGATION;
   413                 flags |= View.SYSTEM_UI_FLAG_IMMERSIVE;
   414                 flags |= View.SYSTEM_UI_FLAG_IMMERSIVE_STICKY;
   415               }
   416 
   417             /* Apply the given flags.  */
   418             view.setSystemUiVisibility (flags);
   419           }
   420         else
   421           view.setSystemUiVisibility (View.SYSTEM_UI_FLAG_VISIBLE);
   422       }
   423   }
