    82 /* Normally use a symbolic link to represent a lock.
    83    The strategy: to lock a file FN, create a symlink .#FN in FN's
    84    directory, with link data USER@HOST.PID:BOOT.  This avoids a single
    85    mount (== failure) point for lock files.  The :BOOT is omitted if
    86    the boot time is not available.
    87 
    88    When the host in the lock data is the current host, we can check if
    89    the pid is valid with kill.
    90 
    91    Otherwise, we could look at a separate file that maps hostnames to
    92    reboot times to see if the remote pid can possibly be valid, since we
    93    don't want Emacs to have to communicate via pipes or sockets or
    94    whatever to other processes, either locally or remotely; rms says
    95    that's too unreliable.  Hence the separate file, which could
    96    theoretically be updated by daemons running separately -- but this
    97    whole idea is unimplemented; in practice, at least in our
    98    environment, it seems such stale locks arise fairly infrequently, and
    99    Emacs' standard methods of dealing with clashes suffice.
   100 
   101    We use symlinks instead of normal files because (1) they can be
   102    stored more efficiently on the filesystem, since the kernel knows
   103    they will be small, and (2) all the info about the lock can be read
   104    in a single system call (readlink).  Although we could use regular
   105    files to be useful on old systems lacking symlinks, nowadays
   106    virtually all such systems are probably single-user anyway, so it
   107    didn't seem worth the complication.
   108 
   109    Similarly, we don't worry about a possible 14-character limit on
   110    file names, because those are all the same systems that don't have
   111    symlinks.
   112 
   113    This is compatible with the locking scheme used by Interleaf (which
   114    has contributed this implementation for Emacs), and was designed by
   115    Karl Berry, Ethan Jacobson, Kimbo Mundy, and others.
   116 
   117    On some file systems, notably those of MS-Windows, symbolic links
   118    do not work well, so instead of a symlink .#FN -> USER@HOST.PID:BOOT,
   119    the lock is a regular file .#FN with contents USER@HOST.PID:BOOT.  To
   120    establish a lock, a nonce file is created and then renamed to .#FN.
   121    On MS-Windows this renaming is atomic unless the lock is forcibly
   122    acquired.  On other systems the renaming is atomic if the lock is
   123    forcibly acquired; if not, the renaming is done via hard links,
   124    which is good enough for lock-file purposes.
   125 
   126    To summarize, race conditions can occur with either:
   127 
   128    * Forced locks on MS-Windows systems.
   129 
   130    * Non-forced locks on non-MS-Windows systems that support neither
   131      hard nor symbolic links.  */

   143 static time_t
   144 get_boot_time (void)
   145 {
   146 #if defined (BOOT_TIME)
   147   int counter;
   148 #endif
   149 
   150   if (boot_time_initialized)
   151     return boot_time;
   152   boot_time_initialized = 1;
   153 
   154 #if defined (CTL_KERN) && defined (KERN_BOOTTIME)
   155   {
   156     int mib[2];
   157     size_t size;
   158     struct timeval boottime_val;
   159 
   160     mib[0] = CTL_KERN;
   161     mib[1] = KERN_BOOTTIME;
   162     size = sizeof (boottime_val);
   163 
   164     if (sysctl (mib, 2, &boottime_val, &size, NULL, 0) >= 0 && size != 0)
   165       {
   166         boot_time = boottime_val.tv_sec;
   167         return boot_time;
   168       }
   169   }
   170 #endif /* defined (CTL_KERN) && defined (KERN_BOOTTIME) */
   171 
   172   if (BOOT_TIME_FILE)
   173     {
   174       struct stat st;
   175       if (stat (BOOT_TIME_FILE, &st) == 0)
   176         {
   177           boot_time = st.st_mtime;
   178           return boot_time;
   179         }
   180     }
   181 
   182 #if defined (BOOT_TIME)
   183   /* The utmp routines maintain static state.  Don't touch that state
   184      if we are going to dump, since it might not survive dumping.  */
   185   if (will_dump_p ())
   186     return boot_time;
   187 
   188   /* Try to get boot time from utmp before wtmp,
   189      since utmp is typically much smaller than wtmp.
   190      Passing a null pointer causes get_boot_time_1
   191      to inspect the default file, namely utmp.  */
   192   get_boot_time_1 (0, 0);
   193   if (boot_time)
   194     return boot_time;
   195 
   196   /* Try to get boot time from the current wtmp file.  */
   197   get_boot_time_1 (WTMP_FILE, 1);
   198 
   199   /* If we did not find a boot time in wtmp, look at wtmp, and so on.  */
   200   for (counter = 0; counter < 20 && ! boot_time; counter++)
   201     {
   202       Lisp_Object filename = Qnil;
   203       bool delete_flag = false;
   204       char cmd_string[sizeof WTMP_FILE ".19.gz"];
   205       AUTO_STRING_WITH_LEN (tempname, cmd_string,
   206                             sprintf (cmd_string, "%s.%d", WTMP_FILE, counter));
   207       if (! NILP (Ffile_exists_p (tempname)))
   208         filename = tempname;
   209       else
   210         {
   211           tempname = make_formatted_string (cmd_string, "%s.%d.gz",
   212                                             WTMP_FILE, counter);
   213           if (! NILP (Ffile_exists_p (tempname)))
   214             {
   215               /* The utmp functions on older systems accept only file
   216                  names up to 8 bytes long.  Choose a 2 byte prefix, so
   217                  the 6-byte suffix does not make the name too long.  */
   218               filename = Fmake_temp_file_internal (build_string ("wt"), Qnil,
   219                                                    empty_unibyte_string, Qnil);
   220               CALLN (Fcall_process, build_string ("gzip"), Qnil,
   221                      list2 (QCfile, filename), Qnil,
   222                      build_string ("-cd"), tempname);
   223               delete_flag = true;
   224             }
   225         }
   226 
   227       if (! NILP (filename))
   228         {
   229           get_boot_time_1 (SSDATA (filename), 1);
   230           if (delete_flag)
   231             unlink (SSDATA (filename));
   232         }
   233     }
   234 
   235   return boot_time;
   236 #else
   237   return 0;
   238 #endif
   239 }

   241 #ifdef BOOT_TIME
   242 /* Try to get the boot time from wtmp file FILENAME.
   243    This succeeds if that file contains a reboot record.
   244 
   245    If FILENAME is zero, use the same file as before;
   246    if no FILENAME has ever been specified, this is the utmp file.
   247    Use the newest reboot record if NEWEST,
   248    the first reboot record otherwise.
   249    Ignore all reboot records on or before BOOT_TIME.
   250    Success is indicated by setting BOOT_TIME to a larger value.  */
   251 
   252 void
   253 get_boot_time_1 (const char *filename, bool newest)
   254 {
   255   struct utmp ut, *utp;
   256 
   257   if (filename)
   258     utmpname (filename);
   259 
   260   setutent ();
   261 
   262   while (1)
   263     {
   264       /* Find the next reboot record.  */
   265       ut.ut_type = BOOT_TIME;
   266       utp = getutid (&ut);
   267       if (! utp)
   268         break;
   269       /* Compare reboot times and use the newest one.  */
   270       if (utp->ut_time > boot_time)
   271         {
   272           boot_time = utp->ut_time;
   273           if (! newest)
   274             break;
   275         }
   276       /* Advance on element in the file
   277          so that getutid won't repeat the same one.  */
   278       utp = getutent ();
   279       if (! utp)
   280         break;
   281     }
   282   endutent ();
   283 }

   290 /* Here is the structure that stores information about a lock.  */
   291 
   292 typedef struct
   293 {
   294   /* Location of '@', '.', and ':' (or equivalent) in USER.  If there's
   295      no colon or equivalent, COLON points to the end of USER.  */
   296   char *at, *dot, *colon;
   297 
   298   /* Lock file contents USER@HOST.PID with an optional :BOOT_TIME
   299      appended.  This memory is used as a lock file contents buffer, so
   300      it needs room for MAX_LFINFO + 1 bytes.  A string " (pid NNNN)"
   301      may be appended to the USER@HOST while generating a diagnostic,
   302      so make room for its extra bytes (as opposed to ".NNNN") too.  */
   303   char user[MAX_LFINFO + 1 + sizeof " (pid )" - sizeof "."];
   304 } lock_info_type;

   306 /* For some reason Linux kernels return EPERM on file systems that do
   307    not support hard or symbolic links.  This symbol documents the quirk.
   308    There is no way to tell whether a symlink call fails due to
   309    permissions issues or because links are not supported, but luckily
   310    the lock file code should work either way.  */
   311 enum { LINKS_MIGHT_NOT_WORK = EPERM };

   313 /* Rename OLD to NEW.  If FORCE, replace any existing NEW.
   314    It is OK if there are temporarily two hard links to OLD.
   315    Return 0 if successful, -1 (setting errno) otherwise.  */
   316 static int
   317 rename_lock_file (char const *old, char const *new, bool force)
   318 {
   319 #ifdef WINDOWSNT
   320   return sys_rename_replace (old, new, force);
   321 #else
   322   if (! force)
   323     {
   324       struct stat st;
   325 
   326       int r = renameat_noreplace (AT_FDCWD, old, AT_FDCWD, new);
   327       if (! (r < 0 && errno == ENOSYS))
   328         return r;
   329       if (link (old, new) == 0)
   330         return unlink (old) == 0 || errno == ENOENT ? 0 : -1;
   331       if (errno != ENOSYS && errno != LINKS_MIGHT_NOT_WORK)
   332         return -1;
   333 
   334       /* 'link' does not work on this file system.  This can occur on
   335          a GNU/Linux host mounting a FAT32 file system.  Fall back on
   336          'rename' after checking that NEW does not exist.  There is a
   337          potential race condition since some other process may create
   338          NEW immediately after the existence check, but it's the best
   339          we can portably do here.  */
   340       if (emacs_fstatat (AT_FDCWD, new, &st, AT_SYMLINK_NOFOLLOW) == 0
   341           || errno == EOVERFLOW)
   342         {
   343           errno = EEXIST;
   344           return -1;
   345         }
   346       if (errno != ENOENT)
   347         return -1;
   348     }
   349 
   350   return rename (old, new);
   351 #endif
   352 }

   354 /* Create the lock file LFNAME with contents LOCK_INFO_STR.  Return 0 if
   355    successful, an errno value on failure.  If FORCE, remove any
   356    existing LFNAME if necessary.  */
   357 
   358 static int
   359 create_lock_file (char *lfname, char *lock_info_str, bool force)
   360 {
   361 #ifdef WINDOWSNT
   362   /* Symlinks are supported only by later versions of Windows, and
   363      creating them is a privileged operation that often triggers
   364      User Account Control elevation prompts.  Avoid the problem by
   365      pretending that 'symlink' does not work.  */
   366   int err = ENOSYS;
   367 #else
   368   int err = symlink (lock_info_str, lfname) == 0 ? 0 : errno;
   369 #endif
   370 
   371   if (err == EEXIST && force)
   372     {
   373       unlink (lfname);
   374       err = symlink (lock_info_str, lfname) == 0 ? 0 : errno;
   375     }
   376 
   377   if (err == ENOSYS || err == LINKS_MIGHT_NOT_WORK || err == ENAMETOOLONG)
   378     {
   379       static char const nonce_base[] = ".#-emacsXXXXXX";
   380       char *last_slash = strrchr (lfname, '/');
   381       ptrdiff_t lfdirlen = last_slash + 1 - lfname;
   382       USE_SAFE_ALLOCA;
   383       char *nonce = SAFE_ALLOCA (lfdirlen + sizeof nonce_base);
   384       int fd;
   385       memcpy (nonce, lfname, lfdirlen);
   386       strcpy (nonce + lfdirlen, nonce_base);
   387 
   388       fd = mkostemp (nonce, O_BINARY | O_CLOEXEC);
   389       if (fd < 0)
   390         err = errno;
   391       else
   392         {
   393           ptrdiff_t lock_info_len;
   394           lock_info_len = strlen (lock_info_str);
   395           err = 0;
   396 
   397           /* Make the lock file readable to others, so that others' sessions
   398              can read it.  Even though nobody should write to the lock file,
   399              keep it user-writable to work around problems on nonstandard file
   400              systems that prohibit unlinking readonly files (Bug#37884).  */
   401           if (emacs_write (fd, lock_info_str, lock_info_len) != lock_info_len
   402               || fchmod (fd, S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH) != 0)
   403             err = errno;
   404 
   405           /* There is no need to call fsync here, as the contents of
   406              the lock file need not survive system crashes.  */
   407           if (emacs_close (fd) != 0)
   408             err = errno;
   409           if (!err && rename_lock_file (nonce, lfname, force) != 0)
   410             err = errno;
   411           if (err)
   412             unlink (nonce);
   413         }
   414 
   415       SAFE_FREE ();
   416     }
   417 
   418   return err;
   419 }

   421 /* Lock the lock file named LFNAME.
   422    If FORCE, do so even if it is already locked.
   423    Return 0 if successful, an error number on failure.  */
   424 
   425 static int
   426 lock_file_1 (Lisp_Object lfname, bool force)
   427 {
   428   intmax_t boot = get_boot_time ();
   429   Lisp_Object luser_name = Fuser_login_name (Qnil);
   430   Lisp_Object lhost_name = Fsystem_name ();
   431 
   432   /* Protect against the extremely unlikely case of the host name
   433      containing an @ character.  */
   434   if (!NILP (lhost_name) && strchr (SSDATA (lhost_name), '@'))
   435     lhost_name = CALLN (Ffuncall, intern ("string-replace"),
   436                         build_string ("@"), build_string ("-"),
   437                         lhost_name);
   438 
   439   char const *user_name = STRINGP (luser_name) ? SSDATA (luser_name) : "";
   440   char const *host_name = STRINGP (lhost_name) ? SSDATA (lhost_name) : "";
   441   char lock_info_str[MAX_LFINFO + 1];
   442   intmax_t pid = getpid ();
   443 
   444   if (boot)
   445     {
   446       if (sizeof lock_info_str
   447           <= snprintf (lock_info_str, sizeof lock_info_str,
   448                        "%s@%s.%"PRIdMAX":%"PRIdMAX,
   449                        user_name, host_name, pid, boot))
   450         return ENAMETOOLONG;
   451     }
   452   else if (sizeof lock_info_str
   453            <= snprintf (lock_info_str, sizeof lock_info_str,
   454                         "%s@%s.%"PRIdMAX,
   455                         user_name, host_name, pid))
   456     return ENAMETOOLONG;
   457 
   458   return create_lock_file (SSDATA (lfname), lock_info_str, force);
   459 }

   474 /* Read the data for the lock file LFNAME into LFINFO.  Read at most
   475    MAX_LFINFO + 1 bytes.  Return the number of bytes read, or -1
   476    (setting errno) on error.  */
   477 
   478 static ptrdiff_t
   479 read_lock_data (char *lfname, char lfinfo[MAX_LFINFO + 1])
   480 {
   481   ptrdiff_t nbytes;
   482 
   483   while ((nbytes = readlinkat (AT_FDCWD, lfname, lfinfo, MAX_LFINFO + 1)) < 0
   484          && errno == EINVAL)
   485     {
   486       int fd = emacs_open (lfname, O_RDONLY | O_NOFOLLOW, 0);
   487       if (0 <= fd)
   488         {
   489           ptrdiff_t read_bytes = emacs_read (fd, lfinfo, MAX_LFINFO + 1);
   490           int read_errno = errno;
   491           if (emacs_close (fd) != 0)
   492             return -1;
   493           errno = read_errno;
   494           return read_bytes;
   495         }
   496 
   497       if (errno != ELOOP)
   498         return -1;
   499 
   500       /* readlinkat saw a non-symlink, but emacs_open saw a symlink.
   501          The former must have been removed and replaced by the latter.
   502          Try again.  */
   503       maybe_quit ();
   504     }
   505 
   506   return nbytes;
   507 }

   509 /* True if errno values are negative.  Although the C standard

   513 /* Nonzero values that are not errno values.  */
   514 enum

   516     /* Another process on this machine owns it.  */
   517     ANOTHER_OWNS_IT = NEGATIVE_ERRNO ? 1 : -1,
   518 
   519     /* This Emacs process owns it.  */
   520     I_OWN_IT = 2 * ANOTHER_OWNS_IT
   521   };

   523 /* Return 0 if nobody owns the lock file LFNAME or the lock is obsolete,
   524    ANOTHER_OWNS_IT if another process owns it
   525      (and set OWNER (if non-null) to info),
   526    I_OWN_IT if the current process owns it,
   527    or an errno value if something is wrong with the locking mechanism.  */
   528 
   529 static int
   530 current_lock_owner (lock_info_type *owner, Lisp_Object lfname)
   531 {
   532   lock_info_type local_owner;
   533   ptrdiff_t lfinfolen;
   534   intmax_t pid, boot_time;
   535   char *at, *dot, *lfinfo_end;
   536 
   537   /* Even if the caller doesn't want the owner info, we still have to
   538      read it to determine return value.  */
   539   if (!owner)
   540     owner = &local_owner;
   541 
   542   /* If nonexistent lock file, all is well; otherwise, got strange error. */
   543   lfinfolen = read_lock_data (SSDATA (lfname), owner->user);
   544   if (lfinfolen < 0)
   545     return errno == ENOENT || errno == ENOTDIR ? 0 : errno;
   546   if (MAX_LFINFO < lfinfolen)
   547     return ENAMETOOLONG;
   548   owner->user[lfinfolen] = 0;
   549 
   550   /* Parse USER@HOST.PID:BOOT_TIME.  If can't parse, return EINVAL.  */
   551   /* The USER is everything before the last @.  */
   552   owner->at = at = memrchr (owner->user, '@', lfinfolen);
   553   if (!at)
   554     return EINVAL;
   555   owner->dot = dot = strrchr (at, '.');
   556   if (!dot)
   557     return EINVAL;
   558 
   559   /* The PID is everything from the last '.' to the ':' or equivalent.  */
   560   if (! c_isdigit (dot[1]))
   561     return EINVAL;
   562   errno = 0;
   563   pid = strtoimax (dot + 1, &owner->colon, 10);
   564   if (errno == ERANGE)
   565     pid = -1;
   566 
   567   /* After the ':' or equivalent, if there is one, comes the boot time.  */
   568   char *boot = owner->colon + 1;
   569   switch (owner->colon[0])
   570     {
   571     case 0:
   572       boot_time = 0;
   573       lfinfo_end = owner->colon;
   574       break;
   575 
   576     case '\357':
   577       /* Treat "\357\200\242" (U+F022 in UTF-8) as if it were ":" (Bug#24656).
   578          This works around a bug in the Linux CIFS kernel client, which can
   579          mistakenly transliterate ':' to U+F022 in symlink contents.
   580          See <https://bugzilla.redhat.com/show_bug.cgi?id=1384153>.  */
   581       if (! (boot[0] == '\200' && boot[1] == '\242'))
   582         return EINVAL;
   583       boot += 2;
   584       FALLTHROUGH;
   585     case ':':
   586       if (! c_isdigit (boot[0]))
   587         return EINVAL;
   588       boot_time = strtoimax (boot, &lfinfo_end, 10);
   589       break;
   590 
   591     default:
   592       return EINVAL;
   593     }
   594   if (lfinfo_end != owner->user + lfinfolen)
   595     return EINVAL;
   596 
   597   Lisp_Object system_name = Fsystem_name ();
   598   /* If `system-name' returns nil, that means we're in a
   599      --no-build-details Emacs, and the name part of the link (e.g.,
   600      .#test.txt -> larsi@.118961:1646577954) is an empty string.  */
   601   if (NILP (system_name))
   602     system_name = build_string ("");
   603   /* Protect against the extremely unlikely case of the host name
   604      containing an @ character.  */
   605   else if (strchr (SSDATA (system_name), '@'))
   606     system_name = CALLN (Ffuncall, intern ("string-replace"),
   607                          build_string ("@"), build_string ("-"),
   608                          system_name);
   609   /* On current host?  */
   610   if (STRINGP (system_name)
   611       && dot - (at + 1) == SBYTES (system_name)
   612       && memcmp (at + 1, SSDATA (system_name), SBYTES (system_name)) == 0)
   613     {
   614       if (pid == getpid ())
   615         return I_OWN_IT;
   616       else if (0 < pid && pid <= TYPE_MAXIMUM (pid_t)
   617                && (kill (pid, 0) >= 0 || errno == EPERM)
   618                && (boot_time == 0
   619                    || (boot_time <= TYPE_MAXIMUM (time_t)
   620                        && within_one_second (boot_time, get_boot_time ()))))
   621         return ANOTHER_OWNS_IT;
   622       /* The owner process is dead or has a strange pid, so try to
   623          zap the lockfile.  */
   624       else
   625         return unlink (SSDATA (lfname)) < 0 ? errno : 0;
   626     }
   627   else
   628     { /* If we wanted to support the check for stale locks on remote machines,
   629          here's where we'd do it.  */
   630       return ANOTHER_OWNS_IT;
   631     }
   632 }
   633 
   634 
   635 /* Lock the lock named LFNAME if possible.
   636    Return 0 in that case.
   637    Return ANOTHER_OWNS_IT if some other process owns the lock, and info about
   638      that process in CLASHER.
   639    Return errno value if cannot lock for any other reason.  */
   640 
   641 static int
   642 lock_if_free (lock_info_type *clasher, Lisp_Object lfname)
   643 {
   644   int err;
   645   while ((err = lock_file_1 (lfname, 0)) == EEXIST)
   646     {
   647       err = current_lock_owner (clasher, lfname);
   648 
   649       /* Return if we locked it, or another process owns it, or it is
   650          a strange error.  */
   651       if (err != 0)
   652         return err == I_OWN_IT ? 0 : err;
   653 
   654       /* We deleted a stale lock or some other process deleted the lock;
   655          try again to lock the file.  */
   656     }
   657 
   658   return err;
   659 }

   661 /* Return the encoded name of the lock file for FN, or nil if none.  */
   662 
   663 static Lisp_Object
   664 make_lock_file_name (Lisp_Object fn)
   665 {
   666   Lisp_Object lock_file_name;
   667 #if defined HAVE_ANDROID && !defined ANDROID_STUBIFY
   668   char *name;
   669 #endif
   670 
   671   fn = Fexpand_file_name (fn, Qnil);
   672 
   673 #if defined HAVE_ANDROID && !defined ANDROID_STUBIFY
   674   /* Files in /assets and /contents can't have lock files on Android
   675      as these directories are fabrications of android.c, and backed by
   676      read only data.  */
   677 
   678   name = SSDATA (fn);
   679 
   680   if (android_is_special_directory (name, "/assets")
   681       || android_is_special_directory (name, "/content"))
   682   return Qnil;
   683 #endif /* defined HAVE_ANDROID && !defined ANDROID_STUBIFY */
   684 
   685   lock_file_name = call1 (Qmake_lock_file_name, fn);
   686 
   687   return !NILP (lock_file_name) ? ENCODE_FILE (lock_file_name) : Qnil;
   688 }

   690 /* lock_file locks file FN,
   691    meaning it serves notice on the world that you intend to edit that file.
   692    This should be done only when about to modify a file-visiting
   693    buffer previously unmodified.
   694    Do not (normally) call this for a buffer already modified,
   695    as either the file is already locked, or the user has already
   696    decided to go ahead without locking.
   697 
   698    When this returns, either the lock is locked for us,
   699    or lock creation failed,
   700    or the user has said to go ahead without locking.
   701 
   702    If the file is locked by someone else, this calls
   703    ask-user-about-lock (a Lisp function) with two arguments,
   704    the file name and info about the user who did the locking.
   705    This function can signal an error, or return t meaning
   706    take away the lock, or return nil meaning ignore the lock.  */
   707 
   708 static Lisp_Object
   709 lock_file (Lisp_Object fn)
   710 {
   711   lock_info_type lock_info;
   712 
   713   /* Don't do locking while dumping Emacs.
   714      Uncompressing wtmp files uses call-process, which does not work
   715      in an uninitialized Emacs.  */
   716   if (will_dump_p ())
   717     return Qnil;
   718 
   719   Lisp_Object lfname = Qnil;
   720   if (create_lockfiles)
   721     {
   722       /* Create the name of the lock-file for file fn */
   723       lfname = make_lock_file_name (fn);
   724       if (NILP (lfname))
   725         return Qnil;
   726     }
   727 
   728   /* See if this file is visited and has changed on disk since it was
   729      visited.  */
   730   Lisp_Object subject_buf = get_truename_buffer (fn);
   731   if (!NILP (subject_buf)
   732       && NILP (Fverify_visited_file_modtime (subject_buf))
   733       && !NILP (Ffile_exists_p (fn))
   734       && !(!NILP (lfname) && current_lock_owner (NULL, lfname) == I_OWN_IT))
   735     call1 (intern ("userlock--ask-user-about-supersession-threat"), fn);
   736 
   737   /* Don't do locking if the user has opted out.  */
   738   if (!NILP (lfname))
   739     {
   740       /* Try to lock the lock.  FIXME: This ignores errors when
   741          lock_if_free returns an errno value.  */
   742       if (lock_if_free (&lock_info, lfname) == ANOTHER_OWNS_IT)
   743         {
   744           /* Someone else has the lock.  Consider breaking it.  */
   745           Lisp_Object attack;
   746           char *dot = lock_info.dot;
   747           ptrdiff_t pidlen = lock_info.colon - (dot + 1);
   748           static char const replacement[] = " (pid ";
   749           int replacementlen = sizeof replacement - 1;
   750           memmove (dot + replacementlen, dot + 1, pidlen);
   751           strcpy (dot + replacementlen + pidlen, ")");
   752           memcpy (dot, replacement, replacementlen);
   753           attack = call2 (intern ("ask-user-about-lock"), fn,
   754                           build_string (lock_info.user));
   755           /* Take the lock if the user said so.  */
   756           if (!NILP (attack))
   757             lock_file_1 (lfname, 1);
   758         }
   759     }
   760   return Qnil;
   761 }

   763 static Lisp_Object
   764 unlock_file (Lisp_Object fn)
   765 {
   766   Lisp_Object lfname = make_lock_file_name (fn);
   767   if (NILP (lfname))
   768     return Qnil;
   769 
   770   int err = current_lock_owner (0, lfname);
   771   if (! (err == 0 || err == ANOTHER_OWNS_IT
   772          || (err == I_OWN_IT
   773              && (unlink (SSDATA (lfname)) == 0 || (err = errno) == ENOENT))))
   774     report_file_errno ("Unlocking file", fn, err);
   775 
   776   return Qnil;
   777 }

   780 unlock_file_handle_error (Lisp_Object err)

   782   call1 (intern ("userlock--handle-unlock-error"), err);
   783   return Qnil;
   784 }

   893 DEFUN ("file-locked-p", Ffile_locked_p, Sfile_locked_p, 1, 1, 0,
   894        doc: /* Return a value indicating whether FILENAME is locked.
   895 The value is nil if the FILENAME is not locked,
   896 t if it is locked by you, else a string saying which user has locked it.  */)
   897   (Lisp_Object filename)

   899 #ifdef MSDOS
   900   return Qnil;
   901 #else

   903   int owner;
   904   lock_info_type locker;

   906   /* If the file name has special constructs in it,
   907      call the corresponding file name handler.  */
   908   Lisp_Object handler;
   909   handler = Ffind_file_name_handler (filename, Qfile_locked_p);
   910   if (!NILP (handler))
   911     {
   912       return call2 (handler, Qfile_locked_p, filename);
   913     }
   914 
   915   Lisp_Object lfname = make_lock_file_name (filename);
   916   if (NILP (lfname))
   917     return Qnil;
   918 
   919   owner = current_lock_owner (&locker, lfname);
   920   switch (owner)
   921     {
   922     case I_OWN_IT: ret = Qt; break;
   923     case ANOTHER_OWNS_IT:
   924       ret = make_string (locker.user, locker.at - locker.user);
   925       break;
   926     case  0: ret = Qnil; break;
   927     default: report_file_errno ("Testing file lock", filename, owner);
   928     }
   929 
   930   return ret;
   931 #endif
   932 }
