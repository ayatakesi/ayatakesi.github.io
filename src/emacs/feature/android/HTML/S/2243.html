    31 /* Selection support on Android is confined to copying and pasting of
    32    plain text and MIME data from the clipboard.  There is no primary
    33    selection.
    34 
    35    While newer versions of Android are supposed to have the necessary
    36    interfaces for transferring other kinds of selection data, doing so
    37    is too complicated, and involves registering ``content providers''
    38    and all kinds of other stuff; for this reason, Emacs does not
    39    support setting the clipboard contents to anything other than plain
    40    text.  */

    91 
    92   FIND_METHOD (set_clipboard, "setClipboard", "([B)V");
    93   FIND_METHOD (owns_clipboard, "ownsClipboard", "()I");

   115 DEFUN ("android-clipboard-owner-p", Fandroid_clipboard_owner_p,
   116        Sandroid_clipboard_owner_p, 0, 0, 0,
   117        doc: /* Return whether or not Emacs owns the clipboard.
   118 Alternatively, return the symbol `lambda' if that could not be
   119 determined.  */)
   120   (void)
   121 {
   122   jint rc;
   123 
   124   if (!android_init_gui)
   125     error ("Accessing clipboard without display connection");
   126 
   127   block_input ();
   128   rc = (*android_java_env)->CallIntMethod (android_java_env,
   129                                            clipboard,
   130                                            clipboard_class.owns_clipboard);
   131   android_exception_check ();
   132   unblock_input ();
   133 
   134   /* If rc is 0 or 1, then Emacs knows whether or not it owns the
   135      clipboard.  If rc is -1, then Emacs does not.  */
   136 
   137   if (rc < 0)
   138     return Qlambda;
   139 
   140   return rc ? Qt : Qnil;
   141 }

   266        Sandroid_get_clipboard_targets, 0, 0, 0,
   267        doc: /* Return a list of data types in the clipboard.
   268 Value is a list of MIME types as strings, each defining a single extra
   269 data type available from the clipboard.  */)
   270   (void)
   271 {
   272   jarray bytes_array;
   273   jbyteArray bytes;
   274   jmethodID method;
   275   size_t length, length1, i;
   276   jbyte *data;
   277   Lisp_Object targets, tem;
   278 
   279   if (!android_init_gui)
   280     error ("No Android display connection!");
   281 
   282   targets = Qnil;
   283   block_input ();
   284   method = clipboard_class.get_clipboard_targets;
   285   bytes_array = (*android_java_env)->CallObjectMethod (android_java_env,
   286                                                        clipboard, method);
   287   android_exception_check ();
   288 
   289   if (!bytes_array)
   290     goto fail;
   291 
   292   length = (*android_java_env)->GetArrayLength (android_java_env,
   293                                                 bytes_array);
   294   for (i = 0; i < length; ++i)
   295     {
   296       /* Retireve the MIME type.  */
   297       bytes
   298         = (*android_java_env)->GetObjectArrayElement (android_java_env,
   299                                                       bytes_array, i);
   300       android_exception_check_nonnull (bytes, bytes_array);
   301 
   302       /* Cons it onto the list of targets.  */
   303       length1 = (*android_java_env)->GetArrayLength (android_java_env,
   304                                                      bytes);
   305       data = (*android_java_env)->GetByteArrayElements (android_java_env,
   306                                                         bytes, NULL);
   307       android_exception_check_nonnull_1 (data, bytes, bytes_array);
   308 
   309       /* Decode the string.  */
   310       tem = make_unibyte_string ((char *) data, length1);
   311       tem = code_convert_string_norecord (tem, Qutf_8, false);
   312       targets = Fcons (tem, targets);
   313 
   314       /* Delete the retrieved data.  */
   315       (*android_java_env)->ReleaseByteArrayElements (android_java_env,
   316                                                      bytes, data,
   317                                                      JNI_ABORT);
   318       ANDROID_DELETE_LOCAL_REF (bytes);
   319     }
   320   unblock_input ();
   321 
   322   ANDROID_DELETE_LOCAL_REF (bytes_array);
   323   return Fnreverse (targets);
   324 
   325  fail:
   326   unblock_input ();
   327   return Qnil;
   328 }

   338 DEFUN ("android-get-clipboard-data", Fandroid_get_clipboard_data,
   339        Sandroid_get_clipboard_data, 1, 1, 0,
   340        doc: /* Return the clipboard data of the given MIME TYPE.
   341 Value is a unibyte string containing the entire contents of the
   342 clipboard, after its owner has converted the data to the given
   343 MIME type.  Value is nil if the conversion fails, or if the data
   344 is not present.
   345 
   346 Value is also nil if the clipboard data consists of a single URL which
   347 does not have any corresponding data.  In that case, use
   348 `android-get-clipboard' instead.  */)
   349   (Lisp_Object type)
   350 {
   351   jlongArray array;
   352   jbyteArray bytes;
   353   jmethodID method;
   354   int fd;
   355   ptrdiff_t rc;
   356   jlong offset, length, *longs;
   357   specpdl_ref ref;
   358   char *buffer, *start;
   359 
   360   if (!android_init_gui)
   361     error ("No Android display connection!");
   362 
   363   /* Encode the string as UTF-8.  */
   364   CHECK_STRING (type);
   365   type = ENCODE_UTF_8 (type);
   366 
   367   /* Then give it to the selection code.  */
   368   block_input ();
   369   bytes = (*android_java_env)->NewByteArray (android_java_env,
   370                                              SBYTES (type));
   371   (*android_java_env)->SetByteArrayRegion (android_java_env, bytes,
   372                                            0, SBYTES (type),
   373                                            (jbyte *) SDATA (type));
   374   android_exception_check ();
   375 
   376   method = clipboard_class.get_clipboard_data;
   377   array = (*android_java_env)->CallObjectMethod (android_java_env,
   378                                                  clipboard, method,
   379                                                  bytes);
   380   android_exception_check_1 (bytes);
   381   ANDROID_DELETE_LOCAL_REF (bytes);
   382 
   383   if (!array)
   384     goto fail;
   385 
   386   longs = (*android_java_env)->GetLongArrayElements (android_java_env,
   387                                                      array, NULL);
   388   android_exception_check_nonnull (longs, array);
   389 
   390   /* longs[0] is the file descriptor.
   391      longs[1] is an offset to apply to the file.
   392      longs[2] is either -1, or the number of bytes to read from the
   393      file.  */
   394   fd = longs[0];
   395   offset = longs[1];
   396   length = longs[2];
   397 
   398   (*android_java_env)->ReleaseLongArrayElements (android_java_env,
   399                                                  array, longs,
   400                                                  JNI_ABORT);
   401   ANDROID_DELETE_LOCAL_REF (array);
   402   unblock_input ();
   403 
   404   /* Now begin reading from longs[0].  */
   405   ref = SPECPDL_INDEX ();
   406   record_unwind_protect_int (close_file_unwind, fd);
   407 
   408   if (length != -1)
   409     {
   410       buffer = xmalloc (MIN (length, PTRDIFF_MAX));
   411       record_unwind_protect_ptr (xfree, buffer);
   412 
   413       rc = emacs_read_quit (fd, buffer,
   414                             MIN (length, PTRDIFF_MAX));
   415 
   416       /* Return nil upon an IO problem.  */
   417       if (rc < 0)
   418         return unbind_to (ref, Qnil);
   419 
   420       /* Return the data as a unibyte string.  */
   421       return unbind_to (ref, make_unibyte_string (buffer, rc));
   422     }
   423 
   424   /* Otherwise, read BUFSIZ bytes at a time.  */
   425   buffer = xmalloc (BUFSIZ);
   426   length = 0;
   427   start = buffer;
   428 
   429   record_unwind_protect_ptr (android_xfree_inside, &buffer);
   430 
   431   /* Seek to the start of the data.  */
   432 
   433   if (offset)
   434     {
   435       if (lseek (fd, offset, SEEK_SET) < 0)
   436         return unbind_to (ref, Qnil);
   437     }
   438 
   439   while (true)
   440     {
   441       rc = emacs_read_quit (fd, start, BUFSIZ);
   442 
   443       if (!INT_ADD_OK (rc, length, &length)
   444           || PTRDIFF_MAX - length < BUFSIZ)
   445         memory_full (PTRDIFF_MAX);
   446 
   447       if (rc < 0)
   448         return unbind_to (ref, Qnil);
   449 
   450       if (rc < BUFSIZ)
   451         break;
   452 
   453       buffer = xrealloc (buffer, length + BUFSIZ);
   454       start = buffer + length;
   455     }
   456 
   457   return unbind_to (ref, make_unibyte_string (buffer, rc));
   458 
   459  fail:
   460   unblock_input ();
   461   return Qnil;
   462 }

   466 void
   467 init_androidselect (void)
   468 {
   469   jobject tem;
   470   jmethodID make_clipboard;
   471 
   472   if (!android_init_gui)
   473     return;
   474 
   475   android_init_emacs_clipboard ();
   476 
   477   make_clipboard = clipboard_class.make_clipboard;
   478   tem
   479     = (*android_java_env)->CallStaticObjectMethod (android_java_env,
   480                                                    clipboard_class.class,
   481                                                    make_clipboard);
   482   if (!tem)
   483     emacs_abort ();
   484 
   485   clipboard = (*android_java_env)->NewGlobalRef (android_java_env, tem);
   486 
   487   if (!clipboard)
   488     emacs_abort ();
   489 
   490   ANDROID_DELETE_LOCAL_REF (tem);
   491 }
