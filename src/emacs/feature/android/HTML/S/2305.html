   171 /* Recursively process events until a ANDROID_CONTEXT_MENU event
   172    arrives.  Then, return the item ID specified in the event in
   173    *ID.  */
   174 
   175 static void
   176 android_process_events_for_menu (int *id)
   177 {
   178   int blocked;
   179 
   180   /* Set menu_event_id to -1; handle_one_android_event will set it to
   181      the event ID upon receiving a context menu event.  This can cause
   182      a non-local exit.  */
   183   x_display_list->menu_event_id = -1;
   184 
   185   /* Unblock input completely.  */
   186   blocked = interrupt_input_blocked;
   187   totally_unblock_input ();
   188 
   189   /* Now wait for the menu event ID to change.  */
   190   while (x_display_list->menu_event_id == -1)
   191     {
   192       /* Wait for events to become available.  */
   193       android_wait_event ();
   194 
   195       /* Process pending signals.  */
   196       process_pending_signals ();
   197 
   198       /* Maybe quit.  This is important because the framework (on
   199          Android 4.0.3) can sometimes fail to deliver context menu
   200          closed events if a submenu was opened, and the user still
   201          needs to be able to quit.  */
   202       maybe_quit ();
   203     }
   204 
   205   /* Restore the input block.  */
   206   interrupt_input_blocked = blocked;
   207 
   208   /* Return the ID.  */
   209   *id = x_display_list->menu_event_id;
   210 }

   241 Lisp_Object
   242 android_menu_show (struct frame *f, int x, int y, int menuflags,
   243                    Lisp_Object title, const char **error_name)
   244 {
   245   jobject context_menu, current_context_menu;
   246   jobject title_string, help_string, temp;
   247   size_t i;
   248   Lisp_Object pane_name, prefix;
   249   const char *pane_string;
   250   specpdl_ref count, count1;
   251   Lisp_Object item_name, enable, def, tem, entry, type, selected;
   252   Lisp_Object help;
   253   jmethodID method;
   254   jobject store;
   255   bool rc;
   256   jobject window;
   257   int id, item_id, submenu_depth;
   258   struct android_dismiss_menu_data data;
   259   struct android_menu_subprefix *subprefix, *temp_subprefix;
   260   struct android_menu_subprefix *subprefix_1;
   261   bool checkmark;
   262   unsigned int serial;
   263 
   264   count = SPECPDL_INDEX ();
   265   serial = ++current_menu_serial;
   266 
   267   block_input ();
   268 
   269   /* Push the first local frame.  */
   270   android_push_local_frame ();
   271 
   272   /* Push the first local frame for the context menu.  */
   273   method = menu_class.create_context_menu;
   274   current_context_menu = context_menu
   275     = (*android_java_env)->CallStaticObjectMethod (android_java_env,
   276                                                    menu_class.class,
   277                                                    method);
   278 
   279   /* Push the second local frame for temporaries.  */
   280   count1 = SPECPDL_INDEX ();
   281   android_push_local_frame ();
   282 
   283   /* Iterate over the menu.  */
   284   i = 0, submenu_depth = 0;
   285 
   286   while (i < menu_items_used)
   287     {
   288       if (NILP (AREF (menu_items, i)))
   289         {
   290           /* This is the start of a new submenu.  However, it can be
   291              ignored here.  */
   292           i += 1;
   293           submenu_depth += 1;
   294         }
   295       else if (EQ (AREF (menu_items, i), Qlambda))
   296         {
   297           /* This is the end of a submenu.  Go back to the previous
   298              context menu.  */
   299           store = current_context_menu;
   300           current_context_menu
   301             = (*android_java_env)->CallObjectMethod (android_java_env,
   302                                                      current_context_menu,
   303                                                      menu_class.parent);
   304           android_exception_check ();
   305 
   306           if (store != context_menu)
   307             ANDROID_DELETE_LOCAL_REF (store);
   308           i += 1;
   309           submenu_depth -= 1;
   310 
   311           if (!current_context_menu || submenu_depth < 0)
   312             {
   313               __android_log_print (ANDROID_LOG_FATAL, __func__,
   314                                    "unbalanced submenu pop in menu_items");
   315               emacs_abort ();
   316             }
   317         }
   318       else if (EQ (AREF (menu_items, i), Qt)
   319                && submenu_depth != 0)
   320         i += MENU_ITEMS_PANE_LENGTH;
   321       else if (EQ (AREF (menu_items, i), Qquote))
   322         i += 1;
   323       else if (EQ (AREF (menu_items, i), Qt))
   324         {
   325           /* This is a new pane.  Switch back to the topmost context
   326              menu.  */
   327           if (current_context_menu != context_menu)
   328             ANDROID_DELETE_LOCAL_REF (current_context_menu);
   329           current_context_menu = context_menu;
   330 
   331           /* Now figure out the title of this pane.  */
   332           pane_name = AREF (menu_items, i + MENU_ITEMS_PANE_NAME);
   333           prefix = AREF (menu_items, i + MENU_ITEMS_PANE_PREFIX);
   334           pane_string = (NILP (pane_name)
   335                          ? "" : SSDATA (pane_name));
   336           if ((menuflags & MENU_KEYMAPS) && !NILP (prefix))
   337             pane_string++;
   338 
   339           /* Add the pane.  */
   340           temp = (*android_java_env)->NewStringUTF (android_java_env,
   341                                                     pane_string);
   342           android_exception_check ();
   343 
   344           (*android_java_env)->CallVoidMethod (android_java_env,
   345                                                current_context_menu,
   346                                                menu_class.add_pane,
   347                                                temp);
   348           android_exception_check ();
   349           ANDROID_DELETE_LOCAL_REF (temp);
   350 
   351           i += MENU_ITEMS_PANE_LENGTH;
   352         }
   353       else
   354         {
   355           item_name = AREF (menu_items, i + MENU_ITEMS_ITEM_NAME);
   356           enable = AREF (menu_items, i + MENU_ITEMS_ITEM_ENABLE);
   357           def = AREF (menu_items, i + MENU_ITEMS_ITEM_DEFINITION);
   358           type = AREF (menu_items, i + MENU_ITEMS_ITEM_TYPE);
   359           selected = AREF (menu_items, i + MENU_ITEMS_ITEM_SELECTED);
   360           help = AREF (menu_items, i + MENU_ITEMS_ITEM_HELP);
   361 
   362           /* This is an actual menu item (or submenu).  Add it to the
   363              menu.  */
   364 
   365           if (i + MENU_ITEMS_ITEM_LENGTH < menu_items_used
   366               && NILP (AREF (menu_items, i + MENU_ITEMS_ITEM_LENGTH)))
   367             {
   368               /* This is a submenu.  Add it.  */
   369               title_string = (!NILP (item_name)
   370                               ? android_build_string (item_name)
   371                               : NULL);
   372               help_string = NULL;
   373 
   374               /* Menu items can have tool tips on Android 26 and
   375                  later.  In this case, set it to the help string.  */
   376 
   377               if (android_get_current_api_level () >= 26
   378                   && STRINGP (help))
   379                 help_string = android_build_string (help);
   380 
   381               store = current_context_menu;
   382               current_context_menu
   383                 = (*android_java_env)->CallObjectMethod (android_java_env,
   384                                                          current_context_menu,
   385                                                          menu_class.add_submenu,
   386                                                          title_string,
   387                                                          help_string);
   388               android_exception_check ();
   389 
   390               if (store != context_menu)
   391                 ANDROID_DELETE_LOCAL_REF (store);
   392 
   393               if (title_string)
   394                 ANDROID_DELETE_LOCAL_REF (title_string);
   395 
   396               if (help_string)
   397                 ANDROID_DELETE_LOCAL_REF (help_string);
   398             }
   399           else if (NILP (def) && menu_separator_name_p (SSDATA (item_name)))
   400             /* Ignore this separator item.  */
   401             ;
   402           else
   403             {
   404               /* Compute the item ID.  This is the index of value.
   405                  Make sure it doesn't overflow.  */
   406 
   407               if (!INT_ADD_OK (0, i + MENU_ITEMS_ITEM_VALUE, &item_id))
   408                 memory_full (i + MENU_ITEMS_ITEM_VALUE * sizeof (Lisp_Object));
   409 
   410               /* Add this menu item with the appropriate state.  */
   411 
   412               title_string = (!NILP (item_name)
   413                               ? android_build_string (item_name)
   414                               : NULL);
   415               help_string = NULL;
   416 
   417               /* Menu items can have tool tips on Android 26 and
   418                  later.  In this case, set it to the help string.  */
   419 
   420               if (android_get_current_api_level () >= 26
   421                   && STRINGP (help))
   422                 help_string = android_build_string (help);
   423 
   424               /* Determine whether or not to display a check box.  */
   425 
   426               checkmark = (EQ (type, QCtoggle)
   427                            || EQ (type, QCradio));
   428 
   429               (*android_java_env)->CallVoidMethod (android_java_env,
   430                                                    current_context_menu,
   431                                                    menu_class.add_item,
   432                                                    (jint) item_id,
   433                                                    title_string,
   434                                                    (jboolean) !NILP (enable),
   435                                                    (jboolean) checkmark,
   436                                                    (jboolean) !NILP (selected),
   437                                                    help_string,
   438                                                    (jboolean) (EQ (type,
   439                                                                    QCradio)));
   440               android_exception_check ();
   441 
   442               if (title_string)
   443                 ANDROID_DELETE_LOCAL_REF (title_string);
   444 
   445               if (help_string)
   446                 ANDROID_DELETE_LOCAL_REF (help_string);
   447             }
   448 
   449           i += MENU_ITEMS_ITEM_LENGTH;
   450         }
   451     }
   452 
   453   /* The menu has now been built.  Pop the second local frame.  */
   454   unbind_to (count1, Qnil);
   455 
   456   /* Now, display the context menu.  */
   457   window = android_resolve_handle (FRAME_ANDROID_WINDOW (f),
   458                                    ANDROID_HANDLE_WINDOW);
   459   rc = (*android_java_env)->CallBooleanMethod (android_java_env,
   460                                                context_menu,
   461                                                menu_class.display,
   462                                                window, (jint) x,
   463                                                (jint) y,
   464                                                (jint) serial);
   465   android_exception_check ();
   466 
   467   if (!rc)
   468     /* This means displaying the menu failed.  */
   469     goto finish;
   470 
   471   /* Make sure the context menu is always dismissed.  */
   472   data.menu = context_menu;
   473   data.window = window;
   474   record_unwind_protect_ptr (android_dismiss_menu, &data);
   475 
   476   /* Next, process events waiting for something to be selected.  */
   477   popup_activated_flag = 1;
   478   android_process_events_for_menu (&id);
   479 
   480   if (!id)
   481     /* This means no menu item was selected.  */
   482     goto finish;
   483 
   484   /* This means the id is invalid.  */
   485   if (id >= ASIZE (menu_items))
   486     goto finish;
   487 
   488   /* Now return the menu item at that location.  */
   489   tem = Qnil;
   490   subprefix = NULL;
   491   record_unwind_protect_ptr (android_free_subprefixes, &subprefix);
   492 
   493   /* Find the selected item, and its pane, to return
   494      the proper value.  */
   495 
   496   prefix = entry = Qnil;
   497   i = 0;
   498   while (i < menu_items_used)
   499     {
   500       if (NILP (AREF (menu_items, i)))
   501         {
   502           temp_subprefix = xmalloc (sizeof *temp_subprefix);
   503           temp_subprefix->last = subprefix;
   504           subprefix = temp_subprefix;
   505           subprefix->subprefix = prefix;
   506 
   507           prefix = entry;
   508           i++;
   509         }
   510       else if (EQ (AREF (menu_items, i), Qlambda))
   511         {
   512           prefix = subprefix->subprefix;
   513           temp_subprefix = subprefix->last;
   514           xfree (subprefix);
   515           subprefix = temp_subprefix;
   516 
   517           i++;
   518         }
   519       else if (EQ (AREF (menu_items, i), Qt))
   520         {
   521           prefix
   522             = AREF (menu_items, i + MENU_ITEMS_PANE_PREFIX);
   523           i += MENU_ITEMS_PANE_LENGTH;
   524         }
   525       /* Ignore a nil in the item list.
   526          It's meaningful only for dialog boxes.  */
   527       else if (EQ (AREF (menu_items, i), Qquote))
   528         i += 1;
   529       else
   530         {
   531           entry = AREF (menu_items, i + MENU_ITEMS_ITEM_VALUE);
   532 
   533           if (i + MENU_ITEMS_ITEM_VALUE == id)
   534             {
   535               if (menuflags & MENU_KEYMAPS)
   536                 {
   537                   entry = list1 (entry);
   538 
   539                   if (!NILP (prefix))
   540                     entry = Fcons (prefix, entry);
   541 
   542                   for (subprefix_1 = subprefix; subprefix_1;
   543                        subprefix_1 = subprefix_1->last)
   544                     if (!NILP (subprefix_1->subprefix))
   545                       entry = Fcons (subprefix_1->subprefix, entry);
   546                 }
   547 
   548               tem = entry;
   549             }
   550           i += MENU_ITEMS_ITEM_LENGTH;
   551         }
   552     }
   553 
   554   unblock_input ();
   555   return unbind_to (count, tem);
   556 
   557  finish:
   558   unblock_input ();
   559   return unbind_to (count, Qnil);
   560 }

   620 static Lisp_Object
   621 android_dialog_show (struct frame *f, Lisp_Object title,
   622                      Lisp_Object header, const char **error_name)
   623 {
   624   specpdl_ref count;
   625   jobject dialog, java_header, java_title, temp;
   626   size_t i;
   627   Lisp_Object item_name, enable, entry;
   628   bool rc;
   629   int id;
   630   jmethodID method;
   631   unsigned int serial;
   632 
   633   /* Generate a unique ID for events from this dialog box.  */
   634   serial = ++current_menu_serial;
   635 
   636   if (menu_items_n_panes > 1)
   637     {
   638       *error_name = "Multiple panes in dialog box";
   639       return Qnil;
   640     }
   641 
   642   /* Do the initial setup.  */
   643   count = SPECPDL_INDEX ();
   644   *error_name = NULL;
   645 
   646   android_push_local_frame ();
   647 
   648   /* Figure out what header to use.  */
   649   java_header = (!NILP (header)
   650                  ? android_build_jstring ("Information")
   651                  : android_build_jstring ("Question"));
   652 
   653   /* And the title.  */
   654   java_title = android_build_string (title);
   655 
   656   /* Now create the dialog.  */
   657   method = dialog_class.create_dialog;
   658   dialog = (*android_java_env)->CallStaticObjectMethod (android_java_env,
   659                                                         dialog_class.class,
   660                                                         method, java_header,
   661                                                         java_title,
   662                                                         (jint) serial);
   663   android_exception_check ();
   664 
   665   /* Delete now unused local references.  */
   666   if (java_header)
   667     ANDROID_DELETE_LOCAL_REF (java_header);
   668   ANDROID_DELETE_LOCAL_REF (java_title);
   669 
   670   /* Create the buttons.  */
   671   i = MENU_ITEMS_PANE_LENGTH;
   672   while (i < menu_items_used)
   673     {
   674       item_name = AREF (menu_items, i + MENU_ITEMS_ITEM_NAME);
   675       enable = AREF (menu_items, i + MENU_ITEMS_ITEM_ENABLE);
   676 
   677       /* Verify that there is no submenu here.  */
   678 
   679       if (NILP (item_name))
   680         {
   681           *error_name = "Submenu in dialog items";
   682           return unbind_to (count, Qnil);
   683         }
   684 
   685       /* Skip past boundaries between buttons on different sides.  The
   686          Android toolkit is too silly to understand this
   687          distinction.  */
   688 
   689       if (EQ (item_name, Qquote))
   690         ++i;
   691       else
   692         {
   693           /* Make sure i is within bounds.  */
   694           if (i > TYPE_MAXIMUM (jint))
   695             {
   696               *error_name = "Dialog box too big";
   697               return unbind_to (count, Qnil);
   698             }
   699 
   700           /* Add the button.  */
   701           temp = android_build_string (item_name);
   702           (*android_java_env)->CallVoidMethod (android_java_env,
   703                                                dialog,
   704                                                dialog_class.add_button,
   705                                                temp, (jint) i,
   706                                                (jboolean) NILP (enable));
   707           android_exception_check ();
   708           ANDROID_DELETE_LOCAL_REF (temp);
   709           i += MENU_ITEMS_ITEM_LENGTH;
   710         }
   711     }
   712 
   713   /* The dialog is now built.  Run it.  */
   714   rc = (*android_java_env)->CallBooleanMethod (android_java_env,
   715                                                dialog,
   716                                                dialog_class.display);
   717   android_exception_check ();
   718 
   719   if (!rc)
   720     quit ();
   721 
   722   /* Wait for the menu ID to arrive.  */
   723   android_process_events_for_menu (&id);
   724 
   725   if (!id)
   726     quit ();
   727 
   728   /* Find the selected item, and its pane, to return
   729      the proper value.  */
   730   i = 0;
   731   while (i < menu_items_used)
   732     {
   733       if (EQ (AREF (menu_items, i), Qt))
   734         i += MENU_ITEMS_PANE_LENGTH;
   735       else if (EQ (AREF (menu_items, i), Qquote))
   736         /* This is the boundary between left-side elts and right-side
   737            elts.  */
   738         ++i;
   739       else
   740         {
   741           entry = AREF (menu_items, i + MENU_ITEMS_ITEM_VALUE);
   742 
   743           if (id == i)
   744             return entry;
   745 
   746           i += MENU_ITEMS_ITEM_LENGTH;
   747         }
   748     }
   749 
   750   return Qnil;
   751 }

   753 Lisp_Object
   754 android_popup_dialog (struct frame *f, Lisp_Object header,
   755                       Lisp_Object contents)
   756 {
   757   Lisp_Object title;
   758   const char *error_name;
   759   Lisp_Object selection;
   760   specpdl_ref specpdl_count = SPECPDL_INDEX ();
   761 
   762   check_window_system (f);
   763 
   764   /* Decode the dialog items from what was specified.  */
   765   title = Fcar (contents);
   766   CHECK_STRING (title);
   767   record_unwind_protect_void (unuse_menu_items);
   768 
   769   if (NILP (Fcar (Fcdr (contents))))
   770     /* No buttons specified, add an "Ok" button so users can pop down
   771        the dialog.  */
   772     contents = list2 (title, Fcons (build_string ("Ok"), Qt));
   773 
   774   list_of_panes (list1 (contents));
   775 
   776   /* Display them in a dialog box.  */
   777   block_input ();
   778   selection = android_dialog_show (f, title, header, &error_name);
   779   unblock_input ();
   780 
   781   unbind_to (specpdl_count, Qnil);
   782   discard_menu_items ();
   783 
   784   if (error_name)
   785     error ("%s", error_name);
   786 
   787   return selection;
   788 }
