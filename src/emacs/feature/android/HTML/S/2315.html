   650       && (X)->face_id == (Y)->face_id                           \
   651       && (X)->padding_p == (Y)->padding_p                       \
   652       && (X)->left_box_line_p == (Y)->left_box_line_p           \
   653       && (X)->right_box_line_p == (Y)->right_box_line_p         \
   654       && (X)->voffset == (Y)->voffset                           \
   655       && (X)->pixel_width == (Y)->pixel_width)

   657 /* Are character codes, faces, padding_ps of glyphs *X and *Y equal?  */
   658 
   659 #define GLYPH_CHAR_AND_FACE_EQUAL_P(X, Y)       \
   660   ((X)->u.ch == (Y)->u.ch                       \
   661    && (X)->face_id == (Y)->face_id              \
   662    && (X)->padding_p == (Y)->padding_p)

   664 /* Fill a character glyph GLYPH.  CODE, FACE_ID, PADDING_P correspond
   665    to the bits defined for the typedef `GLYPH' in lisp.h.  */
   666 
   667 #define SET_CHAR_GLYPH(GLYPH, CODE, FACE_ID, PADDING_P) \
   668      do                                                 \
   669        {                                                \
   670          (GLYPH).u.ch = (CODE);                         \

   743 /* Glyph Matrix.
   744 
   745    Three kinds of glyph matrices exist:
   746 
   747    1. Frame glyph matrices.  These are used for terminal frames whose
   748    redisplay needs a view of the whole screen due to limited terminal
   749    capabilities.  Frame matrices are used only in the update phase
   750    of redisplay.  They are built in update_frame and not used after
   751    the update has been performed.
   752 
   753    2. Window glyph matrices on frames having frame glyph matrices.
   754    Such matrices are sub-matrices of their corresponding frame matrix,
   755    i.e., frame glyph matrices and window glyph matrices share the same
   756    glyph memory, which is allocated in the form of a glyph_pool structure.
   757    Glyph rows in such a window matrix are slices of frame matrix rows.
   758 
   759    3. Free-standing window glyph matrices managing their own glyph
   760    storage.  This form is used in window-based redisplay which
   761    includes variable width and height fonts etc.
   762 
   763    The size of a window's row vector depends on the height of fonts
   764    defined on its frame.  It is chosen so that the vector is large
   765    enough to describe all lines in a window when it is displayed in
   766    the smallest possible character size.  When new fonts are loaded,
   767    or window sizes change, the row vector is adjusted accordingly.  */
   768 
   769 struct glyph_matrix
   770 {
   771   /* The pool from which glyph memory is allocated, if any.  This is
   772      null for frame matrices and for window matrices managing their
   773      own storage.  */
   774   struct glyph_pool *pool;
   775 
   776   /* Vector of glyph row structures.  The row at nrows - 1 is reserved
   777      for the mode line.  */
   778   struct glyph_row *rows;
   779 
   780   /* Number of elements allocated for the vector rows above.  */
   781   ptrdiff_t rows_allocated;
   782 
   783   /* The number of rows used by the window if all lines were displayed
   784      with the smallest possible character height.  */
   785   int nrows;
   786 
   787   /* Origin within the frame matrix if this is a window matrix on a
   788      frame having a frame matrix.  Both values are zero for
   789      window-based redisplay.  */
   790   int matrix_x, matrix_y;
   791 
   792   /* Width and height of the matrix in columns and rows.  */
   793   int matrix_w, matrix_h;
   794 
   795   /* If this structure describes a window matrix of window W,
   796      window_pixel_left is the value of W->pixel_left, window_pixel_top
   797      the value of W->pixel_top, window_height and window_width are width
   798      and height of W, as returned by window_box, and window_vscroll is
   799      the value of W->vscroll at the time the matrix was last adjusted.
   800      Only set for window-based redisplay.  */
   801   int window_pixel_left, window_pixel_top;
   802   int window_height, window_width;
   803   int window_vscroll;
   804 
   805   /* Number of glyphs reserved for left and right marginal areas when
   806      the matrix was last adjusted.  */
   807   int left_margin_glyphs, right_margin_glyphs;
   808 
   809   /* Flag indicating that scrolling should not be tried in
   810      update_window.  This flag is set by functions like try_window_id
   811      which do their own scrolling.  */
   812   bool_bf no_scrolling_p : 1;
   813 
   814   /* True means window displayed in this matrix has a tab line.  */
   815   bool_bf tab_line_p : 1;
   816 
   817   /* True means window displayed in this matrix has a header
   818      line.  */
   819   bool_bf header_line_p : 1;
   820 
   821 #ifdef GLYPH_DEBUG
   822   /* A string identifying the method used to display the matrix.  */
   823   char method[512];
   824 #endif
   825 
   826   /* The buffer this matrix displays.  Set in
   827      mark_window_display_accurate_1.  */
   828   struct buffer *buffer;
   829 
   830   /* Values of BEGV and ZV as of last redisplay.  Set in
   831      mark_window_display_accurate_1.  */
   832   ptrdiff_t begv, zv;
   833 };

   881    window matrices on frames having frame matrices reference slices of
